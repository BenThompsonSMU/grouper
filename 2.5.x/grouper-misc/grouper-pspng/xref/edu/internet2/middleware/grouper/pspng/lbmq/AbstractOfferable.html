3   import java.util.AbstractCollection;
4   import java.util.Collection;
5   import java.util.Queue;
6   
7   /**
8    * This class provides skeletal implementations of some {@link Offerable} operations. The implementations in this class
9    * are appropriate when the base implementation does <em>not</em> allow <tt>null</tt> elements. Method {@link #add add}
10   * is based on {@link #offer offer}, but throws exceptions instead of indicating failure via <tt>false</tt> or
11   * <tt>null</tt> returns.
12   *
13   * <p>
14   * An <tt>Offerable</tt> implementation that extends this class must minimally define a method {@link Queue#offer} which
15   * does not permit insertion of <tt>null</tt> elements, along with methods {@link Collection#size}, and
16   * {@link Collection#iterator}. Typically, additional methods will be overridden as well. If these requirements cannot
17   * be met, consider instead subclassing {@link AbstractCollection}.
18   * 
19   * @param <E>
20   *            the type of elements held in this collection
21   */
22  public abstract class AbstractOfferable<E> extends AbstractCollection<E> implements Offerable<E> {
23  
24      public boolean add(E e) {
25          if (offer(e))
26              return true;
27          else
28              throw new IllegalStateException("Queue full");
29      }
30  
31      /**
32       * Adds all of the elements in the specified collection to this queue. Attempts to addAll of a queue to itself
33       * result in <tt>IllegalArgumentException</tt>. Further, the behavior of this operation is undefined if the
34       * specified collection is modified while the operation is in progress.
35       *
36       * <p>
37       * This implementation iterates over the specified collection, and adds each element returned by the iterator to
38       * this queue, in turn. A runtime exception encountered while trying to add an element (including, in particular, a
39       * <tt>null</tt> element) may result in only some of the elements having been successfully added when the associated
40       * exception is thrown.
41       *
42       * @param c
43       *            collection containing elements to be added to this queue
44       * @return <tt>true</tt> if this queue changed as a result of the call
45       * @throws ClassCastException
46       *             if the class of an element of the specified collection prevents it from being added to this queue
47       * @throws NullPointerException
48       *             if the specified collection contains a null element and this queue does not permit null elements, or
49       *             if the specified collection is null
50       * @throws IllegalArgumentException
51       *             if some property of an element of the specified collection prevents it from being added to this
52       *             queue, or if the specified collection is this queue
53       * @throws IllegalStateException
54       *             if not all the elements can be added at this time due to insertion restrictions
55       * @see #add(Object)
56       */
57      public boolean addAll(Collection<? extends E> c) {
58          if (c == null)
59              throw new NullPointerException();
60          if (c == this)
61              throw new IllegalArgumentException();
62          boolean modified = false;
63          for (E e : c)
64              if (add(e))
65                  modified = true;
66          return modified;
67      }
