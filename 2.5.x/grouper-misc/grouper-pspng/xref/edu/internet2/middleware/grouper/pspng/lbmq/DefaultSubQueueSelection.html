6   package edu.internet2.middleware.grouper.pspng.lbmq;

8   import java.util.ArrayList;
9   
10  /**
11   * Chooses the next queue to be used from the highest priority priority group.
12   * If no queue is found it searches the lower priority groups and so on until
13   * it finds a queue.
14   */
15  public class DefaultSubQueueSelection<K, E> implements LinkedBlockingMultiQueue.SubQueueSelection<K, E> {
16  
17      private ArrayList<LinkedBlockingMultiQueue<K, E>.PriorityGroup> priorityGroups;
18  
19      @Override
20      public LinkedBlockingMultiQueue.SubQueue getNext() {
21          for (int i = 0; i < priorityGroups.size(); i++) {
22              LinkedBlockingMultiQueue.SubQueue subQueue = priorityGroups.get(i).getNextSubQueue();
23              if (subQueue != null) {
24                  return subQueue;
25              }
26          }
27          return null;
28      }
29  
30      @Override
31      public E peek() {
32          // assert takeLock.isHeldByCurrentThread();
33          for (int i = 0; i < priorityGroups.size(); i++) {
34              E dequed = priorityGroups.get(i).peek();
35              if (dequed != null) {
36                  return dequed;
37              }
38          }
39          return null;
40      }
41  
42      @Override
43      public void setPriorityGroups(ArrayList<LinkedBlockingMultiQueue<K, E>.PriorityGroup> priorityGroups) {
44          this.priorityGroups = priorityGroups;
45      }
46  
47  }
