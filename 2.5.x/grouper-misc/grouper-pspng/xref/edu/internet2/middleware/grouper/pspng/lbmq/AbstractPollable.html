3   import java.util.NoSuchElementException;
4   import java.util.Queue;
5   
6   /**
7    * This class provides skeletal implementations of some {@link Pollable} operations. The implementations in this class
8    * are appropriate when the base implementation does <em>not</em> allow <tt>null</tt> elements. Methods {@link #remove
9    * remove}, and {@link #element element} are based on {@link #poll poll}, and {@link #peek peek}, respectively, but
10   * throw exceptions instead of indicating failure via <tt>false</tt> or <tt>null</tt> returns.
11   * 
12   * <p>
13   * A <tt>Pollable</tt> implementation that extends this class must minimally define methods {@link Queue#peek} and
14   * {@link Queue#poll}. Typically, additional methods will be overridden as well.
15   *
16   * @param <E>
17   *            the type of elements held in this collection
18   */
19  public abstract class AbstractPollable<E> implements Pollable<E> {
20  
21      /**
22       * Retrieves and removes the head of this queue. This method differs from {@link #poll poll} only in that it throws
23       * an exception if this queue is empty.
24       *
25       * <p>
26       * This implementation returns the result of <tt>poll</tt> unless the queue is empty.
27       *
28       * @return the head of this queue
29       * @throws NoSuchElementException
30       *             if this queue is empty
31       */
32      public E remove() {
33          E x = poll();
34          if (x != null)
35              return x;
36          else
37              throw new NoSuchElementException();
38      }
39  
40      /**
41       * Retrieves, but does not remove, the head of this queue. This method differs from {@link #peek peek} only in that
42       * it throws an exception if this queue is empty.
43       *
44       * <p>
45       * This implementation returns the result of <tt>peek</tt> unless the queue is empty.
46       *
47       * @return the head of this queue
48       * @throws NoSuchElementException
49       *             if this queue is empty
50       */
51      public E element() {
52          E x = peek();
53          if (x != null)
54              return x;
55          else
56              throw new NoSuchElementException();
57      }
58  
59  }
