12  /**
13   * This class can be used to track activities that are underway, and report
14   * when they are "tardy" according to expectations. You can think of this as
15   * an alternative to jstack that automatically logs what the system is doing
16   * when things are taking longer than expected.
17   *
18   * Without an AOP library, the easiest way to use this class is to create
19   * try-resource blocks around different steps of a process:
20   * try (WatchedTask t=new WatchedTask("FullSync", "FullSync of group %s", groupName) {
21   *   ... code ...
22   * }
23   *
24   * If "...code..." takes too long, then information about the WatchedTask and
25   * its children will be logged.
26   *
27   * Status: Early development, needs the ability to configure timing expectations.
28   *
29   * This is abstract for now to prevent its use. In the end, it will not be abstract.
30   */
31  abstract class WatchedTask implements AutoCloseable{
32    private static final Logger LOG = LoggerFactory.getLogger(WatchedTask.class);
33  
34  
35    // Each thread keeps a sequence for its tasks
36    private static ThreadLocal<AtomicInteger> taskSequences = new ThreadLocal<AtomicInteger>() {
37      @Override
38      protected AtomicInteger initialValue() {
39        return new AtomicInteger();
40      }
41    };
42  
43    // A thread that wakes up (via Timers) and reports about tardy tasks
44    private final static ScheduledExecutorService timerScheduler
45            = Executors.newScheduledThreadPool(1);
46  
47    // Each thread keeps its call stack
48    private static ThreadLocal<Stack<WatchedTask>> taskStacks = new ThreadLocal<Stack<WatchedTask>>() {
49      @Override
50      protected Stack<WatchedTask> initialValue() {
51        Stack<WatchedTask> result = new Stack<>();
52  
53        return result;
54      }
55    };
56  
57    // What stack are is this stack within
58    private final Stack<WatchedTask> ourStack = taskStacks.get();
59  
60    private final String taskId = String.format("%s-%s",
61            PspUtils.getThreadId(), PspUtils.getIdString(taskSequences.get().incrementAndGet()) );
62  
63    // What Task started us
64    private WatchedTask parentTask;
65  
66    // Statistics are accumulated by shortName
67    // Warning Durations are looked up by shortName
68    private String shortName;
69  
70    private String detailedName;
71  
72    private final Date startDate = new Date();
73    private ScheduledFuture timer;
74  
75    private double warningDuration_secs;
76    private AtomicReference<Date> finishTime = new AtomicReference<>();
77  
78  
79    // Keep track of the number of times each subtask has been started,
80    // indexed by shortName
81    private Map<String, AtomicInteger> subTaskCounts = new HashMap<String, AtomicInteger>() {
82      @Override
83      public AtomicInteger get(Object key) {
84        // Default value for missing key is AtomicInteger(0)
85        if ( !containsKey(key) ) {
86          put((String)key, new AtomicInteger(0));
87        }
88        return super.get(key);
89      }
90    };
91  
92  
93    // Used to create a root watchdog
94    private WatchedTask() {
95      shortName = detailedName = null;
96      parentTask = null;
97      warningDuration_secs = -1;
98    }
99  
100 
101   public WatchedTask(String shortName, String detailedNameFormat, Object... detailedNameArgs) {
102     if ( taskStacks.get().isEmpty() ) {
103       taskStacks.get().push(new WatchedTask() {});
104     }
105 
106     parentTask = taskStacks.get().peek();
107     this.shortName = shortName;
108     if ( detailedNameFormat == null ) {
109       this.detailedName = shortName;
110     } else {
111       this.detailedName = String.format(detailedNameFormat, detailedNameArgs);
112     }
113 
114     parentTask.subTaskCounts.get(shortName).incrementAndGet();
115 
116     this.warningDuration_secs = 0.25;
117 
118     if ( warningDuration_secs > 0 ) {
119       timer = scheduleFutureWarning(warningDuration_secs);
120     }
121   }
122 
123 
124   public WatchedTask(String shortName) {
125     this(shortName, null);
126   }
127 
128 
129   private ScheduledFuture scheduleFutureWarning(double timeUntilWarning_secs) {
130     ScheduledFuture result = timerScheduler.schedule(new Runnable() {
131                                         @Override
132                                         public void run() {
133                                           synchronized (ourStack) {
134                                             // Have we finished in the interval between being run
135                                             // by scheduler and synchronizing?
136                                             if ( finishTime.get() != null ) {
137                                               warningTimerExpired();
138                                               timer = scheduleFutureWarning(60);
139                                             }
140                                           }
141                                         }
142                                       },
143             (int)(1000*warningDuration_secs), TimeUnit.MILLISECONDS);
144     return result;
145   }
146 
147   private void warningTimerExpired() {
148     LOG.warn("Task is taking longer than expected: {} has taken {} vs expected {} secs",
149             detailedName, PspUtils.formatElapsedTime(startDate, null), warningDuration_secs);
150     LinkedList<WatchedTask> ourChildrenTasks = getChildrenTasks();
151 
152     ourChildrenTasks.add(0, this);
153   }
154 
155   private LinkedList<WatchedTask> getChildrenTasks() {
156     // Find the stack of tasks from us to our children
157     LinkedList<WatchedTask> ourChildrenTasks = new LinkedList<>();
158 
159     synchronized (ourStack) {
160       WatchedTask task = ourStack.peek();
161 
162       while ( task != this && task != null ) {
163         ourChildrenTasks.add(0, task);
164       }
165     }
166 
167     // We should either have no children or our first child should have us as a parent
168     if ( ourChildrenTasks.size()==0 || ourChildrenTasks.get(0).parentTask!=this ) {
169       LOG.error("Task Watchdog: Invalid task chain... active tasks did not lead to us");
170     }
171     return ourChildrenTasks;
172   }
173 
174 
175   @Override
176   public void close() {
177     finish();
178   }
179 
180   public void finish() {
181     // We synchronize to make sure scheduled task isn't running, otherwise
182     // the scheduled task might reschedule a warning timer after we cancel it
183     synchronized (ourStack) {
184       if (finishTime.get() != null) {
185         return;
186       }
187 
188       // Finish our children if they haven't finished themselves
189       LinkedList<WatchedTask> ourChildren = getChildrenTasks();
190       Iterator<WatchedTask> childIterator = ourChildren.descendingIterator();
191       while (childIterator.hasNext()) {
192         WatchedTask child = childIterator.next();
193 
194         LOG.warn("Task Watchdog: Task not finished properly: {}", child);
195         child.finish();
196       }
197 
198       finishTime.set(new Date());
199       if (timer != null) {
200         timer.cancel(false);
201         timer = null;
202       }
203 
204       if ( ourStack.peek() == this ) {
205         ourStack.pop();
206       } else {
207         LOG.warn("Task Watchdog: Expected {} to be the top task, but {} was instead",
208                 this, ourStack.peek());
209       }
210     }
211   }
