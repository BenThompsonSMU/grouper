3   import org.joda.time.Duration;
4   import org.joda.time.Instant;
5   import org.slf4j.Logger;
6   
7   /**
8    * This classs helps report how much of a long task has been completed
9    * and how long the rest is expected to take
10   */
11  public class ProgressMonitor {
12    final Logger LOG;
13    final boolean minimalLogging;
14    final String taskLabel;
15    final int totalAmountOfWorkExpected;
16    int amountOfWorkCompleted;
17    final Instant timeStarted = Instant.now();
18    Instant timeFinished;
19    Instant lastTimeUpdatePrinted;
20    final Duration logUpdateInterval;
21  
22    int lastLog_amountOfWorkCompleted=0;
23  
24    // Work that has been started but not completed
25    int amountOfWorkBeingWorkedOn=0;
26  
27    public ProgressMonitor(int totalAmountOfWorkExpected, Logger LOG, boolean minimalLogging, int loggingInterval_secs, String taskLabelFormat, Object... taskLabelArgs) {
28      this.LOG = LOG;
29      this.minimalLogging=minimalLogging;
30      this.logUpdateInterval = Duration.standardSeconds(loggingInterval_secs);
31      this.taskLabel = String.format(taskLabelFormat, taskLabelArgs);
32      this.totalAmountOfWorkExpected = totalAmountOfWorkExpected;
33  
34      if ( !minimalLogging ) {
35        if (totalAmountOfWorkExpected >= 0) {
36          LOG.info("{} Started: TotalWorkExpected={}", taskLabel, totalAmountOfWorkExpected);
37        } else {
38          LOG.info("{} Started", taskLabel);
39        }
40      }
41  
42      lastTimeUpdatePrinted = Instant.now();
43    }
44  
45    public ProgressMonitor(Logger LOG, boolean minimalLogging, int loggingInterval_secs, String taskLabelFormat, Object... taskLabelArgs) {
46      this(0, LOG, minimalLogging, loggingInterval_secs, taskLabelFormat, taskLabelArgs);
47    }
48  
49  
50    public void startWork(int amountOfWork) {
51      // Have we been told about other work that had started
52      if ( amountOfWorkBeingWorkedOn > 0 ) {
53        // workCompleted() will also mark amountOfWorkBeingWordOn as complete,
54        // so we have to zero it before calling it
55        int temp = amountOfWorkBeingWorkedOn;
56        amountOfWorkBeingWorkedOn=0;
57        workCompleted(temp);
58      }
59  
60      amountOfWorkBeingWorkedOn = amountOfWork;
61    }
62  
63    public void workCompleted(int amountOfWork) {
64      amountOfWorkCompleted += amountOfWorkBeingWorkedOn;
65      amountOfWorkBeingWorkedOn=0;
66      amountOfWorkCompleted += amountOfWork;
67  
68      logProgressIfNecessary();
69    }
70  
71    public void completelyDone(String completionLabel) {
72      // Only handle completion once
73      if ( timeFinished!=null ) {
74        return;
75      }
76      timeFinished = Instant.now();
77      Duration overallTaskDuration = new Duration(timeStarted, timeFinished);
78  
79      // Log when either not logging minimally or when we've run long enough
80      if ( !minimalLogging || overallTaskDuration.isLongerThan(logUpdateInterval) ) {
81        LOG.info(String.format("%s Completed (%s): %d items in %s (%.1f items/minute)",
82                taskLabel, completionLabel, amountOfWorkCompleted,
83                PspUtils.formatElapsedTime(overallTaskDuration),
84                1.0 * amountOfWorkCompleted / overallTaskDuration.getMillis() * 1000 * 60));
85      }
86    }
87  
88  
89    protected void logProgressIfNecessary() {
90      Duration howLongSinceProgressWasPrinted = new Duration(lastTimeUpdatePrinted, Instant.now());
91      if ( howLongSinceProgressWasPrinted.isLongerThan(logUpdateInterval) ) {
92        Duration howLongSinceStarted = new Duration(timeStarted, Instant.now());
93  
94        long workDone_sinceLastLog = amountOfWorkCompleted - lastLog_amountOfWorkCompleted;
95        lastLog_amountOfWorkCompleted=amountOfWorkCompleted;
96  
97        double workDonePerMinute_sinceLastLog = 60.0*1000.0*workDone_sinceLastLog/howLongSinceProgressWasPrinted.getMillis();
98        double workDonePerMinute_overall = 60.0*1000.0*amountOfWorkCompleted/howLongSinceStarted.getMillis();
99  
100       if ( totalAmountOfWorkExpected>=0 ) {
101         long workLeftToDo = totalAmountOfWorkExpected - amountOfWorkCompleted;
102         long estimatedTimeLeft_seconds = (long) (60.0*workLeftToDo/workDonePerMinute_sinceLastLog);
103 
104         Duration estimatedTimeLeft = Duration.standardSeconds(estimatedTimeLeft_seconds);
105 
106         LOG.info(String.format("%s Progress: %d of %d (%s%%) in %s (%s items/min overall, %s items/min recently). %d work to go, ETA: %s",
107                 taskLabel, amountOfWorkCompleted, totalAmountOfWorkExpected,
108                 PspUtils.formatWithSignificantDigits(100.0 * amountOfWorkCompleted / totalAmountOfWorkExpected, 2),
109                 PspUtils.formatElapsedTime(howLongSinceStarted),
110                 PspUtils.formatWithSignificantDigits(workDonePerMinute_overall, 2),
111                 PspUtils.formatWithSignificantDigits(workDonePerMinute_sinceLastLog, 2),
112                 workLeftToDo,
113                 PspUtils.formatElapsedTime(estimatedTimeLeft)));
114       }
115       else {
116         // Log what we know when we don't know how much work to expect
117 
118         LOG.info(String.format("%s Progress: %d in %s (%s items/min overall, %s items/min recently).",
119                 taskLabel, amountOfWorkCompleted,
120                 PspUtils.formatElapsedTime(howLongSinceStarted),
121                 PspUtils.formatWithSignificantDigits(workDonePerMinute_overall, 2),
122                 PspUtils.formatWithSignificantDigits(workDonePerMinute_sinceLastLog, 2)));
123       }
124 
125       lastTimeUpdatePrinted = Instant.now();
126     }
127   }
