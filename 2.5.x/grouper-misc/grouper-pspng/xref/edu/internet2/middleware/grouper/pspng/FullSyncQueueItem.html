224   public String toJson() {
225     // Manually building map that will be JSONed. Jackson could do this, but that isn't
226     // included in grouper right now, and I don't want to add such a dependency in this upcoming
227     // patch.
228     // Also, net.sf.json seems to be bean-specific and not able to JSONify general objects,
229     // nor to control the attribute names.
230     Map<String, Object> result = new HashMap<>();
231 
232     result.put(ID_PROPERTY, id);
233     result.put(PROVISIONER_NAME_PROPERTY, provisionerName);
234     result.put(COMMAND_PROPERTY, command.name());
235 
236     result.put(SOURCE_PROPERTY, source);
237     result.put(EXTERNAL_REFERENCE, externalReference);
238     result.put(REASON_PROPERTY, reason);
239 
240     result.put(GROUP_PROPERTY, groupName);
241 
242     result.put(FIRST_QUEUED_JEPOCH_PROPERTY, firstQueuedDate.getMillis());
243     result.put(ASOF_JEPOCH_PROPERTY, asofDate.getMillis());
244     result.put(MOST_RECENT_QUEUED_JEPOCH_PROPERTY, mostRecentQueuedDate.getMillis());
245 
246     if ( wakeTimeDate!=null ) {
247       result.put(WAKE_TIME_JEPOCH_PROPERTY, wakeTimeDate.getMillis());
248     }
249 
250     if ( wasSuccessful!= null ) {
251       result.put(WAS_SUCCESSFUL_PROPERTY, wasSuccessful ? "true" : "false");
252     }
253 
254     result.put(REPLY_QUEUE, replyQueue);
255     result.put(RETRY_COUNT_PROPERTY, retryCount);
256 
257     if ( willBeRetried!=null ) {
258       result.put(WILL_BE_RETRIED_PROPERTY, willBeRetried ? "true" : "false");
259     }
260 
261     result.put(PROCESSING_RESULT_MESSAGE_PROPERTY, processingResultMessage);
262 
263     // This jsonutils invocation wraps the result in a "<Class>": <result> envelope
264     // which gets in my way
265     // return JsonUtils.jsonConvertTo(result);
266     // So this is a partial copy of the jsonConvertTo method...
267 
268     JsonConfig jsonConfig = new JsonConfig();
269     jsonConfig.setJsonPropertyFilter(new PropertyFilter() {
270       public boolean apply(Object source, String name, Object value) {
271         return value == null;
272       }
273     });
274     JSONObject jsonObject = JSONObject.fromObject(result, jsonConfig);
275     String json = jsonObject.toString();
276     return json;
277   }

304   public void processingCompletedSuccessfully(String messageFormat, Object... messageArgs) {
305       // Only act the first time we're told this was completed
306       if ( stats.processingCompletedTime != null ) {
307           return;
308       }
309 
310       if (StringUtils.isNotEmpty(messageFormat))
311         processingResultMessage = String.format("Successful: " + messageFormat, messageArgs);
312       else
313         processingResultMessage = "Successful";
314 
315       wasSuccessful = true;
316 
317       processingCompleted();
318   }

320   private void processingCompleted() {
321     stats.processingCompletedTime = new Date();
322     startStep(null);
323     processingCompletionTime = DateTime.now();
324     LOG.info("FullSync Item done ({}). Stats: {}/{}: {}",
325             processingResultMessage, stats, getProcessingTimeBreakdown(), this);
326 
327     acknowledgeMessage();
328   }

330   public void processingCompletedUnsuccessfully(boolean willBeRetried, String messageFormat, Object... messageArgs) {
331     // Only act the first time we're told this was completed
332     if ( stats.processingCompletedTime != null ) {
333         return;
334     }
335     this.willBeRetried=willBeRetried;
336     processingResultMessage = String.format(messageFormat, messageArgs);
337 
338     wasSuccessful = false;

385   public String getProcessingTimeBreakdown() {
386     StringBuilder result = new StringBuilder();
387 
388     Duration totalProcessingTime;
389     if ( mostRecentDequeueTime == null )
390       return "NotYetProcessed";
391 
392     if ( processingCompletionTime==null ) {
393       totalProcessingTime = new Duration(mostRecentDequeueTime, Instant.now());
394       result.append(String.format("ProcTime(SoFar): %s", PspUtils.formatElapsedTime(totalProcessingTime)));
395     } else {
396       totalProcessingTime = new Duration(mostRecentDequeueTime, processingCompletionTime);
397       result.append(String.format("ProcTime: %s", PspUtils.formatElapsedTime(totalProcessingTime)));
398     }
399 
400     if (processingStepTimingMeasurements.size()>0 && totalProcessingTime.getMillis()>0) {
401       result.append(" Timing breakdown: ");
402 
403       for (Map.Entry<String, Duration> processingStep : processingStepTimingMeasurements.entrySet()) {
404         result.append(String.format("%s=%d%%/", processingStep.getKey(), (int) (100 * processingStep.getValue().getMillis() / totalProcessingTime.getMillis())));
405       }
406     }
407 
408     return result.toString();
409   }

412   public void incrementRetryCount() {
413     retryCount++;
414     mostRecentQueuedDate = new DateTime();
415 
416     // Push event out 10 seconds more each retry, up to 15minutes
417     int sleepTime_secs = Math.min(10*retryCount, 900);
418 
419     wakeTimeDate = new DateTime().withDurationAdded(1000*sleepTime_secs, 1);
420   }
