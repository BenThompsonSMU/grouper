55    /**
56     * This adds/removes values from the given user. This is a helper class to package
57     * up the modification into the necessary LDAP ModifyRequest.
58     * 
59     * @param ldapUser
60     * @param modType
61     * @param valuesToChange
62     */
63    protected void scheduleUserModification(LdapUser ldapUser, AttributeModificationType modType, Collection<String> valuesToChange) {
64      String attributeName = config.getProvisionedAttributeName();
65      
66      for ( String value : valuesToChange )
67        // ADD/REMOVE <value> to/from <attribute> of <user>
68        LOG.info("Will change LDAP: {} {} {} {} of {}", 
69            new Object[] {modType, value,
70            modType == AttributeModificationType.ADD ? "to" : "from",
71            attributeName, ldapUser});
72        
73      scheduleLdapModification(
74          new ModifyRequest(
75              ldapUser.getLdapObject().getDn(), 
76              new AttributeModification(modType, new LdapAttribute(attributeName, valuesToChange.toArray(new String[0])))));
77    }

80    @Override
81    protected void addMembership(GrouperGroupInfo grouperGroupInfo, LdapGroup ldapGroup,
82        Subject subject, LdapUser ldapUser) throws PspException {
83  
84      if ( ldapUser == null ) {
85        LOG.warn("{}: Skipping addMembership: LdapUser does not exist for subject {}", getDisplayName(), subject.getId());
86        return;
87      }
88      
89      String attributeValue = getAttributeValueForGroup(grouperGroupInfo);
90      
91      scheduleUserModification(ldapUser, AttributeModificationType.ADD, Arrays.asList(attributeValue));
92    }

104 
105     scheduleUserModification(ldapUser, AttributeModificationType.REMOVE, Arrays.asList(attributeValue));

108   @Override
109   protected boolean doFullSync(GrouperGroupInfo grouperGroupInfo, LdapGroup ldapGroup,
110       Set<Subject> correctSubjects, Map<Subject, LdapUser> tsUserMap, Set<LdapUser> correctTSUsers,
111       JobStatistics stats)
112       throws PspException {
113 
114     stats.totalCount.set(correctTSUsers.size());
115     String attributeName = config.getProvisionedAttributeName();
116     String attributeValue = getAttributeValueForGroup(grouperGroupInfo);
117 
118     LOG.info("Fetching all ldap users that have {}={}", attributeName, attributeValue);
119     Set<String> currentMatches_dnList = getLdapSystem().performLdapSearchRequest_returningValuesOfAnAttribute(
120             correctSubjects.size(), config.getUserSearchBaseDn(), SearchScope.SUBTREE,
121         "dn", attributeName + "={0}",
122         attributeValue);
123 
124 
125     Set<String> correctMembers_dnList = new HashSet<>(correctTSUsers.size());
126     for(LdapUser ldapUser : correctTSUsers) {
127       correctMembers_dnList.add(ldapUser.dn_lc);
128     }
129 
130 
131 
132     LOG.info("{}: Full-sync comparison for {}: Target-subject count: Correct/Actual: {}/{}",
133       new Object[] {getDisplayName(), grouperGroupInfo, correctTSUsers.size(), currentMatches_dnList.size()});
134 
135     LOG.trace("{}: Full-sync comparison: Correct: {}", getDisplayName(), correctMembers_dnList);
136     LOG.trace("{}: Full-sync comparison: Actual: {}", getDisplayName(), currentMatches_dnList);
137 
138 
139     LOG.info("{}: Finding users that need attribute removed", getDisplayName());
140     // EXTRA MATCHES = CURRENT_MATCHES - CORRECT_MATCHES
141     Set<String> extraMatches_dnList = new HashSet<String>(currentMatches_dnList);
142     extraMatches_dnList.removeAll(correctMembers_dnList);
143 
144     LOG.info("{}: There are {} users that need the attribute removed", getDisplayName(), extraMatches_dnList.size());
145     stats.deleteCount.set(extraMatches_dnList.size());
146     
147     for (String extraMatch_dn : extraMatches_dnList) {
148       getLdapSystem().performLdapModify(
149               new ModifyRequest(
150                       extraMatch_dn,
151                       new AttributeModification(
152                               AttributeModificationType.REMOVE,
153                               new LdapAttribute(config.getProvisionedAttributeName(), attributeValue))),
154               true);
155     }
156 
157     LOG.info("{}: Finding users that need attribute added", getDisplayName());
158     // MISSING MATCHES = CORRECT_MATCHES - CURRENT_MATCHES
159     Set<String> missingMatches_dnList = new HashSet<String>(correctMembers_dnList);
160     missingMatches_dnList.removeAll(currentMatches_dnList);
161 
162     LOG.info("{}: There are {} users that need the attribute added", getDisplayName(), missingMatches_dnList.size());
163     stats.insertCount.set(missingMatches_dnList.size());
164 
165     for (String missingMatch_dn : missingMatches_dnList) {
166       getLdapSystem().performLdapModify(
167               new ModifyRequest(
168                       missingMatch_dn,
169                       new AttributeModification(
170                               AttributeModificationType.ADD,
171                               new LdapAttribute(config.getProvisionedAttributeName(), attributeValue))),
172               true);
173 
174     }
175 
176     LOG.info("{}: Brief full-sync summary: Correct={}, Current={}, Extra={}, Missing={}", 
177         new Object[] {getDisplayName(), correctSubjects.size(), currentMatches_dnList.size(),
178         extraMatches_dnList.size(), missingMatches_dnList.size()});
179 
180     return extraMatches_dnList.size()>0 || missingMatches_dnList.size()>0;
181   }

188   @Override
189 	protected void doFullSync_cleanupExtraGroups(JobStatistics stats) throws PspException {
190 
191       String attribute = config.getProvisionedAttributeName();
192 
193       // This will either be
194       //   null or empty: No cleanup will occur
195       //   *: Attribute is entirely determined by grouper and will be completely scrubbed
196       //   a true prefix: Only values that start with that prefix will be inspected/scrubbed
197 
198       String allValuesPrefix = config.getAllProvisionedValuesPrefix();
199       if ( StringUtils.isEmpty(allValuesPrefix) ) {
200         LOG.error("{}: Unable to cleanup extra groups without allProvisionedValuesPrefix being defined. Set the prefix to * if you want grouper to fully control the attribute.", getDisplayName());
201         return;
202       }
203 
204       String allValuesLdapFilter;
205       Pattern allValuesPattern;
206 
207       if ( allValuesPrefix.equals("*") ) {
208           allValuesLdapFilter = String.format("%s=*", attribute);
209           allValuesPattern = Pattern.compile(".*");
210       }
211       else {
212           allValuesLdapFilter = String.format("%s=%s*", attribute, PspJexlUtils.escapeLdapFilter(allValuesPrefix));
213           allValuesPattern = Pattern.compile(Pattern.quote(allValuesPrefix) + ".*");
214       }
215 
216       LOG.debug("{}: Looking for all grouper-sourced values of {}.", getDisplayName(), attribute);
217 
218       Set<String> allValuesOfAttribute = new TreeSet<>(String.CASE_INSENSITIVE_ORDER);
219 
220       List<LdapObject> usersWithGrouperValues 
221         = getLdapSystem().performLdapSearchRequest(-1, config.getUserSearchBaseDn(), SearchScope.SUBTREE,
222             Arrays.asList(attribute), allValuesLdapFilter);
223       
224       // We're going to go through all the values of all the ldap objects. 
225       // We're going to save all those values that come from grouper (because they match 'pattern')
226       Set<String> grouperSourcedValuesUsedInTargetSystem = new TreeSet<String>(String.CASE_INSENSITIVE_ORDER);
227       for ( LdapObject user : usersWithGrouperValues ) {
228         Collection<String> values = user.getStringValues(attribute);
229         allValuesOfAttribute.addAll(values);
230 
231         for ( String value : values ) {
232           if ( allValuesPattern.matcher(value).matches() )
233             grouperSourcedValuesUsedInTargetSystem.add(value);
234         }
235       }
236 
237       // Now we go through all the grouper groups and see what their attribute values
238       // would be. 
239       Set<String> valuesDefinedByGrouperGroups = new TreeSet<String>(String.CASE_INSENSITIVE_ORDER);
240       for ( GrouperGroupInfo groupInfo : getAllGroupsForProvisioner() ) {
241         String value = getAttributeValueForGroup(groupInfo);
242         valuesDefinedByGrouperGroups.add(value);
243       }
244 
245       LOG.debug("All values of attributes found in LDAP: {}", allValuesOfAttribute);
246       LOG.debug("Values of attribute found in LDAP that match grouper-responsible pattern: {}", grouperSourcedValuesUsedInTargetSystem);
247       LOG.debug("Values of attribute corresponding to existing grouper groups: {}", valuesDefinedByGrouperGroups);
248       
249       // Now we know what values are used by current groups and what values are in the ldap server
250       // Subtract these and we know what values to remove
251       Set<String> extraValues = new TreeSet<String>(String.CASE_INSENSITIVE_ORDER);
252       extraValues.addAll(grouperSourcedValuesUsedInTargetSystem);
253       extraValues.removeAll(valuesDefinedByGrouperGroups);
254       
255       LOG.info("{}: There are {} values that should be purged from the target system", getDisplayName(), extraValues.size());
256       LOG.debug("{}: Values that should be purged: {}", getDisplayName(), extraValues);
257       
258       for (String extraValue : extraValues ) {
259         LOG.info("{}: Purging attribute value {}", getDisplayName(), extraValue);
260         purgeAttributeValue(attribute, extraValue, stats);
261       }
262 	}

285   protected void purgeAttributeValue(String attributeName, String valueToPurge, JobStatistics stats) throws PspException {
286     Set<String> objectsWithAttribute_dnList = getLdapSystem().performLdapSearchRequest_returningValuesOfAnAttribute(
287             -1, config.getUserSearchBaseDn(), SearchScope.SUBTREE,
288             "dn",  attributeName + "={0}", valueToPurge);
289 
290     LOG.info("{}: There are {} ldap objects with attribute value={}", 
291         new Object[] {getDisplayName(), objectsWithAttribute_dnList.size(), valueToPurge});
292 
293     stats.deleteCount.addAndGet(objectsWithAttribute_dnList.size());
294 
295     for ( String objectWithAttribute_dn : objectsWithAttribute_dnList )
296       scheduleUserModification(new LdapUser(objectWithAttribute_dn), AttributeModificationType.REMOVE, Arrays.asList(valueToPurge));
297   }
