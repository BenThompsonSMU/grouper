3   import java.util.Collection;
4   import java.util.NoSuchElementException;
5   import java.util.concurrent.TimeUnit;
6   import java.util.concurrent.BlockingQueue;
7   
8   /**
9    * This interface captures the "head side" of the {@link BlockingQueue} interface
10   */
11  public interface Pollable<E> {
12  
13      /**
14       * Retrieves and removes the head of this queue. This method differs from {@link #poll poll} only in that it throws
15       * an exception if this queue is empty.
16       *
17       * @return the head of this queue
18       * @throws NoSuchElementException
19       *             if this queue is empty
20       */
21      E remove();
22  
23      /**
24       * Retrieves and removes the head of this queue, or returns {@code null} if this queue is empty.
25       *
26       * @return the head of this queue, or {@code null} if this queue is empty
27       */
28      E poll();
29  
30      /**
31       * Retrieves, but does not remove, the head of this queue. This method differs from {@link #peek peek} only in that
32       * it throws an exception if this queue is empty.
33       *
34       * @return the head of this queue
35       * @throws NoSuchElementException
36       *             if this queue is empty
37       */
38      E element();
39  
40      /**
41       * Retrieves, but does not remove, the head of this queue, or returns {@code null} if this queue is empty.
42       *
43       * @return the head of this queue, or {@code null} if this queue is empty
44       */
45      E peek();
46  
47      /**
48       * Retrieves and removes the head of this queue, waiting if necessary until an element becomes available.
49       *
50       * @return the head of this queue
51       * @throws InterruptedException
52       *             if interrupted while waiting
53       */
54      E take() throws InterruptedException;
55  
56      /**
57       * Retrieves and removes the head of this queue, waiting up to the specified wait time if necessary for an element
58       * to become available.
59       *
60       * @param timeout
61       *            how long to wait before giving up, in units of {@code unit}
62       * @param unit
63       *            a {@code TimeUnit} determining how to interpret the {@code timeout} parameter
64       * @return the head of this queue, or {@code null} if the specified waiting time elapses before an element is
65       *         available
66       * @throws InterruptedException
67       *             if interrupted while waiting
68       */
69      E poll(long timeout, TimeUnit unit) throws InterruptedException;
70  
71      /**
72       * Removes all available elements from this queue and adds them to the given collection. This operation may be more
73       * efficient than repeatedly polling this queue. A failure encountered while attempting to add elements to
74       * collection {@code c} may result in elements being in neither, either or both collections when the associated
75       * exception is thrown. Attempts to drain a queue to itself result in {@link IllegalArgumentException}. Further, the
76       * behavior of this operation is undefined if the specified collection is modified while the operation is in
77       * progress.
78       *
79       * @param c
80       *            the collection to transfer elements into
81       * @return the number of elements transferred
82       * @throws UnsupportedOperationException
83       *             if addition of elements is not supported by the specified collection
84       * @throws ClassCastException
85       *             if the class of an element of this queue prevents it from being added to the specified collection
86       * @throws NullPointerException
87       *             if the specified collection is null
88       * @throws IllegalArgumentException
89       *             if the specified collection is this queue, or some property of an element of this queue prevents it
90       *             from being added to the specified collection
91       */
92      int drainTo(Collection<? super E> c);
93  
94      /**
95       * Removes at most the given number of available elements from this queue and adds them to the given collection. A
96       * failure encountered while attempting to add elements to collection {@code c} may result in elements being in
97       * neither, either or both collections when the associated exception is thrown. Attempts to drain a queue to itself
98       * result in {@link IllegalArgumentException}. Further, the behavior of this operation is undefined if the specified
99       * collection is modified while the operation is in progress.
100      *
101      * @param c
102      *            the collection to transfer elements into
103      * @param maxElements
104      *            the maximum number of elements to transfer
105      * @return the number of elements transferred
106      * @throws UnsupportedOperationException
107      *             if addition of elements is not supported by the specified collection
108      * @throws ClassCastException
109      *             if the class of an element of this queue prevents it from being added to the specified collection
110      * @throws NullPointerException
111      *             if the specified collection is null
112      * @throws IllegalArgumentException
113      *             if the specified collection is this queue, or some property of an element of this queue prevents it
114      *             from being added to the specified collection
115      */
116     int drainTo(Collection<? super E> c, int maxElements);
117 
118 }
