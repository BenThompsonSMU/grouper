59     * Return a (unique) id for the current thread. This is both useful for logging because it is
60     * short and differentiates between two threads that might have the same name.

62     * @return

64    public static String getThreadId() {
65      if ( threadId.get() == null )
66        threadId.set("t-" + threadCounter.incrementAndGet());
67  
68      return threadId.get();

79    public static String getIdString(final long n) {
80      int idCharacterCount = idCharacters.length;
81      StringBuilder resultInReverseOrder = new StringBuilder();
82  
83      long curVal = n;
84  
85      while ( curVal > 0 ) {
86        resultInReverseOrder.append(idCharacters[(int)(curVal % idCharacterCount)]);
87        curVal /= idCharacterCount;
88      }
89  
90      return resultInReverseOrder.reverse().toString();
91    }

104   /**
105    * chops a list into sublists of length L
106    * 
107    * Adapted from: polygenelubricants at StackOverflow
108    * http://stackoverflow.com/questions/2895342/java-how-can-i-split-an-arraylist-in-multiple-small-arraylists
109    * @param items
110    * @param L
111    * @return
112    */
113   //
114   static <T> List<List<T>> chopped(Collection<T> items, final int L) {
115      List<T> list;
116      
117      // Convert items to a List if it isn't one already
118      if ( items instanceof List )
119        list = (List<T>) items;
120      else
121        list = new ArrayList<T>(items);
122      
123      List<List<T>> parts = new ArrayList<List<T>>();
124      final int N = list.size();
125      for (int i = 0; i < N; i += L) {
126          parts.add(list.subList(i, Math.min(N, i + L)));
127      }
128      return parts;
129   }

131   public static Map<String, Object> getStemAttributes(Group group) {
132     // In order for stems closer to the group to take precedence, 
133     // we need the stem path in reverse order (from root to group's parent)
134     // We do this by walking up the stem path from the group to the root
135     // and save them and then reverse them. 
136     List<Stem> groupStemPath = new ArrayList<Stem>();
137     Stem stem = group.getParentStem();
138     while ( stem != null ) {
139       groupStemPath.add(stem);
140       
141       if ( stem.isRootStem() )
142         stem = null;
143       else
144         stem = stem.getParentStem();
145     }
146     Collections.reverse(groupStemPath);
147     
148     
149     // OverallAttributeValues: Stores all Stem attributes from root to parent stem
150     // If an attribute appears in multiple parent stems: 
151     //   Single-Valued: stem closest to the group wins (parent attributes take prec over grandparent)
152     //   Multi-Valued: All the attribute values are merged into a list of values for the attribute
153     Map<String, Object> stemPathAttributes = new HashMap<String, Object>();
154     
155     for ( Stem aStem : groupStemPath ) {
156       Set<AttributeAssign> attributeAssigns = aStem.getAttributeDelegate().getAttributeAssigns();
157       for ( AttributeAssign attributeAssign : attributeAssigns ) {
158         AttributeDef attributeDef = attributeAssign.getAttributeDef();
159         AttributeDefName attributeDefName = attributeAssign.getAttributeDefName();
160         String attributeName = attributeDefName.getName();
161         
162         Set<AttributeAssignValue> attributeAssignValues = attributeAssign.getValueDelegate().getAttributeAssignValues();
163         
164         List<String> attributeValues = new ArrayList<String>();
165         
166         for ( AttributeAssignValue attributeAssignValue : attributeAssignValues ) {
167           String value = attributeAssignValue.getValueFriendly();
168           attributeValues.add(value);
169         }
170         
171         // Skip the attribute if it doesn't actually have any values
172         if ( attributeValues.size() == 0 )
173           continue;
174         
175         if ( ! (attributeDef.isMultiValued() || attributeDef.isMultiAssignable()) ) 
176           // Single-Valued: Put the first value into map, replacing whatever was there
177           stemPathAttributes.put(attributeName, attributeValues.iterator().next());
178         else {
179           // Multi-valued: Put all the values into a collection
180           Collection<Object> valueArray = (Collection<Object>) stemPathAttributes.get(attributeName);
181           if ( valueArray == null ) 
182             stemPathAttributes.put(attributeName, attributeValues);
183           else
184             valueArray.addAll(attributeValues);
185         }
186       }
187     }
188     return stemPathAttributes;
189   }

235   /**
236    * This method returns the first RDN of an LDAP DN. If there weren't commas for other reasons, this
237    * would mean that it returns the part of the DN before the first comma. However, there can be commas
238    * for different reasons, so this is more complicated
239    *
240    * Here are some examples:
241    *   Easy:
242    *   cn=xyz,ou=department,dc=example,dc=edu would return cn=xyz
243    *   JEXL, no comma:
244    *   cn=${group.name},ou=department,dc=example,dc=edu would return cn=${group.name}
245    *
246    *   JEXL, commas in method parameter lists:
247    *   cn=${group.name.substring(1,3)},ou=department,dc=example,dc=edu would return cn=${group.name.substring(1,3)}
248    *
249    *   Nested JEXL:
250    *   cn=${Hash.md5(${group.name})},ou=department,dc=example,dc=edu would return cn=${Hash.md5(${group.name}}
251    *
252    *  There are other ways commas need to be ignored (escaped, etc):
253    *    cn=abc\,123,ou=department,dc=example,dc=edu would return cn=abc\,123
254    *
255    * @param dnAttributeValue
256    * @return
257    */
258   public static String getFirstRdnString(final String dnAttributeValue) {
259     // We are going to replace JEXL and comma-escaping strings with jibberish, equal-length strings
260     // until there aren't any more of them. Then, the first comma will actually be the separator
261     // between the first and second RDN components
262     //
263     // This approach was chosen because java doesn't have support for recursive regexes and this is
264     // simpler than a real parser
265 
266     String s = dnAttributeValue;
267 
268     // Removed some escaped patterns: \\ and \,
269     s = replaceMatchesWithUnderscores(s, Pattern.compile("\\\\"));
270     s = replaceMatchesWithUnderscores(s, Pattern.compile("\\,"));
271 
272     // Match ${function()} but not nested ${function(${value})} (fwiw, the inner ${value} would be matched)
273     s = replaceMatchesWithUnderscores(s, Pattern.compile("\\$\\{[^{}]*\\}"));
274 
275     int indexOfFirstComma = s.indexOf(',');
276     if ( indexOfFirstComma < 0 ) {
277       return dnAttributeValue;
278     }
279     else {
280       // This is why we've kept s to be the same length of dnAttributeValue:
281       //   we can use the index of the first comma in s to find the first DN-relevant comma in
282       //   the original dnAttributeValue
283       return dnAttributeValue.substring(0, indexOfFirstComma);
284     }
285   }

287   /**
288    * This is used by getFirstRdnString() in order to iteratively remove parts of a string
289    * that match a pattern. However, this method does not just delete the matched characters,
290    * it instead replaces them with an equal number of underscores, KEEPING THE INPUT AND
291    * RESULT STRINGS THE SAME LENGTH
292    *
293    * @param s
294    * @param jexlPattern
295    * @return
296    */
297   private static String replaceMatchesWithUnderscores(String s, Pattern jexlPattern) {
298     Matcher m = jexlPattern.matcher(s);
299     while (m.find()) {
300       String matched = m.group();
301 
302       // Replace the occurrence with an equal-length string of underscores
303       String newString = s.replace(matched, StringUtils.repeat("_", matched.length()) );
304 
305       s=newString;
306       m=jexlPattern.matcher(s);
307     }
308 
309     return s;
310   }

317   static void hibernateRefresh(final Object objectToHibernateRefresh) {
318       LOG.debug("Rereading group information from database: {}/{}", objectToHibernateRefresh.getClass(), objectToHibernateRefresh);
319 
320       try {
321         HibernateSession.callbackHibernateSession(GrouperTransactionType.READONLY_OR_USE_EXISTING,
322                 AuditControl.WILL_NOT_AUDIT,
323                 new HibernateHandler() {
324                   @Override
325                   public Object callback(HibernateHandlerBean hibernateHandlerBean) throws GrouperDAOException {
326                     hibernateHandlerBean.getHibernateSession().getSession().refresh(objectToHibernateRefresh);
327                     return objectToHibernateRefresh;
328                   }
329                 }
330         );
331       } catch (GrouperDAOException e) {
332         LOG.warn("Unable to refresh object from database, probably because it has been deleted: {}", objectToHibernateRefresh);
333         // Ignoring error as deleted objects do not need to be refreshed
334       }
335     }

352     static String formatElapsedTime(final long milliseconds) {
353       StringBuilder result = new StringBuilder();
354 
355       long millisecondsRemaining = milliseconds;
356 
357       // Used to grab days/hours/minutes/etc of the starting time
358       long chunk;
359 
360       // Seconds and milliseconds are distracting when the ElapsedTime is big
361       // This is roughly removing units that are <1% of the total
362       boolean ignoreSeconds=     milliseconds > Duration.standardHours(2).getMillis();
363       boolean ignoreMilliseconds=milliseconds > Duration.standardMinutes(2).getMillis();
364 
365 
366       // Days
367       chunk=24*60*60*1000;
368       if ( result.length()>0 || millisecondsRemaining > chunk ) {
369         result.append(String.format("%dd+", millisecondsRemaining/chunk));
370         millisecondsRemaining=millisecondsRemaining % chunk;
371       }
372 
373       // Hours
374       chunk=60*60*1000;
375       if ( result.length()>0 || millisecondsRemaining > chunk ) {
376         result.append(String.format("%02dh:", millisecondsRemaining/chunk));
377         millisecondsRemaining=millisecondsRemaining % chunk;
378       }
379       // Minutes
380       chunk=60*1000;
381       if ( result.length()>0 || millisecondsRemaining > chunk ) {
382         result.append(String.format("%02dm:", millisecondsRemaining/chunk));
383         millisecondsRemaining=millisecondsRemaining % chunk;
384       }
385       if ( !ignoreSeconds )
386       {
387         result.append(String.format("%02d", millisecondsRemaining/1000));
388 
389         if (!ignoreMilliseconds && millisecondsRemaining%1000 != 0)
390           result.append(String.format(".%03d", millisecondsRemaining%1000));
391 
392         result.append("s");
393       }
394 
395       return result.toString();
396     }

398   /**
399    * Format the number with an adjustable number of decimal places, so decimal places are
400    * only used if they're needed to show significant values. This also rounds the number
401    * to significant digits
402    * @param number
403    * @param significantDigits
404    * @return
405    */
406     public static String formatWithSignificantDigits(double number, int significantDigits) {
407       BigDecimal bdNumber = new BigDecimal(number);
408       BigDecimal bdRounded = bdNumber.round(new MathContext(significantDigits));
409       double rounded = bdRounded.doubleValue();
410 
411       // This is from https://stackoverflow.com/a/25308216
412       DecimalFormat df = new DecimalFormat("0");
413       df.setMaximumFractionDigits(340); //340 = DecimalFormat.DOUBLE_FRACTION_DIGITS
414 
415       return df.format(rounded);
416     }

453    * Returns a new set that is c1 - c2
454    * @param caseSensitiveValues
455    * @param c1
456    * @param c2
457    * @return
458    */
459   public static  Set<String> subtractStringCollections(boolean caseSensitiveValues, Collection<String> c1, Collection<String> c2) {
460     Set<String> set1 = getStringSet(caseSensitiveValues, c1);
461 
462     // We wish we didn't have to make this Set, but a jvm bug makes set1.removeAll(c2) sometimes
463     // use set1's contains() semantics and sometimes use c2's. Therefore, we need to make sure
464     // both possibilities use consistent rules
465     // See: https://bugs.openjdk.java.net/browse/JDK-6394757
466     // See: https://bugs.openjdk.java.net/browse/JDK-8180409
467     Set<String> set2 = getStringSet(caseSensitiveValues, c2);
468 
469     set1.removeAll(set2);
470     return set1;
471   }

488     Set<String> set2 = getStringSet(caseSensitiveValues, c2);
489 
490     set1.retainAll(set2);
491     return set1;
492   }
