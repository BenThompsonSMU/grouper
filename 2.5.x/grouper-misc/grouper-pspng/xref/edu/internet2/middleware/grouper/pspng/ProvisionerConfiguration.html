103     /**
104      * This expression says that the provisionerName has to be in a group or stem provision_to attribute
105      * and NOT in neither a group or stem do_not_provision_to attribute
106      */
107     protected String groupSelectionExpression_defaultValue() {
108       // GRP-1356: pspng should use the default configuration folder
109       // = "${utils.containedWithin(provisionerName, stemAttributes['etc:pspng:provision_to'], groupAttributes['etc:pspng:provision_to']) " 
110       //    + "&& !utils.containedWithin(provisionerName, stemAttributes['etc:pspng:do_not_provision_to'], groupAttributes['etc:pspng:do_not_provision_to'])}";
111       String rootStem = GrouperConfig.retrieveConfig().propertyValueString(
112           "grouper.rootStemForBuiltinObjects", "etc");
113       return "${utils.containedWithin(provisionerName, stemAttributes['" + rootStem 
114           + ":pspng:provision_to'], groupAttributes['" + rootStem + ":pspng:provision_to']) " 
115           + "&& !utils.containedWithin(provisionerName, stemAttributes['" + rootStem 
116           + ":pspng:do_not_provision_to'], groupAttributes['" + rootStem + ":pspng:do_not_provision_to'])}";
117       
118     }
119     protected Collection<String> attributesUsedInGroupSelectionExpression;
120     protected String attributesUsedInGroupSelectionExpression_defaultValue = "provision_to,do_not_provision_to";
121 
122     // attributesUsedInGroupSelectionExpression are referenced within the groupSelectionExpression
123     // However, what are they compared to? By default, they are compared to the name of the provisioner
124     // and we can use database filtering to find matching groups faster. However, if a different
125     // selection expression is used, then we can't use database filtering, and we have to run the jexl
126     // expression on each group/folder that refers to a selection-relevant attribute.
127     //
128     // This configuration item enables database filtering when looking for possible matching groups.
129     // The groups are still run through the groupSelectionExpression, but this allows fewer
130     // groups to be checked against that expression.
131     protected boolean attributesUsedInGroupSelectionExpressionAreComparedToProvisionerName;
132     protected boolean attributesUsedInGroupSelectionExpressionAreComparedToProvisionerName_defaultValue = true;
133

171     /**
172      * Populate the various settings from the configuration source(s). This is separated from our 
173      * constructor so that our subclasses can override default values. In other words, a subclass
174      * might construct us, change a _defaultValue ivar and then ask us to do 'readConfiguration'.
175      */
176     public void readConfiguration() {
177         final String qualifiedParameterNamespace = PARAMETER_NAMESPACE + provisionerName + ".";
178 
179         LOG.debug("Ldap Group Provisioner - Setting properties for {} consumer/provisioner.", provisionerName);
180 
181         enabled =
182             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "enabled", enabled_defaultValue);
183         LOG.debug("Provisioner {} - Setting enabled to {}", provisionerName, enabled);
184 
185         grouperIsAuthoritative =
186             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "grouperIsAuthoritative", grouperIsAuthoritative_defaultValue);
187         LOG.debug("Provisioner {} - Setting grouperIsAuthoritative to {}", provisionerName, grouperIsAuthoritative);
188 
189         // This attribute used to have a poorly-named configuration property. Keeping compatibility with it.
190         if (GrouperLoaderConfig.retrieveConfig().containsKey(qualifiedParameterNamespace + "grouperGroupCacheTime_secs") ) {
191             dataCacheTime_secs =
192                     GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "grouperGroupCacheTime_secs", dataCacheTime_secs_defaultValue);
193         } else {
194             dataCacheTime_secs =
195                     GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "dataCacheTime_secs", dataCacheTime_secs_defaultValue);
196         }
197 
198         LOG.debug("Provisioner {} - Setting dataCacheTime_secs to {}", provisionerName, dataCacheTime_secs);
199     
200         sleepTimeAfterError_ms =
201             GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "sleepTimeAfterError_ms", sleepTimeAfterError_ms_defaultValue);
202         LOG.debug("Provisioner {} - Setting sleepTimeAfterError_ms to {}", provisionerName, sleepTimeAfterError_ms);
203     
204         grouperGroupCacheSize =
205                 GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "grouperGroupCacheSize", grouperGroupCacheSize_defaultValue);
206         LOG.debug("Provisioner {} - Setting grouperGroupCacheSize to {}", provisionerName, grouperGroupCacheSize);
207 
208         grouperSubjectCacheSize =
209             GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "grouperSubjectCacheSize", grouperSubjectCacheSize_defaultValue);
210         LOG.debug("Provisioner {} - Setting grouperSubjectCacheSize to {}", provisionerName, grouperSubjectCacheSize);
211 
212         targetSystemGroupCacheSize =
213                 GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "targetSystemGroupCacheSize", targetSystemGroupCacheSize_defaultValue);
214         LOG.debug("Provisioner {} - Setting targetSystemGroupCacheSize to {}", provisionerName, targetSystemGroupCacheSize);
215 
216         targetSystemUserCacheSize =
217                 GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "targetSystemUserCacheSize", targetSystemUserCacheSize_defaultValue);
218         LOG.debug("Provisioner {} - Setting targetSystemUserCacheSize to {}", provisionerName, targetSystemUserCacheSize);
219 
220         createMissingUsers =
221             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "createMissingUsers", createMissingUsers_defaultValue);
222         LOG.debug("Provisioner {} - Setting createMissingUsers to {}", provisionerName, createMissingUsers);
223         
224         needsTargetSystemUsers =
225             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "needsTargetSystemUsers", needsTargetSystemUsers_defaultValue);
226         LOG.debug("Provisioner {} - Setting needsTargetSystemUsers to {}", provisionerName, needsTargetSystemUsers);
227 
228         needsTargetSystemGroups =
229             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "needsTargetSystemGroups", needsTargetSystemGroups_defaultValue);
230         LOG.debug("Provisioner {} - Setting needsTargetSystemGroups to {}", provisionerName, needsTargetSystemGroups);
231 
232         supportsEmptyGroups =
233             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "supportsEmptyGroups", supportsEmptyGroups_defaultValue);
234         LOG.debug("Provisioner {} - Setting supportsEmptyGroups to {}", provisionerName, supportsEmptyGroups);
235 
236         groupSelectionExpression =
237             GrouperLoaderConfig.retrieveConfig().propertyValueString(qualifiedParameterNamespace + "groupSelectionExpression", groupSelectionExpression_defaultValue());
238         LOG.debug("Provisioner {} - Setting groupSelectionExpression to {}", provisionerName, groupSelectionExpression);
239 
240         // List of attributes used in GroupSelection expression
241         String attributesUsedInGroupSelectionExpression_string =
242             GrouperLoaderConfig.retrieveConfig().propertyValueString(qualifiedParameterNamespace + "attributesUsedInGroupSelectionExpression", attributesUsedInGroupSelectionExpression_defaultValue);
243 
244         String attributesUsed[] = attributesUsedInGroupSelectionExpression_string.split(" *, *");
245         
246         attributesUsedInGroupSelectionExpression = new ArrayList<String>();
247         String rootStem = GrouperConfig.retrieveConfig().propertyValueString(
248             "grouper.rootStemForBuiltinObjects", "etc");
249         for ( String attribute: attributesUsed ) {
250           if ( attribute.contains(":") ) {
251             attributesUsedInGroupSelectionExpression.add(attribute);
252           }
253           else {
254             attributesUsedInGroupSelectionExpression.add(String.format("%s:pspng:%s", rootStem, attribute));
255           }
256         }
257         LOG.debug("Provisioner {} - Setting attributesUsedInGroupSelectionExpression to {}", provisionerName, attributesUsedInGroupSelectionExpression);
258         
259         attributesUsedInGroupSelectionExpressionAreComparedToProvisionerName =
260             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "attributesUsedInGroupSelectionExpressionAreComparedToProvisionerName", attributesUsedInGroupSelectionExpressionAreComparedToProvisionerName_defaultValue);
261         LOG.debug("Provisioner {} - Setting attributesUsedInGroupSelectionExpressionAreComparedToProvisionerName to {}", provisionerName, attributesUsedInGroupSelectionExpressionAreComparedToProvisionerName);
262 
263         userSearch_batchSize =
264             GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "userSearch_batchSize", userSearch_batchSize_defaultValue);
265         LOG.debug("Provisioner {} - Setting userSearch_batchSize to {}", provisionerName, userSearch_batchSize);
266 
267         groupSearch_batchSize =
268             GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "groupSearch_batchSize", groupSearch_batchSize_defaultValue);
269         LOG.debug("Provisioner {} - Setting groupSearch_batchSize to {}", provisionerName, groupSearch_batchSize);
270 
271         coordinationTimeout_secs =
272                 GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "coordinationTimeout_secs", coordinationTimeout_secs_defaultValue);
273         LOG.debug("Provisioner {} - Setting coordinationTimeout_secs to {}", provisionerName, coordinationTimeout_secs);
274         coordinationUpdateInterval_secs =
275                 GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "coordinationUpdateInterval_secs", coordinationUpdateInterval_secs_defaultValue);
276         LOG.debug("Provisioner {} - Setting coordinationUpdateInterval_secs to {}", provisionerName, coordinationUpdateInterval_secs);
277 
278         areChangesToInternalGrouperSubjectsIgnored =
279                 GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "areChangesToInternalGrouperSubjectsIgnored", areChangesToInternalGrouperSubjectsIgnored_defaultValue);
280         LOG.debug("Provisioner {} - Setting areChangesToInternalGrouperSubjectsIgnored to {}", provisionerName, areChangesToInternalGrouperSubjectsIgnored);
281 
282         grouperMessagingSystemName =
283                 GrouperLoaderConfig.retrieveConfig().propertyValueString(qualifiedParameterNamespace + "grouperMessagingSystemName", grouperMessagingSystemName_defaultValue);
284         LOG.debug("Provisioner {} - Setting grouperMessagingSystemName to {}", provisionerName, grouperMessagingSystemName);
285 
286         numberOfDataFetchingWorkers =
287                 GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "numberOfDataFetchingWorkers", numberOfDataFetchingWorkers_defaultValue);
288         LOG.debug("Provisioner {} - Setting numberOfDataFetchingWorkers to {}", provisionerName, numberOfDataFetchingWorkers);
289 
290         missingSubjectsWarningThreshold_percentage =
291                 GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "missingSubjectsWarningThreshold_percentage", missingSubjectsWarningThreshold_percentage_defaultValue);
292         LOG.debug("Provisioner {} - Setting missingSubjectsWarningThreshold_percentage to {}", provisionerName, missingSubjectsWarningThreshold_percentage);
293 
294 
295         areCacheSizeWarningsEnabled =
296                 GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "areCacheSizeWarningsEnabled", areCacheSizeWarningsEnabled_defaultValue);
297         LOG.debug("Provisioner {} - Setting areCacheSizeWarningsEnabled to {}", provisionerName, areCacheSizeWarningsEnabled);
298 
299 
300         cacheFullnessWarningThreshold_percentage =
301                 GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "cacheFullnessWarningThreshold_percentage", cacheFullnessWarningThreshold_percentage_defaultValue);
302         LOG.debug("Provisioner {} - Setting cacheFullnessWarningThreshold_percentage to {}", provisionerName, cacheFullnessWarningThreshold_percentage);
303 
304         maxNumberOfTimesToRepeatedlyFullSyncGroup =
305             GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "maxNumberOfTimesToRepeatedlyFullSyncGroup", maxNumberOfTimesToRepeatedlyFullSyncGroup_defaultValue);
306         LOG.debug("Provisioner {} - Setting maxNumberOfTimesToRepeatedlyFullSyncGroup to {}", provisionerName, maxNumberOfTimesToRepeatedlyFullSyncGroup);
307         if ( maxNumberOfTimesToRepeatedlyFullSyncGroup<1 ) {
308             LOG.warn("Provisioner {} - maxNumberOfTimesToRepeatedlyFullSyncGroup must be at least 1", provisionerName);
309             maxNumberOfTimesToRepeatedlyFullSyncGroup = 1;
310         }
311 
312         timeToSleepBetweenRepeatedFullSyncs_ms =
313                 GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "timeToSleepBetweenRepeatedFullSyncs_ms", timeToSleepBetweenRepeatedFullSyncs_ms_defaultValue);
314         LOG.debug("Provisioner {} - Setting timeToSleepBetweenRepeatedFullSyncs_ms to {}", provisionerName, timeToSleepBetweenRepeatedFullSyncs_ms);
315     }
