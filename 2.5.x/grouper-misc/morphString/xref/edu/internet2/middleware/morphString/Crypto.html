16  package edu.internet2.middleware.morphString;
17  
18  import java.io.InputStream;
19  import java.io.OutputStream;
20  import java.security.NoSuchAlgorithmException;
21  
22  import javax.crypto.Cipher;
23  import javax.crypto.CipherInputStream;
24  import javax.crypto.CipherOutputStream;
25  import javax.crypto.KeyGenerator;
26  import javax.crypto.SecretKey;
27  import javax.crypto.spec.SecretKeySpec;
28  
29  import edu.internet2.middleware.morphString.apache.codec.binary.Base64;
30  
31  /**
32   * The purpose of this class is to provide encryption 
33   * and decryption using standard Java libraries, for potentially 
34   * large amounts of data.
35   * <p>
36   * This class provides default encryption using AES with a constant
37   * 128 bit key.  If you want something more secure feel free to 
38   * override the defaults however you please.
39   * <p>
40   * This class works in one of two ways, (1) in memory using Strings, or (2) via 
41   * I/O streams (preferred for large amounts of data).
42   * <p>
43   * Crypo objects, or more specifically the default ciphers they create, are not 
44   * threadsafe and are not computationally cheap, so a threadlocal factory 
45   * method is provided for convenience.  This is the preferred means of usage,
46   * but feel free to create these objects however you please.
47   * <p>
48   * Note that you can encrypt BLOB fields by specifying encryption in the 
49   * configurator (Crypto is the default encryption mechanism for that).
50   * <p> 
51   */
52  public class Crypto {
53    /** threadlocal provided for conveniency */
54    private static final ThreadLocal<Crypto> threadLocalCrypto = new ThreadLocal<Crypto>();
55    
56    /** @return a non-null thread-safe crypto object from a ThreadLocal */
57    public static Crypto getThreadLocalCrypto() {
58      Crypto crypto = threadLocalCrypto.get();
59      if (crypto == null) {
60        crypto = new Crypto();
61        threadLocalCrypto.set(crypto);
62      }
63      return crypto;
64    }
65    
66    /**
67     * Generate a key.
68     * @param cipherName the name of the cipher, if null will default to "AES"
69     * @param keybits the number of bits in the key, if null will default to 128
70     * @return the bytes comprising the key
71     */
72    public static byte[] generateKeyBytes(String cipherName, Integer keybits) {
73      KeyGenerator keyGenerator = null;
74      cipherName = cipherName == null ? "AES" : cipherName;
75      try {
76        keyGenerator = KeyGenerator.getInstance(cipherName);
77      } catch (NoSuchAlgorithmException e) {
78        throw new RuntimeException("Failed to create KeyGenerator for cipherName="+cipherName, e);
79      }
80      keyGenerator.init(keybits == null ? 128 : keybits); // 192 and 256 bits may not be available
81      
82      // Generate the secret key specs.
83      SecretKey secretKey = keyGenerator.generateKey();
84      byte[] keyBytes = secretKey.getEncoded();
85      
86      return keyBytes;    
87    }
88  
89    
90    /** SecretKeySpec */
91    private SecretKeySpec key;
92    
93    /** lazily constructed cipher */
94    private Cipher cipher = null;
95      
96    /**
97     * Create the default cipher
98     * @return the default cipher
99     */
100   public Cipher createDefaultCipher() {
101     try {
102       return Cipher.getInstance("AES");
103     } catch(Exception x) {
104       throw new RuntimeException("Failed to create the default cipher!", x);
105     }
106   }
107   
108   /** Default crypto object */
109   public Crypto() {
110     this(Morph.key());
111   }
112   
113   /** Default crypto object 
114    * @param theKey used to encrypt/decrypt 
115    */
116   public Crypto(String theKey) {
117     super();
118     init(theKey);
119   }
120   
121   /** only warn once */
122   private static boolean warned = false;
123   
124   /** initialize the key and cipher 
125    * @param secret
126    */
127   protected void init(String secret) {
128     
129     if (MorphStringUtils.isBlank(secret)) {
130       throw new NullPointerException("Must supply a non blank encrypt.key");
131     }
132     StringBuilder secretBuilder = new StringBuilder(secret);
133     if (!warned && secret.length() < 8) {
134       //note we dont have a logger
135       System.out.println("morphString warning: secret.key in morphString.properties should be at least 8 chars");
136       warned = true;
137     }
138     //secret must be length 16 or 32.  pad if not
139     while (secretBuilder.length() < 16) {
140       secretBuilder.append("x");
141     }
142     if (secretBuilder.length() > 16) {
143       while (secretBuilder.length() < 32) {
144         secretBuilder.append("x");
145       }
146     }
147     if (secretBuilder.length() > 32) {
148       secretBuilder.delete(32, secretBuilder.length());
149     }
150     secret = secretBuilder.toString();
151     this.key = new SecretKeySpec(secret.getBytes(), "AES");
152     this.cipher = this.createDefaultCipher();
153   }
154   
155   /**
156    * Encrypt the string
157    * @param clearText
158    * @return the encrypted String
159    */
160   public String encrypt(String clearText) {
161     byte[] input = clearText.getBytes();
162     try {
163       this.initCipher(true);
164       byte[] output = this.cipher.doFinal(input);
165       byte[] encoded = Base64.encodeBase64(output);      
166       return new String(encoded);
167     } catch(Exception x) {
168       throw new RuntimeException("Failed to encrypt string", x);
169     }
170   } 
171    
172   /**
173    * Decrypt the string
174    * @param cipherText
175    * @return the decrypted string
176    */
177   public String decrypt(String cipherText) {
178     try {      
179         byte[] cipherBytes = cipherText.getBytes();
180         byte[] decodedBytes = Base64.decodeBase64(cipherBytes);
181         this.initCipher(false);
182         byte[] clearBytes = this.cipher.doFinal(decodedBytes);
183         return new String(clearBytes);
184     } catch(Exception x) {
185       throw new RuntimeException("Failed to decrypt the cipherText", x);
186     }
187   }
188   
189   /**
190    * Initialize the cipher for encryption or decryption
191    * @param encrypt true to encrypt, false to decrypt
192    */
193   private void initCipher(boolean encrypt) {
194     try {
195       if ( encrypt ) {
196         this.cipher.init(Cipher.ENCRYPT_MODE, this.key);
197       } else {
198         this.cipher.init(Cipher.DECRYPT_MODE, this.key);
199       }
200     } catch(Exception x) {
201       throw new RuntimeException("Failed to init cipher for "+(encrypt ? "encrypt" : "decrypt"), x);
202     }
203   }
204  
205   /**
206    * Get the encrypted input stream
207    * @param in
208    * @return the encrypted input stream
209    */
210   public InputStream encrypt(InputStream in) {
211     this.initCipher(true);
212     return new CipherInputStream(in, this.cipher);
213   }
214   
215   /**
216    * the decrypted input stream
217    * @param in
218    * @return the decrypted input stream
219    */
220   public InputStream decrypt(InputStream in) {
221     this.initCipher(false);
222     return new CipherInputStream(in, this.cipher);
223   }
224   
225   /**
226    * the encrypted output stream
227    * @param out
228    * @return the encrypted output stream
229    */
230   public OutputStream encrypt(OutputStream out) {
231     this.initCipher(true);
232     return new CipherOutputStream(out, this.cipher);
233   }
234   
235   /**
236    * the decrypted output stream
237    * @param out
238    * @return the decrypted output stream
239    */
240   public OutputStream decrypt(OutputStream out) {
241     this.initCipher(false);
242     return new CipherOutputStream(out, this.cipher);
243   }
244 }
