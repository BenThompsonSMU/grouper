61    /**
62     * @see edu.internet2.middleware.grouperClient.messaging.GrouperMessagingSystem#send(edu.internet2.middleware.grouperClient.messaging.GrouperMessageSendParam)
63     */
64    @Override
65    public GrouperMessageSendResult send(GrouperMessageSendParam grouperMessageSendParam) {
66          
67      if (grouperMessageSendParam.getGrouperMessageQueueParam() == null) {
68        throw new IllegalArgumentException("grouperMessageQueueParam is required.");
69      }
70      
71      String queueOrTopicName = grouperMessageSendParam.getGrouperMessageQueueParam().getQueueOrTopicName();
72      String exchangeType = grouperMessageSendParam.getExchangeType();
73      
74      if (StringUtils.isBlank(queueOrTopicName)) {
75        throw new IllegalArgumentException("queueOrTopicName is required.");
76      }
77      
78      GrouperMessageSystemParam grouperMessageSystemParam = grouperMessageSendParam.getGrouperMessageSystemParam();
79      if (grouperMessageSystemParam == null || StringUtils.isBlank(grouperMessageSystemParam.getMessageSystemName())) {
80        throw new IllegalArgumentException("grouperMessageSystemParam.messageSystemName is a required field.");
81      }
82      
83      try {
84        
85        Connection connection = connectionFactory.getConnection(grouperMessageSystemParam.getMessageSystemName());
86        Channel channel = connection.createChannel();
87        
88        String error = createQueueOrExchange(grouperMessageSystemParam, channel, queueOrTopicName, 
89            exchangeType, grouperMessageSendParam.getGrouperMessageQueueParam().getQueueType());
90        
91        if (error != null) {
92          throw new IllegalArgumentException(error);
93        }
94        
95        for (GrouperMessage grouperMessage: GrouperClientUtils.nonNull(grouperMessageSendParam.getGrouperMessages())) {
96          String message = grouperMessage.getMessageBody();
97          if (grouperMessageSendParam.getGrouperMessageQueueParam().getQueueType() == GrouperMessageQueueType.topic) {
98            channel.basicPublish(queueOrTopicName, StringUtils.defaultString(grouperMessageSendParam.getRoutingKey(), ""), MessageProperties.PERSISTENT_BASIC, message.getBytes("UTF-8"));
99          } else {
100           channel.basicPublish("", queueOrTopicName, MessageProperties.PERSISTENT_BASIC, message.getBytes("UTF-8"));
101         }
102         LOG.info("Sent message: "+message);
103       }
104       channel.close();
105       
106     } catch(IOException e) {
107       throw new RuntimeException("Error occurred while sending message to messaging system: "+grouperMessageSystemParam.getMessageSystemName(), e);
108     } catch(TimeoutException e) {
109       throw new RuntimeException("Error occurred while closing channel for messaging system: "+grouperMessageSystemParam.getMessageSystemName(), e);
110     }
111     return new GrouperMessageSendResult();
112   }

122   /**
123    * @see edu.internet2.middleware.grouperClient.messaging.GrouperMessagingSystem#receive(edu.internet2.middleware.grouperClient.messaging.GrouperMessageReceiveParam)
124    */
125   @Override
126   public GrouperMessageReceiveResult receive(GrouperMessageReceiveParam grouperMessageReceiveParam) {
127     
128     GrouperMessageSystemParam grouperMessageSystemParam = grouperMessageReceiveParam.getGrouperMessageSystemParam();
129     if (grouperMessageSystemParam == null || StringUtils.isBlank(grouperMessageSystemParam.getMessageSystemName())) {
130       throw new IllegalArgumentException("grouperMessageSystemParam.messageSystemName is required.");
131     }
132     GrouperMessagingConfig grouperMessagingConfig = GrouperClientConfig.retrieveConfig().retrieveGrouperMessagingConfigNonNull(grouperMessageSystemParam.getMessageSystemName());
133     int defaultPageSize = grouperMessagingConfig.propertyValueInt(GrouperClientConfig.retrieveConfig(), "defaultPageSize", 5);
134     int maxPageSize = grouperMessagingConfig.propertyValueInt(GrouperClientConfig.retrieveConfig(), "maxPageSize", 5);
135         
136     Integer maxMessagesToReceiveAtOnce = grouperMessageReceiveParam.getMaxMessagesToReceiveAtOnce();
137     
138     if (maxMessagesToReceiveAtOnce == null) {
139       maxMessagesToReceiveAtOnce = defaultPageSize;
140     }
141     
142     if (maxMessagesToReceiveAtOnce > maxPageSize) {
143       maxMessagesToReceiveAtOnce = maxPageSize;
144     }
145     
146     final Integer pageSize = maxMessagesToReceiveAtOnce;
147     
148     String queueOrTopicName = grouperMessageReceiveParam.getGrouperMessageQueueParam().getQueueOrTopicName();
149     String exchangeType = grouperMessageReceiveParam.getExchangeType();
150     
151     if (StringUtils.isBlank(queueOrTopicName)) {
152       throw new IllegalArgumentException("queueOrTopicName is required.");
153     }
154     
155     Integer longPollMillis = grouperMessageReceiveParam.getLongPollMilis();
156     
157     if (longPollMillis == null || longPollMillis < 0) {
158       longPollMillis = 1000;
159     }
160     
161     GrouperMessageReceiveResult result = new GrouperMessageReceiveResult();
162     final Collection<GrouperMessage> messages = new ArrayList<GrouperMessage>();
163     result.setGrouperMessages(messages);
164     
165     try {
166 
167       Connection connection = connectionFactory.getConnection(grouperMessageSystemParam.getMessageSystemName());
168       final Channel channel = connection.createChannel();
169       Consumer consumer = new DefaultConsumer(channel) {
170         @Override
171         public void handleDelivery(String consumerTag, Envelope envelope, AMQP.BasicProperties properties, byte[] body)
172             throws IOException {
173           String message = new String(body, "UTF-8");
174           GrouperMessageRabbitmquperMessageRabbitmq.html#GrouperMessageRabbitmq">GrouperMessageRabbitmq rabbitmqMessage = new GrouperMessageRabbitmq(message, properties.getMessageId());
175           messages.add(rabbitmqMessage);
176           channel.basicAck(envelope.getDeliveryTag(), false);
177           if (messages.size() >= pageSize) {
178             try {
179               channel.close();
180             } catch (TimeoutException e) {
181               LOG.error("Error occurred while closing channel", e);
182             }
183           }
184           LOG.info("Received: "+message);
185         }
186       };
187       
188       String error = createQueueOrExchange(grouperMessageSystemParam, channel, queueOrTopicName, 
189           exchangeType, grouperMessageReceiveParam.getGrouperMessageQueueParam().getQueueType());
190       
191       if (error != null) {
192         throw new IllegalArgumentException(error);
193       }
194       
195       if (grouperMessageReceiveParam.getGrouperMessageQueueParam().getQueueType() == GrouperMessageQueueType.topic) {
196         DeclareOk declareOk = channel.queueDeclare();
197         channel.queueBind(declareOk.getQueue(), queueOrTopicName, StringUtils.defaultString(grouperMessageReceiveParam.getRoutingKey(), ""));
198         channel.basicConsume(declareOk.getQueue(), false, consumer);
199       } else if (grouperMessageReceiveParam.getGrouperMessageQueueParam().getQueueType() == GrouperMessageQueueType.queue) {
200         channel.basicConsume(queueOrTopicName, false, consumer);
201       }
202       
203       new Timer().schedule(
204         new java.util.TimerTask() {
205           @Override
206           public void run() {
207             try {
208               if (channel.isOpen()) {
209                 channel.close();
210               }
211             } catch (Exception e) {
212               LOG.error("Error occurred while closing channel", e); 
213             }
214           }
215         }, longPollMillis);
216     } catch(IOException e) {
217       throw new RuntimeException("Error occurred while trying to receive messages for "+grouperMessageSystemParam.getMessageSystemName(), e);
218     }
219     
220     return result;
221   }

224   /**
225    * @param grouperMessageSystemParam
226    * @param channel
227    * @param queueOrTopicName
228    * @param queueType
229    * @return
230    * @throws IOException
231    */
232   private String createQueueOrExchange(GrouperMessageSystemParam grouperMessageSystemParam,
233       Channel channel, String queueOrTopicName, String exchangeType, GrouperMessageQueueType queueType) throws IOException {
234     
235     String error = null;
236     
237     if (queueType == GrouperMessageQueueType.topic) {
238       if (grouperMessageSystemParam.isAutocreateObjects()) {
239         
240         BuiltinExchangeType exchange;
241         try {
242           exchange = BuiltinExchangeType.valueOf(exchangeType.toUpperCase());
243         } catch(Exception e) {
244           String validExchangeTypes = StringUtils.join(BuiltinExchangeType.values(), ",");
245           error = "exchange type "+exchangeType+" is not valid. Valid options are: "+validExchangeTypes;
246           return error;
247         }
248         channel.exchangeDeclare(queueOrTopicName, exchange, true);
249         
250       } else {
251         try {
252           channel.exchangeDeclarePassive(queueOrTopicName);
253         } catch (IOException e) {
254           error = "exchange "+queueOrTopicName+" doesn't exist. Either create the exchange or set the autoCreateObjects to true.";
255         }
256       }
257     } else if (queueType == GrouperMessageQueueType.queue) {
258         if (grouperMessageSystemParam.isAutocreateObjects()) {
259           channel.queueDeclare(queueOrTopicName, true, false, false, null);
260         } else {
261           try {
262             channel.queueDeclarePassive(queueOrTopicName);
263           } catch (IOException e) {
264             error = "queue "+queueOrTopicName+" doesn't exist. Either create the queue or set the autoCreateObjects to true.";
265           }
266         }
267     }
268     return error;
269   }
