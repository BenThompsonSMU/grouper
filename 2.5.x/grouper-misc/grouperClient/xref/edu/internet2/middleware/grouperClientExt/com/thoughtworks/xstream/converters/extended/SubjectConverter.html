26  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.extended;
27  
28  import java.util.Collections;
29  import java.util.HashSet;
30  import java.util.Iterator;
31  import java.util.Set;
32  
33  import javax.security.auth.Subject;
34  
35  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.MarshallingContext;
36  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.UnmarshallingContext;
37  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.collections.AbstractCollectionConverter;
38  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamReader;
39  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamWriter;
40  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.mapper.Mapper;
41  
42  /**
43   * Converts a {@link Subject} instance. Note, that this Converter does only convert the contained Principals as
44   * it is done by JDK serialization, but not any credentials. For other behaviour you can derive your own converter,
45   * overload the appropriate methods and register it in the {@link edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.XStream}.
46   *
47   * @author J&ouml;rg Schaible
48   * @since 1.1.3
49   */
50  public class SubjectConverter extends AbstractCollectionConverter {
51  
52      public SubjectConverter(Mapper mapper) {
53          super(mapper);
54      }
55  
56      public boolean canConvert(Class type) {
57          return type.equals(Subject.class);
58      }
59  
60      public void marshal(Object source, HierarchicalStreamWriter writer, MarshallingContext context) {
61          Subject subject = (Subject) source;
62          marshalPrincipals(subject.getPrincipals(), writer, context);
63          marshalPublicCredentials(subject.getPublicCredentials(), writer, context);
64          marshalPrivateCredentials(subject.getPrivateCredentials(), writer, context);
65          marshalReadOnly(subject.isReadOnly(), writer);
66      }
67      
68      protected void marshalPrincipals(Set principals, HierarchicalStreamWriter writer, MarshallingContext context) {
69          writer.startNode("principals");
70          for (final Iterator iter = principals.iterator(); iter.hasNext();) {
71              final Object principal = iter.next(); // pre jdk 1.4 a Principal was also in javax.security
72              writeItem(principal, context, writer);
73          }
74          writer.endNode();
75      };
76      
77      protected void marshalPublicCredentials(Set pubCredentials, HierarchicalStreamWriter writer, MarshallingContext context) {
78      };
79  
80      protected void marshalPrivateCredentials(Set privCredentials, HierarchicalStreamWriter writer, MarshallingContext context) {
81      };
82    
83      protected void marshalReadOnly(boolean readOnly, HierarchicalStreamWriter writer) {
84          writer.startNode("readOnly");
85          writer.setValue(String.valueOf(readOnly));
86          writer.endNode();
87      };
88  
89      public Object unmarshal(HierarchicalStreamReader reader, UnmarshallingContext context) {
90          Set principals = unmarshalPrincipals(reader, context);
91          Set publicCredentials = unmarshalPublicCredentials(reader, context);
92          Set privateCredentials = unmarshalPrivateCredentials(reader, context);
93          boolean readOnly = unmarshalReadOnly(reader);
94          return new Subject(readOnly, principals, publicCredentials, privateCredentials);
95      }
96      
97      protected Set unmarshalPrincipals(HierarchicalStreamReader reader, UnmarshallingContext context) {
98          return populateSet(reader, context);
99      };
100     
101     protected Set unmarshalPublicCredentials(HierarchicalStreamReader reader, UnmarshallingContext context) {
102         return Collections.EMPTY_SET;
103     };
104 
105     protected Set unmarshalPrivateCredentials(HierarchicalStreamReader reader, UnmarshallingContext context) {
106         return Collections.EMPTY_SET;
107     };
108 
109     protected boolean unmarshalReadOnly(HierarchicalStreamReader reader) {
110         reader.moveDown();
111         boolean readOnly = Boolean.getBoolean(reader.getValue());
112         reader.moveUp();
113         return readOnly;
114     };
115 
116     protected Set populateSet(HierarchicalStreamReader reader, UnmarshallingContext context) {
117         Set set = new HashSet();
118         reader.moveDown();
119         while (reader.hasMoreChildren()) {
120             reader.moveDown();
121             Object elementl = readItem(reader, context, set);
122             reader.moveUp();
123             set.add(elementl);
124         }
125         reader.moveUp();
126         return set;
127     }
128 }
