27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.extended;
28  
29  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.ConversionException;
30  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.basic.AbstractSingleValueConverter;
31  
32  /**
33   * Converts a java.lang.Class to XML.
34   * 
35   * @author Aslak Helles&oslash;y
36   * @author Joe Walnes
37   * @author Matthew Sandoz
38   * @author J&ouml;rg Schaible
39   */
40  public class JavaClassConverter extends AbstractSingleValueConverter {
41  
42      private ClassLoader classLoader;
43  
44      /**
45       * @deprecated As of 1.1.1 - use other constructor and explicitly supply a ClassLoader.
46       */
47      public JavaClassConverter() {
48          this(Thread.currentThread().getContextClassLoader());
49      }
50  
51      public JavaClassConverter(ClassLoader classLoader) {
52          this.classLoader = classLoader;
53      }
54  
55      public boolean canConvert(Class clazz) {
56          return Class.class.equals(clazz); // :)
57      }
58  
59      public String toString(Object obj) {
60          return ((Class) obj).getName();
61      }
62  
63      public Object fromString(String str) {
64          try {
65              return loadClass(str);
66          } catch (ClassNotFoundException e) {
67              throw new ConversionException("Cannot load java class " + str, e);
68          }
69      }
70  
71      private Class loadClass(String className) throws ClassNotFoundException {
72          Class resultingClass = primitiveClassForName(className);
73          if( resultingClass != null ){
74              return resultingClass;
75          }
76          int dimension;
77          for(dimension = 0; className.charAt(dimension) == '['; ++dimension);
78          if (dimension > 0) {
79              final ClassLoader classLoaderToUse;
80              if (className.charAt(dimension) == 'L') {
81                  String componentTypeName = className.substring(dimension + 1, className.length() - 1);
82                  classLoaderToUse = classLoader.loadClass(componentTypeName).getClassLoader();
83              } else {
84                  classLoaderToUse = null;
85              }
86              return Class.forName(className, false, classLoaderToUse);
87          }
88          return classLoader.loadClass(className);
89      }
90  
91      /**
92       * Lookup table for primitive types.
93       */
94      private Class primitiveClassForName(String name) {
95          return  name.equals("void") ? Void.TYPE :
96                  name.equals("boolean") ? Boolean.TYPE :
97                  name.equals("byte") ? Byte.TYPE :
98                  name.equals("char") ? Character.TYPE :
99                  name.equals("short") ? Short.TYPE :
100                 name.equals("int") ? Integer.TYPE :
101                 name.equals("long") ? Long.TYPE :
102                 name.equals("float") ? Float.TYPE :
103                 name.equals("double") ? Double.TYPE :
104                 null;
105     }
106 
107 }
