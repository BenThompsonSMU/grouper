42  /**
43   * A {@link SAXSource JAXP TrAX Source} that enables using XStream object serialization as
44   * direct input for XSLT processors without resorting to an intermediate representation such as
45   * text XML, DOM or DOM4J. <p/> The following example shows how to apply an XSL Transformation
46   * to a set of Java objects gathered into a List (<code>source</code>):
47   * </p>
48   * 
49   * <pre><code>
50   * public static String transform(List source, String stylesheet) {
51   *     try {
52   *         Transformer transformer = TransformerFactory.newInstance().newTransformer(
53   *             new StreamSource(stylesheet));
54   *         TraxSource in = new TraxSource(source);
55   *         Writer out = new StringWriter();
56   *         transformer.transform(in, new StreamResult(out));
57   *         return out.toString();
58   *     } catch (TransformerException e) {
59   *         throw new RuntimeException(&quot;XSLT Transformation failed&quot;, e);
60   *     }
61   * }
62   * </code></pre>
63   * 
64   * @author Laurent Bihanic
65   */
66  public class TraxSource extends SAXSource {
67  
68      /**
69       * If {@link javax.xml.transform.TransformerFactory#getFeature} returns <code>true</code>
70       * when passed this value as an argument, the Transformer natively supports XStream. <p/>
71       * <strong>Note</strong>: This implementation does not override the
72       * {@link SAXSource#FEATURE} value defined by its superclass to be considered as a SAXSource
73       * by Transformer implementations not natively supporting this XStream-specific source
74       * </p>
75       */
76      public final static String XSTREAM_FEATURE = "http://com.thoughtworks.xstream/XStreamSource/feature";
77  
78      /**
79       * The XMLReader object associated to this source or <code>null</code> if no XMLReader has
80       * yet been requested.
81       * 
82       * @see #getXMLReader
83       */
84      private XMLReader xmlReader = null;
85  
86      /**
87       * The configured XStream facade to use for serializing objects.
88       */
89      private XStream xstream = null;
90  
91      /**
92       * The list of Java objects to be serialized.
93       */
94      private List source = null;
95  
96      // -------------------------------------------------------------------------
97      // Constructors
98      // -------------------------------------------------------------------------
99  
100     /**
101      * Creates a XStream TrAX source.
102      */
103     public TraxSource() {
104         super(new InputSource());
105     }
106 
107     /**
108      * Creates a XStream TrAX source, specifying the object to marshal.
109      * 
110      * @param source the object to marshal.
111      * @throws IllegalArgumentException if <code>source</code> is <code>null</code>.
112      * @see #setSource(java.lang.Object)
113      */
114     public TraxSource(Object source) {
115         super(new InputSource());
116 
117         this.setSource(source);
118     }
119 
120     /**
121      * Creates a XStream TrAX source, specifying the object to marshal and a configured (with
122      * aliases) XStream facade.
123      * 
124      * @param source the object to marshal.
125      * @param xstream a configured XStream facade.
126      * @throws IllegalArgumentException if <code>source</code> or <code>xstream</code> is
127      *                 <code>null</code>.
128      * @see #setSource(java.lang.Object)
129      * @see #setXStream(edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.XStream)
130      */
131     public TraxSource(Object source, XStream xstream) {
132         super(new InputSource());
133 
134         this.setSource(source);
135         this.setXStream(xstream);
136     }
137 
138     /**
139      * Creates a XStream TrAX source, setting the objects to marshal.
140      * 
141      * @param source the list of objects to marshal.
142      * @throws IllegalArgumentException if <code>source</code> is <code>null</code> or
143      *                 empty.
144      * @see #setSourceAsList(java.util.List)
145      */
146     public TraxSource(List source) {
147         super(new InputSource());
148 
149         this.setSourceAsList(source);
150     }
151 
152     /**
153      * Creates a XStream TrAX source, setting the objects to marshal and a configured (with
154      * aliases) XStream facade.
155      * 
156      * @param source the list of objects to marshal.
157      * @param xstream a configured XStream facade.
158      * @throws IllegalArgumentException if <code>source</code> or <code>xstream</code> is
159      *                 <code>null</code> or <code>source</code> is empty.
160      * @see #setSourceAsList(java.util.List)
161      * @see #setXStream(edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.XStream)
162      */
163     public TraxSource(List source, XStream xstream) {
164         super(new InputSource());
165 
166         this.setSourceAsList(source);
167         this.setXStream(xstream);
168     }

171     // SAXSource overwritten methods
172     // -------------------------------------------------------------------------
173 
174     /**
175      * Sets the SAX InputSource to be used for the Source. <p/> As this implementation only
176      * supports object lists as data source, this method always throws an
177      * {@link UnsupportedOperationException}.
178      * </p>
179      * 
180      * @param inputSource a valid InputSource reference.
181      * @throws UnsupportedOperationException always!
182      */
183     public void setInputSource(InputSource inputSource) {
184         throw new UnsupportedOperationException();
185     }
186 
187     /**
188      * Set the XMLReader to be used for the Source. <p/> As this implementation only supports
189      * object lists as data source, this method throws an {@link UnsupportedOperationException}
190      * if the provided reader object does not implement the SAX {@link XMLFilter} interface.
191      * Otherwise, a {@link SaxWriter} instance will be attached as parent of the filter chain.
192      * </p>
193      * 
194      * @param reader a valid XMLReader or XMLFilter reference.
195      * @throws UnsupportedOperationException if <code>reader</code> is not a SAX
196      *                 {@link XMLFilter}.
197      * @see #getXMLReader
198      */
199     public void setXMLReader(XMLReader reader) {
200         this.createXMLReader(reader);
201     }
202 
203     /**
204      * Returns the XMLReader to be used for the Source. <p/> This implementation returns a
205      * specific XMLReader ({@link SaxWriter}) generating the XML from a list of input objects.
206      * </p>
207      * 
208      * @return an XMLReader generating the XML from a list of input objects.
209      */
210     public XMLReader getXMLReader() {
211         if (this.xmlReader == null) {
212             this.createXMLReader(null);
213         }
214         return this.xmlReader;
215     }
216 
217     // -------------------------------------------------------------------------
218     // Specific implementation
219     // -------------------------------------------------------------------------
220 
221     /**
222      * Sets the XStream facade to use when marshalling objects.
223      * 
224      * @param xstream a configured XStream facade.
225      * @throws IllegalArgumentException if <code>xstream</code> is <code>null</code>.
226      */
227     public void setXStream(XStream xstream) {
228         if (xstream == null) {
229             throw new IllegalArgumentException("xstream");
230         }
231         this.xstream = xstream;
232 
233         this.configureXMLReader();
234     }
235 
236     /**
237      * Sets the object to marshal.
238      * 
239      * @param obj the object to marshal.
240      * @throws IllegalArgumentException if <code>source</code> is <code>null</code>.
241      */
242     public void setSource(Object obj) {
243         if (obj == null) {
244             throw new IllegalArgumentException("obj");
245         }
246         List list = new ArrayList(1);
247         list.add(obj);
248 
249         this.setSourceAsList(list);
250     }
251 
252     /**
253      * Sets the list of objects to marshal. <p/> When dealing with non-text input (such as SAX
254      * or DOM), XSLT processors support multiple root node children for the source tree (see <a
255      * href="http://www.w3.org/TR/xslt#root-node-children">section 3.1</a> of the &quot;XSL
256      * Transformations (XSLT) Version 1.0&quot; specification. Using a list of objects as source
257      * makes use of this feature and allows creating XML documents merging the XML serialization
258      * of several Java objects.
259      * 
260      * @param list the list of objects to marshal.
261      * @throws IllegalArgumentException if <code>source</code> is <code>null</code> or
262      *                 empty.
263      */
264     public void setSourceAsList(List list) {
265         if ((list == null) || (list.isEmpty())) {
266             throw new IllegalArgumentException("list");
267         }
268         this.source = list;
269 
270         this.configureXMLReader();
271     }
272 
273     private void createXMLReader(XMLReader filterChain) {
274         if (filterChain == null) {
275             this.xmlReader = new SaxWriter();
276         } else {
277             if (filterChain instanceof XMLFilter) {
278                 // Connect the filter chain to a document reader.
279                 XMLFilter filter = (XMLFilter)filterChain;
280                 while (filter.getParent() instanceof XMLFilter) {
281                     filter = (XMLFilter)(filter.getParent());
282                 }
283                 if (!(filter.getParent() instanceof SaxWriter)) {
284                     filter.setParent(new SaxWriter());
285                 }
286 
287                 // Read XML data from filter chain.
288                 this.xmlReader = filterChain;
289             } else {
290                 throw new UnsupportedOperationException();
291             }
292         }
293         this.configureXMLReader();
294     }
295 
296     private void configureXMLReader() {
297         if (this.xmlReader != null) {
298             try {
299                 if (this.xstream != null) {
300                     this.xmlReader.setProperty(
301                         SaxWriter.CONFIGURED_XSTREAM_PROPERTY, this.xstream);
302                 }
303                 if (this.source != null) {
304                     this.xmlReader.setProperty(
305                         SaxWriter.SOURCE_OBJECT_LIST_PROPERTY, this.source);
306                 }
307             } catch (SAXException e) {
308                 throw new IllegalArgumentException(e.getMessage());
309             }
310         }
311     }
