46  package edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient;
47  
48  import java.io.IOException;
49  import java.io.InputStream;
50  
51  import edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.params.HttpConnectionManagerParams;
52  import edu.internet2.middleware.grouperClientExt.org.apache.commons.logging.Log;
53  import edu.internet2.middleware.grouperClientExt.org.apache.commons.logging.LogFactory;
54  
55  /**
56   * A connection manager that provides access to a single HttpConnection.  This
57   * manager makes no attempt to provide exclusive access to the contained
58   * HttpConnection.
59   *
60   * @author <a href="mailto:becke@u.washington.edu">Michael Becke</a>
61   * @author Eric Johnson
62   * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
63   * @author <a href="mailto:oleg@ural.ru">Oleg Kalnichevski</a>
64   * @author Laura Werner
65   * 
66   * @since 2.0
67   */
68  public class SimpleHttpConnectionManager implements HttpConnectionManager {
69  
70      private static final Log LOG = LogFactory.getLog(SimpleHttpConnectionManager.class);
71  
72      private static final String MISUSE_MESSAGE = 
73          "SimpleHttpConnectionManager being used incorrectly.  Be sure that"
74          + " HttpMethod.releaseConnection() is always called and that only one thread"
75          + " and/or method is using this connection manager at a time.";
76      
77      /**
78       * Since the same connection is about to be reused, make sure the
79       * previous request was completely processed, and if not
80       * consume it now.
81       * @param conn The connection
82       */
83      static void finishLastResponse(HttpConnection conn) {
84          InputStream lastResponse = conn.getLastResponseInputStream();
85          if (lastResponse != null) {
86              conn.setLastResponseInputStream(null);
87              try {
88                  lastResponse.close();
89              } catch (IOException ioe) {
90                  conn.close();
91              }
92          }
93      }
94      
95      /** The http connection */
96      protected HttpConnection httpConnection;
97  
98      /**
99       * Collection of parameters associated with this connection manager.
100      */
101     private HttpConnectionManagerParamstExt/org/apache/commons/httpclient/params/HttpConnectionManagerParams.html#HttpConnectionManagerParams">HttpConnectionManagerParams params = new HttpConnectionManagerParams(); 
102 
103     /**
104      * The time the connection was made idle.
105      */
106     private long idleStartTime = Long.MAX_VALUE;
107     
108     /**
109      * Used to test if {@link #httpConnection} is currently in use 
110      * (i.e. checked out).  This is only used as a sanity check to help
111      * debug cases where this connection manager is being used incorrectly.
112      * It will not be used to enforce thread safety.
113      */
114     private volatile boolean inUse = false;
115 
116     private boolean alwaysClose = false;
117 
118     /**
119      * The connection manager created with this constructor will try to keep the 
120      * connection open (alive) between consecutive requests if the alwaysClose 
121      * parameter is set to <tt>false</tt>. Otherwise the connection manager will 
122      * always close connections upon release.
123      * 
124      * @param alwaysClose if set <tt>true</tt>, the connection manager will always
125      *    close connections upon release.
126      */
127     public SimpleHttpConnectionManager(boolean alwaysClose) {
128         super();
129         this.alwaysClose = alwaysClose;
130     }
131     
132     /**
133      * The connection manager created with this constructor will always try to keep 
134      * the connection open (alive) between consecutive requests.
135      */
136     public SimpleHttpConnectionManager() {
137         super();
138     }
139     
140     /**
141      * @see HttpConnectionManager#getConnection(HostConfiguration)
142      */
143     public HttpConnection getConnection(HostConfiguration hostConfiguration) {
144         return getConnection(hostConfiguration, 0);
145     }
146 
147     /**
148      * Gets the staleCheckingEnabled value to be set on HttpConnections that are created.
149      * 
150      * @return <code>true</code> if stale checking will be enabled on HttpConections
151      * 
152      * @see HttpConnection#isStaleCheckingEnabled()
153      * 
154      * @deprecated Use {@link HttpConnectionManagerParams#isStaleCheckingEnabled()},
155      * {@link HttpConnectionManager#getParams()}.
156      */
157     public boolean isConnectionStaleCheckingEnabled() {
158         return this.params.isStaleCheckingEnabled();
159     }
160 
161     /**
162      * Sets the staleCheckingEnabled value to be set on HttpConnections that are created.
163      * 
164      * @param connectionStaleCheckingEnabled <code>true</code> if stale checking will be enabled 
165      * on HttpConections
166      * 
167      * @see HttpConnection#setStaleCheckingEnabled(boolean)
168      * 
169      * @deprecated Use {@link HttpConnectionManagerParams#setStaleCheckingEnabled(boolean)},
170      * {@link HttpConnectionManager#getParams()}.
171      */
172     public void setConnectionStaleCheckingEnabled(boolean connectionStaleCheckingEnabled) {
173         this.params.setStaleCheckingEnabled(connectionStaleCheckingEnabled);
174     }
175     
176     /**
177      * This method always returns the same connection object. If the connection is already
178      * open, it will be closed and the new host configuration will be applied.
179      * 
180      * @param hostConfiguration The host configuration specifying the connection
181      *        details.
182      * @param timeout this parameter has no effect. The connection is always returned
183      *        immediately.
184      * @since 3.0
185      */
186     public HttpConnection getConnectionWithTimeout(
187         HostConfiguration hostConfiguration, long timeout) {
188 
189         if (httpConnection == null) {
190             httpConnection = new HttpConnection(hostConfiguration);
191             httpConnection.setHttpConnectionManager(this);
192             httpConnection.getParams().setDefaults(this.params);
193         } else {
194 
195             // make sure the host and proxy are correct for this connection
196             // close it and set the values if they are not
197             if (!hostConfiguration.hostEquals(httpConnection)
198                 || !hostConfiguration.proxyEquals(httpConnection)) {
199                     
200                 if (httpConnection.isOpen()) {
201                     httpConnection.close();
202                 }
203 
204                 httpConnection.setHost(hostConfiguration.getHost());
205                 httpConnection.setPort(hostConfiguration.getPort());
206                 httpConnection.setProtocol(hostConfiguration.getProtocol());
207                 httpConnection.setLocalAddress(hostConfiguration.getLocalAddress());
208 
209                 httpConnection.setProxyHost(hostConfiguration.getProxyHost());
210                 httpConnection.setProxyPort(hostConfiguration.getProxyPort());
211             } else {
212                 finishLastResponse(httpConnection);
213             }
214         }
215 
216         // remove the connection from the timeout handler
217         idleStartTime = Long.MAX_VALUE;
218 
219         if (inUse) LOG.warn(MISUSE_MESSAGE);
220         inUse = true;
221         
222         return httpConnection;
223     }
224 
225     /**
226      * @see HttpConnectionManager#getConnection(HostConfiguration, long)
227      * 
228      * @deprecated Use #getConnectionWithTimeout(HostConfiguration, long)
229      */
230     public HttpConnection getConnection(
231         HostConfiguration hostConfiguration, long timeout) {
232         return getConnectionWithTimeout(hostConfiguration, timeout);
233     }
234 
235     /**
236      * @see HttpConnectionManager#releaseConnection(edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.HttpConnection)
237      */
238     public void releaseConnection(HttpConnection conn) {
239         if (conn != httpConnection) {
240             throw new IllegalStateException("Unexpected release of an unknown connection.");
241         }
242         if (this.alwaysClose) {
243             httpConnection.close();
244         } else {
245             // make sure the connection is reuseable
246             finishLastResponse(httpConnection);
247         }
248         
249         inUse = false;
250 
251         // track the time the connection was made idle
252         idleStartTime = System.currentTimeMillis();
253     }
254 
255     /**
256      * Returns {@link HttpConnectionManagerParams parameters} associated 
257      * with this connection manager.
258      * 
259      * @since 2.1
260      * 
261      * @see HttpConnectionManagerParams
262      */
263     public HttpConnectionManagerParams getParams() {
264         return this.params;
265     }
266 
267     /**
268      * Assigns {@link HttpConnectionManagerParams parameters} for this 
269      * connection manager.
270      * 
271      * @since 2.1
272      * 
273      * @see HttpConnectionManagerParams
274      */
275     public void setParams(final HttpConnectionManagerParams params) {
276         if (params == null) {
277             throw new IllegalArgumentException("Parameters may not be null");
278         }
279         this.params = params;
280     }
281     
282     /**
283      * @since 3.0
284      */
285     public void closeIdleConnections(long idleTimeout) {
286         long maxIdleTime = System.currentTimeMillis() - idleTimeout;
287         if (idleStartTime <= maxIdleTime) {
288             httpConnection.close();
289         }
290     }
291     
292     /**
293      * since 3.1
294      */
295     public void shutdown() {
296         httpConnection.close();
297     }
