27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.mapper;
28  
29  import java.util.HashMap;
30  import java.util.Iterator;
31  import java.util.Map;
32  
33  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.InitializationException;
34  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.alias.ClassMapper;
35  
36  
37  /**
38   * Mapper that resolves default implementations of classes. For example,
39   * mapper.serializedClass(ArrayList.class) will return java.util.List. Calling
40   * mapper.defaultImplementationOf(List.class) will return ArrayList.
41   * 
42   * @author Joe Walnes
43   */
44  public class DefaultImplementationsMapper extends MapperWrapper {
45  
46      private final Map typeToImpl = new HashMap();
47      private transient Map implToType = new HashMap();
48  
49      public DefaultImplementationsMapper(Mapper wrapped) {
50          super(wrapped);
51          addDefaults();
52      }
53  
54      /**
55       * @deprecated As of 1.2, use {@link #DefaultImplementationsMapper(Mapper)}
56       */
57      public DefaultImplementationsMapper(ClassMapper wrapped) {
58          this((Mapper)wrapped);
59      }
60  
61      protected void addDefaults() {
62          // null handling
63          addDefaultImplementation(null, Mapper.Null.class);
64          // register primitive types
65          addDefaultImplementation(Boolean.class, boolean.class);
66          addDefaultImplementation(Character.class, char.class);
67          addDefaultImplementation(Integer.class, int.class);
68          addDefaultImplementation(Float.class, float.class);
69          addDefaultImplementation(Double.class, double.class);
70          addDefaultImplementation(Short.class, short.class);
71          addDefaultImplementation(Byte.class, byte.class);
72          addDefaultImplementation(Long.class, long.class);
73      }
74  
75      public void addDefaultImplementation(Class defaultImplementation, Class ofType) {
76          if (defaultImplementation != null && defaultImplementation.isInterface()) {
77              throw new InitializationException(
78                  "Default implementation is not a concrete class: "
79                      + defaultImplementation.getName());
80          }
81          typeToImpl.put(ofType, defaultImplementation);
82          implToType.put(defaultImplementation, ofType);
83      }
84  
85      public String serializedClass(Class type) {
86          Class baseType = (Class)implToType.get(type);
87          return baseType == null ? super.serializedClass(type) : super.serializedClass(baseType);
88      }
89  
90      public Class defaultImplementationOf(Class type) {
91          if (typeToImpl.containsKey(type)) {
92              return (Class)typeToImpl.get(type);
93          } else {
94              return super.defaultImplementationOf(type);
95          }
96      }
97  
98      private Object readResolve() {
99          implToType = new HashMap();
100         for (final Iterator iter = typeToImpl.keySet().iterator(); iter.hasNext();) {
101             final Object type = iter.next();
102             implToType.put(typeToImpl.get(type), type);
103         }
104         return this;
105     }
106 
107 }
