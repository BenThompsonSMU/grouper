61  public class UsernamePasswordCredentials implements Credentials {
62  
63      // ----------------------------------------------------------- Constructors
64  
65      /**
66       * Default constructor.
67       * 
68       * @deprecated Do not use. Null user name no longer allowed
69       */
70      public UsernamePasswordCredentials() {
71          super();
72      }
73  
74  
75      /**
76       * The constructor with the username and password combined string argument.
77       *
78       * @param usernamePassword the username:password formed string
79       * @see #toString
80       */
81      public UsernamePasswordCredentials(String usernamePassword) {
82          super();
83          if (usernamePassword == null) {
84              throw new IllegalArgumentException("Username:password string may not be null");            
85          }
86          int atColon = usernamePassword.indexOf(':');
87          if (atColon >= 0) {
88              this.userName = usernamePassword.substring(0, atColon);
89              this.password = usernamePassword.substring(atColon + 1);
90          } else {
91              this.userName = usernamePassword;
92          }
93      }
94  
95  
96      /**
97       * The constructor with the username and password arguments.
98       *
99       * @param userName the user name
100      * @param password the password
101      */
102     public UsernamePasswordCredentials(String userName, String password) {
103         super();
104         if (userName == null) {
105             throw new IllegalArgumentException("Username may not be null");            
106         }
107         this.userName = userName;
108         this.password = password;
109     }
110

116     private String userName;

118 
119     /**
120      * Password.
121      */
122     private String password;

128     /**
129      * User name property setter. User name may not be null.
130      *
131      * @param userName
132      * @see #getUserName()
133      * 
134      * @deprecated Do not use. The UsernamePasswordCredentials objects should be immutable
135      */
136     public void setUserName(String userName) {
137         if (userName == null) {
138             throw new IllegalArgumentException("Username may not be null");            
139         }
140         this.userName = userName;
141     }
142 
143 
144     /**
145      * User name property getter.
146      *
147      * @return the userName
148      * @see #setUserName(String)
149      */

151         return userName;
152     }
153 
154 
155     /**
156      * Password property setter.
157      *
158      * @param password
159      * @see #getPassword()
160      *

163     public void setPassword(String password) {
164         this.password = password;
165     }
166 
167 
168     /**
169      * Password property getter.
170      *
171      * @return the password
172      * @see #setPassword(String)
173      */
174     public String getPassword() {
175         return password;
176     }
177 
178     
179     /**
180      * Get this object string.
181      *
182      * @return the username:password formed string
183      */
184     public String toString() {
185         StringBuffer result = new StringBuffer();
186         result.append(this.userName);
187         result.append(":");
188         result.append((this.password == null) ? "null" : this.password);
189         return result.toString();
190     }
191 
192     /**
193      * Does a hash of both user name and password.
194      *
195      * @return The hash code including user name and password.
196      */
197     public int hashCode() {
198         int hash = LangUtils.HASH_SEED;
199         hash = LangUtils.hashCode(hash, this.userName);
200         hash = LangUtils.hashCode(hash, this.password);
201         return hash;
202     }
