82      /**
83       * Construct an AnnotationMapper.
84       * 
85       * @param wrapped the next {@link Mapper} in the chain
86       * @since 1.3
87       */
88      public AnnotationMapper(
89          final Mapper wrapped, final ConverterRegistry converterRegistry,
90          final ClassLoader classLoader, final ReflectionProvider reflectionProvider,
91          final JVM jvm) {
92          super(wrapped);
93          this.converterRegistry = converterRegistry;
94          annotatedTypes.add(Object.class);
95          classAliasingMapper = (ClassAliasingMapper)lookupMapperOfType(ClassAliasingMapper.class);
96          defaultImplementationsMapper = (DefaultImplementationsMapper)lookupMapperOfType(DefaultImplementationsMapper.class);
97          implicitCollectionMapper = (ImplicitCollectionMapper)lookupMapperOfType(ImplicitCollectionMapper.class);
98          fieldAliasingMapper = (FieldAliasingMapper)lookupMapperOfType(FieldAliasingMapper.class);
99          attributeMapper = (AttributeMapper)lookupMapperOfType(AttributeMapper.class);
100         localConversionMapper = (LocalConversionMapper)lookupMapperOfType(LocalConversionMapper.class);
101         locked = true;
102         arguments = new Object[]{this, classLoader, reflectionProvider, jvm};
103     }

105     @Override
106     public String realMember(Class type, String serialized) {
107         if (!locked) {
108             processAnnotations(type);
109         }
110         return super.realMember(type, serialized);

145     public void processAnnotations(final Class[] initialTypes) {
146         if (initialTypes == null || initialTypes.length == 0) {
147             return;
148         }
149         locked = true;
150         synchronized (annotatedTypes) {
151             final Set<Class<?>> types = new UnprocessedTypesSet();
152             for (Class initialType : initialTypes) {
153                 types.add(initialType);
154             }
155             processTypes(types);
156         }
157     }
158 
159     private void processAnnotations(final Class initialType) {
160         if (initialType == null) {
161             return;
162         }
163         synchronized (annotatedTypes) {
164             final Set<Class<?>> types = new UnprocessedTypesSet();
165             types.add(initialType);
166             processTypes(types);
167         }
168     }
169 
170     private void processTypes(final Set<Class<?>> types) {
171         while (!types.isEmpty()) {
172             final Iterator<Class<?>> iter = types.iterator();
173             final Class<?> type = iter.next();
174             iter.remove();
175 
176             if (annotatedTypes.add(type)) {
177                 if (type.isPrimitive()) {
178                     continue;
179                 }
180 
181                 addParametrizedTypes(type, types);
182 
183                 processConverterAnnotations(type);
184                 processAliasAnnotation(type, types);
185 
186                 if (type.isInterface()) {
187                     continue;
188                 }
189 
190                 processImplicitCollectionAnnotation(type);
191 
192                 final Field[] fields = type.getDeclaredFields();
193                 for (int i = 0; i < fields.length; i++ ) {
194                     final Field field = fields[i];
195                     if (field.isEnumConstant()
196                         || (field.getModifiers() & (Modifier.STATIC | Modifier.TRANSIENT)) > 0) {
197                         continue;
198                     }
199 
200                     addParametrizedTypes(field.getGenericType(), types);
201 
202                     if (field.isSynthetic()) {
203                         continue;
204                     }
205 
206                     processFieldAliasAnnotation(field);
207                     processAsAttributeAnnotation(field);
208                     processImplicitAnnotation(field);
209                     processOmitFieldAnnotation(field);
210                     processLocalConverterAnnotation(field);
211                 }
212             }
213         }
214     }

216     private void addParametrizedTypes(Type type, final Set<Class<?>> types) {
217         final Set<Type> processedTypes = new HashSet<Type>();
218         Set<Type> localTypes = new LinkedHashSet<Type>() {
219 
220             @Override
221             public boolean add(Type o) {
222                 if (o instanceof Class) {
223                     return types.add((Class<?>)o);
224                 }
225                 return o == null || processedTypes.contains(o) ? false : super.add(o);
226             }
227 
228         };
229         while (type != null) {
230             processedTypes.add(type);
231             if (type instanceof Class) {
232                 Class<?> clazz = (Class<?>)type;
233                 types.add(clazz);
234                 if (!clazz.isPrimitive()) {
235                     final TypeVariable<?>[] typeParameters = clazz.getTypeParameters();
236                     for (TypeVariable<?> typeVariable : typeParameters) {
237                         localTypes.add(typeVariable);
238                     }
239                     localTypes.add(clazz.getGenericSuperclass());
240                     for (final Type iface : clazz.getGenericInterfaces()) {
241                         localTypes.add(iface);
242                     }
243                 }
244             } else if (type instanceof TypeVariable) {
245                 final TypeVariable<?> typeVariable = (TypeVariable<?>)type;
246                 final Type[] bounds = typeVariable.getBounds();
247                 for (Type bound : bounds) {
248                     localTypes.add(bound);
249                 }
250             } else if (type instanceof ParameterizedType) {
251                 final ParameterizedType parametrizedType = (ParameterizedType)type;
252                 localTypes.add(parametrizedType.getRawType());
253                 Type[] actualArguments = parametrizedType.getActualTypeArguments();
254                 for (Type actualArgument : actualArguments) {
255                     localTypes.add(actualArgument);
256                 }
257             } else if (type instanceof GenericArrayType) {
258                 final GenericArrayType arrayType = (GenericArrayType)type;
259                 localTypes.add(arrayType.getGenericComponentType());
260             }
261 
262             if (!localTypes.isEmpty()) {
263                 Iterator<Type> iter = localTypes.iterator();
264                 type = iter.next();
265                 iter.remove();
266             } else {
267                 type = null;
268             }
269         }
270     }

272     private void processConverterAnnotations(final Class<?> type) {
273         if (converterRegistry != null) {
274             final XStreamConverters convertersAnnotation = type
275                 .getAnnotation(XStreamConverters.class);
276             final XStreamConverter converterAnnotation = type.getAnnotation(XStreamConverter.class);
277             final List<XStreamConverter> annotations = convertersAnnotation != null
278                 ? new ArrayList<XStreamConverter>(Arrays.asList(convertersAnnotation.value()))
279                 : new ArrayList<XStreamConverter>();
280             if (converterAnnotation != null) {
281                 annotations.add(converterAnnotation);
282             }
283             for (final XStreamConverter annotation : annotations) {
284                 final Class<? extends Converter> converterType = annotation.value();
285                 final Converter converter = cacheConverter(converterType);
286                 if (converter != null) {
287                     if (converter != converterAnnotation || converter.canConvert(type)) {
288                         converterRegistry.registerConverter(converter, XStream.PRIORITY_NORMAL);
289                     } else {
290                         throw new InitializationException("Converter "
291                             + converterType.getName()
292                             + " cannot handle annotated class "
293                             + type.getName());
294                     }
295                 }
296             }
297         }
298     }

300     private void processAliasAnnotation(final Class<?> type, final Set<Class<?>> types) {
301         final XStreamAlias aliasAnnotation = type.getAnnotation(XStreamAlias.class);
302         if (aliasAnnotation != null) {
303             if (classAliasingMapper == null) {
304                 throw new InitializationException("No "
305                     + ClassAliasingMapper.class.getName()
306                     + " available");
307             }
308             if (aliasAnnotation.impl() != Void.class) {
309                 // Alias for Interface/Class with an impl
310                 classAliasingMapper.addClassAlias(aliasAnnotation.value(), type);
311                 defaultImplementationsMapper.addDefaultImplementation(
312                     aliasAnnotation.impl(), type);
313                 if (type.isInterface()) {
314                     types.add(aliasAnnotation.impl()); // alias Interface's impl
315                 }
316             } else {
317                 classAliasingMapper.addClassAlias(aliasAnnotation.value(), type);
318             }
319         }
320     }

322     @Deprecated
323     private void processImplicitCollectionAnnotation(final Class<?> type) {
324         final XStreamImplicitCollection implicitColAnnotation = type
325             .getAnnotation(XStreamImplicitCollection.class);
326         if (implicitColAnnotation != null) {
327             if (implicitCollectionMapper == null) {
328                 throw new InitializationException("No "
329                     + ImplicitCollectionMapper.class.getName()
330                     + " available");
331             }
332             final String fieldName = implicitColAnnotation.value();
333             final String itemFieldName = implicitColAnnotation.item();
334             final Field field;
335             try {
336                 field = type.getDeclaredField(fieldName);
337             } catch (final NoSuchFieldException e) {
338                 throw new InitializationException(type.getName()
339                     + " does not have a field named '"
340                     + fieldName
341                     + "' as required by "
342                     + XStreamImplicitCollection.class.getName());
343             }
344             Class itemType = null;
345             final Type genericType = field.getGenericType();
346             if (genericType instanceof ParameterizedType) {
347                 final Type typeArgument = ((ParameterizedType)genericType)
348                     .getActualTypeArguments()[0];
349                 itemType = getClass(typeArgument);
350             }
351             if (itemType == null) {
352                 implicitCollectionMapper.add(type, fieldName, null, Object.class);
353             } else {
354                 if (itemFieldName.equals("")) {
355                     implicitCollectionMapper.add(type, fieldName, null, itemType);
356                 } else {
357                     implicitCollectionMapper.add(type, fieldName, itemFieldName, itemType);
358                 }
359             }
360         }
361     }

363     private void processFieldAliasAnnotation(final Field field) {
364         final XStreamAlias aliasAnnotation = field.getAnnotation(XStreamAlias.class);
365         if (aliasAnnotation != null) {
366             if (fieldAliasingMapper == null) {
367                 throw new InitializationException("No "
368                     + FieldAliasingMapper.class.getName()
369                     + " available");
370             }
371             fieldAliasingMapper.addFieldAlias(aliasAnnotation.value(), field
372                 .getDeclaringClass(), field.getName());
373         }
374     }

376     private void processAsAttributeAnnotation(final Field field) {
377         final XStreamAsAttribute asAttributeAnnotation = field
378             .getAnnotation(XStreamAsAttribute.class);
379         if (asAttributeAnnotation != null) {
380             if (attributeMapper == null) {
381                 throw new InitializationException("No "
382                     + AttributeMapper.class.getName()
383                     + " available");
384             }
385             attributeMapper.addAttributeFor(field);
386         }
387     }

389     private void processImplicitAnnotation(final Field field) {
390         final XStreamImplicit implicitAnnotation = field.getAnnotation(XStreamImplicit.class);
391         if (implicitAnnotation != null) {
392             if (implicitCollectionMapper == null) {
393                 throw new InitializationException("No "
394                     + ImplicitCollectionMapper.class.getName()
395                     + " available");
396             }
397             String fieldName = field.getName();
398             String itemFieldName = implicitAnnotation.itemFieldName();
399             Class itemType = null;
400             final Type genericType = field.getGenericType();
401             if (genericType instanceof ParameterizedType) {
402                 final Type typeArgument = ((ParameterizedType)genericType)
403                     .getActualTypeArguments()[0];
404                 itemType = getClass(typeArgument);
405             }
406             if (itemFieldName != null && !"".equals(itemFieldName)) {
407                 implicitCollectionMapper.add(
408                     field.getDeclaringClass(), fieldName, itemFieldName, itemType);
409             } else {
410                 implicitCollectionMapper.add(field.getDeclaringClass(), fieldName, itemType);
411             }
412         }
413     }
414 
415     private void processOmitFieldAnnotation(final Field field) {
416         final XStreamOmitField omitFieldAnnotation = field
417             .getAnnotation(XStreamOmitField.class);
418         if (omitFieldAnnotation != null) {
419             if (fieldAliasingMapper == null) {
420                 throw new InitializationException("No "
421                     + FieldAliasingMapper.class.getName()
422                     + " available");
423             }
424             fieldAliasingMapper.omitField(field.getDeclaringClass(), field.getName());
425         }
426     }
427 
428     private void processLocalConverterAnnotation(final Field field) {
429         final XStreamConverter annotation = field.getAnnotation(XStreamConverter.class);
430         if (annotation != null) {
431             final Class<? extends Converter> converterType = annotation.value();
432             final Converter converter = cacheConverter(converterType);
433             if (converter != null) {
434                 if (localConversionMapper == null) {
435                     throw new InitializationException("No "
436                         + LocalConversionMapper.class.getName()
437                         + " available");
438                 }
439                 localConversionMapper.registerLocalConverter(field.getDeclaringClass(), field
440                     .getName(), converter);
441             }
442         }
443     }
444 
445     private Converter cacheConverter(final Class<? extends Converter> converterType) {
446         Converter converter = converterCache.get(converterType);
447         if (converter == null) {
448             try {
449                 converter = (Converter)DependencyInjectionFactory.newInstance(
450                     converterType, arguments);
451                 converterCache.put(converterType, converter);
452             } catch (final Exception e) {
453                 throw new InitializationException("Cannot instantiate converter "
454                     + converterType.getName(), e);
455             }
456         }
457         return converter;
458     }

460     private Class<?> getClass(final Type typeArgument) {
461         Class<?> type = null;
462         if (typeArgument instanceof ParameterizedType) {
463             type = (Class<?>)((ParameterizedType)typeArgument).getRawType();
464         } else if (typeArgument instanceof Class) {
465             type = (Class<?>)typeArgument;
466         }
467         return type;
468     }

470     private final class UnprocessedTypesSet extends LinkedHashSet<Class<?>> {
471         @Override
472         public boolean add(Class<?> type) {
473             if (type == null) {
474                 return false;
475             }
476             while (type.isArray()) {
477                 type = type.getComponentType();
478             }
479             final String name = type.getName();
480             if (name.startsWith("java.") || name.startsWith("java.")) {
481                 return false;
482             }
483             return annotatedTypes.contains(type) ? false : super.add(type);
484         }
485     }

487     private static class WeakHashSet<K> implements Set<K> {
488 
489         private static Object NULL = new Object();
490         private WeakHashMap<K, Object> map = new WeakHashMap<K, Object>();
491 
492         public boolean add(K o) {
493             return map.put(o, NULL) == null;
494         }
495 
496         public boolean addAll(Collection<? extends K> c) {
497             boolean ret = false;
498             for (K k : c) {
499                 ret = add(k) | false;
500             }
501             return ret;
502         }
503 
504         public void clear() {
505             map.clear();
506         }
507 
508         public boolean contains(Object o) {
509             return map.containsKey(o);
510         }
511 
512         public boolean containsAll(Collection<?> c) {
513             return map.keySet().containsAll(c);
514         }
515 
516         public boolean isEmpty() {
517             return map.isEmpty();
518         }
519 
520         public Iterator<K> iterator() {
521             return map.keySet().iterator();
522         }
523 
524         public boolean remove(Object o) {
525             return map.remove(o) != null;
526         }
527 
528         public boolean removeAll(Collection<?> c) {
529             boolean ret = false;
530             for (Object object : c) {
531                 ret = remove(object) | false;
532             }
533             return ret;
534         }
535 
536         public boolean retainAll(Collection<?> c) {
537             boolean ret = false;
538             for (final Iterator<K> iter = iterator(); iter.hasNext();) {
539                 final K element = iter.next();
540                 if (!c.contains(element)) {
541                     iter.remove();
542                     ret = true;
543                 }
544             }
545             return ret;
546         }
547 
548         public int size() {
549             return map.size();
550         }
551 
552         public Object[] toArray() {
553             return map.keySet().toArray();
554         }
555 
556         public <T> T[] toArray(T[] a) {
557             return map.keySet().toArray(a);
558         }
