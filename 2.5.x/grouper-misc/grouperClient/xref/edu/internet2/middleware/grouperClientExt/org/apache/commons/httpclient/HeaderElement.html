55  /**
56   * <p>One element of an HTTP header's value.</p>
57   * <p>
58   * Some HTTP headers (such as the set-cookie header) have values that
59   * can be decomposed into multiple elements.  Such headers must be in the
60   * following form:
61   * </p>
62   * <pre>
63   * header  = [ element ] *( "," [ element ] )
64   * element = name [ "=" [ value ] ] *( ";" [ param ] )
65   * param   = name [ "=" [ value ] ]
66   *
67   * name    = token
68   * value   = ( token | quoted-string )
69   *
70   * token         = 1*&lt;any char except "=", ",", ";", &lt;"&gt; and
71   *                       white space&gt;
72   * quoted-string = &lt;"&gt; *( text | quoted-char ) &lt;"&gt;
73   * text          = any char except &lt;"&gt;
74   * quoted-char   = "\" char
75   * </pre>
76   * <p>
77   * Any amount of white space is allowed between any part of the
78   * header, element or param and is ignored. A missing value in any
79   * element or param will be stored as the empty {@link String};
80   * if the "=" is also missing <var>null</var> will be stored instead.
81   * </p>
82   * <p>
83   * This class represents an individual header element, containing
84   * both a name/value pair (value may be <tt>null</tt>) and optionally
85   * a set of additional parameters.
86   * </p>
87   * <p>
88   * This class also exposes a {@link #parse} method for parsing a
89   * {@link Header} value into an array of elements.
90   * </p>
91   *
92   * @see Header
93   *
94   * @author <a href="mailto:bcholmes@interlog.com">B.C. Holmes</a>
95   * @author <a href="mailto:jericho@thinkfree.com">Park, Sung-Gu</a>
96   * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
97   * @author <a href="mailto:oleg@ural.com">Oleg Kalnichevski</a>
98   * 
99   * @since 1.0
100  * @version $Revision: 1.1 $ $Date: 2008-11-30 10:57:19 $
101  */

197     /**
198      * This parses the value part of a header. The result is an array of
199      * HeaderElement objects.
200      *
201      * @param headerValue  the array of char representation of the header value
202      *                     (as received from the web server).
203      * @return array of {@link HeaderElement}s.
204      * 
205      * @since 3.0
206      */
207     public static final HeaderElement[] parseElements(char[] headerValue) {
208             
209         LOG.trace("enter HeaderElement.parseElements(char[])");
210 
211         if (headerValue == null) {
212             return new HeaderElement[] {};
213         }
214         List elements = new ArrayList(); 
215         
216         int i = 0;
217         int from = 0;
218         int len = headerValue.length;
219         boolean qouted = false;
220         while (i < len) {
221             char ch = headerValue[i];
222             if (ch == '"') {
223                 qouted = !qouted;
224             }
225             HeaderElement element = null;
226             if ((!qouted) && (ch == ',')) {
227                 element = new HeaderElement(headerValue, from, i);
228                 from = i + 1;
229             } else if (i == len - 1) {
230                 element = new HeaderElement(headerValue, from, len);
231             }
232             if ((element != null) && (element.getName() != null)) {
233                 elements.add(element);
234             }
235             i++;
236         }
237         return (HeaderElement[])
238             elements.toArray(new HeaderElement[elements.size()]);
239     }

284     /**
285      * Returns parameter with the given name, if found. Otherwise null 
286      * is returned
287      *
288      * @param name The name to search by.
289      * @return NameValuePair parameter with the given name
290      */
291 
292     public NameValuePair getParameterByName(String name) {
293 
294         LOG.trace("enter HeaderElement.getParameterByName(String)");
295 
296         if (name == null) {
297             throw new IllegalArgumentException("Name may not be null");
298         } 
299         NameValuePair found = null;
300         NameValuePair parameters[] = getParameters();
301         if (parameters != null) {
302             for (int i = 0; i < parameters.length; i++) {
303                 NameValuePair current = parameters[ i ];
304                 if (current.getName().equalsIgnoreCase(name)) {
305                     found = current;
306                     break;
307                 }
308             }
309         }
310         return found;
311     }
