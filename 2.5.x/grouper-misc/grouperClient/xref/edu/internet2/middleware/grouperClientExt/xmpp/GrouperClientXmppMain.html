82    /**
83    * see if the esb event matches an EL filter. Note the available objects are
84    * event for the EsbEvent, and grouperUtil for the GrouperUtil class which has
85    * a lot of utility methods
86    * @param filterString
87    * @param esbEvent
88    * @return true if matches, false if doesnt
89    */
90    public static boolean matchesFilter(EsbEvent esbEvent, String filterString) {
91      try {
92        Map<String, Object> variableMap = new HashMap<String, Object>();
93        variableMap.put("event", esbEvent);
94        variableMap.put("grouperClientUtils", new GrouperClientUtils());
95        
96        String elResultString = GrouperClientUtils.substituteExpressionLanguage(filterString, variableMap, true, false, false, true);
97        return GrouperClientUtils.booleanValue(elResultString);
98        
99      } catch (Exception e) {
100       throw new RuntimeException("Problem seeing if matches filter for sequence: "
101           + (esbEvent == null ? null : esbEvent.getSequenceNumber()) + ", '" + filterString + "'", e);
102     }
103   }

108   /**
109   * get an xmpp connection
110   * @return xmpp connection
111   */
112   private static synchronized XMPPConnection xmppConnection() {
113     if (xmppConnection == null || !xmppConnection.isAuthenticated() || !xmppConnection.isConnected()) {
114       String user = null;
115       String pass = null;
116       String resource = null;
117       String server = null;
118       int port = -1;
119       try {
120         if (xmppConnection != null) {
121           try {
122             xmppConnection.disconnect();
123           } catch (Exception e) {
124             //this is ok
125           }
126         }
127         server = xmppServer();
128         port = xmppPort();
129         ConnectionConfiguration config = new ConnectionConfiguration(server, port);
130 
131         boolean debuggerEnabled = GrouperClientConfig.retrieveConfig().propertyValueBoolean("grouperClient.xmpp.debuggerEnabled", false);
132 
133         config.setDebuggerEnabled(debuggerEnabled);
134         config.setReconnectionAllowed(true);
135 
136         config.setSASLAuthenticationEnabled(true);
137         SASLAuthentication.supportSASLMechanism("PLAIN");
138         config.setSecurityMode(ConnectionConfiguration.SecurityMode.enabled);
139         xmppConnection = new XMPPConnection(config);
140         xmppConnection.connect();
141 
142         user = xmppUser();
143         pass = xmppPass();
144         resource = xmppResource();
145         xmppConnection.login(user, pass, resource);
146       } catch (XMPPException xe) {
147         throw new RuntimeException("Problem connecting: server: " + server + ", port: "
148             + port + ", user: " + user + ", pass not included, "
149             //+ GrouperClientUtils.repeat("*", GrouperClientUtils.defaultString(pass).length())
150             + ", resource: " + resource, xe);
151       }
152     }
153     return xmppConnection;
154   }

222   /**
223   * connect to xmpp
224    * @param grouperClientXmppMessageHandler the handler for the message
225    */
226    public static void xmppConnect(final GrouperClientXmppMessageHandler grouperClientXmppMessageHandler) {
227      
228      XMPPConnection theXmppConnection = xmppConnection();
229      theXmppConnection.addPacketListener(new PacketListener() {
230 
231        // @Override
232        public void processPacket(Packet packet) {
233          Message message = null;
234          try {
235            message = (Message) packet;
236            if (log.isDebugEnabled()) {
237              log.debug(message == null ? null : message.toXML());
238            }
239            grouperClientXmppMessageHandler.handleMessage(message);
240 
241          } catch (Throwable re) {
242            String messageXml = message == null ? null : message.toXML();
243            log.error("Problem with message: " + messageXml, re);
244            throw new RuntimeException(re);
245          }
246 
247        }
248      }, new PacketFilter() {
249 
250        // @Override
251        public boolean accept(Packet packet) {
252          if (packet instanceof Message) {
253            Message message = (Message) packet;
254            Type type = message.getType();
255            if (type == Type.chat && !GrouperClientUtils.isBlank(message.getBody())) {
256              if (allowFromJabberIds().contains(message.getFrom())) {
257                return true;
258              }
259              if (log.isDebugEnabled()) {
260                log.debug("Not expecting message from: " + message.getFrom());
261              }
262            }
263          }
264          return false;
265        }
266      });
267    }

496   /**
497    * note, this doesnt return, and you should only call this once...
498    * @param grouperClientXmppMessageHandler handle the message
499    */
500    public static void xmppLoop(GrouperClientXmppMessageHandler grouperClientXmppMessageHandler) {
501      //are we already in this loop?
502      if (xmppConnection != null) {
503        return;
504      }
505      while (true) {
506        try {
507          if (xmppConnection == null || !xmppConnection.isConnected() || !xmppConnection.isAuthenticated()) {
508 
509            //if not starting, this is a problem
510            if (xmppConnection != null) {
511              log.error("xmpp connection is not connected");
512              try {
513                xmppConnection.disconnect();
514              } catch (Exception e) {
515                log.error("error", e);
516              }
517              xmppConnection = null;
518            }
519            xmppConnect(grouperClientXmppMessageHandler);
520          }
521        } catch (Exception e) {
522          log.error("Problem with xmpp", e);
523        }
524        GrouperClientUtils.sleep(60000);
525      }
526    }

528    /**
529     * schedule a cron job
530     * @param jobName something unique and descriptive
531     * @param quartzCronString
532     * @param jobClass
533     */
534    public static void scheduleJob(String jobName, String quartzCronString, Class<? extends Job> jobClass) {
535      //no cron string, dont run the cron
536      if (GrouperClientUtils.isBlank(quartzCronString)) {
537        return;
538      }
539      
540      String jobGroup = Scheduler.DEFAULT_GROUP;
541      JobDetail jobDetail = JobBuilder.newJob(jobClass)
542          .withIdentity(jobName, jobGroup)
543          .storeDurably(true)
544          .build();
545 
546      Scheduler scheduler = GrouperClientXmppMain.scheduler();
547 
548      boolean uniqueTriggerNames = GrouperClientConfig.retrieveConfig().propertyValueBoolean("grouperClient.xmpp.uniqueQuartzTriggerNames", false);
549 
550      //in old versions of quartz, the trigger group cannot be null
551      String triggerName = "trigger_" + jobName;
552 
553      if (uniqueTriggerNames) {
554        triggerName += GrouperClientUtils.uniqueId();
555      }
556 
557      CronTrigger cronTrigger = null;
558      if (!uniqueTriggerNames) {
559        try {
560          cronTrigger = (CronTrigger) scheduler.getTrigger(new TriggerKey(triggerName, jobGroup));
561        } catch (SchedulerException se) {
562          throw new RuntimeException("Problem with trigger: " + jobName, se);
563        }
564      }
565      if (cronTrigger == null) {
566        try {
567          cronTrigger = TriggerBuilder.newTrigger()
568              .withIdentity(triggerName)
569              .withSchedule(CronScheduleBuilder.cronSchedule(quartzCronString))
570              .build();   
571        } catch (RuntimeException pe) {
572          throw new RuntimeException("Problems parsing: '" + quartzCronString + "'", pe);
573        }
574      }
575 
576      try {
577        scheduler.scheduleJob(jobDetail, cronTrigger);
578      } catch (SchedulerException se) {
579        throw new RuntimeException("Problem with job: " + jobName, se);
580      }
581 
582 
583    }

646 
647     if (!SUPPORTED_EVENT_TYPES.contains(eventType)) {
648       throw new RuntimeException("Not expecting action: '" + eventType + "'");
649     }
650 
651     List<GrouperClientXmppSubject> oldList = java.util.Collections.EMPTY_LIST;
652 
653     if (eventType.equals("MEMBERSHIP_ADD") || eventType.equals("MEMBERSHIP_DELETE")) {
654       oldList = groupMemberships.get(groupName);
655       if (oldList == null) {
656         // Do a full refresh and then ask for the group membership again
657         fullRefreshGroup(grouperClientXmppJob, groupName);
658         oldList = groupMemberships.get(groupName);
659       }
660       if (oldList == null) {
661         throw new NullPointerException("Why is old list null????");
662       }
663     }
664 
665     List<GrouperClientXmppSubject> newList = new ArrayList<GrouperClientXmppSubject>(oldList);
666 
667     if (GrouperClientUtils.equals(eventType, "MEMBERSHIP_ADD")) {
668 
669       boolean changed = newList.add(grouperClientXmppSubject);
670       if (changed && log.isDebugEnabled()) {
671         log.debug("Group " + groupName + " already contains subject: " + grouperClientXmppSubject.getSubjectId());
672       }
673 
674     } else if (GrouperClientUtils.equals(eventType, "MEMBERSHIP_DELETE")) {
675 
676       Collection<GrouperClientXmppSubject> toRemove = GrouperClientUtils.toList(grouperClientXmppSubject);
677       boolean changed = newList.removeAll(toRemove);
678       if (!changed && log.isDebugEnabled()) {
679         log.debug("Group " + groupName + " already doesnt contain subject: " + grouperClientXmppSubject.getSubjectId());
680       }
681     }
682 
683     groupMemberships.put(groupName, newList);
684 
685     if (log.isDebugEnabled()) {
686       log.debug("Event: " + eventType + " for " + groupName + ", memberships list was "
687           + GrouperClientUtils.length(oldList)
688           + " and is now " + GrouperClientUtils.length(newList) + " subjects");
689     }
690 
691     // Instantiate the handler for this job and call the handleIncremental method
692     Class<GrouperClientXmppHandler> handlerClass = GrouperClientUtils
693         .forName(grouperClientXmppJob.getHandlerClass());
694     GrouperClientXmppHandler grouperClientXmppHandler = GrouperClientUtils
695         .newInstance(handlerClass);
696     grouperClientXmppHandler.handleIncremental(grouperClientXmppJob,
697         groupName, GrouperClientUtils.extensionFromName(groupName),
698          newList, oldList, grouperClientXmppSubject, eventType);
699   }
