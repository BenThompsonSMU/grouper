27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.collections;

29  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.Converter;
30  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.MarshallingContext;
31  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.UnmarshallingContext;
32  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamReader;
33  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamWriter;
34  
35  import java.util.BitSet;
36  import java.util.StringTokenizer;
37  
38  /**
39   * Converts a java.util.BitSet to XML, as a compact
40   * comma delimited list of ones and zeros.
41   *
42   * @author Joe Walnes
43   */
44  public class BitSetConverter implements Converter {
45  
46      public boolean canConvert(Class type) {
47          return type.equals(BitSet.class);
48      }
49  
50      public void marshal(Object source, HierarchicalStreamWriter writer, MarshallingContext context) {
51          BitSet bitSet = (BitSet) source;
52          StringBuffer buffer = new StringBuffer();
53          boolean seenFirst = false;
54          for (int i = 0; i < bitSet.length(); i++) {
55              if (bitSet.get(i)) {
56                  if (seenFirst) {
57                      buffer.append(',');
58                  } else {
59                      seenFirst = true;
60                  }
61                  buffer.append(i);
62              }
63          }
64          writer.setValue(buffer.toString());
65      }
66  
67      public Object unmarshal(HierarchicalStreamReader reader, UnmarshallingContext context) {
68          BitSet result = new BitSet();
69          StringTokenizer tokenizer = new StringTokenizer(reader.getValue(), ",", false);
70          while (tokenizer.hasMoreTokens()) {
71              int index = Integer.parseInt(tokenizer.nextToken());
72              result.set(index);
73          }
74          return result;
75      }
76  }
