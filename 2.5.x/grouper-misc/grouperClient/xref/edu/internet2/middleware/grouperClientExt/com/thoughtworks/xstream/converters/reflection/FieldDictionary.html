43  /**
44   * A field dictionary instance caches information about classes fields.
45   * 
46   * @author Joe Walnes
47   * @author J&ouml;rg Schaible
48   * @author Guilherme Silveira
49   */
50  public class FieldDictionary {
51  
52      private transient Map keyedByFieldNameCache;
53      private transient Map keyedByFieldKeyCache;
54      private final FieldKeySorter sorter;
55  
56      public FieldDictionary() {
57          this(new ImmutableFieldKeySorter());
58      }
59  
60      public FieldDictionary(FieldKeySorter sorter) {
61          this.sorter = sorter;
62          init();
63      }
64  
65      private void init() {
66          keyedByFieldNameCache = new WeakHashMap();
67          keyedByFieldKeyCache = new WeakHashMap();
68          keyedByFieldNameCache.put(Object.class, Collections.EMPTY_MAP);
69          keyedByFieldKeyCache.put(Object.class, Collections.EMPTY_MAP);
70      }
71  
72      /**
73       * Returns an iterator for all fields for some class
74       * 
75       * @param cls the class you are interested on
76       * @return an iterator for its fields
77       * @deprecated since 1.3, use {@link #fieldsFor(Class)} instead
78       */
79      public Iterator serializableFieldsFor(Class cls) {
80          return fieldsFor(cls);
81      }
82  
83      /**
84       * Returns an iterator for all fields for some class
85       * 
86       * @param cls the class you are interested on
87       * @return an iterator for its fields
88       */
89      public Iterator fieldsFor(Class cls) {
90          return buildMap(cls, true).values().iterator();
91      }
92  
93      /**
94       * Returns an specific field of some class. If definedIn is null, it searches for the field
95       * named 'name' inside the class cls. If definedIn is different than null, tries to find the
96       * specified field name in the specified class cls which should be defined in class
97       * definedIn (either equals cls or a one of it's superclasses)
98       * 
99       * @param cls the class where the field is to be searched
100      * @param name the field name
101      * @param definedIn the superclass (or the class itself) of cls where the field was defined
102      * @return the field itself
103      */
104     public Field field(Class cls, String name, Class definedIn) {
105         Map fields = buildMap(cls, definedIn != null);
106         Field field = (Field)fields.get(definedIn != null ? (Object)new FieldKey(
107             name, definedIn, 0) : (Object)name);
108         if (field == null) {
109             throw new ObjectAccessException("No such field " + cls.getName() + "." + name);
110         } else {
111             return field;
112         }
113     }
114 
115     private Map buildMap(final Class type, boolean tupleKeyed) {
116         Class cls = type;
117         synchronized (this) {
118             if (!keyedByFieldNameCache.containsKey(type)) {
119                 final List superClasses = new ArrayList();
120                 while (!Object.class.equals(cls)) {
121                     superClasses.add(0, cls);
122                     cls = cls.getSuperclass();
123                 }
124                 Map lastKeyedByFieldName = Collections.EMPTY_MAP;
125                 Map lastKeyedByFieldKey = Collections.EMPTY_MAP;
126                 for (final Iterator iter = superClasses.iterator(); iter.hasNext();) {
127                     cls = (Class)iter.next();
128                     if (!keyedByFieldNameCache.containsKey(cls)) {
129                         final Map keyedByFieldName = new HashMap(lastKeyedByFieldName);
130                         final Map keyedByFieldKey = new OrderRetainingMap(lastKeyedByFieldKey);
131                         Field[] fields = cls.getDeclaredFields();
132                         if (JVM.reverseFieldDefinition()) {
133                             for (int i = fields.length >> 1; i-- > 0;) {
134                                 final int idx = fields.length - i - 1;
135                                 final Field field = fields[i];
136                                 fields[i] = fields[idx];
137                                 fields[idx] = field;
138                             }
139                         }
140                         for (int i = 0; i < fields.length; i++) {
141                             Field field = fields[i];
142                             FieldKeydleware/grouperClientExt/com/thoughtworks/xstream/converters/reflection/FieldKey.html#FieldKey">FieldKey fieldKey = new FieldKey(field.getName(), field
143                                 .getDeclaringClass(), i);
144                             field.setAccessible(true);
145                             Field existent = (Field)keyedByFieldName.get(field.getName());
146                             if (existent == null
147                             // do overwrite statics
148                                 || ((existent.getModifiers() & Modifier.STATIC) != 0)
149                                 // overwrite non-statics with non-statics only
150                                 || (existent != null && ((field.getModifiers() & Modifier.STATIC) == 0))) {
151                                 keyedByFieldName.put(field.getName(), field);
152                             }
153                             keyedByFieldKey.put(fieldKey, field);
154                         }
155                         keyedByFieldNameCache.put(cls, keyedByFieldName);
156                         keyedByFieldKeyCache.put(cls, sorter.sort(type, keyedByFieldKey));
157                     }
158                     lastKeyedByFieldName = (Map)keyedByFieldNameCache.get(cls);
159                     lastKeyedByFieldKey = (Map)keyedByFieldKeyCache.get(cls);
160                 }
161             }
162         }
163         return (Map)(tupleKeyed ? keyedByFieldKeyCache.get(type) : keyedByFieldNameCache
164             .get(type));
165     }

167     protected Object readResolve() {
168         init();
169         return this;
170     }
