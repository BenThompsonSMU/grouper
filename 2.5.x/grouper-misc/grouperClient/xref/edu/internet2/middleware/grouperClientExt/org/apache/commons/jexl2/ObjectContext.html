32  package edu.internet2.middleware.grouperClientExt.org.apache.commons.jexl2;
33  
34  /**
35   * Wraps an Object as a Jexl context.
36   * @param <T> the wrapped object type to use
37   * @since 2.1
38   */
39  public class ObjectContext<T> implements JexlContext {
40      /** The property solving jexl engine. */
41      private final JexlEngine jexl;
42      /** The object serving as context provider. */
43      private final T object;
44  
45      /**
46       * Creates a new ObjectContext.
47       * @param engine the jexl engine to use to solve properties
48       * @param wrapped the object to wrap in this context
49       */
50      public ObjectContext(JexlEngine engine, T wrapped) {
51          this.jexl = engine;
52          this.object = wrapped;
53      }
54  
55      /** {@inheritDoc} */
56      public Object get(String name) {
57          return jexl.getProperty(object, name);
58      }
59  
60      /** {@inheritDoc} */
61      public void set(String name, Object value) {
62          jexl.setProperty(object, name, value);
63      }
64  
65      /** {@inheritDoc} */
66      public boolean has(String name) {
67          return jexl.getUberspect().getPropertyGet(object, name, null) != null;
68      }
69  }
