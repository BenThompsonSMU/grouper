46  package edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.cookie;
47  
48  import java.util.Date;
49  
50  import edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.Cookie;
51  
52  /**
53   * <p>
54   * Cookie class for {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.cookie.RFC2965Spec}
55   * cookie specification. It extends {@link Cookie} class and adds newer cookie
56   * attributes and functions required for this specification.
57   * </p>
58   *
59   * @author Samit Jain (jain.samit@gmail.com)
60   * 
61   * @since 3.1
62   */
63  public class Cookie2 extends Cookie {
64  
65      // string constants for cookie attributes
66      public static final String DOMAIN = "domain";
67      public static final String PATH = "path";
68      public static final String PORT = "port";
69      public static final String VERSION = "version";
70      public static final String SECURE = "secure";
71      public static final String MAXAGE = "max-age";
72      public static final String COMMENT = "comment";
73      public static final String COMMENTURL = "commenturl";
74      public static final String DISCARD = "discard";
75  
76      /**
77       * Default constructor. Creates a blank cookie
78       */
79      public Cookie2() {
80          super(null, "noname", null, null, null, false);
81      }
82  
83      /**
84       * Creates a cookie with the given name, value and domain attribute.
85       *
86       * @param name    the cookie name
87       * @param value   the cookie value
88       * @param domain  the domain this cookie can be sent to
89       */
90      public Cookie2(String domain, String name, String value) {
91          super(domain, name, value);
92      }
93  
94      /**
95       * Creates a cookie with the given name, value, domain attribute,
96       * path attribute, expiration attribute, and secure attribute
97       *
98       * @param name    the cookie name
99       * @param value   the cookie value
100      * @param domain  the domain this cookie can be sent to
101      * @param path    the path prefix for which this cookie can be sent
102      * @param expires the {@link Date} at which this cookie expires,
103      *                or <tt>null</tt> if the cookie expires at the end
104      *                of the session
105      * @param secure if true this cookie can only be sent over secure
106      * connections
107      * @throws IllegalArgumentException If cookie name is null or blank,
108      *   cookie name contains a blank, or cookie name starts with character $
109      *
110      */
111     public Cookie2(String domain, String name, String value,
112                    String path, Date expires, boolean secure) {
113         super(domain, name, value, path, expires, secure);
114     }
115 
116     /**
117      * Creates a cookie with the given name, value, domain attribute,
118      * path attribute, expiration attribute, secure attribute, and ports
119      * attribute.
120      *
121      * @param name    the cookie name
122      * @param value   the cookie value
123      * @param domain  the domain this cookie can be sent to
124      * @param path    the path prefix for which this cookie can be sent
125      * @param expires the {@link Date} at which this cookie expires,
126      *                or <tt>null</tt> if the cookie expires at the end
127      *                of the session
128      * @param secure if true this cookie can only be sent over secure
129      * connections
130      * @param ports   the ports for which this cookie can be sent
131      * @throws IllegalArgumentException If cookie name is null or blank,
132      *   cookie name contains a blank, or cookie name starts with character $
133      *
134      */
135     public Cookie2(String domain, String name, String value,
136                    String path, Date expires, boolean secure, int[] ports) {
137         super(domain, name, value, path, expires, secure);
138         setPorts(ports);
139     }
140 
141    /**
142      * If a user agent (web browser) presents this cookie to a user, the
143      * cookie's purpose will be described by the information at this URL.
144      *
145      * @see #setCommentURL(String)
146      */
147     public String getCommentURL() {
148         return cookieCommentURL;
149     }
150 
151    /**
152      * If a user agent (web browser) presents this cookie to a user, the
153      * cookie's purpose will be described by the information at this URL.
154      *
155      * @param commentURL
156      *
157      * @see #getCommentURL()
158      */
159     public void setCommentURL(String commentURL) {
160         this.cookieCommentURL = commentURL;
161     }
162 
163     /**
164      * Get the Port attribute. It restricts the ports to which a cookie
165      * may be returned in a Cookie request header.
166      *
167      * @see #setPorts(int[])
168      */
169     public int[] getPorts() {
170         return cookiePorts;
171     }
172 
173    /**
174      * Set the Port attribute. It restricts the ports to which a cookie
175      * may be returned in a Cookie request header.
176      *
177      * @param ports
178      *
179      * @see #getPorts()
180      */
181     public void setPorts(int[] ports) {
182         this.cookiePorts = ports;
183     }
184 
185     /**
186      * Set the Discard attribute.
187      *
188      * Note: <tt>Discard</tt> attribute overrides <tt>Max-age</tt>.
189      *
190      * @see #isPersistent()
191      */
192     public void setDiscard(boolean toDiscard) {
193         discard = toDiscard;
194     }
195 
196     /**
197      * Returns <tt>false</tt> if the cookie should be discarded at the end
198      * of the "session"; <tt>true</tt> otherwise.
199      *
200      * @return <tt>false</tt> if the cookie should be discarded at the end
201      *         of the "session"; <tt>true</tt> otherwise
202      */
203     public boolean isPersistent() {
204         return (null != getExpiryDate()) && !discard;
205     }
206 
207     /**
208      * Indicates whether the cookie had a port attribute specified in the
209      * <tt>Set-Cookie2</tt> response header. 
210      *
211      * @param value <tt>true</tt> if port attribute is specified in response
212      * header.
213      *
214      * @see #isPortAttributeSpecified
215      */
216     public void setPortAttributeSpecified(boolean value) {
217         hasPortAttribute = value;
218     }
219 
220     /**
221      * @return <tt>true</tt> if cookie port attribute was specified in the
222      * <tt>Set-Cookie2</tt> header.
223      *
224      * @see #setPortAttributeSpecified
225      */
226     public boolean isPortAttributeSpecified() {
227         return hasPortAttribute;
228     }
229 
230     /**
231      * Indicates whether the Port attribute in <tt>Set-Cookie2</tt> header
232      * contains no value (is of the form Port="").
233      * <p>
234      * This value is required for generating
235      * the <tt>Cookie</tt> request header because the specification requires that if
236      * <tt>Set-Cookie2</tt> header contains a blank value for port attribute,
237      * the <tt>Cookie</tt> header should also contain a port attribute with no value.
238      *
239      * @param value <tt>true</tt> if port attribute is specified as blank in response
240      * header.
241      *
242      * @see #isPortAttributeBlank
243      */
244     public void setPortAttributeBlank(boolean value) {
245         isPortAttributeBlank = value;
246     }
247 
248     /**
249      * @return <tt>true</tt> if the port attribute in <tt>Set-Cookie2</tt> header
250      * had no value (was of the form Port="").
251      *
252      * @see #setPortAttributeBlank
253      */
254     public boolean isPortAttributeBlank() {
255         return isPortAttributeBlank;
256     }
257 
258     /**
259      * Indicates whether the cookie had a version attribute specified in the
260      * <tt>Set-Cookie2</tt> response header. 
261      *
262      * @param value <tt>true</tt> if version attribute is specified in response
263      * header.
264      * @see #isVersionAttributeSpecified()
265      */
266     public void setVersionAttributeSpecified(boolean value) {
267         hasVersionAttribute = value;
268     }
269 
270     /**
271      * @return <tt>true</tt> if cookie version attribute was specified in the
272      * <tt>Set-Cookie2</tt> header.
273      *
274      * @see #setVersionAttributeSpecified
275      */
276     public boolean isVersionAttributeSpecified() {
277         return hasVersionAttribute;
278     }
279 
280     /**
281      * Return a textual representation of the cookie.
282      *
283      * @return string.
284      */
285     public String toExternalForm() {
286         CookieSpec spec =
287                 CookiePolicy.getCookieSpec(CookiePolicy.RFC_2965);
288         return spec.formatCookie(this);
289     }
290 
291     /**
292      * Comment URL attribute
293      */
294     private String cookieCommentURL;
295 
296     /**
297      * Port attribute.
298      */
299     private int[] cookiePorts;
300 
301     /**
302      * Discard attribute.
303      */
304     private boolean discard = false;
305 
306     /**
307      * Indicates if the set-cookie2 header included a Port attribute for this
308      * cookie
309      */
310     private boolean hasPortAttribute = false;
311 
312     /**
313      * Indicates if the set-cookie2 header's Port attribute did not have
314      * any value.
315      */
316     private boolean isPortAttributeBlank = false;
317 
318     /**
319      * Indicates if the set-cookie2 header included a Version attribute
320      */
321     private boolean hasVersionAttribute = false;
322 
323 
324 }
