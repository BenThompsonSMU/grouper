83      /**
84        * Parses the Set-Cookie value into an array of <tt>Cookie</tt>s.
85        *
86        * <p>Syntax of the Set-Cookie HTTP Response Header:</p>
87        * 
88        * <p>This is the format a CGI script would use to add to 
89        * the HTTP headers a new piece of data which is to be stored by 
90        * the client for later retrieval.</p>
91        *  
92        * <PRE>
93        *  Set-Cookie: NAME=VALUE; expires=DATE; path=PATH; domain=DOMAIN_NAME; secure
94        * </PRE>
95        *
96        * <p>Please note that Netscape draft specification does not fully 
97        * conform to the HTTP header format. Netscape draft does not specify 
98        * whether multiple cookies may be sent in one header. Hence, comma 
99        * character may be present in unquoted cookie value or unquoted 
100       * parameter value.</p>
101       * 
102       * @link http://wp.netscape.com/newsref/std/cookie_spec.html
103       * 
104       * @param host the host from which the <tt>Set-Cookie</tt> value was
105       * received
106       * @param port the port from which the <tt>Set-Cookie</tt> value was
107       * received
108       * @param path the path from which the <tt>Set-Cookie</tt> value was
109       * received
110       * @param secure <tt>true</tt> when the <tt>Set-Cookie</tt> value was
111       * received over secure conection
112       * @param header the <tt>Set-Cookie</tt> received from the server
113       * @return an array of <tt>Cookie</tt>s parsed from the Set-Cookie value
114       * @throws MalformedCookieException if an exception occurs during parsing
115       * 
116       * @since 3.0
117       */
118     public Cookie[] parse(String host, int port, String path, 
119         boolean secure, final String header) 
120         throws MalformedCookieException {
121             
122         LOG.trace("enter NetscapeDraftSpec.parse(String, port, path, boolean, Header)");
123 
124         if (host == null) {
125             throw new IllegalArgumentException("Host of origin may not be null");
126         }
127         if (host.trim().equals("")) {
128             throw new IllegalArgumentException("Host of origin may not be blank");
129         }
130         if (port < 0) {
131             throw new IllegalArgumentException("Invalid port: " + port);
132         }
133         if (path == null) {
134             throw new IllegalArgumentException("Path of origin may not be null.");
135         }
136         if (header == null) {
137             throw new IllegalArgumentException("Header may not be null.");
138         }
139 
140         if (path.trim().equals("")) {
141             path = PATH_DELIM;
142         }
143         host = host.toLowerCase();
144 
145         String defaultPath = path;    
146         int lastSlashIndex = defaultPath.lastIndexOf(PATH_DELIM);
147         if (lastSlashIndex >= 0) {
148             if (lastSlashIndex == 0) {
149                 //Do not remove the very first slash
150                 lastSlashIndex = 1;
151             }
152             defaultPath = defaultPath.substring(0, lastSlashIndex);
153         }

173     /**
174       * Parse the cookie attribute and update the corresponsing {@link Cookie}
175       * properties as defined by the Netscape draft specification
176       *
177       * @param attribute {@link NameValuePair} cookie attribute from the
178       * <tt>Set- Cookie</tt>
179       * @param cookie {@link Cookie} to be updated
180       * @throws MalformedCookieException if an exception occurs during parsing
181       */
182     public void parseAttribute(
183         final NameValuePair attribute, final Cookie cookie)
184         throws MalformedCookieException {
185             
186         if (attribute == null) {
187             throw new IllegalArgumentException("Attribute may not be null.");
188         }
189         if (cookie == null) {
190             throw new IllegalArgumentException("Cookie may not be null.");
191         }
192         final String paramName = attribute.getName().toLowerCase();
193         final String paramValue = attribute.getValue();
194 
195         if (paramName.equals("expires")) {
196 
197             if (paramValue == null) {
198                 throw new MalformedCookieException(
199                     "Missing value for expires attribute");
200             }
201             try {
202                 DateFormat expiryFormat = new SimpleDateFormat(
203                     "EEE, dd-MMM-yyyy HH:mm:ss z", Locale.US);
204                 Date date = expiryFormat.parse(paramValue);
205                 cookie.setExpiryDate(date);
206             } catch (ParseException e) {
207                 throw new MalformedCookieException("Invalid expires "
208                     + "attribute: " + e.getMessage());
209             }
210         } else {
211             super.parseAttribute(attribute, cookie);
212         }
213     }

225     /**
226       * Performs Netscape draft compliant {@link Cookie} validation
227       *
228       * @param host the host from which the {@link Cookie} was received
229       * @param port the port from which the {@link Cookie} was received
230       * @param path the path from which the {@link Cookie} was received
231       * @param secure <tt>true</tt> when the {@link Cookie} was received 
232       * using a secure connection
233       * @param cookie The cookie to validate.
234       * @throws MalformedCookieException if an exception occurs during
235       * validation
236       */
237     public void validate(String host, int port, String path, 
238         boolean secure, final Cookie cookie) 
239         throws MalformedCookieException {
240             
241         LOG.trace("enterNetscapeDraftCookieProcessor "
242             + "RCF2109CookieProcessor.validate(Cookie)");
243         // Perform generic validation
244         super.validate(host, port, path, secure, cookie);
245         // Perform Netscape Cookie draft specific validation
246         if (host.indexOf(".") >= 0) {
247             int domainParts = new StringTokenizer(cookie.getDomain(), ".")
248                 .countTokens();
249 
250             if (isSpecialDomain(cookie.getDomain())) {
251                 if (domainParts < 2) {
252                     throw new MalformedCookieException("Domain attribute \""
253                         + cookie.getDomain() 
254                         + "\" violates the Netscape cookie specification for "
255                         + "special domains");
256                 }
257             } else {
258                 if (domainParts < 3) {
259                     throw new MalformedCookieException("Domain attribute \""
260                         + cookie.getDomain() 
261                         + "\" violates the Netscape cookie specification");
262                 }            
263             }
264         }
265     }

267     /**
268      * Checks if the given domain is in one of the seven special
269      * top level domains defined by the Netscape cookie specification.
270      * @param domain The domain.
271      * @return True if the specified domain is "special"
272      */
273     private static boolean isSpecialDomain(final String domain) {
274         final String ucDomain = domain.toUpperCase();
275         if (ucDomain.endsWith(".COM") 
276            || ucDomain.endsWith(".EDU")
277            || ucDomain.endsWith(".NET")
278            || ucDomain.endsWith(".GOV")
279            || ucDomain.endsWith(".MIL")
280            || ucDomain.endsWith(".ORG")
281            || ucDomain.endsWith(".INT")) {
282             return true;
283         }
284         return false;
285     }
