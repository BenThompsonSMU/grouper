101     /**
102      * Creates a cookie with the given name, value, domain attribute,
103      * path attribute, expiration attribute, and secure attribute 
104      *
105      * @param name    the cookie name
106      * @param value   the cookie value
107      * @param domain  the domain this cookie can be sent to
108      * @param path    the path prefix for which this cookie can be sent
109      * @param expires the {@link Date} at which this cookie expires,
110      *                or <tt>null</tt> if the cookie expires at the end
111      *                of the session
112      * @param secure if true this cookie can only be sent over secure
113      * connections
114      * @throws IllegalArgumentException If cookie name is null or blank,
115      *   cookie name contains a blank, or cookie name starts with character $
116      *   
117      */
118     public Cookie(String domain, String name, String value, 
119         String path, Date expires, boolean secure) {
120             
121         super(name, value);
122         LOG.trace("enter Cookie(String, String, String, String, Date, boolean)");
123         if (name == null) {
124             throw new IllegalArgumentException("Cookie name may not be null");
125         }
126         if (name.trim().equals("")) {
127             throw new IllegalArgumentException("Cookie name may not be blank");
128         }
129         this.setPath(path);
130         this.setDomain(domain);
131         this.setExpiryDate(expires);
132         this.setSecure(secure);
133     }

135     /**
136      * Creates a cookie with the given name, value, domain attribute,
137      * path attribute, maximum age attribute, and secure attribute 
138      *
139      * @param name   the cookie name
140      * @param value  the cookie value
141      * @param domain the domain this cookie can be sent to
142      * @param path   the path prefix for which this cookie can be sent
143      * @param maxAge the number of seconds for which this cookie is valid.
144      *               maxAge is expected to be a non-negative number. 
145      *               <tt>-1</tt> signifies that the cookie should never expire.
146      * @param secure if <tt>true</tt> this cookie can only be sent over secure
147      * connections
148      */
149     public Cookie(String domain, String name, String value, String path, 
150         int maxAge, boolean secure) {
151             
152         this(domain, name, value, path, null, secure);
153         if (maxAge < -1) {
154             throw new IllegalArgumentException("Invalid max age:  " + Integer.toString(maxAge));
155         }            
156         if (maxAge >= 0) {
157             setExpiryDate(new Date(System.currentTimeMillis() + maxAge * 1000L));
158         }
159     }

445     /**
446      * Return a textual representation of the cookie.
447      * 
448      * @return string.
449      */
450     public String toExternalForm() {
451         CookieSpec spec = null;
452         if (getVersion() > 0) {
453             spec = CookiePolicy.getDefaultSpec(); 
454         } else {
455             spec = CookiePolicy.getCookieSpec(CookiePolicy.NETSCAPE); 
456         }
457         return spec.formatCookie(this); 
458     }

460     /**
461      * <p>Compares two cookies to determine order for cookie header.</p>
462      * <p>Most specific should be first. </p>
463      * <p>This method is implemented so a cookie can be used as a comparator for
464      * a SortedSet of cookies. Specifically it's used above in the 
465      * createCookieHeader method.</p>
466      * @param o1 The first object to be compared
467      * @param o2 The second object to be compared
468      * @return See {@link java.util.Comparator#compare(Object,Object)}
469      */
470     public int compare(Object o1, Object o2) {
471         LOG.trace("enter Cookie.compare(Object, Object)");
472 
473         if (!(o1 instanceof Cookie)) {
474             throw new ClassCastException(o1.getClass().getName());
475         }
476         if (!(o2 instanceof Cookie)) {
477             throw new ClassCastException(o2.getClass().getName());
478         }
479         Cookieef="../../../../../../../../edu/internet2/middleware/grouperClientExt/org/apache/commons/httpclient/Cookie.html#Cookie">Cookie c1 = (Cookie) o1;
480         Cookieef="../../../../../../../../edu/internet2/middleware/grouperClientExt/org/apache/commons/httpclient/Cookie.html#Cookie">Cookie c2 = (Cookie) o2;
481         if (c1.getPath() == null && c2.getPath() == null) {
482             return 0;
483         } else if (c1.getPath() == null) {
484             // null is assumed to be "/"
485             if (c2.getPath().equals(CookieSpec.PATH_DELIM)) {
486                 return 0;
487             } else {
488                 return -1;
489             }
490         } else if (c2.getPath() == null) {
491             // null is assumed to be "/"
492             if (c1.getPath().equals(CookieSpec.PATH_DELIM)) {
493                 return 0;
494             } else {
495                 return 1;
496             }
497         } else {
498             return c1.getPath().compareTo(c2.getPath());
499         }
500     }
