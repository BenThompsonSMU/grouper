46  package edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.methods.multipart;
47  
48  import java.io.ByteArrayOutputStream;
49  import java.io.IOException;
50  import java.io.OutputStream;
51  
52  
53  import edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.util.EncodingUtil;
54  import edu.internet2.middleware.grouperClientExt.org.apache.commons.logging.Log;
55  import edu.internet2.middleware.grouperClientExt.org.apache.commons.logging.LogFactory;

57  /**
58   * Abstract class for one Part of a multipart post object.
59   *
60   * @author <a href="mailto:mattalbright@yahoo.com">Matthew Albright</a>
61   * @author <a href="mailto:jsdever@apache.org">Jeff Dever</a>
62   * @author <a href="mailto:adrian@ephox.com">Adrian Sutton</a>
63   * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
64   * @author <a href="mailto:oleg@ural.ru">Oleg Kalnichevski</a>
65   *
66   * @since 2.0
67   */
68  public abstract class Part {
69  
70      /** Log object for this class. */
71      private static final Log LOG = LogFactory.getLog(Part.class);
72  
73      /** 
74       * The boundary 
75       * @deprecated use {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.httpclient.params.HttpMethodParams#MULTIPART_BOUNDARY}
76       */
77      protected static final String BOUNDARY = "----------------314159265358979323846";
78      
79      /** 
80       * The boundary as a byte array.
81       * @deprecated
82       */
83      protected static final byte[] BOUNDARY_BYTES = EncodingUtil.getAsciiBytes(BOUNDARY);
84  
85      /**
86       * The default boundary to be used if {@link #setBoundaryBytes(byte[])) has not
87       * been called.
88       */
89      private static final byte[] DEFAULT_BOUNDARY_BYTES = BOUNDARY_BYTES;    
90      
91      /** Carriage return/linefeed */
92      protected static final String CRLF = "\r\n";
93      
94      /** Carriage return/linefeed as a byte array */
95      protected static final byte[] CRLF_BYTES = EncodingUtil.getAsciiBytes(CRLF);
96      
97      /** Content dispostion characters */
98      protected static final String QUOTE = "\"";
99      
100     /** Content dispostion as a byte array */
101     protected static final byte[] QUOTE_BYTES = 
102       EncodingUtil.getAsciiBytes(QUOTE);
103 
104     /** Extra characters */
105     protected static final String EXTRA = "--";
106     
107     /** Extra characters as a byte array */
108     protected static final byte[] EXTRA_BYTES = 
109       EncodingUtil.getAsciiBytes(EXTRA);
110     
111     /** Content dispostion characters */
112     protected static final String CONTENT_DISPOSITION = "Content-Disposition: form-data; name=";
113     
114     /** Content dispostion as a byte array */
115     protected static final byte[] CONTENT_DISPOSITION_BYTES = 
116       EncodingUtil.getAsciiBytes(CONTENT_DISPOSITION);
117 
118     /** Content type header */
119     protected static final String CONTENT_TYPE = "Content-Type: ";
120 
121     /** Content type header as a byte array */
122     protected static final byte[] CONTENT_TYPE_BYTES = 
123       EncodingUtil.getAsciiBytes(CONTENT_TYPE);
124 
125     /** Content charset */
126     protected static final String CHARSET = "; charset=";
127 
128     /** Content charset as a byte array */
129     protected static final byte[] CHARSET_BYTES = 
130       EncodingUtil.getAsciiBytes(CHARSET);
131 
132     /** Content type header */
133     protected static final String CONTENT_TRANSFER_ENCODING = "Content-Transfer-Encoding: ";
134 
135     /** Content type header as a byte array */
136     protected static final byte[] CONTENT_TRANSFER_ENCODING_BYTES = 
137       EncodingUtil.getAsciiBytes(CONTENT_TRANSFER_ENCODING);
138 
139     /**
140      * Return the boundary string.
141      * @return the boundary string
142      * @deprecated uses a constant string. Rather use {@link #getPartBoundary}
143      */
144     public static String getBoundary() {
145         return BOUNDARY;
146     }
147 
148     /**
149      * The ASCII bytes to use as the multipart boundary.
150      */
151     private byte[] boundaryBytes;
152     
153     /**
154      * Return the name of this part.
155      * @return The name.
156      */
157     public abstract String getName();
158     
159     /**
160      * Returns the content type of this part.
161      * @return the content type, or <code>null</code> to exclude the content type header
162      */
163     public abstract String getContentType();
164 
165     /**
166      * Return the character encoding of this part.
167      * @return the character encoding, or <code>null</code> to exclude the character 
168      * encoding header
169      */
170     public abstract String getCharSet();
171 
172     /**
173      * Return the transfer encoding of this part.
174      * @return the transfer encoding, or <code>null</code> to exclude the transfer encoding header
175      */
176     public abstract String getTransferEncoding();
177 
178     /**
179      * Gets the part boundary to be used.
180      * @return the part boundary as an array of bytes.
181      * 
182      * @since 3.0
183      */
184     protected byte[] getPartBoundary() {
185         if (boundaryBytes == null) {
186             // custom boundary bytes have not been set, use the default.
187             return DEFAULT_BOUNDARY_BYTES;
188         } else {
189             return boundaryBytes;            
190         }
191     }
192     
193     /**
194      * Sets the part boundary.  Only meant to be used by 
195      * {@link Part#sendParts(OutputStream, Part[], byte[])}
196      * and {@link Part#getLengthOfParts(Part[], byte[])}
197      * @param boundaryBytes An array of ASCII bytes.
198      * @since 3.0
199      */
200     void setPartBoundary(byte[] boundaryBytes) {
201         this.boundaryBytes = boundaryBytes;
202     }
203     
204     /**
205      * Tests if this part can be sent more than once.
206      * @return <code>true</code> if {@link #sendData(OutputStream)} can be successfully called 
207      * more than once.
208      * @since 3.0
209      */
210     public boolean isRepeatable() {
211         return true;
212     }
213     
214     /**
215      * Write the start to the specified output stream
216      * @param out The output stream
217      * @throws IOException If an IO problem occurs.
218      */
219     protected void sendStart(OutputStream out) throws IOException {
220         LOG.trace("enter sendStart(OutputStream out)");
221         out.write(EXTRA_BYTES);
222         out.write(getPartBoundary());
223         out.write(CRLF_BYTES);
224     }
225     
226     /**
227      * Write the content disposition header to the specified output stream
228      * 
229      * @param out The output stream
230      * @throws IOException If an IO problem occurs.
231      */
232     protected void sendDispositionHeader(OutputStream out) throws IOException {
233         LOG.trace("enter sendDispositionHeader(OutputStream out)");
234         out.write(CONTENT_DISPOSITION_BYTES);
235         out.write(QUOTE_BYTES);
236         out.write(EncodingUtil.getAsciiBytes(getName()));
237         out.write(QUOTE_BYTES);
238     }
239     
240     /**
241      * Write the content type header to the specified output stream
242      * @param out The output stream
243      * @throws IOException If an IO problem occurs.
244      */
245      protected void sendContentTypeHeader(OutputStream out) throws IOException {
246         LOG.trace("enter sendContentTypeHeader(OutputStream out)");
247         String contentType = getContentType();
248         if (contentType != null) {
249             out.write(CRLF_BYTES);
250             out.write(CONTENT_TYPE_BYTES);
251             out.write(EncodingUtil.getAsciiBytes(contentType));
252             String charSet = getCharSet();
253             if (charSet != null) {
254                 out.write(CHARSET_BYTES);
255                 out.write(EncodingUtil.getAsciiBytes(charSet));
256             }
257         }
258     }
259 
260     /**
261      * Write the content transfer encoding header to the specified 
262      * output stream
263      * 
264      * @param out The output stream
265      * @throws IOException If an IO problem occurs.
266      */
267      protected void sendTransferEncodingHeader(OutputStream out) throws IOException {
268         LOG.trace("enter sendTransferEncodingHeader(OutputStream out)");
269         String transferEncoding = getTransferEncoding();
270         if (transferEncoding != null) {
271             out.write(CRLF_BYTES);
272             out.write(CONTENT_TRANSFER_ENCODING_BYTES);
273             out.write(EncodingUtil.getAsciiBytes(transferEncoding));
274         }
275     }
276 
277     /**
278      * Write the end of the header to the output stream
279      * @param out The output stream
280      * @throws IOException If an IO problem occurs.
281      */
282     protected void sendEndOfHeader(OutputStream out) throws IOException {
283         LOG.trace("enter sendEndOfHeader(OutputStream out)");
284         out.write(CRLF_BYTES);
285         out.write(CRLF_BYTES);
286     }
287     
288     /**
289      * Write the data to the specified output stream
290      * @param out The output stream
291      * @throws IOException If an IO problem occurs.
292      */
293     protected abstract void sendData(OutputStream out) throws IOException;
294     
295     /**
296      * Return the length of the main content
297      * 
298      * @return long The length.
299      * @throws IOException If an IO problem occurs
300      */
301     protected abstract long lengthOfData() throws IOException;
302     
303     /**
304      * Write the end data to the output stream.
305      * @param out The output stream
306      * @throws IOException If an IO problem occurs.
307      */
308     protected void sendEnd(OutputStream out) throws IOException {
309         LOG.trace("enter sendEnd(OutputStream out)");
310         out.write(CRLF_BYTES);
311     }
312     
313     /**
314      * Write all the data to the output stream.
315      * If you override this method make sure to override 
316      * #length() as well
317      * 
318      * @param out The output stream
319      * @throws IOException If an IO problem occurs.
320      */
321     public void send(OutputStream out) throws IOException {
322         LOG.trace("enter send(OutputStream out)");
323         sendStart(out);
324         sendDispositionHeader(out);
325         sendContentTypeHeader(out);
326         sendTransferEncodingHeader(out);
327         sendEndOfHeader(out);
328         sendData(out);
329         sendEnd(out);
330     }
331 
332 
333     /**
334      * Return the full length of all the data.
335      * If you override this method make sure to override 
336      * #send(OutputStream) as well
337      * 
338      * @return long The length.
339      * @throws IOException If an IO problem occurs
340      */
341     public long length() throws IOException {
342         LOG.trace("enter length()");
343         if (lengthOfData() < 0) {
344             return -1;
345         }
346         ByteArrayOutputStream overhead = new ByteArrayOutputStream();
347         sendStart(overhead);
348         sendDispositionHeader(overhead);
349         sendContentTypeHeader(overhead);
350         sendTransferEncodingHeader(overhead);
351         sendEndOfHeader(overhead);
352         sendEnd(overhead);
353         return overhead.size() + lengthOfData();
354     }
355 
356     /**
357      * Return a string representation of this object.
358      * @return A string representation of this object.
359      * @see java.lang.Object#toString()
360      */    
361     public String toString() {
362         return this.getName();
363     }
364 
365     /**
366      * Write all parts and the last boundary to the specified output stream.
367      * 
368      * @param out The stream to write to.
369      * @param parts The parts to write.
370      * 
371      * @throws IOException If an I/O error occurs while writing the parts.
372      */
373     public static void sendParts(OutputStream out, final Part[] parts)
374         throws IOException {
375         sendParts(out, parts, DEFAULT_BOUNDARY_BYTES);
376     }
377 
378     /**
379      * Write all parts and the last boundary to the specified output stream.
380      * 
381      * @param out The stream to write to.
382      * @param parts The parts to write.
383      * @param partBoundary The ASCII bytes to use as the part boundary.
384      * 
385      * @throws IOException If an I/O error occurs while writing the parts.
386      * 
387      * @since 3.0
388      */
389     public static void sendParts(OutputStream out, Part[] parts, byte[] partBoundary)
390         throws IOException {
391         
392         if (parts == null) {
393             throw new IllegalArgumentException("Parts may not be null"); 
394         }
395         if (partBoundary == null || partBoundary.length == 0) {
396             throw new IllegalArgumentException("partBoundary may not be empty");
397         }
398         for (int i = 0; i < parts.length; i++) {
399             // set the part boundary before the part is sent
400             parts[i].setPartBoundary(partBoundary);
401             parts[i].send(out);
402         }
403         out.write(EXTRA_BYTES);
404         out.write(partBoundary);
405         out.write(EXTRA_BYTES);
406         out.write(CRLF_BYTES);
407     }
408     
409     /**
410      * Return the total sum of all parts and that of the last boundary
411      * 
412      * @param parts The parts.
413      * @return The total length
414      * 
415      * @throws IOException If an I/O error occurs while writing the parts.
416      */
417     public static long getLengthOfParts(Part[] parts)
418     throws IOException {
419         return getLengthOfParts(parts, DEFAULT_BOUNDARY_BYTES);
420     }
421     
422     /**
423      * Gets the length of the multipart message including the given parts.
424      * 
425      * @param parts The parts.
426      * @param partBoundary The ASCII bytes to use as the part boundary.
427      * @return The total length
428      * 
429      * @throws IOException If an I/O error occurs while writing the parts.
430      * 
431      * @since 3.0
432      */
433     public static long getLengthOfParts(Part[] parts, byte[] partBoundary) throws IOException {
434         LOG.trace("getLengthOfParts(Parts[])");
435         if (parts == null) {
436             throw new IllegalArgumentException("Parts may not be null"); 
437         }
438         long total = 0;
439         for (int i = 0; i < parts.length; i++) {
440             // set the part boundary before we calculate the part's length
441             parts[i].setPartBoundary(partBoundary);
442             long l = parts[i].length();
443             if (l < 0) {
444                 return -1;
445             }
446             total += l;
447         }
448         total += EXTRA_BYTES.length;
449         total += partBoundary.length;
450         total += EXTRA_BYTES.length;
451         total += CRLF_BYTES.length;
452         return total;
453     }
