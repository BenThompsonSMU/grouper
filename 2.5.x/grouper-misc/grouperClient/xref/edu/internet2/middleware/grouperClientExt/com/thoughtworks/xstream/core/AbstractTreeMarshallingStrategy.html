37  /**
38   * Basic functionality of a tree based marshalling strategy.
39   * 
40   * @author Joe Walnes
41   * @author J&ouml;rg Schaible
42   * @since 1.3
43   */
44  public abstract class AbstractTreeMarshallingStrategy implements MarshallingStrategy {
45  
46      public Object unmarshal(Object root, HierarchicalStreamReader reader, DataHolder dataHolder, ConverterLookup converterLookup, Mapper mapper) {
47          TreeUnmarshaller context = createUnmarshallingContext(root, reader, converterLookup, mapper);
48          return context.start(dataHolder);
49      }
50  
51      public void marshal(HierarchicalStreamWriter writer, Object obj, ConverterLookup converterLookup, Mapper mapper, DataHolder dataHolder) {
52          TreeMarshaller context = createMarshallingContext(writer, converterLookup, mapper);
53          context.start(obj, dataHolder);
54      }
55      
56      protected abstract TreeUnmarshaller createUnmarshallingContext(Object root,
57          HierarchicalStreamReader reader, ConverterLookup converterLookup, Mapper mapper);
58  
59      protected abstract TreeMarshaller createMarshallingContext(
60          HierarchicalStreamWriter writer, ConverterLookup converterLookup, Mapper mapper);
61  
62      /**
63       * @deprecated As of 1.2, use {@link #unmarshal(Object, HierarchicalStreamReader, DataHolder, ConverterLookup, Mapper)}
64       */
65      public Object unmarshal(Object root, HierarchicalStreamReader reader, DataHolder dataHolder, DefaultConverterLookup converterLookup, ClassMapper classMapper) {
66          return unmarshal(root, reader, dataHolder, (ConverterLookup)converterLookup, (Mapper)classMapper);
67      }
68  
69      /**
70       * @deprecated As of 1.2, use {@link #marshal(HierarchicalStreamWriter, Object, ConverterLookup, Mapper, DataHolder)}
71       */
72      public void marshal(HierarchicalStreamWriter writer, Object obj, DefaultConverterLookup converterLookup, ClassMapper classMapper, DataHolder dataHolder) {
73          marshal(writer, obj, converterLookup, (Mapper)classMapper, dataHolder);
74      }
75  
76  }
