27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.path;
28  
29  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.ErrorWriter;
30  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.HierarchicalStreamReader;
31  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.io.ReaderWrapper;
32  
33  /**
34   * Wrapper for HierarchicalStreamReader that tracks the path (a subset of XPath) of the current node that is being read.
35   *
36   * @see PathTracker
37   * @see Path
38   *
39   * @author Joe Walnes
40   */
41  public class PathTrackingReader extends ReaderWrapper {
42  
43      private final PathTracker pathTracker;
44  
45      public PathTrackingReader(HierarchicalStreamReader reader, PathTracker pathTracker) {
46          super(reader);
47          this.pathTracker = pathTracker;
48          pathTracker.pushElement(getNodeName());
49      }
50  
51      public void moveDown() {
52          super.moveDown();
53          pathTracker.pushElement(getNodeName());
54      }
55  
56      public void moveUp() {
57          super.moveUp();
58          pathTracker.popElement();
59      }
60  
61      public void appendErrors(ErrorWriter errorWriter) {
62          errorWriter.add("path", pathTracker.getPath().toString());
63          super.appendErrors(errorWriter);
64      }
65  
66  }
