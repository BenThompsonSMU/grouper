30  /**
31   * Mapper that ensures that all names in the serialization stream are read in an XML friendly way.
32   * <ul>
33   * <li><b>_</b> (underscore) chars appearing in class names are replaced with <b>$<br> (dollar)</li>
34   * <li><b>_DOLLAR_</b> string appearing in field names are replaced with <b>$<br> (dollar)</li>
35   * <li><b>__</b> string appearing in field names are replaced with <b>_<br> (underscore)</li>
36   * <li><b>default</b> is the prefix for class names with no package.</li>
37   * </ul>
38   * 
39   * @author Joe Walnes
40   * @author Mauro Talevi
41   */
42  public class XStream11XmlFriendlyMapper extends AbstractXmlFriendlyMapper {
43  
44      public XStream11XmlFriendlyMapper(Mapper wrapped) {
45          super(wrapped);
46      }
47  
48      public Class realClass(String elementName) {
49          return super.realClass(unescapeClassName(elementName));
50      }
51  
52      public String realMember(Class type, String serialized) {
53          return unescapeFieldName(super.realMember(type, serialized));
54      }
55  
56      public String mapNameFromXML(String xmlName) {
57          return unescapeFieldName(xmlName);
58      }
59  
60      
61  }
