27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.basic;
28  
29  import java.text.ParseException;
30  import java.text.SimpleDateFormat;
31  import java.util.Date;
32  
33  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.ConversionException;
34  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.core.util.ThreadSafeSimpleDateFormat;
35  
36  /**
37   * Converts a java.util.Date to a String as a date format,
38   * retaining precision down to milliseconds.
39   *
40   * @author Joe Walnes
41   * @author J&ouml;rg Schaible
42   */
43  public class DateConverter extends AbstractSingleValueConverter {
44  
45      private final ThreadSafeSimpleDateFormat defaultFormat;
46      private final ThreadSafeSimpleDateFormat[] acceptableFormats;
47  
48      /**
49       * Construct a DateConverter with standard formats and lenient set off.
50       */
51      public DateConverter() {
52          this(false);
53      }
54  
55      /**
56       * Construct a DateConverter with standard formats.
57       * 
58       * @param lenient the lenient setting of {@link SimpleDateFormat#setLenient(boolean)}
59       * @since 1.3
60       */
61      public DateConverter(boolean lenient) {
62          this("yyyy-MM-dd HH:mm:ss.S z",
63              new String[] { 
64                  "yyyy-MM-dd HH:mm:ss.S a", 
65                  "yyyy-MM-dd HH:mm:ssz", "yyyy-MM-dd HH:mm:ss z", // JDK 1.3 needs both versions
66                  "yyyy-MM-dd HH:mm:ssa" },  // backwards compatibility
67                  lenient);
68      }
69  
70      /**
71       * Construct a DateConverter with lenient set off.
72       * 
73       * @param defaultFormat the default format
74       * @param acceptableFormats fallback formats
75       */
76      public DateConverter(String defaultFormat, String[] acceptableFormats) {
77          this(defaultFormat, acceptableFormats, false);
78      }
79  
80      /**
81       * Construct a DateConverter.
82       * 
83       * @param defaultFormat the default format
84       * @param acceptableFormats fallback formats
85       * @param lenient the lenient setting of {@link SimpleDateFormat#setLenient(boolean)}
86       * @since 1.3
87       */
88      public DateConverter(String defaultFormat, String[] acceptableFormats, boolean lenient) {
89          this.defaultFormat = new ThreadSafeSimpleDateFormat(defaultFormat, 4, 20, lenient);
90          this.acceptableFormats = new ThreadSafeSimpleDateFormat[acceptableFormats.length];
91          for (int i = 0; i < acceptableFormats.length; i++) {
92              this.acceptableFormats[i] = new ThreadSafeSimpleDateFormat(acceptableFormats[i], 1, 20, lenient);
93          }
94      }
95  
96      public boolean canConvert(Class type) {
97          return type.equals(Date.class);
98      }
99  
100     public Object fromString(String str) {
101         try {
102             return defaultFormat.parse(str);
103         } catch (ParseException e) {
104             for (int i = 0; i < acceptableFormats.length; i++) {
105                 try {
106                     return acceptableFormats[i].parse(str);
107                 } catch (ParseException e2) {
108                     // no worries, let's try the next format.
109                 }
110             }
111             // no dateFormats left to try
112             throw new ConversionException("Cannot parse date " + str);
113         }
114     }
115 
116     public String toString(Object obj) {
117         return defaultFormat.format((Date) obj);
118     }
119 
120 }
