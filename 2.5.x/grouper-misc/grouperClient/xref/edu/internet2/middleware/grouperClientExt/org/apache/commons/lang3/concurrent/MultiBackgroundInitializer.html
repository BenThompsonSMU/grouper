32  package edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.concurrent;
33  
34  import java.util.Collections;
35  import java.util.HashMap;
36  import java.util.Map;
37  import java.util.NoSuchElementException;
38  import java.util.Set;
39  import java.util.concurrent.ExecutorService;
40  
41  /**
42   * <p>
43   * A specialized {@link BackgroundInitializer} implementation that can deal with
44   * multiple background initialization tasks.
45   * </p>
46   * <p>
47   * This class has a similar purpose as {@link BackgroundInitializer}. However,
48   * it is not limited to a single background initialization task. Rather it
49   * manages an arbitrary number of {@code BackgroundInitializer} objects,
50   * executes them, and waits until they are completely initialized. This is
51   * useful for applications that have to perform multiple initialization tasks
52   * that can run in parallel (i.e. that do not depend on each other). This class
53   * takes care about the management of an {@code ExecutorService} and shares it
54   * with the {@code BackgroundInitializer} objects it is responsible for; so the
55   * using application need not bother with these details.
56   * </p>
57   * <p>
58   * The typical usage scenario for {@code MultiBackgroundInitializer} is as
59   * follows:
60   * <ul>
61   * <li>Create a new instance of the class. Optionally pass in a pre-configured
62   * {@code ExecutorService}. Alternatively {@code MultiBackgroundInitializer} can
63   * create a temporary {@code ExecutorService} and delete it after initialization
64   * is complete.</li>
65   * <li>Create specialized {@link BackgroundInitializer} objects for the
66   * initialization tasks to be performed and add them to the {@code
67   * MultiBackgroundInitializer} using the
68   * {@link #addInitializer(String, BackgroundInitializer)} method.</li>
69   * <li>After all initializers have been added, call the {@link #start()} method.
70   * </li>
71   * <li>When access to the result objects produced by the {@code
72   * BackgroundInitializer} objects is needed call the {@link #get()} method. The
73   * object returned here provides access to all result objects created during
74   * initialization. It also stores information about exceptions that have
75   * occurred.</li>
76   * </ul>
77   * </p>
78   * <p>
79   * {@code MultiBackgroundInitializer} starts a special controller task that
80   * starts all {@code BackgroundInitializer} objects added to the instance.
81   * Before the an initializer is started it is checked whether this initializer
82   * already has an {@code ExecutorService} set. If this is the case, this {@code
83   * ExecutorService} is used for running the background task. Otherwise the
84   * current {@code ExecutorService} of this {@code MultiBackgroundInitializer} is
85   * shared with the initializer.
86   * </p>
87   * <p>
88   * The easiest way of using this class is to let it deal with the management of
89   * an {@code ExecutorService} itself: If no external {@code ExecutorService} is
90   * provided, the class creates a temporary {@code ExecutorService} (that is
91   * capable of executing all background tasks in parallel) and destroys it at the
92   * end of background processing.
93   * </p>
94   * <p>
95   * Alternatively an external {@code ExecutorService} can be provided - either at
96   * construction time or later by calling the
97   * {@link #setExternalExecutor(ExecutorService)} method. In this case all
98   * background tasks are scheduled at this external {@code ExecutorService}.
99   * <strong>Important note:</strong> When using an external {@code
100  * ExecutorService} be sure that the number of threads managed by the service is
101  * large enough. Otherwise a deadlock can happen! This is the case in the
102  * following scenario: {@code MultiBackgroundInitializer} starts a task that
103  * starts all registered {@code BackgroundInitializer} objects and waits for
104  * their completion. If for instance a single threaded {@code ExecutorService}
105  * is used, none of the background tasks can be executed, and the task created
106  * by {@code MultiBackgroundInitializer} waits forever.
107  * </p>
108  *
109  * @since 3.0
110  * @version $Id: MultiBackgroundInitializer.java 1082301 2011-03-16 21:02:15Z oheger $
111  */
112 public class MultiBackgroundInitializer
113         extends
114         BackgroundInitializer<MultiBackgroundInitializer.MultiBackgroundInitializerResults> {
115     /** A map with the child initializers. */
116     private final Map<String, BackgroundInitializer<?>> childInitializers =
117         new HashMap<String, BackgroundInitializer<?>>();
118 
119     /**
120      * Creates a new instance of {@code MultiBackgroundInitializer}.
121      */
122     public MultiBackgroundInitializer() {
123         super();
124     }
125 
126     /**
127      * Creates a new instance of {@code MultiBackgroundInitializer} and
128      * initializes it with the given external {@code ExecutorService}.
129      *
130      * @param exec the {@code ExecutorService} for executing the background
131      * tasks
132      */
133     public MultiBackgroundInitializer(ExecutorService exec) {
134         super(exec);
135     }
136 
137     /**
138      * Adds a new {@code BackgroundInitializer} to this object. When this
139      * {@code MultiBackgroundInitializer} is started, the given initializer will
140      * be processed. This method must not be called after {@link #start()} has
141      * been invoked.
142      *
143      * @param name the name of the initializer (must not be <b>null</b>)
144      * @param init the {@code BackgroundInitializer} to add (must not be
145      * <b>null</b>)
146      * @throws IllegalArgumentException if a required parameter is missing
147      * @throws IllegalStateException if {@code start()} has already been called
148      */
149     public void addInitializer(String name, BackgroundInitializer<?> init) {
150         if (name == null) {
151             throw new IllegalArgumentException(
152                     "Name of child initializer must not be null!");
153         }
154         if (init == null) {
155             throw new IllegalArgumentException(
156                     "Child initializer must not be null!");
157         }
158 
159         synchronized (this) {
160             if (isStarted()) {
161                 throw new IllegalStateException(
162                         "addInitializer() must not be called after start()!");
163             }
164             childInitializers.put(name, init);
165         }
166     }
167 
168     /**
169      * Returns the number of tasks needed for executing all child {@code
170      * BackgroundInitializer} objects in parallel. This implementation sums up
171      * the required tasks for all child initializers (which is necessary if one
172      * of the child initializers is itself a {@code MultiBackgroundInitializer}
173      * ). Then it adds 1 for the control task that waits for the completion of
174      * the children.
175      *
176      * @return the number of tasks required for background processing
177      */
178     @Override
179     protected int getTaskCount() {
180         int result = 1;
181 
182         for (BackgroundInitializer<?> bi : childInitializers.values()) {
183             result += bi.getTaskCount();
184         }
185 
186         return result;
187     }
188 
189     /**
190      * Creates the results object. This implementation starts all child {@code
191      * BackgroundInitializer} objects. Then it collects their results and
192      * creates a {@code MultiBackgroundInitializerResults} object with this
193      * data. If a child initializer throws a checked exceptions, it is added to
194      * the results object. Unchecked exceptions are propagated.
195      *
196      * @return the results object
197      * @throws Exception if an error occurs
198      */
199     @Override
200     protected MultiBackgroundInitializerResults initialize() throws Exception {
201         Map<String, BackgroundInitializer<?>> inits;
202         synchronized (this) {
203             // create a snapshot to operate on
204             inits = new HashMap<String, BackgroundInitializer<?>>(
205                     childInitializers);
206         }
207 
208         // start the child initializers
209         ExecutorService exec = getActiveExecutor();
210         for (BackgroundInitializer<?> bi : inits.values()) {
211             if (bi.getExternalExecutor() == null) {
212                 // share the executor service if necessary
213                 bi.setExternalExecutor(exec);
214             }
215             bi.start();
216         }
217 
218         // collect the results
219         Map<String, Object> results = new HashMap<String, Object>();
220         Map<String, ConcurrentException> excepts = new HashMap<String, ConcurrentException>();
221         for (Map.Entry<String, BackgroundInitializer<?>> e : inits.entrySet()) {
222             try {
223                 results.put(e.getKey(), e.getValue().get());
224             } catch (ConcurrentException cex) {
225                 excepts.put(e.getKey(), cex);
226             }
227         }
228 
229         return new MultiBackgroundInitializerResults(inits, results, excepts);
230     }
231 
232     /**
233      * A data class for storing the results of the background initialization
234      * performed by {@code MultiBackgroundInitializer}. Objects of this inner
235      * class are returned by {@link MultiBackgroundInitializer#initialize()}.
236      * They allow access to all result objects produced by the
237      * {@link BackgroundInitializer} objects managed by the owning instance. It
238      * is also possible to retrieve status information about single
239      * {@link BackgroundInitializer}s, i.e. whether they completed normally or
240      * caused an exception.
241      */
242     public static class MultiBackgroundInitializerResults {
243         /** A map with the child initializers. */
244         private final Map<String, BackgroundInitializer<?>> initializers;
245 
246         /** A map with the result objects. */
247         private final Map<String, Object> resultObjects;
248 
249         /** A map with the exceptions. */
250         private final Map<String, ConcurrentException> exceptions;
251 
252         /**
253          * Creates a new instance of {@code MultiBackgroundInitializerResults}
254          * and initializes it with maps for the {@code BackgroundInitializer}
255          * objects, their result objects and the exceptions thrown by them.
256          *
257          * @param inits the {@code BackgroundInitializer} objects
258          * @param results the result objects
259          * @param excepts the exceptions
260          */
261         private MultiBackgroundInitializerResults(
262                 Map<String, BackgroundInitializer<?>> inits,
263                 Map<String, Object> results,
264                 Map<String, ConcurrentException> excepts) {
265             initializers = inits;
266             resultObjects = results;
267             exceptions = excepts;
268         }
269 
270         /**
271          * Returns the {@code BackgroundInitializer} with the given name. If the
272          * name cannot be resolved, an exception is thrown.
273          *
274          * @param name the name of the {@code BackgroundInitializer}
275          * @return the {@code BackgroundInitializer} with this name
276          * @throws NoSuchElementException if the name cannot be resolved
277          */
278         public BackgroundInitializer<?> getInitializer(String name) {
279             return checkName(name);
280         }
281 
282         /**
283          * Returns the result object produced by the {@code
284          * BackgroundInitializer} with the given name. This is the object
285          * returned by the initializer's {@code initialize()} method. If this
286          * {@code BackgroundInitializer} caused an exception, <b>null</b> is
287          * returned. If the name cannot be resolved, an exception is thrown.
288          *
289          * @param name the name of the {@code BackgroundInitializer}
290          * @return the result object produced by this {@code
291          * BackgroundInitializer}
292          * @throws NoSuchElementException if the name cannot be resolved
293          */
294         public Object getResultObject(String name) {
295             checkName(name);
296             return resultObjects.get(name);
297         }
298 
299         /**
300          * Returns a flag whether the {@code BackgroundInitializer} with the
301          * given name caused an exception.
302          *
303          * @param name the name of the {@code BackgroundInitializer}
304          * @return a flag whether this initializer caused an exception
305          * @throws NoSuchElementException if the name cannot be resolved
306          */
307         public boolean isException(String name) {
308             checkName(name);
309             return exceptions.containsKey(name);
310         }
311 
312         /**
313          * Returns the {@code ConcurrentException} object that was thrown by the
314          * {@code BackgroundInitializer} with the given name. If this
315          * initializer did not throw an exception, the return value is
316          * <b>null</b>. If the name cannot be resolved, an exception is thrown.
317          *
318          * @param name the name of the {@code BackgroundInitializer}
319          * @return the exception thrown by this initializer
320          * @throws NoSuchElementException if the name cannot be resolved
321          */
322         public ConcurrentException getException(String name) {
323             checkName(name);
324             return exceptions.get(name);
325         }
326 
327         /**
328          * Returns a set with the names of all {@code BackgroundInitializer}
329          * objects managed by the {@code MultiBackgroundInitializer}.
330          *
331          * @return an (unmodifiable) set with the names of the managed {@code
332          * BackgroundInitializer} objects
333          */
334         public Set<String> initializerNames() {
335             return Collections.unmodifiableSet(initializers.keySet());
336         }
337 
338         /**
339          * Returns a flag whether the whole initialization was successful. This
340          * is the case if no child initializer has thrown an exception.
341          *
342          * @return a flag whether the initialization was successful
343          */
344         public boolean isSuccessful() {
345             return exceptions.isEmpty();
346         }
347 
348         /**
349          * Checks whether an initializer with the given name exists. If not,
350          * throws an exception. If it exists, the associated child initializer
351          * is returned.
352          *
353          * @param name the name to check
354          * @return the initializer with this name
355          * @throws NoSuchElementException if the name is unknown
356          */
357         private BackgroundInitializer<?> checkName(String name) {
358             BackgroundInitializer<?> init = initializers.get(name);
359             if (init == null) {
360                 throw new NoSuchElementException(
361                         "No child initializer with name " + name);
362             }
363 
364             return init;
365         }
366     }
