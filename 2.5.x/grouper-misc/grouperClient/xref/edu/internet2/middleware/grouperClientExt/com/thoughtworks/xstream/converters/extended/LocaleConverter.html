27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.extended;
28  
29  import edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.basic.AbstractSingleValueConverter;
30  
31  import java.util.Locale;
32  
33  /**
34   * Converts a java.util.Locale to a string.
35   *
36   * @author Jose A. Illescas
37   * @author Joe Walnes
38   */
39  public class LocaleConverter extends AbstractSingleValueConverter {
40  
41      public boolean canConvert(Class type) {
42          return type.equals(Locale.class);
43      }
44  
45      public Object fromString(String str) {
46          int[] underscorePositions = underscorePositions(str);
47          String language, country, variant;
48          if (underscorePositions[0] == -1) { // "language"
49              language = str;
50              country = "";
51              variant = "";
52          } else if (underscorePositions[1] == -1) { // "language_country"
53              language = str.substring(0, underscorePositions[0]);
54              country = str.substring(underscorePositions[0] + 1);
55              variant = "";
56          } else { // "language_country_variant"
57              language = str.substring(0, underscorePositions[0]);
58              country = str.substring(underscorePositions[0] + 1, underscorePositions[1]);
59              variant = str.substring(underscorePositions[1] + 1);
60          }
61          return new Locale(language, country, variant);
62      }
63  
64      private int[] underscorePositions(String in) {
65          int[] result = new int[2];
66          for (int i = 0; i < result.length; i++) {
67              int last = i == 0 ? 0 : result[i - 1];
68              result[i] = in.indexOf('_', last + 1);
69          }
70          return result;
71      }
72  
73  }
