27  package edu.internet2.middleware.grouperClientExt.com.thoughtworks.xstream.converters.basic;
28  
29  /**
30   * Converts a long primitive or java.lang.Long wrapper to
31   * a String.
32   *
33   * @author Joe Walnes
34   */
35  public class LongConverter extends AbstractSingleValueConverter {
36  
37      public boolean canConvert(Class type) {
38          return type.equals(long.class) || type.equals(Long.class);
39      }
40  
41      public Object fromString(String str) {
42          return Long.decode(str);
43      }
44  
45  }
