34  import java.util.Calendar;

36  import java.util.Locale;

39  /**
40   * <p>Date and time formatting utilities and constants.</p>
41   *
42   * <p>Formatting is performed using the thread-safe
43   * {@link edu.internet2.middleware.grouperClientExt.org.apache.commons.lang3.time.FastDateFormat} class.</p>
44   *
45   * @since 2.0
46   * @version $Id: DateFormatUtils.java 1088899 2011-04-05 05:31:27Z bayard $
47   */
48  public class DateFormatUtils {
49  
50      /**
51       * The UTC time zone (often referred to as GMT).
52       * This is private as it is mutable.
53       */
54      private static final TimeZone UTC_TIME_ZONE = TimeZone.getTimeZone("GMT");
55      /**
56       * ISO8601 formatter for date-time without time zone.
57       * The format used is <tt>yyyy-MM-dd'T'HH:mm:ss</tt>.
58       */
59      public static final FastDateFormat ISO_DATETIME_FORMAT
60              = FastDateFormat.getInstance("yyyy-MM-dd'T'HH:mm:ss");
61  
62      /**
63       * ISO8601 formatter for date-time with time zone.
64       * The format used is <tt>yyyy-MM-dd'T'HH:mm:ssZZ</tt>.
65       */
66      public static final FastDateFormat ISO_DATETIME_TIME_ZONE_FORMAT
67              = FastDateFormat.getInstance("yyyy-MM-dd'T'HH:mm:ssZZ");
68  
69      /**
70       * ISO8601 formatter for date without time zone.
71       * The format used is <tt>yyyy-MM-dd</tt>.
72       */
73      public static final FastDateFormat ISO_DATE_FORMAT
74              = FastDateFormat.getInstance("yyyy-MM-dd");
75  
76      /**
77       * ISO8601-like formatter for date with time zone.
78       * The format used is <tt>yyyy-MM-ddZZ</tt>.
79       * This pattern does not comply with the formal ISO8601 specification
80       * as the standard does not allow a time zone  without a time.
81       */
82      public static final FastDateFormat ISO_DATE_TIME_ZONE_FORMAT
83              = FastDateFormat.getInstance("yyyy-MM-ddZZ");
84  
85      /**
86       * ISO8601 formatter for time without time zone.
87       * The format used is <tt>'T'HH:mm:ss</tt>.
88       */
89      public static final FastDateFormat ISO_TIME_FORMAT
90              = FastDateFormat.getInstance("'T'HH:mm:ss");
91  
92      /**
93       * ISO8601 formatter for time with time zone.
94       * The format used is <tt>'T'HH:mm:ssZZ</tt>.
95       */
96      public static final FastDateFormat ISO_TIME_TIME_ZONE_FORMAT
97              = FastDateFormat.getInstance("'T'HH:mm:ssZZ");
98  
99      /**
100      * ISO8601-like formatter for time without time zone.
101      * The format used is <tt>HH:mm:ss</tt>.
102      * This pattern does not comply with the formal ISO8601 specification
103      * as the standard requires the 'T' prefix for times.
104      */
105     public static final FastDateFormat ISO_TIME_NO_T_FORMAT
106             = FastDateFormat.getInstance("HH:mm:ss");
107 
108     /**
109      * ISO8601-like formatter for time with time zone.
110      * The format used is <tt>HH:mm:ssZZ</tt>.
111      * This pattern does not comply with the formal ISO8601 specification
112      * as the standard requires the 'T' prefix for times.
113      */
114     public static final FastDateFormat ISO_TIME_NO_T_TIME_ZONE_FORMAT
115             = FastDateFormat.getInstance("HH:mm:ssZZ");
116 
117     /**
118      * SMTP (and probably other) date headers.
119      * The format used is <tt>EEE, dd MMM yyyy HH:mm:ss Z</tt> in US locale.
120      */
121     public static final FastDateFormat SMTP_DATETIME_FORMAT
122             = FastDateFormat.getInstance("EEE, dd MMM yyyy HH:mm:ss Z", Locale.US);

135     /**
136      * <p>Formats a date/time into a specific pattern using the UTC time zone.</p>
137      * 
138      * @param millis  the date to format expressed in milliseconds
139      * @param pattern  the pattern to use to format the date, not null
140      * @return the formatted date
141      */
142     public static String formatUTC(long millis, String pattern) {
143         return format(new Date(millis), pattern, UTC_TIME_ZONE, null);
144     }

305     /**
306      * <p>Formats a date/time into a specific pattern in a time zone  and locale.</p>
307      * 
308      * @param date  the date to format, not null
309      * @param pattern  the pattern to use to format the date, not null, not null
310      * @param timeZone  the time zone  to use, may be <code>null</code>
311      * @param locale  the locale to use, may be <code>null</code>
312      * @return the formatted date
313      */
314     public static String format(Date date, String pattern, TimeZone timeZone, Locale locale) {
315         FastDateFormat df = FastDateFormat.getInstance(pattern, timeZone, locale);
316         return df.format(date);
317     }
