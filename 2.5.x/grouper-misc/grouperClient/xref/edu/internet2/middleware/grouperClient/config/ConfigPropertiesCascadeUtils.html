20  import java.util.Map;
21  
22  import edu.internet2.middleware.grouperClient.util.GrouperClientUtils;
23  
24  /**
25   * utility methods specific to grouper client
26   */
27  public class ConfigPropertiesCascadeUtils extends ConfigPropertiesCascadeCommonUtils {
28  
29    /**
30     * substitute an EL for objects.  Dont worry if something returns null
31     * @param stringToParse
32     * @param variableMap
33     * @return the string
34     */
35    public static String substituteExpressionLanguage(String stringToParse, Map<String, Object> variableMap) {
36      
37      return substituteExpressionLanguage(stringToParse, variableMap, true, true, true, false);
38      
39    }
40    
41  
42    /**
43     * substitute an EL for objects
44     * @param stringToParse
45     * @param variableMap
46     * @param allowStaticClasses if true allow static classes not registered with context
47     * @param silent if silent mode, swallow exceptions (warn), and dont warn when variable not found
48     * @param lenient false if undefined variables should throw an exception.  if lenient is true (default)
49     * then undefined variables are null
50     * @param logOnNull if null output of substitution should be logged
51     * @return the string
52     */
53    public static String substituteExpressionLanguage(String stringToParse, 
54        Map<String, Object> variableMap, boolean allowStaticClasses, boolean silent, boolean lenient, boolean logOnNull) {
55  
56      //  //we dont have jexl so dont do this logic
57      //  return stringToParse;
58  
59      return GrouperClientUtils.substituteExpressionLanguage(stringToParse, variableMap, allowStaticClasses, silent, lenient, logOnNull);
60    }
61  
62  }
