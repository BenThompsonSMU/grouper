59    /**
60     * retrieve dircontext
61     * @param ldapUrl 
62     * @return the context
63     */
64    @SuppressWarnings("unchecked")
65    public static DirContext retrieveContext(String ldapUrl) {
66  
67      Map<String, Object> debugMap = LOG.isDebugEnabled() ? new LinkedHashMap<String, Object>() : null;
68  
69      if (debugMap != null) {
70        debugMap.put("method", "GrouperClientLdapUtils.retrieveContext");
71      }
72      
73      String ldapUser = null;
74      String ldapPass = null;
75      try {
76        // Set up the environment for creating the initial context
77        Hashtable<String, String> env = new Hashtable<String, String>();
78        
79        //ldapUrl = GrouperClientUtils.propertiesValue("grouperClient.ldap.url", false);
80        
81        //see if invalid SSL
82        String ldapsSocketFactoryName = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.ldaps.customSocketFactory");
83        
84        if (!GrouperClientUtils.isBlank(ldapsSocketFactoryName)) {
85          if (ldapUrl.startsWith("ldaps")) {
86            Class<? extends SocketFactory> ldapsSocketFactoryClass = GrouperClientUtils.forName(ldapsSocketFactoryName);
87            env.put("java.naming.ldap.factory.socket", ldapsSocketFactoryClass.getName());
88          }
89        }
90        
91        env.put(Context.INITIAL_CONTEXT_FACTORY, "com.sun.jndi.ldap.LdapCtxFactory");
92        
93        
94        env.put(Context.PROVIDER_URL, ldapUrl);
95  
96        if (debugMap != null) {
97          debugMap.put("LDAP url", ldapUrl);
98          debugMap.put("LDAP authentication type", "simple");
99        }
100 
101       String userLabel = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.ldap.user.label");
102 
103       ldapUser = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.ldap." + userLabel);
104   
105       String ldapUserPrefix = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.ldap.user.prefix");
106       String ldapUserSuffix = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.ldap.user.suffix");
107       
108       //put all these together
109       ldapUser = ldapUserPrefix + ldapUser + ldapUserSuffix;
110       
111       //env.put(Context.SECURITY_PRINCIPAL, "uid=" + ldapUser + ",ou=entities,dc=upenn,dc=edu");
112       if (debugMap != null) {
113         debugMap.put("LDAP user", ldapUser);
114       }
115 
116       env.put(Context.SECURITY_PRINCIPAL, ldapUser);
117   
118       boolean disableExternalFileLookup = GrouperClientConfig.retrieveConfig().propertyValueBooleanRequired(
119           "encrypt.disableExternalFileLookup");
120       
121       //lets lookup if file
122       ldapPass = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.ldap.password");
123       String ldapPassFromFile = GrouperClientUtils.readFromFileIfFile(ldapPass, disableExternalFileLookup);
124 
125       String passPrefix = null;
126       
127       if (!GrouperClientUtils.equals(ldapPass, ldapPassFromFile)) {
128         
129         passPrefix = "LDAP pass: reading encrypted value from file: " + ldapPass;
130         
131         String encryptKey = GrouperClientUtils.encryptKey();
132         
133         ldapPass = new Crypto(encryptKey).decrypt(ldapPassFromFile);
134         
135       } else {
136         passPrefix = "LDAP pass: reading scalar value from grouper.client.properties";
137       }
138       
139       if (GrouperClientConfig.retrieveConfig().propertyValueBoolean("grouperClient.logging.logMaskedPassword", false)) {
140         if (debugMap != null) {
141           debugMap.put("Pass", passPrefix + ": " + GrouperClientUtils.repeat("*", ldapPass.length()));
142         }
143       }
144       
145       env.put(Context.SECURITY_CREDENTIALS, ldapPass);
146        
147       // Create the initial context
148       DirContext context = new InitialDirContext(env);
149       return context;
150     } catch (NamingException ne) {
151       throw new RuntimeException("Problem connecting to ldap: url: " + ldapUrl + ", user: " + ldapUser, ne );
152     } finally {
153       if (debugMap != null) {
154         LOG.debug(GrouperClientUtils.mapToString(debugMap));
155       }
156     }
157 
158   }

170   /**
171    * print out a naming enumeration
172    * @param namingEnumeration
173    * @throws NamingException 
174    */
175   public static void printNamingEnumeration(NamingEnumeration<?> namingEnumeration) throws NamingException {
176     while (namingEnumeration.hasMore()) {
177       Object nextElement = namingEnumeration.next();
178       if (nextElement instanceof Attribute) {
179         Attribute attribute = (Attribute)nextElement;
180         printAttribute(attribute);
181       } else if (nextElement instanceof SearchResult) {
182         SearchResult searchResult = (SearchResult)nextElement;
183         System.out.println("Search result: " + searchResult.getNameInNamespace());
184         Attributes attributes = searchResult.getAttributes();
185         printAttributes(attributes);
186       } else if (nextElement instanceof NameClassPair) {
187         NameClassPair nameClassPair = (NameClassPair)namingEnumeration.nextElement();
188         System.out.println("Name class pair: " + nameClassPair.getClassName() + ", " + nameClassPair.getNameInNamespace());
189       } else {
190         throw new RuntimeException("Not expecting type: " + nextElement);
191       }
192       
193     }

197   /**
198    * retrieve a single valued attribute as string
199    * @param object
200    * @param attributeName 
201    * @throws NamingException 
202    * @return the attribute value or null if not there
203    */
204   public static String retrieveAttributeStringValue(Object object, 
205       String attributeName) throws NamingException {
206     if (object == null) {
207       return null;
208     }
209     Map<String, Object> debugLog = LOG.isDebugEnabled() ? new LinkedHashMap<String, Object>() : null;
210     if (debugLog != null) {
211       debugLog.put("method", "GrouperClientLdapUtils.retrieveAttributeStringValue");
212     }
213     try {
214       if (object instanceof Attribute) {
215         Attribute attribute = (Attribute)object;
216         if (GrouperClientUtils.equals(attribute.getID(), attributeName)) {
217           String value = (String)attribute.get();
218           if (debugLog != null) {
219             debugLog.put("LDAP found attribute: '" + attributeName + "' with value", "'" + value + "'");
220           }
221           return value;
222         }
223         if (debugLog != null) {
224           debugLog.put("LDAP didnt find attribute: '" + attributeName + "'", "instead found attribute: '" + attribute.getID() + "'");
225         }
226         return null;
227       } else if (object instanceof SearchResult) {
228         if (debugLog != null) {
229           debugLog.put("LDAP found ", "SearchResult");
230         }
231         SearchResult searchResult = (SearchResult)object;
232         Attributes attributes = searchResult.getAttributes();
233         Attribute attribute = attributes.get(attributeName);
234         if (debugLog != null) {
235           debugLog.put("LDAP SearchResult attributes has attribute", "'" + attributeName + "'? " + (attribute!=null));
236         }
237         return retrieveAttributeStringValue(attribute, attributeName);
238       } else if (object instanceof NamingEnumeration<?>) {
239         NamingEnumeration<?> namingEnumeration = (NamingEnumeration<?>)object;
240         if (!namingEnumeration.hasMore()) {
241           if (debugLog != null) {
242             debugLog.put("LDAP found empty", "NamingEnumeration");
243           }
244           return null;
245         }
246         Object next = namingEnumeration.next();
247         if (namingEnumeration.hasMore()) {
248           if (debugLog != null) {
249             debugLog.put("LDAP found more than one element in", "NamingEnumeration");
250           }
251           throw new RuntimeException("Expecting one result");
252         }
253         if (debugLog != null) {
254           debugLog.put("LDAP processing NamingEnumeration of size", "one");
255         }
256         return retrieveAttributeStringValue(next, attributeName);
257       } else {
258         throw new RuntimeException("Not expecting type: " + object);
259       }
260     } finally {
261       if (debugLog != null) {
262         LOG.debug(GrouperClientUtils.mapToString(debugLog));
263       }
264     }
265   }

267   /**
268    * retrieve a string array of values
269    * @param object
270    * @param attributeName 
271    * @throws NamingException 
272    * @return the attribute value or null if not there
273    */
274   public static List<String> retrieveAttributeStringListValue(Object object, 
275       String attributeName) throws NamingException {
276     if (object == null) {
277       LOG.debug("LDAP object is null looking for list of string attributes: '" + attributeName + "'");
278       return null;
279     }
280     if (object instanceof Attribute) {
281       LOG.debug("LDAP found attribute: '" + attributeName + "'");
282       Attribute attribute = (Attribute)object;
283       NamingEnumeration<?> namingEnumeration = attribute.getAll();
284       return retrieveAttributeStringListValue(namingEnumeration, attributeName);
285     } else if (object instanceof SearchResult) {
286       SearchResult searchResult = (SearchResult)object;
287       Attributes attributes = searchResult.getAttributes();
288       Attribute attribute = attributes.get(attributeName);
289       LOG.debug("LDAP found SearchResult for attribute: '" + attributeName + "', found attribute? " + (attribute!= null));
290       return retrieveAttributeStringListValue(attribute, attributeName);
291     } else if (object instanceof NamingEnumeration<?>) {
292       LOG.debug("LDAP found NamingEnumeration for attribute: '" + attributeName + "'");
293       int size = 0;
294       NamingEnumeration<?> namingEnumeration = (NamingEnumeration<?>)object;
295       List<String> resultList = new ArrayList<String>();
296       while (namingEnumeration.hasMore()) {
297         Object next = namingEnumeration.next();
298         if (next instanceof SearchResult) {
299           if (size == 0 && !namingEnumeration.hasMore()) {
300             LOG.debug("LDAP found SearchResult in NamingEnumeration for attribute: '" + attributeName + "'");
301             return retrieveAttributeStringListValue(next, attributeName);
302           }
303           LOG.debug("LDAP found multiple SearchResults in NamingEnumeration for attribute: '" + attributeName + "'");
304           throw new RuntimeException("Error: multiple search results found!");
305         }
306         resultList.add((String)next);
307         size++;
308       }
309       if (size == 0) {
310         LOG.debug("LDAP did not found SearchResult in NamingEnumeration for attribute: '" + attributeName + "'");
311         return null;
312       }
313       return resultList;
314     } else {
315       throw new RuntimeException("Not expecting type: " + object.getClass() + ", " + object);
316     }
317   }

334   /** how often should we reconfigure the failover client */
335   private static Integer configureEverySeconds = null;
336   /** when was the failover client last configured */
337   private static Long lastFailoverConfigure = null;
338   /** readonly failover config name */
339   public static final String LDAP_FAILOVER_CONFIG_NAME = "grouperLdap";
340   /** cache this so we know if we need to reconfigure */
341   public static File lastDiscoveryConfigFile = null;

343   /**
344    * see if needs reconfigure
345    * @return true or false
346    */
347   private static boolean needsReconfigure() {
348     boolean needsReconfigure = lastFailoverConfigure == null || (System.currentTimeMillis() - lastFailoverConfigure) / 1000 > configureEverySeconds;
349     if (!DiscoveryClient.hasDiscovery() && lastFailoverConfigure != null) {
350       needsReconfigure = false;
351     }
352     return needsReconfigure;
353   }
