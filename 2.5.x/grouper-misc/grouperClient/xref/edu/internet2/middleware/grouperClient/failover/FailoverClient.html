98     * file where state is read or stored to or null if not saving state
99     * @return the file or null if not saving
100    */
101   static File fileSaveFailoverClientState() {
102 
103     //cache this
104     if (failoverStateFileSet) {
105       return failoverStateFile;
106     }
107 
108     String cacheDirectoryName = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.cacheDirectory");
109     boolean hasCacheDirectory = !GrouperClientUtils.isBlank(cacheDirectoryName);
110 
111     if (hasCacheDirectory) {
112       cacheDirectoryName = GrouperClientUtils.cacheDirectoryName();
113     }
114 
115     int saveStateEverySeconds = GrouperClientConfig.retrieveConfig().propertyValueInt("grouperClient.saveFailoverStateEverySeconds", -1);
116 
117     boolean hasSaveStateEverySeconds = saveStateEverySeconds >= 0;
118 
119     //you need both
120     if (hasSaveStateEverySeconds && !hasCacheDirectory) {
121 
122       throw new RuntimeException("You have grouperClient.saveFailoverStateEverySeconds set in the grouper.properties " +
123       "but you do not have grouperClient.cacheDirectory set, you need a directory to save the file");
124 
125     }
126 
127     failoverStateFileSet = true;
128     if (!hasSaveStateEverySeconds) {
129       return null;
130     }
131 
132     failoverStateFile = new File(cacheDirectoryName + File.separator + "grouperClientFailoverState.bin");
133 
134     return failoverStateFile;
135   }

213               debugMap.put("success", "false");
214             }
215             //delete the file, log the exception.  note, it was deleted in the unserialize call probably
216             LOG.error("Cant unserialize failover state file, deleting it: "

287   private void incrementErrorForConnection(String connectionName) {
288 
289     ExpirableCache<Long, Long> errorCount = errorCountPerMinute(connectionName);
290     Long minutesSince1970 = minutesSince1970();
291     Long currentCount = GrouperClientUtils.defaultIfNull(errorCount.get(minutesSince1970), 0L);
292     currentCount++;
293     errorCount.put(minutesSince1970, currentCount);

298   /**
299    * see how many errors in the failover connection
300    * @param connectionName
301    * @param minutesSince1970
302    * @param minutesToCheck 
303    * @return the number of errors
304    */
305   private long errorsForConnection(String connectionName, long minutesSince1970) {
306 
307     long errors = 0;
308 
309     ExpirableCache<Long, Long> errorCache = errorCountPerMinute(connectionName);
310 
311     //get the errors from the last X minutes
312     for (int i=0;i<this.failoverConfig.getMinutesToKeepErrors()+1;i++) {
313 
314       Long currentErrors = errorCache.get(minutesSince1970-i);
315 
316       if (currentErrors != null) {
317         errors += currentErrors;
318       }
319     }
320     return errors;
321   }

359   /**
360    * get a failover client from memory or disk or make a new one
361    * @param failoverConfig config with the type we are talking about
362    */
363   public static synchronized void initFailoverClient(FailoverConfig failoverConfig) {
364     FailoverClient failoverClient = instanceMapFromType().get(failoverConfig.getConnectionType());
365 
366     if (failoverClient == null) {
367 
368       failoverClient = new FailoverClient();
369       instanceMapFromType().put(failoverConfig.getConnectionType(), failoverClient);
370     }
371     failoverClient.setFailoverConfig(failoverConfig);
372 
373     failoverConfigMap.put(failoverConfig.getConnectionType(), failoverConfig);
374 
375     failoverConfig.getAffinitySeconds();
376     int affinitySeconds = failoverConfig.getAffinitySeconds();
377     //error if less than 1
378     if (affinitySeconds < 1) {
379       affinitySeconds = 1;
380     }
381     
382     //see if affinity is correct
383     if (failoverClient.connectionAffinityCache == null || failoverClient.connectionAffinityCache.getDefaultTimeToLiveInMillis() / 1000 != affinitySeconds) {
384       failoverClient.connectionAffinityCache = new ExpirableCache<Boolean, String>(ExpirableCacheUnit.SECOND, affinitySeconds);
385     }
386 
387 
388   }

390   /**
391    * based on the connection type, get the list of connection names to try
392    * @param failoverConfig configuration of the failover
393    * @param connectionType
394    * @return the list, if not pooled, just return list of size one
395    */
396   private List<String> orderedListOfConnectionNames() {
397 
398     //set so that connections cant be added twice
399     Set<String> resultConnectionNames = new LinkedHashSet<String>();
400 
401     //all connection names
402     Set<String> connectionNamesInPool = new LinkedHashSet<String>();
403     connectionNamesInPool.addAll(GrouperClientUtils.nonNull(this.failoverConfig.getConnectionNames()));
404     connectionNamesInPool.addAll(GrouperClientUtils.nonNull(this.failoverConfig.getConnectionNamesSecondTier()));
405 
406 
407     //see if there is only one
408     if (GrouperClientCommonUtils.length(this.failoverConfig.getConnectionNames()) 
409         + GrouperClientCommonUtils.length(this.failoverConfig.getConnectionNamesSecondTier()) == 1) {
410 
411       resultConnectionNames.addAll(connectionNamesInPool);
412       return new ArrayList<String>(resultConnectionNames);
413 
414     }
415 
416     Map<String, Object> debugMap = LOG.isDebugEnabled() ? new LinkedHashMap<String, Object>() : null;
417     if (debugMap != null) {
418       debugMap.put("method", "FailoverClient.orderedListOfConnections");
419       if (this.failoverConfig != null) {
420         debugMap.put("connectionType", this.failoverConfig.getConnectionType());
421       }
422       debugMap.put("failoverClient hash", Integer.toHexString(this.hashCode()));
423       debugMap.put("failoverClient cache hash", Integer.toHexString(this.connectionAffinityCache.hashCode()));
424 
425     }
426 
427     try {
428       //see if there is affinity
429       String affinityConnectionName = this.connectionAffinityCache.get(Boolean.TRUE);
430 
431       if (debugMap != null) {
432         debugMap.put("affinityConnection", affinityConnectionName);
433       }
434 
435       //we have a pool
436       Set<String> availableConnectionsFewestErrorsFirst = new LinkedHashSet<String>();
437 
438       long minutesSince1970 = minutesSince1970();
439 
440       Map<Long, Set<String>> connectionsForErrorCount = new HashMap<Long, Set<String>>();
441       Set<Long> numberOfErrors = new TreeSet<Long>();
442 
443       int i = 0;
444       for (String connectionName : connectionNamesInPool) {
445 
446         long errors = errorsForConnection(connectionName, minutesSince1970);
447 
448         //keep track of all errors
449         numberOfErrors.add(errors);
450 
451         Set<String> connectionNames = connectionsForErrorCount.get(errors);
452 
453         if (connectionNames == null) {
454 
455           connectionNames = new LinkedHashSet<String>();
456           connectionsForErrorCount.put(errors, connectionNames);
457 
458         }
459 
460         connectionNames.add(connectionName);
461 
462         if (debugMap != null) {
463           debugMap.put("connection." + i, connectionName + ", errors: " + errors);
464         }
465         i++;
466       }
467 
468       //for each number of errors, pick the best one
469       for (long errors : numberOfErrors) {
470 
471         Set<String> connectionNames = connectionsForErrorCount.get(errors);
472 
473         //if one, all set
474         if (connectionNames.size() == 1) {
475           availableConnectionsFewestErrorsFirst.add(connectionNames.iterator().next());
476           continue;
477         }
478 
479         //multiple with the same error count
480         if (availableConnectionsFewestErrorsFirst.size() == 0) {
481 
482           //see if there is affinity
483           if (!GrouperClientUtils.isBlank(affinityConnectionName) && connectionNames.contains(affinityConnectionName)) {
484             if (debugMap != null) {
485 
486               debugMap.put("found affinity", affinityConnectionName);
487             }
488             availableConnectionsFewestErrorsFirst.add(affinityConnectionName);
489             connectionNames.remove(affinityConnectionName);
490             availableConnectionsFewestErrorsFirst.addAll(connectionNames);
491             continue;
492           }
493 
494           //if active/active, pick one
495           if (this.failoverConfig.getFailoverStrategy() == null || this.failoverConfig.getFailoverStrategy() == FailoverStrategy.activeActive) {
496 
497             Set<String> bestTierConnectionNames = new HashSet<String>();
498             if (GrouperClientUtils.length(this.failoverConfig.getConnectionNamesSecondTier()) == 0) {
499 
500               bestTierConnectionNames.addAll(this.failoverConfig.getConnectionNames());
501 
502             } else if (GrouperClientUtils.length(this.failoverConfig.getConnectionNames()) == 0) {
503               bestTierConnectionNames.addAll(this.failoverConfig.getConnectionNamesSecondTier());
504             } else {
505 
506               //we have both... see if there are first tier ones there
507               for (String firstTierName : GrouperClientUtils.nonNull(this.failoverConfig.getConnectionNames())) {
508                 if (connectionNames.contains(firstTierName)) {
509                   bestTierConnectionNames.add(firstTierName);
510                 }
511               }
512               //if there arent any, just add all
513               if (bestTierConnectionNames.size() == 0) {
514                 bestTierConnectionNames.addAll(connectionNames);
515               }
516 
517             }
518 
519             int index = random.nextInt(bestTierConnectionNames.size());
520             String bestConnection = (String)GrouperClientUtils.get(bestTierConnectionNames, index);
521 
522             availableConnectionsFewestErrorsFirst.add(bestConnection);
523             connectionNames.remove(bestConnection);
524 
525             //if we have affinity, set it
526             this.connectionAffinityCache.put(Boolean.TRUE, bestConnection);
527             if (debugMap != null) {
528               debugMap.put("Setting affinity", bestConnection);
529             }
530           } else {
531             //if not active/active, then use the primary one... or the one furthest from right
532             for (String connectionName : connectionNamesInPool) {
533 
534               if (connectionNames.contains(connectionName)) {
535 
536                 if (availableConnectionsFewestErrorsFirst.size() == 0) {
537                   //affinity in active/standby
538                   if (debugMap != null) {
539                     debugMap.put("Setting affinity", connectionName);
540                   }
541                   //i guess these have affinity too, not sure if it is best, maybe it should always just be preferred... hmmm
542                   this.connectionAffinityCache.put(Boolean.TRUE, connectionName);
543                 }
544 
545                 availableConnectionsFewestErrorsFirst.add(connectionName);
546                 connectionNames.remove(connectionName);
547               }
548             }
549           }
550         }
551         //just add the rest in there based on the number of errors
552         for (String connectionName : connectionNames) {
553           if (!availableConnectionsFewestErrorsFirst.contains(connectionName)) {
554             availableConnectionsFewestErrorsFirst.add(connectionName);
555           }
556         }
557 
558       }
559 
560       //make sure they are all there
561       for (String connectionName : connectionNamesInPool) {
562         if (!availableConnectionsFewestErrorsFirst.contains(connectionName)) {
563           availableConnectionsFewestErrorsFirst.add(connectionName);
564         }
565       }
566 
567       if (debugMap != null) {
568         debugMap.put("affinityConnectionPost", this.connectionAffinityCache.get(Boolean.TRUE));
569       }
570 
571       return new ArrayList<String>(availableConnectionsFewestErrorsFirst);
572     } finally {
573       if (debugMap != null) {
574         LOG.debug(GrouperClientUtils.mapToString(debugMap));
575       }
576     }
577   }

590   /**
591    * run failover logic, return the result from the logic
592    * @param <T>
593    * @param connectionType is the type of connection
594    * @param useThreads is true if we should use threads, maybe pass false e.g. 
595    * if the system is not initted and it relies on something to get configuration.
596    * generally true though
597    * @param failoverLogic
598    * @return the result from the logic
599    */
600   public static <T> T failoverLogic(String connectionType, boolean useThreads, FailoverLogic<T> failoverLogic) {
601 
602     Map<String, Object> debugMap = LOG.isDebugEnabled() ? new LinkedHashMap<String, Object>() : null; 
603 
604     if (debugMap != null) {
605       debugMap.put("method", "FailoverClient.failoverLogic");
606     }
607     try {
608       FailoverClient failoverClient = failoverClient(connectionType);
609 
610       try {
611         return failoverClient.internal_failoverLogic(useThreads, failoverLogic);
612       } finally {
613 
614         //save state to file if necessary
615         int saveStateEverySeconds = GrouperClientConfig.retrieveConfig().propertyValueInt("grouperClient.saveFailoverStateEverySeconds", -1);
616 
617         if (debugMap != null) {
618           debugMap.put("saveStateEverySeconds", saveStateEverySeconds);
619         }
620 
621         File saveStateFile = fileSaveFailoverClientState();
622 
623         if (saveStateEverySeconds > -1) {
624 
625           //if we are saving each time, or if we havent saved in the period that we need to save
626           boolean saveStateNow = saveStateEverySeconds == 0 || saveStateFile.lastModified() < System.currentTimeMillis() - (saveStateEverySeconds * 1000);
627 
628           if (saveStateNow) {
629             if (debugMap != null) {
630               debugMap.put("savingStateToFile", saveStateFile.getAbsolutePath());
631             }
632             //long now = System.nanoTime();
633             try {
634               GrouperClientUtils.serializeObjectToFile(instanceMapFromType, saveStateFile);
635             } catch (Throwable t) {
636               LOG.warn("Problem saving grouperClientFailoverState.bin: " + t.getMessage());
637             }
638             //System.out.println("Serialized in " + ((System.nanoTime() - now) / 1000000) + "ms");
639           }
640 
641         }
642 
643       }
644     } finally {
645       if (debugMap != null) {
646         LOG.debug(GrouperClientUtils.mapToString(debugMap));
647       }
648     }
649   }

651   /**
652    * if we are starting up, add time to the timeout so we have time for classes to load etc...
653    */
654   private static Long startupMillis = null;
655 
656   /**
657    * run failover logic, return the result from the logic
658    * @param useThreads is true if we should use threads, maybe pass false e.g. 
659    * if the system is not initted and it relies on something to get configuration.
660    * generally true though
661    * @param <T>
662    * @param failoverLogic
663    * @return the result from the logic
664    */
665   private <T> T internal_failoverLogic(boolean useThreads, final FailoverLogic<T> failoverLogic) {
666 
667     final Map<String, Object> debugMap = LOG.isDebugEnabled() ? Collections.synchronizedMap(new LinkedHashMap<String, Object>()) : null;
668 
669     if (debugMap != null) {
670       debugMap.put("method", "FailoverClient.internal_failoverLogic");
671     }
672 
673     try {
674 
675       final List<String> orderedConnections = this.orderedListOfConnectionNames();
676 
677       int timeoutSeconds = this.failoverConfig.getTimeoutSeconds();
678 
679       if (startupMillis == null) {
680         startupMillis = System.currentTimeMillis();
681       }
682 
683       if (this.failoverConfig.getSecondsForClassesToLoad() > 0) {
684         //see if that amount of time has passed yet
685         long secondsSinceStartup = (System.currentTimeMillis() - startupMillis) / 1000;
686         if (secondsSinceStartup < this.failoverConfig.getSecondsForClassesToLoad() ) {
687           //add that amount of time to the timeout so we can wait for things to get started
688           timeoutSeconds += (this.failoverConfig.getSecondsForClassesToLoad() - secondsSinceStartup);
689         }
690       }
691 
692       if (debugMap != null) {
693         debugMap.put("type", this.failoverConfig.getConnectionType());
694         debugMap.put("connections", GrouperClientUtils.length(orderedConnections));
695       }
696 
697       if (GrouperClientUtils.length(orderedConnections) == 0) {
698         throw new RuntimeException("Why are there no connections for type: " + this.failoverConfig.getConnectionType());
699       }
700 
701       if (GrouperClientUtils.length(orderedConnections) == 1) {
702         String theFailoverConnectionName = orderedConnections.get(0);
703         if (debugMap != null) {
704           debugMap.put("Not load balancing", "connection type: " + this.failoverConfig.getConnectionType() + ", name: " + theFailoverConnectionName);
705         }
706         return failoverLogic.logic(new FailoverLogicBean(false, theFailoverConnectionName, true));
707       }
708 
709       final Object[] results = new Object[orderedConnections.size()];
710       final Boolean[] successes = new Boolean[orderedConnections.size()];
711 
712       if (!GrouperClientConfig.retrieveConfig().propertyValueBoolean("grouperClient.failoverClientUseThreads", true)) {
713         useThreads = false;
714       }
715 
716       //lets run the logic with a timeout for each connection
717       for (int i = 0; i < orderedConnections.size(); i++) {
718 
719         final int I = i;
720         final String connectionName = orderedConnections.get(i);
721         final String id = LOG.isDebugEnabled() ? GrouperClientUtils.uniqueId() : null;
722 
723         //if not initted dont run in thread
724         if (!useThreads) {
725           try {
726             long startNanos = -1;
727             if (debugMap != null) {
728               debugMap.put("Try " + I + " conn", connectionName);
729               startNanos = System.nanoTime();
730             }
731             results[I] = failoverLogic.logic(new FailoverLogicBean(false, connectionName, i == orderedConnections.size()-1));
732             if (debugMap != null) {
733               debugMap.put("Finish " + I + " conn in " + ((System.nanoTime() - startNanos) / 1000000) + "ms", connectionName);
734             }
735             successes[I] = true;
736           } catch (Throwable e) {
737             if (debugMap != null) {
738               debugMap.put("Error " + I + " conn", e.getMessage());
739             }
740             results[I] = e;
741             successes[I] = false;
742           }
743 
744 
745 
746         } else {
747 
748           final GrouperClientState GROUPER_CLIENT_STATE = GrouperClientState.retrieveGrouperClientState(false);
749           
750           Callable<T> callable = new Callable<T>() {
751 
752             @Override
753             public T call() throws Exception {
754               
755               GrouperClientState.assignGrouperClientState(GROUPER_CLIENT_STATE);
756               
757               try {
758 
759                 long startNanos = -1;
760                 if (debugMap != null) {
761                   debugMap.put("Try " + I + " thread conn", connectionName + ", id: " + id);
762                   startNanos = System.nanoTime();
763                 }
764                 results[I] = failoverLogic.logic(new FailoverLogicBean(true, connectionName, I == orderedConnections.size()-1));
765                 if (debugMap != null) {
766                   debugMap.put("Finish " + I + " thread conn in " + ((System.nanoTime() - startNanos) / 1000000) + "ms", connectionName + ", id: " + id);
767                 }
768                 successes[I] = true;
769               } catch (Throwable e) {
770                 if (debugMap != null) {
771                   debugMap.put("Error " + I + " thread conn", connectionName + ", id: " + id + ", " + e.getMessage());
772                 }
773                 results[I] = e;
774                 successes[I] = false;
775               } finally {
776                 GrouperClientState.removeGrouperClientState();
777               }
778               return null;
779             }
780           };
781 
782           Future<T> future = GrouperClientCommonUtils.retrieveExecutorService().submit(callable);
783           try {
784             if (debugMap != null) {
785               debugMap.put("Wait for " + timeoutSeconds + " secs: " + I, connectionName + ", id: " + id);
786             }
787             future.get(timeoutSeconds, TimeUnit.SECONDS);
788           } catch (Exception e) {
789             LOG.debug("error:", e);
790           }
791 
792         }
793 
794         if (successes[i] != null && successes[i]) {
795           if (debugMap != null) {
796             debugMap.put("Success: " + I, connectionName + ", id: " + id + ", add to affinity cache");
797           }
798           this.connectionAffinityCache.put(Boolean.TRUE, connectionName);
799           return (T)results[i];
800         }
801 
802         //either way this is a problem
803         incrementErrorForConnection(connectionName);
804 
805         boolean isFatal = i == orderedConnections.size() - 1;
806         boolean isException = results[i] instanceof Throwable;
807         String typeString = isException ? "error" : "timeout";
808         RuntimeException re = null;
809         if (isException) {
810           errorCountForTesting++;
811         } else {
812           timeoutCountForTesting++;
813         }
814         if (isException) {
815           if (results[i] instanceof RuntimeException) {
816             re = (RuntimeException)results[i];
817           } else {
818             re = new RuntimeException((Throwable)results[i]);
819           }
820         }
821 
822         if (isFatal) {
823           //see if we can wait a little more...  see if any timeouts have come back yet
824           int waitMore = this.failoverConfig.getExtraTimeoutSeconds();
825 
826           if (waitMore >= 0) {
827             WAIT_MORE: for (int seconds=0;seconds<waitMore;seconds++) {
828               {
829                 boolean hasTimeouts = false;
830                 for (int index = 0; index <= i; index++) {
831                   if (successes[index] != null && successes[index]) {
832                     isFatal = false;
833                     //go out, it will find it below
834                     break WAIT_MORE;
835                   }
836                   if (successes[index] == null) {
837                     hasTimeouts = true;
838                   }
839                 }
840                 //if there are no more timeouts, then all errors, we are done
841                 if (!hasTimeouts) {
842                   break WAIT_MORE;
843                 }
844               }
845               //wait for a second at a time so we can end as soon as possible
846               GrouperClientCommonUtils.sleep(1000);
847             }
848           }
849 
850         }
851 
852         String severityString = isFatal ? "FATAL" : "NON-FATAL";
853 
854         //we need to log where this came from:
855         RuntimeException thisStack = new RuntimeException("this stack");
856 
857         //log everything
858         String objectToLog = severityString + " " + typeString + " in failover connection: " + connectionName
859         + "," + (isFatal ? "" : " will try others in pool") + " (" + errorCountForTesting
860         + " total errors, "
861         + timeoutCountForTesting + " total timeouts, "
862         + fatalProblemCountForTesting + " total fatal errors): [THIS STACK]: " + GrouperClientUtils.getFullStackTrace(thisStack);
863         if (isException) {
864           if (isFatal) {
865             LOG.error(objectToLog, re);
866           } else {
867             LOG.warn(objectToLog, re);
868           }
869         } else {
870           if (isFatal) {
871             LOG.error(objectToLog);
872           } else {
873             LOG.warn(objectToLog);
874           }
875         }
876         //if fatal, throw something...
877         if (isFatal) {
878           fatalProblemCountForTesting++;
879           if (isException) {
880             GrouperClientUtils.injectInException(re, objectToLog);
881             throw re;
882           }
883           throw new RuntimeException(objectToLog);
884         }
885 
886         //see if any work
887         for (int index = 0; index < successes.length; index++) {
888           if (successes[index] != null && successes[index]) {
889             if (debugMap != null) {
890               debugMap.put("Final success for " + index, "add to affinity cache: " + orderedConnections.get(index));
891             }
892             this.connectionAffinityCache.put(Boolean.TRUE, orderedConnections.get(index));
893             return (T)results[index];
894           }
895         }
896 
897       }
898 
899       throw new RuntimeException("Shouldnt get here");
900     } finally {
901       if (debugMap != null) {
902         LOG.debug(GrouperClientUtils.mapToString(debugMap));
903       }
904     }
905   }
