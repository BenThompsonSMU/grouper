147     if (GrouperClientUtils.length(this.subjectLookups) == 0) {
148       throw new RuntimeException("Need at least one subject to add to group: " + this);
149     }
150     
151     if (pointInTimeFrom != null || pointInTimeTo != null) {
152       if (this.includeGroupDetail != null && this.includeGroupDetail) {
153         throw new RuntimeException("Cannot specify includeGroupDetail for point in time queries.");
154       }
155       
156       if (this.memberFilter != null && !this.memberFilter.equals(WsMemberFilter.All)) {
157         throw new RuntimeException("Cannot specify a member filter for point in time queries.");
158       }
159       
160       if (this.enabled == null || !this.enabled) {
161         throw new RuntimeException("Cannot search for disabled memberships for point in time queries.");
162       }
163       
164       if (sortString != null && !sortString.equals("name")) {
165         throw new RuntimeException("Can only sort by name for point in time queries.");
166       }
167     }
