36  /**
37   * class to run a stem save web service call
38   */
39  public class GcStemSave {
40  
41    /** client version */
42    private String clientVersion;
43  
44    /**
45     * assign client version
46     * @param theClientVersion
47     * @return this for chaining
48     */
49    public GcStemSave assignClientVersion(String theClientVersion) {
50      this.clientVersion = theClientVersion;
51      return this;
52    }
53    
54    /** stems to save */
55    private List<WsStemToSave> stemsToSave = new ArrayList<WsStemToSave>();
56  
57    /**
58     * add a stem to save
59     * @param wsStemToSave
60     * @return this for chaining
61     */
62    public GcStemSave addStemToSave(WsStemToSave wsStemToSave) {
63      this.stemsToSave.add(wsStemToSave);
64      return this;
65    }
66    
67    /** params */
68    private List<WsParam> params = new ArrayList<WsParam>();
69  
70    /**
71     * add a param to the list
72     * @param paramName
73     * @param paramValue
74     * @return this for chaining
75     */
76    public GcStemSave addParam(String paramName, String paramValue) {
77      this.params.add(new WsParam(paramName, paramValue));
78      return this;
79    }
80    
81    /**
82     * add a param to the list
83     * @param wsParam
84     * @return this for chaining
85     */
86    public GcStemSave addParam(WsParam wsParam) {
87      this.params.add(wsParam);
88      return this;
89    }
90    
91    /** act as subject if any */
92    private WsSubjectLookup actAsSubject;
93  
94    /**
95     * assign the act as subject if any
96     * @param theActAsSubject
97     * @return this for chaining
98     */
99    public GcStemSave assignActAsSubject(WsSubjectLookup theActAsSubject) {
100     this.actAsSubject = theActAsSubject;
101     return this;
102   }
103   
104   /**
105    * validate this call
106    */
107   private void validate() {
108     if (GrouperClientUtils.length(this.stemsToSave) == 0) {
109       throw new RuntimeException("Stem name is required: " + this);
110     }
111   }
112   
113   /**
114    * tx type for request 
115    */
116   private GcTransactionType txType;
117   
118   /**
119    * execute the call and return the results.  If there is a problem calling the service, an
120    * exception will be thrown
121    * 
122    * @return the results
123    * @throws GcWebServiceError if there is a problem
124    */
125   public WsStemSaveResults execute() {
126     this.validate();
127     WsStemSaveResults wsStemSaveResults = null;
128     try {
129       //Make the body of the request, in this case with beans and marshaling, but you can make
130       //your request document in whatever language or way you want
131       WsRestStemSaveRequestbeans/WsRestStemSaveRequest.html#WsRestStemSaveRequest">WsRestStemSaveRequest stemSave = new WsRestStemSaveRequest();
132 
133       stemSave.setActAsSubjectLookup(this.actAsSubject);
134 
135       stemSave.setTxType(this.txType == null ? null : this.txType.name());
136 
137       stemSave.setWsStemToSaves(GrouperClientUtils.toArray(this.stemsToSave, WsStemToSave.class));
138       
139       //add params if there are any
140       if (this.params.size() > 0) {
141         stemSave.setParams(GrouperClientUtils.toArray(this.params, WsParam.class));
142       }
143       
144       GrouperClientWsrouperClientWs.html#GrouperClientWs">GrouperClientWs grouperClientWs = new GrouperClientWs();
145       
146       //kick off the web service
147       wsStemSaveResults = (WsStemSaveResults)
148         grouperClientWs.executeService("stems", stemSave, "stemSave", this.clientVersion, false);
149       
150       String stemSaveResultMessage = "";
151       
152       //try to get the inner message
153       try {
154         stemSaveResultMessage = wsStemSaveResults.getResults()[0].getResultMetadata().getResultMessage();
155 
156       } catch (Exception e) {}
157       
158       String resultMessage = wsStemSaveResults.getResultMetadata().getResultMessage() + "\n"
159         + stemSaveResultMessage;
160       
161       grouperClientWs.handleFailure(wsStemSaveResults, wsStemSaveResults.getResults(), resultMessage);
162       
163     } catch (Exception e) {
164       GrouperClientUtils.convertToRuntimeException(e);
165     }
166     return wsStemSaveResults;
167     
168   }
169 
170   /**
171    * assign the tx type
172    * @param gcTransactionType
173    * @return self for chaining
174    */
175   public GcStemSave assignTxType(GcTransactionType gcTransactionType) {
176     this.txType = gcTransactionType;
177     return this;
178   }
