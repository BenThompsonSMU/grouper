1   package edu.internet2.middleware.grouperClient.jdbc;
2   
3   import java.sql.Connection;
4   import java.sql.SQLException;
5   
6   /**
7   * bean that wraps connections
8   */
9   public interface GcJdbcConnectionBean {
10  
11   /**
12    * get a connection (dont close this when done, just call "doneWithConnection()"
13    * @return the connection
14    * @throws SQLException if there is a problem
15    */
16   public Connection connection() throws SQLException;
17   
18   /**
19    * call this when the connection is done.  This will do any cleanup
20    * this is a null-safe method.  
21    * @throws SQLException if there is a problem
22    */
23   public void doneWithConnection() throws SQLException;
24  
25   /**
26    * call this when the connection is not needed, in the finally block
27    * this might return to pool.  In general, this shouldnt throw exceptions
28    * since it is done in a finally block, it should only log them
29    */
30   public void doneWithConnectionFinally();
31  
32   /**
33    * call this when the connection is done but there was an error, will pass
34    * an exception.  This should do whatever and rethrow the exception as runtime
35    * @param t 
36    */
37   public void doneWithConnectionError(Throwable t);
38  }
