1   package edu.internet2.middleware.grouperClient.jdbc;

3   import java.sql.SQLException;
4   import java.util.Properties;
5   
6   /**
7    * gives connections and allows returns.  e.b. dbcp pooling, c3p0 pooling,
8    * grouper built in db
9    */
10  public interface GcJdbcConnectionProvider {
11  
12    /**
13     * if the provider requires jdbc data in the subject.properties
14     * @return true if provider requires data in subject.properties
15     */
16    public boolean requiresJdbcConfigInSourcesXml();
17    
18    /** 
19     * init the pool 
20     * @param properties 
21     * @param sourceId mainly for logging
22     * @param driver driver
23     * @param maxActive max active connection if pool
24     * @param defaultMaxActive if max active is needed, and blank, then use this
25     * @param maxIdle max idle connections if pool
26     * @param defaultMaxIdle if max is needed, and blank, then use this
27     * @param maxWaitSeconds max wait if pool empty in seconds.  -1 means long time
28     * @param defaultMaxWaitSeconds if max wait is needed, and blank, then use this, -1 mean long time
29     * @param dbUrl jdbc url of the database
30     * @param dbUser user to login to the database
31     * @param dbPassword password for the database (unencrypted)
32     * @param readOnly if conn should be readonly
33     * @param readOnlyDefault default if not specified and needed
34     * @throws SourceUnavailableException if something is not right or not available
35     */
36    public void init(Properties properties, String sourceId, String driver, Integer maxActive, int defaultMaxActive, Integer maxIdle, int defaultMaxIdle,
37        Integer maxWaitSeconds, int defaultMaxWaitSeconds, String dbUrl, String dbUser, 
38        String dbPassword, Boolean readOnly, boolean readOnlyDefault);
39    
40    /**
41     * get a connection (dont close this when done, just call "doneWithConnection()"
42     * @return the connection
43     * @throws SQLException if there is a problem
44     */
45    public GcJdbcConnectionBean connectionBean() throws SQLException;
46    
47  }
