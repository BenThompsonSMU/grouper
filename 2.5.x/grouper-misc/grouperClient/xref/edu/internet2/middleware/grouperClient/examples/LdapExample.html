60      // Set up the environment for creating the initial context
61      Hashtable env = new Hashtable();
62      env.put(Context.INITIAL_CONTEXT_FACTORY, 
63          "com.sun.jndi.ldap.LdapCtxFactory");
64      env.put(Context.PROVIDER_URL, url);
65  
66      env.put(Context.SECURITY_AUTHENTICATION, "simple");
67      env.put(Context.SECURITY_PRINCIPAL, "uid=" + user + ",ou=entities,dc=upenn,dc=edu");
68      env.put(Context.SECURITY_CREDENTIALS, pass);
69  
70      // Create the initial context
71      DirContext context = new InitialDirContext(env);
72      return context;
73    }

163   public static String pennnameToPennid(DirContext context, String pennname) throws Exception {
164     SearchControls searchControls = new SearchControls();
165     searchControls.setReturningAttributes(new String[]{"pennid"});
166     searchControls.setReturningObjFlag(false);
167     searchControls.setSearchScope(SearchControls.ONELEVEL_SCOPE);
168     NamingEnumeration namingEnumeration = context.search("ou=pennnames", "name=" + pennname, searchControls);
169     //printNamingEnumeration(namingEnumeration);
170     return retrieveAttributeStringValue(namingEnumeration, "pennid");
171   }

190   /**
191    * see if a user is in a group
192    * @param context
193    * @param groupName
194    * @param pennname
195    * @return true or false
196    * @throws NamingException 
197    */
198   public static boolean isInGroup(DirContext context, String groupName, String pennname) throws NamingException {
199     Attributes searchAttributes = new BasicAttributes();
200     searchAttributes.put(new BasicAttribute("cn", groupName));
201     searchAttributes.put(new BasicAttribute("hasMember", pennname));
202     
203     NamingEnumeration namingEnumeration = context.search(
204         "ou=groups",
205         searchAttributes, new String[]{"cn"});
206     String cn = retrieveAttributeStringValue(namingEnumeration, "cn");
207     //printNamingEnumeration(namingEnumeration);
208     boolean isInGroup = GrouperClientUtils.equals(groupName, cn);
209     return isInGroup;
210   }

329   /**
330    * print out a naming enumeration
331    * @param namingEnumeration
332    * @throws NamingException 
333    */
334   public static void printNamingEnumeration(NamingEnumeration namingEnumeration) throws NamingException {
335     while (namingEnumeration.hasMore()) {
336       Object nextElement = namingEnumeration.next();
337       if (nextElement instanceof Attribute) {
338         Attribute attribute = (Attribute)nextElement;
339         printAttribute(attribute);
340       } else if (nextElement instanceof SearchResult) {
341         SearchResult searchResult = (SearchResult)nextElement;
342         System.out.println("Search result: " + searchResult.getNameInNamespace());
343         Attributes attributes = searchResult.getAttributes();
344         printAttributes(attributes);
345       } else if (nextElement instanceof NameClassPair) {
346         NameClassPair nameClassPair = (NameClassPair)namingEnumeration.nextElement();
347         System.out.println("Name class pair: " + nameClassPair.getClassName() + ", " + nameClassPair.getNameInNamespace());
348       } else {
349         throw new RuntimeException("Not expecting type: " + nextElement);
350       }
351       
352     }
353     
354   }

356   /**
357    * retrieve a single valued attribute as string
358    * @param object
359    * @param attributeName 
360    * @throws NamingException 
361    * @return the attribute value or null if not there
362    */
363   public static String retrieveAttributeStringValue(Object object, 
364       String attributeName) throws NamingException {
365     if (object == null) {
366       return null;
367     }
368     if (object instanceof Attribute) {
369       Attribute attribute = (Attribute)object;
370       if (GrouperClientUtils.equals(attribute.getID(), attributeName)) {
371         return (String)attribute.get();
372       }
373       return null;
374     } else if (object instanceof SearchResult) {
375       SearchResult searchResult = (SearchResult)object;
376       Attributes attributes = searchResult.getAttributes();
377       Attribute attribute = attributes.get(attributeName);
378       return retrieveAttributeStringValue(attribute, attributeName);
379     } else if (object instanceof NamingEnumeration) {
380       NamingEnumeration namingEnumeration = (NamingEnumeration)object;
381       if (!namingEnumeration.hasMore()) {
382         return null;
383       }
384       Object next = namingEnumeration.next();
385       if (namingEnumeration.hasMore()) {
386         throw new RuntimeException("Expecting one result");
387       }
388       return retrieveAttributeStringValue(next, attributeName);
389     } else {
390       throw new RuntimeException("Not expecting type: " + object);
391     }
392   }

394   /**
395    * retrieve a string array of values
396    * @param object
397    * @param attributeName 
398    * @throws NamingException 
399    * @return the attribute value or null if not there
400    */
401   public static List<String> retrieveAttributeStringListValue(Object object, 
402       String attributeName) throws NamingException {
403     if (object == null) {
404       return null;
405     }
406     if (object instanceof Attribute) {
407       Attribute attribute = (Attribute)object;
408       NamingEnumeration namingEnumeration = attribute.getAll();
409       return retrieveAttributeStringListValue(namingEnumeration, attributeName);
410     } else if (object instanceof SearchResult) {
411       SearchResult searchResult = (SearchResult)object;
412       Attributes attributes = searchResult.getAttributes();
413       Attribute attribute = attributes.get(attributeName);
414       return retrieveAttributeStringListValue(attribute, attributeName);
415     } else if (object instanceof NamingEnumeration) {
416       int size = 0;
417       NamingEnumeration namingEnumeration = (NamingEnumeration)object;
418       List<String> resultList = new ArrayList<String>();
419       while (namingEnumeration.hasMore()) {
420         Object next = namingEnumeration.next();
421         if (next instanceof SearchResult) {
422           if (size == 0 && !namingEnumeration.hasMore()) {
423             return retrieveAttributeStringListValue(next, attributeName);
424           }
425           throw new RuntimeException("Error: multiple search results found!");
426         }
427         resultList.add((String)next);
428         size++;
429       }
430       if (size == 0) {
431         return null;
432       }
433       return resultList;
434     } else {
435       throw new RuntimeException("Not expecting type: " + object);
436     }
437   }
