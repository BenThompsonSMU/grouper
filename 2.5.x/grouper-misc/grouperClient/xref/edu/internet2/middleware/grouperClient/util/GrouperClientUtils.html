89    /**
90     * @param theClass
91     * @return the log
92     */
93    public static Log retrieveLog(Class<?> theClass) {
94  
95      Log theLog = LogFactory.getLog(theClass);
96      
97      //if this isnt here, dont configure yet
98      if (isBlank(GrouperClientConfig.retrieveConfig().propertyValueString("encrypt.disableExternalFileLookup"))
99          || theClass.equals(GrouperClientCommonUtils.class)) {
100       return new GrouperClientLog(theLog);
101     }
102     
103     if (!configuredLogs) {
104       String logLevel = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.logging.logLevel");
105       String logFile = GrouperClientConfig.retrieveConfig().propertyValueString("grouperClient.logging.logFile");
106       String grouperClientLogLevel = GrouperClientConfig.retrieveConfig().propertyValueString(
107           "grouperClient.logging.grouperClientOnly.logLevel");
108       
109       boolean hasLogLevel = !isBlank(logLevel);
110       boolean hasLogFile = !isBlank(logFile);
111       boolean hasGrouperClientLogLevel = !isBlank(grouperClientLogLevel);
112       
113       if (hasLogLevel || hasLogFile) {
114         if (theLog instanceof Jdk14Logger) {
115           Jdk14Logger jdkLogger = (Jdk14Logger) theLog;
116           Logger logger = jdkLogger.getLogger();
117           long timeToLive = 60;
118           while (logger.getParent() != null && timeToLive-- > 0) {
119             //this should be root logger
120             logger = logger.getParent();
121           }
122   
123           if (length(logger.getHandlers()) == 1) {
124   
125             //remove console appender if only one
126             if (logger.getHandlers()[0].getClass() == ConsoleHandler.class) {
127               logger.removeHandler(logger.getHandlers()[0]);
128             }
129           }
130   
131           if (length(logger.getHandlers()) == 0) {
132             Handler handler = null;
133             if (hasLogFile) {
134               try {
135                 handler = new FileHandler(logFile, true);
136               } catch (IOException ioe) {
137                 throw new RuntimeException(ioe);
138               }
139             } else {
140               handler = new ConsoleHandler();
141             }
142             handler.setFormatter(new SimpleFormatter());
143             handler.setLevel(Level.ALL);
144             logger.addHandler(handler);
145 
146             logger.setUseParentHandlers(false);
147           }
148           
149           if (hasLogLevel) {
150             Level level = Level.parse(logLevel);
151             
152             logger.setLevel(level);
153 
154           }
155         }
156       }
157       
158       if (hasGrouperClientLogLevel) {
159         Level level = Level.parse(grouperClientLogLevel);
160         Log grouperClientLog = LogFactory.getLog("edu.internet2.middleware.grouperClient");
161         if (grouperClientLog instanceof Jdk14Logger) {
162           Jdk14Logger jdkLogger = (Jdk14Logger) grouperClientLog;
163           Logger logger = jdkLogger.getLogger();
164           logger.setLevel(level);
165         }
166       }
167       
168       configuredLogs = true;
169     }
170     
171     return new GrouperClientLog(theLog);
172     
173   }

262     /**
263      * retrieve class if class
264      * @param name
265      * @return class
266      */
267     private static Object retrieveClass(String name) {
268       if (isBlank(name)) {
269         return null;
270       }
271       
272       //see if fully qualified class
273       
274       Boolean knowsIfClass = jexlKnowsIfClass.get(name);
275       
276       //see if knows answer
277       if (knowsIfClass != null) {
278         //return class or null
279         return jexlClass.get(name);
280       }
281       
282       //see if valid class
283       if (jexlClassPattern.matcher(name).matches()) {
284         
285         jexlKnowsIfClass.put(name, true);
286         //try to load
287         try {
288           Class<?> theClass = Class.forName(name);
289           jexlClass.put(name, theClass);
290           return theClass;
291         } catch (Exception e) {
292           LOG.info("Cant load what looks like class: " + name, e);
293           //this is ok I guess, dont rethrow, not sure it is a class
294         }
295       }
296       return null;
297       
298     }

305       
306       //see if registered      
307       Object object = super.get(name);

344   /**
345    * substitute an EL for objects
346    * @param stringToParse
347    * @param variableMap
348    * @param allowStaticClasses if true allow static classes not registered with context
349    * @param silent if silent mode, swallow exceptions (warn), and dont warn when variable not found
350    * @param lenient false if undefined variables should throw an exception.  if lenient is true (default)
351    * then undefined variables are null
352    * @param logOnNull if null output of substitution should be logged
353    * @return the string
354    */
355   public static String substituteExpressionLanguage(String stringToParse, 
356       Map<String, Object> variableMap, boolean allowStaticClasses, boolean silent, boolean lenient, boolean logOnNull) {
357     if (isBlank(stringToParse)) {
358       return stringToParse;
359     }
360     String overallResult = null;
361     Exception exception = null;
362     try {
363       JexlContext jc = allowStaticClasses ? new ElMapContext() : new MapContext();
364 
365       int index = 0;
366       
367       variableMap = nonNull(variableMap);
368       
369       for (String key: variableMap.keySet()) {
370         jc.set(key, variableMap.get(key));
371       }
372       
373       //allow utility methods
374       jc.set("elUtils", new GcElUtilsSafe());
375       //if you add another one here, add it in the logs below
376       
377       // matching ${ exp }   (non-greedy)
378       Pattern pattern = Pattern.compile("\\$\\{(.*?)\\}");
379       Matcher matcher = pattern.matcher(stringToParse);
380       
381       StringBuilder result = new StringBuilder();
382   
383       //loop through and find each script
384       while(matcher.find()) {
385         result.append(stringToParse.substring(index, matcher.start()));
386         
387         //here is the script inside the curlies
388         String script = matcher.group(1);
389         
390         index = matcher.end();
391   
392         if (script.contains("{")) {
393           //we need to match up some curlies here...
394           int scriptStart = matcher.start(1);
395           int openCurlyCount = 0;
396           for (int i=scriptStart; i<stringToParse.length();i++) {
397             char curChar = stringToParse.charAt(i);
398             if (curChar == '{') {
399               openCurlyCount++;
400             }
401             if (curChar == '}') {
402               openCurlyCount--;
403               //negative 1 since we need to get to the close of the parent one...
404               if (openCurlyCount <= -1) {
405                 script = stringToParse.substring(scriptStart, i);
406                 index = i+1;
407                 break;
408               }
409             }
410           }
411         }
412         
413         JexlEngine jexlEngine = new JexlEngine();
414         jexlEngine.setSilent(silent);
415         jexlEngine.setLenient(lenient);
416   
417         Expression e = jexlEngine.createExpression(script);
418   
419         //this is the result of the evaluation
420         Object o = null;
421         
422         try {
423           o = e.evaluate(jc);
424         } catch (JexlException je) {
425           //exception-scrape to see if missing variable
426           if (!lenient && trimToEmpty(je.getMessage()).contains("undefined variable")) {
427             //clean up the message a little bit
428             // e.g. edu.internet2.middleware.grouper.util.GrouperUtil.substituteExpressionLanguage@8846![0,6]: 'amount < 50000 && amount2 < 23;' undefined variable amount
429             String message = je.getMessage();
430             //Pattern exceptionPattern = Pattern.compile("^" + GrouperUtil.class.getName() + "\\.substituteExpressionLanguage.*?]: '(.*)");
431             Pattern exceptionPattern = Pattern.compile("^.*undefined variable (.*)");
432             Matcher exceptionMatcher = exceptionPattern.matcher(message);
433             if (exceptionMatcher.matches()) {
434               //message = "'" + exceptionMatcher.group(1);
435               message = "variable '" + exceptionMatcher.group(1) + "' is not defined in script: '" + script + "'";
436             }
437             throw new GcExpressionLanguageMissingVariableException(message, je);
438           }
439           throw je;
440         }
441           
442         if (o == null) {
443           if (logOnNull) {
444             LOG.warn("expression returned null: " + script + ", in pattern: '" + stringToParse + "', available variables are: "
445                 + toStringForLog(variableMap.keySet()));
446           } else {
447             if (LOG.isDebugEnabled()) {
448               LOG.debug("expression returned null: " + script + ", in pattern: '" + stringToParse + "', available variables are: "
449                   + toStringForLog(variableMap.keySet()));
450             }            
451           }
452           o = "";
453         }
454         
455         if (o instanceof RuntimeException) {
456           throw (RuntimeException)o;
457         }
458         
459         result.append(o);
460         
461       }
462       
463       result.append(stringToParse.substring(index, stringToParse.length()));
464       overallResult = result.toString();
465       return overallResult;
466       
467     } catch (Exception e) {
468       exception = e;
469       if (e instanceof GcExpressionLanguageMissingVariableException) {
470         throw (GcExpressionLanguageMissingVariableException)e;
471       }
472       throw new RuntimeException("Error substituting string: '" + stringToParse + "'", e);
473     } finally {
474       if (LOG.isDebugEnabled()) {
475         Set<String> keysSet = new LinkedHashSet<String>(nonNull(variableMap).keySet());
476         keysSet.add("elUtils");
477         StringBuilder logMessage = new StringBuilder();
478         logMessage.append("Subsituting EL: '").append(stringToParse).append("', and with env vars: ");
479         String[] keys = keysSet.toArray(new String[]{});
480         for (int i=0;i<keys.length;i++) {
481           logMessage.append(keys[i]);
482           if (i != keys.length-1) {
483             logMessage.append(", ");
484           }
485         }
486         logMessage.append(" with result: '" + overallResult + "'");
487         if (exception != null) {
488           logMessage.append(", and exception: " + exception + ", " + getFullStackTrace(exception));
489         }
490         LOG.debug(logMessage.toString());
491       }
492     }
493   }

497    * @param wsSubject subject
498    * @param attributeNames list of attribute names in the subject
499    * @param attributeName to query
500    * @return the value or null

502   public static String subjectAttributeValue(WsSubject wsSubject, String[] attributeNames, String attributeName) {
503     
504     if (GrouperClientUtils.equals("subject__id", attributeName)) {
505       return wsSubject.getId();
506     }
507     
508     if (GrouperClientUtils.equals("subject__name", attributeName)) {
509       return wsSubject.getName();
510     }
511     
512     for (int i=0;i<GrouperClientUtils.length(attributeNames);i++) {
513       
514       if (GrouperClientUtils.equalsIgnoreCase(attributeName, attributeNames[i])
515           && GrouperClientUtils.length(wsSubject.getAttributeValues()) > i) {
516         //got it
517         return wsSubject.getAttributeValue(i);
518       }
519     }
520     return null;
521   }

547    * @param pass
548    * @param logInfo if not null, put log info in there
549    * @return the pass

551   public static String decryptFromFileIfFileExists(String pass, StringBuilder logInfo) {
552 
553     boolean disableExternalFileLookup = GrouperClientUtils.propertiesValueBoolean(
554         "encrypt.disableExternalFileLookup", false, false);
555     
556     //lets lookup if file
557     String passFromFile = GrouperClientUtils.readFromFileIfFileExists(pass, disableExternalFileLookup);
558

560

562         logInfo.append("reading encrypted value from file: " + pass);
563       }
564       
565       String encryptKey = encryptKey();
566       pass = new Crypto(encryptKey).decrypt(passFromFile);
567       
568     } else {
569       if (logInfo != null) {
570         logInfo.append("pass is a scalar");
571       }
572     }
573     
574     return pass;
575   }

578   /**
579    * name of the cache directory without trailing slash
580    * @return the name of the cache directory
581    */
582   public static String cacheDirectoryName() {
583     
584     if (cacheDirectoryName == null) {
585       String directoryName = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.cacheDirectory");
586       if (GrouperClientCommonUtils.isBlank(directoryName)) {
587         throw new RuntimeException("grouperClient.cacheDirectory is required in grouper.client.properties");
588       }
589   
590       directoryName = GrouperClientCommonUtils.stripEnd(directoryName, "/");
591       directoryName = GrouperClientCommonUtils.stripEnd(directoryName, "\\");
592       
593       File discoveryDir = new File(directoryName);
594       directoryName = discoveryDir.getAbsolutePath();
595       if (directoryName.endsWith("/.")) {
596         directoryName = directoryName.substring(0, directoryName.length()-2);
597       }
598       if (directoryName.endsWith("\\.")) {
599         directoryName = directoryName.substring(0, directoryName.length()-2);
600       }
601       cacheDirectoryName = directoryName;
602     }    
603     return cacheDirectoryName;
604   }

626   /**
627    * call HTTP with a url, optional request body, get a response body.  this assumes json by default
628    * @param urlSuffix is after the configured URL
629    * @param serviceAuthn is the config string that identifies the user/pass/url
630    * @param httpCallMethod HTTP
631    * @param body to send if applicable
632    * @return the response
633    */
634   public static HttpCallResponse httpCall(String urlSuffix, String serviceAuthn, HttpCallMethod httpCallMethod, String body) {
635     
636     //make sure right content type is in request (e.g. application/xhtml+xml
637     HttpClient httpClient = new HttpClient();
638 
639     DefaultHttpParams.getDefaultParams().setParameter(
640         HttpMethodParams.RETRY_HANDLER, new DefaultHttpMethodRetryHandler(0, false));
641 
642     httpClient.getParams().setAuthenticationPreemptive(true);
643     
644     int soTimeoutMillis = GrouperClientConfig.retrieveConfig().propertyValueIntRequired(
645         "grouperClient.webService.httpSocketTimeoutMillis");
646 
647     httpClient.getParams().setSoTimeout(soTimeoutMillis);
648     httpClient.getParams().setParameter(HttpMethodParams.HEAD_BODY_CHECK_TIMEOUT, soTimeoutMillis);
649 
650     int connectionManagerMillis = GrouperClientConfig.retrieveConfig().propertyValueIntRequired(
651         "grouperClient.webService.httpConnectionManagerTimeoutMillis");
652 
653     httpClient.getParams().setConnectionManagerTimeout(connectionManagerMillis);
654 
655     String user = GrouperClientConfig.retrieveConfig().propertyValueStringRequired("grouperClient.webService." + serviceAuthn + ".user");
656 
657     boolean disableExternalFileLookup = GrouperClientConfig.retrieveConfig().propertyValueBooleanRequired(
658         "encrypt.disableExternalFileLookup");
659
