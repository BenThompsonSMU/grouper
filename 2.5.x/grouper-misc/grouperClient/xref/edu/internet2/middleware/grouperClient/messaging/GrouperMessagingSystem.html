9   /**
10   * Represents the methods that a messaging system
11   * needs to support
12   */
13  public interface GrouperMessagingSystem {
14  
15    /**
16     * send a message to a queue name.  Note, the recipient could be a 
17     * queue or a topic (generally always one or the other) based on the 
18     * implementation of the messaging system.  Messages must be delievered
19     * in the order that collection iterator designates.  If there is a problem
20     * delivering the messages, the implementation should log, wait (back off)
21     * and retry until it is successful.
22     * @param grouperMessageSendParam has the queue or topic, and the message(s) and perhaps args
23     * @return result
24     */
25    public GrouperMessageSendResult send(GrouperMessageSendParam grouperMessageSendParam);
26  
27    /**
28     * tell the messaging system that these messages are processed
29     * generally the message system will use the message id.  Note, the objects
30     * sent to this method must be the same that were received in the
31     * receiveMessages method.  If there is a problem
32     * delivering the messages, the implementation should wait (back off)
33     * and retry until it is successful.  Alternatively the message should be 
34     * returned to queue, returned to end of queue, or sent to another queue
35     * @param grouperMessageAcknowledgeParam
36     * @return result
37     */
38    public GrouperMessageAcknowledgeResult acknowledge(GrouperMessageAcknowledgeParam grouperMessageAcknowledgeParam);
39  
40    /**
41     * this will generally block until there are messages to process.  These messages
42     * are ordered in the order that they were sent.
43     * @param grouperMessageReceiveParam grouper messaging receive param
44     * @return a message or multiple messages.  It will block until there are messages
45     * available for this recipient to process
46     */
47    public GrouperMessageReceiveResult receive(GrouperMessageReceiveParam grouperMessageReceiveParam);
48  
49  }
