529   /**
530    * see if needs reconfigure
531    * @return true or false
532    */
533   private static boolean needsReconfigure() {
534     boolean needsReconfigure = lastFailoverConfigure == null || (System.currentTimeMillis() - lastFailoverConfigure) / 1000 > configureEverySeconds;
535     if (!DiscoveryClient.hasDiscovery() && lastFailoverConfigure != null) {
536       needsReconfigure = false;
537     }
538     return needsReconfigure;
539   }

549   public Object executeService(final String urlSuffix, final Object toSend, 
550       final String labelForLog, final String clientVersion, final boolean readOnly)  {
551     
552     //configure the failover client (every 30 seconds)

554

557     //copy the grouper client ws instance to this
558

560     
561     GrouperClientWs grouperClientWs = FailoverClient.failoverLogic(connectionType, new FailoverLogic<GrouperClientWs>() {
562 
563       @Override
564       public GrouperClientWs logic(FailoverLogicBean failoverLogicBean) {
565         
566         //if not last connection then throw exception if not success.  If last connection then return the object

701     }

705     }
706     grouperClientWs.result = resultObject;
707     return grouperClientWs;

710   /**
711    * if failure, handle it
712    * @param responseContainer is the object that everything marshaled to
713    * @param resultMetadataHolders
714    * @param resultMessage
715    * @throws GcWebServiceError if there is a problem
716    */
717   public void handleFailure(Object responseContainer, ResultMetadataHolder[] resultMetadataHolders, String resultMessage) {
718     // see if request worked or not
719     if (!this.success) {
720       StringBuilder error = new StringBuilder("Bad response from web service: resultCode: " + this.resultCode
721         + ", " + resultMessage);
722       int errorIndex = 0;
723       for (int i=0;i<GrouperClientUtils.length(resultMetadataHolders);i++) {
724         try {
725           WsResultMeta resultMetadata = resultMetadataHolders[i].getResultMetadata();
726           if (!GrouperClientUtils.equals(resultMetadata.getSuccess(), "T")) {
727             error.append("\nError ").append(errorIndex).append(", result index: ").append(i).append(", code: ").append(resultMetadata.getResultCode())
728               .append(", message: ").append(resultMetadata.getResultMessage());
729             errorIndex++;
730           }
731         } catch (Exception e) {
732           //object not there
733           LOG.debug("issue with error message: ", e);
734         }
735       }
736       throw new GcWebServiceError(responseContainer, error.toString());
737     }
