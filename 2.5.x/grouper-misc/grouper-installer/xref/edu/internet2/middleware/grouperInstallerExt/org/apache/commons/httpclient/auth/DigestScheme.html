64  /**
65   * <p>
66   * Digest authentication scheme as defined in RFC 2617.
67   * Both MD5 (default) and MD5-sess are supported.
68   * Currently only qop=auth or no qop is supported. qop=auth-int
69   * is unsupported. If auth and auth-int are provided, auth is
70   * used.
71   * </p>
72   * <p>
73   * Credential charset is configured via the 
74   * {@link edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.params.HttpMethodParams#CREDENTIAL_CHARSET credential
75   * charset} parameter.  Since the digest username is included as clear text in the generated 
76   * Authentication header, the charset of the username must be compatible with the 
77   * {@link edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.params.HttpMethodParams#HTTP_ELEMENT_CHARSET http element 
78   * charset}.
79   * </p>
80   * TODO: make class more stateful regarding repeated authentication requests
81   * 
82   * @author <a href="mailto:remm@apache.org">Remy Maucherat</a>
83   * @author Rodney Waldhoff
84   * @author <a href="mailto:jsdever@apache.org">Jeff Dever</a>
85   * @author Ortwin Gl?ck
86   * @author Sean C. Sullivan
87   * @author <a href="mailto:adrian@ephox.com">Adrian Sutton</a>
88   * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>

167     /**
168      * Processes the Digest challenge.
169      *  
170      * @param challenge the challenge string
171      * 
172      * @throws MalformedChallengeException is thrown if the authentication challenge
173      * is malformed
174      * 
175      * @since 3.0
176      */
177     public void processChallenge(final String challenge) 
178       throws MalformedChallengeException {
179         super.processChallenge(challenge);
180         
181         if (getParameter("realm") == null) {
182             throw new MalformedChallengeException("missing realm in challange");
183         }
184         if (getParameter("nonce") == null) {
185             throw new MalformedChallengeException("missing nonce in challange");   
186         }
187         
188         boolean unsupportedQop = false;
189         // qop parsing
190         String qop = getParameter("qop");
191         if (qop != null) {
192             StringTokenizer tok = new StringTokenizer(qop,",");
193             while (tok.hasMoreTokens()) {
194                 String variant = tok.nextToken().trim();
195                 if (variant.equals("auth")) {
196                     qopVariant = QOP_AUTH;
197                     break; //that's our favourite, because auth-int is unsupported
198                 } else if (variant.equals("auth-int")) {
199                     qopVariant = QOP_AUTH_INT;               
200                 } else {
201                     unsupportedQop = true;
202                     LOG.warn("Unsupported qop detected: "+ variant);   
203                 }     
204             }
205         }        
206         
207         if (unsupportedQop && (qopVariant == QOP_MISSING)) {
208             throw new MalformedChallengeException("None of the qop methods is supported");   
209         }
210         
211         cnonce = createCnonce();   
212         this.complete = true;
213     }

215     /**
216      * Tests if the Digest authentication process has been completed.
217      * 
218      * @return <tt>true</tt> if Digest authorization has been processed,
219      *   <tt>false</tt> otherwise.
220      * 
221      * @since 3.0
222      */
223     public boolean isComplete() {
224         String s = getParameter("stale");
225         if ("true".equalsIgnoreCase(s)) {
226             return false;
227         } else {
228             return this.complete;
229         }
230     }

252     /**
253      * Produces a digest authorization string for the given set of 
254      * {@link Credentials}, method name and URI.
255      * 
256      * @param credentials A set of credentials to be used for athentication
257      * @param method the name of the method that requires authorization. 
258      * @param uri The URI for which authorization is needed. 
259      * 
260      * @throws InvalidCredentialsException if authentication credentials
261      *         are not valid or not applicable for this authentication scheme
262      * @throws AuthenticationException if authorization string cannot 
263      *   be generated due to an authentication failure
264      * 
265      * @return a digest authorization string
266      * 
267      * @see edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.HttpMethod#getName()
268      * @see edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.HttpMethod#getPath()
269      * 
270      * @deprecated Use {@link #authenticate(Credentials, HttpMethod)}
271      */
272     public String authenticate(Credentials credentials, String method, String uri)
273       throws AuthenticationException {
274 
275         LOG.trace("enter DigestScheme.authenticate(Credentials, String, String)");
276 
277         UsernamePasswordCredentials usernamepassword = null;
278         try {
279             usernamepassword = (UsernamePasswordCredentials) credentials;
280         } catch (ClassCastException e) {
281             throw new InvalidCredentialsException(
282              "Credentials cannot be used for digest authentication: " 
283               + credentials.getClass().getName());
284         }
285         getParameters().put("methodname", method);
286         getParameters().put("uri", uri);
287         String digest = createDigest(
288             usernamepassword.getUserName(),
289             usernamepassword.getPassword());
290         return "Digest " + createDigestHeader(usernamepassword.getUserName(), digest);
291     }

293     /**
294      * Produces a digest authorization string for the given set of 
295      * {@link Credentials}, method name and URI.
296      * 
297      * @param credentials A set of credentials to be used for athentication
298      * @param method The method being authenticated
299      * 
300      * @throws InvalidCredentialsException if authentication credentials
301      *         are not valid or not applicable for this authentication scheme
302      * @throws AuthenticationException if authorization string cannot 
303      *   be generated due to an authentication failure
304      * 
305      * @return a digest authorization string
306      * 
307      * @since 3.0
308      */
309     public String authenticate(Credentials credentials, HttpMethod method)
310     throws AuthenticationException {
311 
312         LOG.trace("enter DigestScheme.authenticate(Credentials, HttpMethod)");
313 
314         UsernamePasswordCredentials usernamepassword = null;
315         try {
316             usernamepassword = (UsernamePasswordCredentials) credentials;
317         } catch (ClassCastException e) {
318             throw new InvalidCredentialsException(
319                     "Credentials cannot be used for digest authentication: " 
320                     + credentials.getClass().getName());
321         }
322         getParameters().put("methodname", method.getName());
323         StringBuffer buffer = new StringBuffer(method.getPath());
324         String query = method.getQueryString();
325         if (query != null) {
326             if (query.indexOf("?") != 0) {
327                 buffer.append("?");
328             }
329             buffer.append(method.getQueryString());
330         }
331         getParameters().put("uri", buffer.toString());
332         String charset = getParameter("charset");
333         if (charset == null) {
334             getParameters().put("charset", method.getParams().getCredentialCharset());
335         }
336         String digest = createDigest(
337             usernamepassword.getUserName(),
338             usernamepassword.getPassword());
339         return "Digest " + createDigestHeader(usernamepassword.getUserName(),
340                 digest);
341     }

343     /**
344      * Creates an MD5 response digest.
345      * 
346      * @param uname Username
347      * @param pwd Password
348      * @param charset The credential charset
349      * 
350      * @return The created digest as string. This will be the response tag's
351      *         value in the Authentication HTTP header.
352      * @throws AuthenticationException when MD5 is an unsupported algorithm
353      */
354     private String createDigest(final String uname, final String pwd) throws AuthenticationException {
355 
356         LOG.trace("enter DigestScheme.createDigest(String, String, Map)");
357 
358         final String digAlg = "MD5";
359 
360         // Collecting required tokens
361         String uri = getParameter("uri");
362         String realm = getParameter("realm");
363         String nonce = getParameter("nonce");
364         String qop = getParameter("qop");
365         String method = getParameter("methodname");
366         String algorithm = getParameter("algorithm");
367         // If an algorithm is not specified, default to MD5.
368         if (algorithm == null) {
369             algorithm = "MD5";
370         }
371         // If an charset is not specified, default to ISO-8859-1.
372         String charset = getParameter("charset");
373         if (charset == null) {
374             charset = "ISO-8859-1";
375         }
376 
377         if (qopVariant == QOP_AUTH_INT) {
378             LOG.warn("qop=auth-int is not supported");
379             throw new AuthenticationException(
380                 "Unsupported qop in HTTP Digest authentication");   
381         }
382 
383         MessageDigest md5Helper;
384 
385         try {
386             md5Helper = MessageDigest.getInstance(digAlg);
387         } catch (Exception e) {
388             throw new AuthenticationException(
389               "Unsupported algorithm in HTTP Digest authentication: "
390                + digAlg);
391         }
392 
393         // 3.2.2.2: Calculating digest
394         StringBuffer tmp = new StringBuffer(uname.length() + realm.length() + pwd.length() + 2);
395         tmp.append(uname);
396         tmp.append(':');
397         tmp.append(realm);
398         tmp.append(':');
399         tmp.append(pwd);
400         // unq(username-value) ":" unq(realm-value) ":" passwd
401         String a1 = tmp.toString();
402         //a1 is suitable for MD5 algorithm
403         if(algorithm.equals("MD5-sess")) {
404             // H( unq(username-value) ":" unq(realm-value) ":" passwd )
405             //      ":" unq(nonce-value)
406             //      ":" unq(cnonce-value)
407 
408             String tmp2=encode(md5Helper.digest(EncodingUtil.getBytes(a1, charset)));
409             StringBuffer tmp3 = new StringBuffer(tmp2.length() + nonce.length() + cnonce.length() + 2);
410             tmp3.append(tmp2);
411             tmp3.append(':');
412             tmp3.append(nonce);
413             tmp3.append(':');
414             tmp3.append(cnonce);
415             a1 = tmp3.toString();
416         } else if(!algorithm.equals("MD5")) {
417             LOG.warn("Unhandled algorithm " + algorithm + " requested");
418         }
419         String md5a1 = encode(md5Helper.digest(EncodingUtil.getBytes(a1, charset)));
420 
421         String a2 = null;
422         if (qopVariant == QOP_AUTH_INT) {
423             LOG.error("Unhandled qop auth-int");
424             //we do not have access to the entity-body or its hash
425             //TODO: add Method ":" digest-uri-value ":" H(entity-body)      
426         } else {
427             a2 = method + ":" + uri;
428         }
429         String md5a2 = encode(md5Helper.digest(EncodingUtil.getAsciiBytes(a2)));
430 
431         // 3.2.2.1
432         String serverDigestValue;
433         if (qopVariant == QOP_MISSING) {
434             LOG.debug("Using null qop method");
435             StringBuffer tmp2 = new StringBuffer(md5a1.length() + nonce.length() + md5a2.length());
436             tmp2.append(md5a1);
437             tmp2.append(':');
438             tmp2.append(nonce);
439             tmp2.append(':');
440             tmp2.append(md5a2);
441             serverDigestValue = tmp2.toString();
442         } else {
443             if (LOG.isDebugEnabled()) {
444                 LOG.debug("Using qop method " + qop);
445             }
446             String qopOption = getQopVariantString();
447             StringBuffer tmp2 = new StringBuffer(md5a1.length() + nonce.length()
448                 + NC.length() + cnonce.length() + qopOption.length() + md5a2.length() + 5);
449             tmp2.append(md5a1);
450             tmp2.append(':');
451             tmp2.append(nonce);
452             tmp2.append(':');
453             tmp2.append(NC);
454             tmp2.append(':');
455             tmp2.append(cnonce);
456             tmp2.append(':');
457             tmp2.append(qopOption);
458             tmp2.append(':');
459             tmp2.append(md5a2); 
460             serverDigestValue = tmp2.toString();
461         }
462 
463         String serverDigest =
464             encode(md5Helper.digest(EncodingUtil.getAsciiBytes(serverDigestValue)));
465 
466         return serverDigest;
467     }

469     /**
470      * Creates digest-response header as defined in RFC2617.
471      * 
472      * @param uname Username
473      * @param digest The response tag's value as String.
474      * 
475      * @return The digest-response as String.
476      */
477     private String createDigestHeader(final String uname, final String digest) 
478         throws AuthenticationException {
479 
480         LOG.trace("enter DigestScheme.createDigestHeader(String, Map, "
481             + "String)");
482 
483         String uri = getParameter("uri");
484         String realm = getParameter("realm");
485         String nonce = getParameter("nonce");
486         String opaque = getParameter("opaque");
487         String response = digest;
488         String algorithm = getParameter("algorithm");
489 
490         List params = new ArrayList(20);
491         params.add(new NameValuePair("username", uname));
492         params.add(new NameValuePair("realm", realm));
493         params.add(new NameValuePair("nonce", nonce));
494         params.add(new NameValuePair("uri", uri));
495         params.add(new NameValuePair("response", response));
496         
497         if (qopVariant != QOP_MISSING) {
498             params.add(new NameValuePair("qop", getQopVariantString()));
499             params.add(new NameValuePair("nc", NC));
500             params.add(new NameValuePair("cnonce", this.cnonce));
501         }
502         if (algorithm != null) {
503             params.add(new NameValuePair("algorithm", algorithm));
504         }    
505         if (opaque != null) {
506             params.add(new NameValuePair("opaque", opaque));
507         }
508 
509         StringBuffer buffer = new StringBuffer();
510         for (int i = 0; i < params.size(); i++) {
511             NameValuePair../../../../../../../edu/internet2/middleware/grouperInstallerExt/org/apache/commons/httpclient/NameValuePair.html#NameValuePair">NameValuePair param = (NameValuePair) params.get(i);
512             if (i > 0) {
513                 buffer.append(", ");
514             }
515             boolean noQuotes = "nc".equals(param.getName()) ||
516                                "qop".equals(param.getName());
517             this.formatter.setAlwaysUseQuotes(!noQuotes);
518             this.formatter.format(buffer, param);
519         }
520         return buffer.toString();
521     }
522 
523     private String getQopVariantString() {
524         String qopOption;
525         if (qopVariant == QOP_AUTH_INT) {
526             qopOption = "auth-int";   
527         } else {
528             qopOption = "auth";
529         }
530         return qopOption;            
531     }

533     /**
534      * Encodes the 128 bit (16 bytes) MD5 digest into a 32 characters long 
535      * <CODE>String</CODE> according to RFC 2617.
536      * 
537      * @param binaryData array containing the digest
538      * @return encoded MD5, or <CODE>null</CODE> if encoding failed
539      */
540     private static String encode(byte[] binaryData) {
541         LOG.trace("enter DigestScheme.encode(byte[])");
542 
543         if (binaryData.length != 16) {
544             return null;
545         } 
546 
547         char[] buffer = new char[32];
548         for (int i = 0; i < 16; i++) {
549             int low = (int) (binaryData[i] & 0x0f);
550             int high = (int) ((binaryData[i] & 0xf0) >> 4);
551             buffer[i * 2] = HEXADECIMAL[high];
552             buffer[(i * 2) + 1] = HEXADECIMAL[low];
553         }
554 
555         return new String(buffer);
556     }

559     /**
560      * Creates a random cnonce value based on the current time.
561      * 
562      * @return The cnonce value as String.
563      * @throws HttpClientError if MD5 algorithm is not supported.
564      */
565     public static String createCnonce() {
566         LOG.trace("enter DigestScheme.createCnonce()");
567 
568         String cnonce;
569         final String digAlg = "MD5";
570         MessageDigest md5Helper;
571 
572         try {
573             md5Helper = MessageDigest.getInstance(digAlg);
574         } catch (NoSuchAlgorithmException e) {
575             throw new HttpClientError(
576               "Unsupported algorithm in HTTP Digest authentication: "
577                + digAlg);
578         }
579 
580         cnonce = Long.toString(System.currentTimeMillis());
581         cnonce = encode(md5Helper.digest(EncodingUtil.getAsciiBytes(cnonce)));
582 
583         return cnonce;
584     }
