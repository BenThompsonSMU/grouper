46  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.protocol;
47  
48  import java.io.IOException;
49  import java.lang.reflect.Constructor;
50  import java.lang.reflect.InvocationTargetException;
51  import java.lang.reflect.Method;
52  import java.net.InetAddress;
53  import java.net.Socket;
54  import java.net.UnknownHostException;
55  
56  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.ConnectTimeoutException;
57  
58  /**
59   * This helper class uses refelction in order to execute Socket methods
60   * available in Java 1.4 and above  
61   * 
62   * @author Oleg Kalnichevski
63   * 
64   * @since 3.0
65   */
66  public final class ReflectionSocketFactory {
67  
68      private static boolean REFLECTION_FAILED = false;
69      
70      private static Constructor INETSOCKETADDRESS_CONSTRUCTOR = null;
71      private static Method SOCKETCONNECT_METHOD = null;
72      private static Method SOCKETBIND_METHOD = null;
73      private static Class SOCKETTIMEOUTEXCEPTION_CLASS = null;
74  
75      private ReflectionSocketFactory() {
76          super();
77      }
78  
79      /**
80       * This method attempts to execute Socket method available since Java 1.4
81       * using reflection. If the methods are not available or could not be executed
82       * <tt>null</tt> is returned
83       *   
84       * @param socketfactoryName name of the socket factory class
85       * @param host the host name/IP
86       * @param port the port on the host
87       * @param localAddress the local host name/IP to bind the socket to
88       * @param localPort the port on the local machine
89       * @param timeout the timeout value to be used in milliseconds. If the socket cannot be
90       *        completed within the given time limit, it will be abandoned
91       * 
92       * @return a connected Socket
93       * 
94       * @throws IOException if an I/O error occurs while creating the socket
95       * @throws UnknownHostException if the IP address of the host cannot be
96       * determined
97       * @throws ConnectTimeoutException if socket cannot be connected within the
98       *  given time limit
99       * 
100      */
101     public static Socket createSocket(
102         final String socketfactoryName, 
103         final String host,
104         final int port,
105         final InetAddress localAddress,
106         final int localPort,
107         int timeout)
108      throws IOException, UnknownHostException, ConnectTimeoutException
109     {
110         if (REFLECTION_FAILED) {
111             //This is known to have failed before. Do not try it again
112             return null;
113         }
114         // This code uses reflection to essentially do the following:
115         //
116         //  SocketFactory socketFactory = Class.forName(socketfactoryName).getDefault();
117         //  Socket socket = socketFactory.createSocket();
118         //  SocketAddress localaddr = new InetSocketAddress(localAddress, localPort);
119         //  SocketAddress remoteaddr = new InetSocketAddress(host, port);
120         //  socket.bind(localaddr);
121         //  socket.connect(remoteaddr, timeout);
122         //  return socket;
123         try {
124             Class socketfactoryClass = Class.forName(socketfactoryName);
125             Method method = socketfactoryClass.getMethod("getDefault", 
126                 new Class[] {});
127             Object socketfactory = method.invoke(null, 
128                 new Object[] {});
129             method = socketfactoryClass.getMethod("createSocket", 
130                 new Class[] {});
131             Socket socket = (Socket) method.invoke(socketfactory, new Object[] {});
132             
133             if (INETSOCKETADDRESS_CONSTRUCTOR == null) {
134                 Class addressClass = Class.forName("java.net.InetSocketAddress");
135                 INETSOCKETADDRESS_CONSTRUCTOR = addressClass.getConstructor(
136                     new Class[] { InetAddress.class, Integer.TYPE });
137             }
138                 
139             Object remoteaddr = INETSOCKETADDRESS_CONSTRUCTOR.newInstance(
140                 new Object[] { InetAddress.getByName(host), new Integer(port)});
141 
142             Object localaddr = INETSOCKETADDRESS_CONSTRUCTOR.newInstance(
143                     new Object[] { localAddress, new Integer(localPort)});
144 
145             if (SOCKETCONNECT_METHOD == null) {
146                 SOCKETCONNECT_METHOD = Socket.class.getMethod("connect", 
147                     new Class[] {Class.forName("java.net.SocketAddress"), Integer.TYPE});
148             }
149 
150             if (SOCKETBIND_METHOD == null) {
151                 SOCKETBIND_METHOD = Socket.class.getMethod("bind", 
152                     new Class[] {Class.forName("java.net.SocketAddress")});
153             }
154             SOCKETBIND_METHOD.invoke(socket, new Object[] { localaddr});
155             SOCKETCONNECT_METHOD.invoke(socket, new Object[] { remoteaddr, new Integer(timeout)});
156             return socket;
157         }
158         catch (InvocationTargetException e) {
159             Throwable cause = e.getTargetException(); 
160             if (SOCKETTIMEOUTEXCEPTION_CLASS == null) {
161                 try {
162                     SOCKETTIMEOUTEXCEPTION_CLASS = Class.forName("java.net.SocketTimeoutException");
163                 } catch (ClassNotFoundException ex) {
164                     // At this point this should never happen. Really.
165                     REFLECTION_FAILED = true;
166                     return null;
167                 }
168             }
169             if (SOCKETTIMEOUTEXCEPTION_CLASS.isInstance(cause)) {
170                 throw new ConnectTimeoutException(
171                     "The host did not accept the connection within timeout of " 
172                     + timeout + " ms", cause);
173             }
174             if (cause instanceof IOException) {
175                 throw (IOException)cause;
176             }
177             return null;
178         }
179         catch (Exception e) {
180             REFLECTION_FAILED = true;
181             return null;
182         }
183     }
184 }
