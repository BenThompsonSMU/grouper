102     private void init(final HostConfiguration hostConfiguration) {
103         // wrap all of the assignments in a synchronized block to avoid
104         // having to negotiate the monitor for each method call
105         synchronized (hostConfiguration) {
106             try {
107                 if (hostConfiguration.host != null) {
108                     this.host = (HttpHost) hostConfiguration.host.clone();
109                 } else {
110                     this.host = null;
111                 }
112                 if (hostConfiguration.proxyHost != null) {
113                     this.proxyHost = (ProxyHost) hostConfiguration.proxyHost.clone();
114                 } else {
115                     this.proxyHost = null;
116                 }
117                 this.localAddress = hostConfiguration.getLocalAddress();
118                 this.params = (HostParams)hostConfiguration.getParams().clone();
119             } catch (CloneNotSupportedException e) {
120                 throw new IllegalArgumentException("Host configuration could not be cloned");
121             }
122         }        
123     }

142     public synchronized String toString() {
143         
144         boolean appendComma = false;
145         StringBuffer b = new StringBuffer(50);        
146         b.append("HostConfiguration[");
147         
148         if (this.host != null) {
149             appendComma = true;
150             b.append("host=").append(this.host);
151         }
152         if (this.proxyHost != null) {
153             if (appendComma) {
154                 b.append(", ");
155             } else {
156                 appendComma = true;
157             }
158             b.append("proxyHost=").append(this.proxyHost);
159         }
160         if (this.localAddress != null) {
161             if (appendComma) {
162                 b.append(", ");
163             } else {
164                 appendComma = true;
165             }
166             b.append("localAddress=").append(this.localAddress);
167             if (appendComma) {
168                 b.append(", ");
169             } else {
170                 appendComma = true;
171             }
172             b.append("params=").append(this.params);
173         }
174         b.append("]");
175         return b.toString();
176     }

178     /**
179      * Tests if the host configuration equals the configuration set on the
180      * connection. True only if the host, port, protocol, local address and virtual address
181      * are equal.  If no host configuration has been set false will be returned.
182      * 
183      * @param connection the connection to test against
184      * @return <code>true</code> if the connection's host information equals that of this
185      * configuration
186      * 
187      * @see #proxyEquals(HttpConnection)
188      */
189     public synchronized boolean hostEquals(final HttpConnection connection) {
190         if (connection == null) {
191             throw new IllegalArgumentException("Connection may not be null");
192         }
193         if (this.host != null) {
194             if (!this.host.getHostName().equalsIgnoreCase(connection.getHost())) {
195                 return false;
196             }
197             if (this.host.getPort() != connection.getPort()) {
198                 return false;
199             }
200             if (!this.host.getProtocol().equals(connection.getProtocol())) {
201                 return false;
202             }
203             if (this.localAddress != null) {
204                 if (!this.localAddress.equals(connection.getLocalAddress())) {
205                     return false;
206                 }
207             } else {
208                 if (connection.getLocalAddress() != null) {
209                     return false; 
210                 }
211             }
212             return true;
213         } else {
214             return false;   
215         }
216     }

218     /**
219      * Tests if the proxy configuration equals the configuration set on the
220      * connection. True only if the proxyHost and proxyPort are equal.
221      *
222      * @param connection the connection to test against
223      * @return <code>true</code> if the connection's proxy information equals that of this
224      * configuration
225      *
226      * @see #hostEquals(HttpConnection)
227      */
228     public synchronized boolean proxyEquals(final HttpConnection connection) {
229         if (connection == null) {
230             throw new IllegalArgumentException("Connection may not be null");
231         }
232         if (this.proxyHost != null) {
233             return
234                 this.proxyHost.getHostName().equalsIgnoreCase(connection.getProxyHost())
235                 && this.proxyHost.getPort() == connection.getProxyPort();
236         } else {
237             return connection.getProxyHost() == null;
238         }
239     }
