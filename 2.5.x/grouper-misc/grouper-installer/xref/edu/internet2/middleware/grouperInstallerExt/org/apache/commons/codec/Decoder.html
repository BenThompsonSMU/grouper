32  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.codec;
33  
34  /**
35   * <p>Provides the highest level of abstraction for Decoders.
36   * This is the sister interface of {@link Encoder}.  All
37   * Decoders implement this common generic interface.</p>
38   * 
39   * <p>Allows a user to pass a generic Object to any Decoder 
40   * implementation in the codec package.</p>
41   * 
42   * <p>One of the two interfaces at the center of the codec package.</p>
43   * 
44   * @author Apache Software Foundation
45   * @version $Id: Decoder.java,v 1.1 2008-11-30 10:57:21 mchyzer Exp $
46   */
47  public interface Decoder {
48  
49      /**
50       * Decodes an "encoded" Object and returns a "decoded"
51       * Object.  Note that the implementation of this
52       * interface will try to cast the Object parameter
53       * to the specific type expected by a particular Decoder
54       * implementation.  If a {@link java.lang.ClassCastException} occurs
55       * this decode method will throw a DecoderException.
56       * 
57       * @param pObject an object to "decode"
58       * 
59       * @return a 'decoded" object
60       * 
61       * @throws DecoderException a decoder exception can
62       * be thrown for any number of reasons.  Some good
63       * candidates are that the parameter passed to this
64       * method is null, a param cannot be cast to the
65       * appropriate type for a specific encoder.
66       */
67      Object decode(Object pObject) throws DecoderException;
68  }
