95      /**
96       * Constructor for the NTLM authentication scheme.
97       * 
98       * @param challenge The authentication challenge
99       * 
100      * @throws MalformedChallengeException is thrown if the authentication challenge
101      * is malformed
102      */
103     public NTLMScheme(final String challenge) throws MalformedChallengeException {
104         super();
105         processChallenge(challenge);
106     }
107 
108     /**
109      * Processes the NTLM challenge.
110      *  
111      * @param challenge the challenge string
112      * 
113      * @throws MalformedChallengeException is thrown if the authentication challenge
114      * is malformed
115      * 
116      * @since 3.0
117      */
118     public void processChallenge(final String challenge) throws MalformedChallengeException {
119         String s = AuthChallengeParser.extractScheme(challenge);
120         if (!s.equalsIgnoreCase(getSchemeName())) {
121             throw new MalformedChallengeException("Invalid NTLM challenge: " + challenge);
122         }
123         int i = challenge.indexOf(' ');
124         if (i != -1) {
125             s = challenge.substring(i, challenge.length());
126             this.ntlmchallenge = s.trim();
127             this.state = TYPE2_MSG_RECEIVED;
128         } else {
129             this.ntlmchallenge = "";
130             if (this.state == UNINITIATED) {
131                 this.state = INITIATED;
132             } else {
133                 this.state = FAILED;
134             }
135         }
136     }

138     /**
139      * Tests if the NTLM authentication process has been completed.
140      * 
141      * @return <tt>true</tt> if Basic authorization has been processed,
142      *   <tt>false</tt> otherwise.
143      * 
144      * @since 3.0
145      */
146     public boolean isComplete() {
147         return this.state == TYPE3_MSG_GENERATED || this.state == FAILED;
148     }

169     /**
170      * Returns a String identifying the authentication challenge.  This is
171      * used, in combination with the host and port to determine if
172      * authorization has already been attempted or not.  Schemes which
173      * require multiple requests to complete the authentication should
174      * return a different value for each stage in the request.
175      * 
176      * <p>Additionally, the ID should take into account any changes to the
177      * authentication challenge and return a different value when appropriate.
178      * For example when the realm changes in basic authentication it should be
179      * considered a different authentication attempt and a different value should
180      * be returned.</p>
181      * 
182      * @return String a String identifying the authentication challenge.  The
183      * returned value may be null.
184      * 
185      * @deprecated no longer used
186      */
187     public String getID() {
188         return ntlmchallenge;
189     }

286     /**
287      * Produces NTLM authorization string for the given set of 
288      * {@link Credentials}.
289      * 
290      * @param credentials The set of credentials to be used for athentication
291      * @param method Method name is ignored by the NTLM authentication scheme
292      * @param uri URI is ignored by the NTLM authentication scheme
293      * @throws InvalidCredentialsException if authentication credentials
294      *         are not valid or not applicable for this authentication scheme
295      * @throws AuthenticationException if authorization string cannot 
296      *   be generated due to an authentication failure
297      * 
298      * @return an NTLM authorization string
299      * 
300      * @deprecated Use {@link #authenticate(Credentials, HttpMethod)}
301      */
302     public String authenticate(Credentials credentials, String method, String uri) 
303       throws AuthenticationException {
304         LOG.trace("enter NTLMScheme.authenticate(Credentials, String, String)");
305 
306         NTCredentials ntcredentials = null;
307         try {
308             ntcredentials = (NTCredentials) credentials;
309         } catch (ClassCastException e) {
310             throw new InvalidCredentialsException(
311              "Credentials cannot be used for NTLM authentication: " 
312               + credentials.getClass().getName());
313         }
314         return NTLMScheme.authenticate(ntcredentials, this.ntlmchallenge);
315     }

317     /**
318      * Produces NTLM authorization string for the given set of 
319      * {@link Credentials}.
320      * 
321      * @param credentials The set of credentials to be used for athentication
322      * @param method The method being authenticated
323      * 
324      * @throws InvalidCredentialsException if authentication credentials
325      *         are not valid or not applicable for this authentication scheme
326      * @throws AuthenticationException if authorization string cannot 
327      *   be generated due to an authentication failure
328      * 
329      * @return an NTLM authorization string
330      * 
331      * @since 3.0
332      */
333     public String authenticate(
334         Credentials credentials, 
335         HttpMethod method
336     ) throws AuthenticationException {
337         LOG.trace("enter NTLMScheme.authenticate(Credentials, HttpMethod)");
338 
339         if (this.state == UNINITIATED) {
340             throw new IllegalStateException("NTLM authentication process has not been initiated");
341         }
342 
343         NTCredentials ntcredentials = null;
344         try {
345             ntcredentials = (NTCredentials) credentials;
346         } catch (ClassCastException e) {
347             throw new InvalidCredentialsException(
348                     "Credentials cannot be used for NTLM authentication: " 
349                     + credentials.getClass().getName());
350         }
351         NTLMnet2/middleware/grouperInstallerExt/org/apache/commons/httpclient/auth/NTLM.html#NTLM">NTLM ntlm = new NTLM();
352         ntlm.setCredentialCharset(method.getParams().getCredentialCharset());
353         String response = null;
354         if (this.state == INITIATED || this.state == FAILED) {
355             response = ntlm.getType1Message(
356                 ntcredentials.getHost(), 
357                 ntcredentials.getDomain());
358             this.state = TYPE1_MSG_GENERATED;
359         } else {
360             response = ntlm.getType3Message(
361                 ntcredentials.getUserName(), 
362                 ntcredentials.getPassword(),
363                 ntcredentials.getHost(), 
364                 ntcredentials.getDomain(),
365                 ntlm.parseType2Message(this.ntlmchallenge));
366             this.state = TYPE3_MSG_GENERATED;
367         }
368         return "NTLM " + response;
369     }
