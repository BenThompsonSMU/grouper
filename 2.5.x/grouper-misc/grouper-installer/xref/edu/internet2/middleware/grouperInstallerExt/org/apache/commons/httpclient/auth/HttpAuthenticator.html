60  /**
61   * Utility methods for HTTP authorization and authentication.  This class
62   * provides utility methods for generating responses to HTTP www and proxy
63   * authentication challenges.
64   * 
65   * <blockquote>
66   * A client SHOULD assume that all paths at or deeper than the depth of the
67   * last symbolic element in the path field of the Request-URI also are within
68   * the protection space specified by the basic realm value of the current
69   * challenge. A client MAY preemptively send the corresponding Authorization
70   * header with requests for resources in that space without receipt of another
71   * challenge from the server. Similarly, when a client sends a request to a
72   * proxy, it may reuse a userid and password in the Proxy-Authorization header
73   * field without receiving another challenge from the proxy server.
74   * </blockquote>
75   * </p>

113     /** Chooses the strongest authentication scheme supported from the
114      * array of authentication challenges. Currently only <code>NTLM</code>,
115      * <code>Digest</code>, <code>Basic</code> schemes are recognized. 
116      * The <code>NTLM</code> scheme is considered the strongest and is 
117      * preferred to all others. The <code>Digest</code> scheme is preferred to 
118      * the <code>Basic</code> one which provides no encryption for credentials.
119      * The <code>Basic</code> scheme is used only if it is the only one 
120      * supported.
121      * 
122      * @param challenges The array of authentication challenges
123      * 
124      * @return The strongest authentication scheme supported
125      * 
126      * @throws MalformedChallengeException is thrown if an authentication 
127      *  challenge is malformed
128      * @throws UnsupportedOperationException when none of challenge types
129      *  available is supported.
130      * 
131      * @deprecated Use {@link AuthChallengeParser#parseChallenges(Header[])} and 
132      *      {@link AuthPolicy#getAuthScheme(String)}
133      */
134     public static AuthScheme selectAuthScheme(final Header[] challenges)
135       throws MalformedChallengeException {
136         LOG.trace("enter HttpAuthenticator.selectAuthScheme(Header[])");
137         if (challenges == null) {
138             throw new IllegalArgumentException("Array of challenges may not be null");
139         }
140         if (challenges.length == 0) {
141             throw new IllegalArgumentException("Array of challenges may not be empty");
142         }
143         String challenge = null;
144         Map challengemap = new HashMap(challenges.length); 
145         for (int i = 0; i < challenges.length; i++) {
146             challenge = challenges[i].getValue();
147             String s = AuthChallengeParser.extractScheme(challenge);
148             challengemap.put(s, challenge);
149         }
150         challenge = (String) challengemap.get("ntlm");
151         if (challenge != null) {
152             return new NTLMScheme(challenge);
153         }
154         challenge = (String) challengemap.get("digest");
155         if (challenge != null) {
156             return new DigestScheme(challenge);
157         }
158         challenge = (String) challengemap.get("basic");
159         if (challenge != null) {
160             return new BasicScheme(challenge);
161         }
162         throw new UnsupportedOperationException(
163           "Authentication scheme(s) not supported: " + challengemap.toString()); 
164     }

166     private static boolean doAuthenticateDefault(
167         HttpMethod method, 
168         HttpConnection conn,
169         HttpState state, 
170         boolean proxy)
171       throws AuthenticationException {
172         if (method == null) {
173             throw new IllegalArgumentException("HTTP method may not be null");
174         }
175         if (state == null) {
176             throw new IllegalArgumentException("HTTP state may not be null");
177         }
178         String host = null;
179         if (conn != null) {
180             host = proxy ? conn.getProxyHost() : conn.getHost();
181         }
182         Credentials credentials = proxy 
183             ? state.getProxyCredentials(null, host) : state.getCredentials(null, host);
184         if (credentials == null) {
185             return false;
186         }
187         if (!(credentials instanceof UsernamePasswordCredentials)) {
188             throw new InvalidCredentialsException(
189              "Credentials cannot be used for basic authentication: " 
190               + credentials.toString());
191         }
192         String auth = BasicScheme.authenticate(
193             (UsernamePasswordCredentials) credentials,
194             method.getParams().getCredentialCharset());
195         if (auth != null) {
196             String s = proxy ? PROXY_AUTH_RESP : WWW_AUTH_RESP;
197             Header/middleware/grouperInstallerExt/org/apache/commons/httpclient/Header.html#Header">Header header = new Header(s, auth, true);
198             method.addRequestHeader(header);
199             return true;
200         } else {
201             return false;
202         }
203     }

271     private static boolean doAuthenticate(
272         AuthScheme authscheme, 
273         HttpMethod method, 
274         HttpConnection conn,
275         HttpState state, 
276         boolean proxy)
277        throws AuthenticationException {
278         if (authscheme == null) {
279             throw new IllegalArgumentException("Authentication scheme may not be null");
280         }
281         if (method == null) {
282             throw new IllegalArgumentException("HTTP method may not be null");
283         }
284         if (state == null) {
285             throw new IllegalArgumentException("HTTP state may not be null");
286         }
287         String host = null;
288         if (conn != null) {
289             if (proxy) {
290                 host = conn.getProxyHost();
291             } else {
292                 host = method.getParams().getVirtualHost();
293                 if (host == null) {
294                     host = conn.getHost();
295                 }
296             }
297         }
298         String realm = authscheme.getRealm();
299         if (LOG.isDebugEnabled()) {
300             StringBuffer buffer = new StringBuffer();
301             buffer.append("Using credentials for ");
302             if (realm == null) {
303                 buffer.append("default");
304             } else {
305                 buffer.append('\'');
306                 buffer.append(realm);
307                 buffer.append('\'');
308             }
309             buffer.append(" authentication realm at "); 
310             buffer.append(host); 
311             LOG.debug(buffer.toString());
312         }
313         Credentials credentials = proxy 
314             ? state.getProxyCredentials(realm, host) 
315             : state.getCredentials(realm, host);
316         if (credentials == null) {
317             StringBuffer buffer = new StringBuffer();
318             buffer.append("No credentials available for the "); 
319             if (realm == null) {
320                 buffer.append("default");
321             } else {
322                 buffer.append('\'');
323                 buffer.append(realm);
324                 buffer.append('\'');
325             }
326             buffer.append(" authentication realm at "); 
327             buffer.append(host); 
328             throw new CredentialsNotAvailableException(buffer.toString());
329         }
330         String auth = authscheme.authenticate(credentials, method);
331         if (auth != null) {
332             String s = proxy ? PROXY_AUTH_RESP : WWW_AUTH_RESP;
333             Header/middleware/grouperInstallerExt/org/apache/commons/httpclient/Header.html#Header">Header header = new Header(s, auth, true);
334             method.addRequestHeader(header);
335             return true;
336         } else {
337             return false;
338         }
339     }
