30   * <p>This file is a copy of the implementation at the Apache Hadoop project.</p>

40    /** Create a new PureJavaCrc32 object. */
41    public PureJavaCrc32C() {
42      reset();
43    }
44  
45    public long getValue() {
46      long ret = crc;
47      return (~ret) & 0xffffffffL;
48    }
49  
50    public void reset() {
51      crc = 0xffffffff;
52    }

54    public void update(byte[] b, int off, int len) {
55      int localCrc = crc;
56  
57      while(len > 7) {
58        final int c0 =(b[off+0] ^ localCrc) & 0xff;
59        final int c1 =(b[off+1] ^ (localCrc >>>= 8)) & 0xff;
60        final int c2 =(b[off+2] ^ (localCrc >>>= 8)) & 0xff;
61        final int c3 =(b[off+3] ^ (localCrc >>>= 8)) & 0xff;
62        localCrc = (T[T8_7_start + c0] ^ T[T8_6_start + c1])
63            ^ (T[T8_5_start + c2] ^ T[T8_4_start + c3]);
64  
65        final int c4 = b[off+4] & 0xff;
66        final int c5 = b[off+5] & 0xff;
67        final int c6 = b[off+6] & 0xff;
68        final int c7 = b[off+7] & 0xff;
69  
70        localCrc ^= (T[T8_3_start + c4] ^ T[T8_2_start + c5])
71             ^ (T[T8_1_start + c6] ^ T[T8_0_start + c7]);
72  
73        off += 8;
74        len -= 8;
75      }
76  
77      /* loop unroll - duff's device style */
78      switch(len) {
79        case 7: localCrc = (localCrc >>> 8) ^ T[T8_0_start + ((localCrc ^ b[off++]) & 0xff)];
80        case 6: localCrc = (localCrc >>> 8) ^ T[T8_0_start + ((localCrc ^ b[off++]) & 0xff)];
81        case 5: localCrc = (localCrc >>> 8) ^ T[T8_0_start + ((localCrc ^ b[off++]) & 0xff)];
82        case 4: localCrc = (localCrc >>> 8) ^ T[T8_0_start + ((localCrc ^ b[off++]) & 0xff)];
83        case 3: localCrc = (localCrc >>> 8) ^ T[T8_0_start + ((localCrc ^ b[off++]) & 0xff)];
84        case 2: localCrc = (localCrc >>> 8) ^ T[T8_0_start + ((localCrc ^ b[off++]) & 0xff)];
85        case 1: localCrc = (localCrc >>> 8) ^ T[T8_0_start + ((localCrc ^ b[off++]) & 0xff)];
86        default:
87          /* nothing */
88      }
89      
90      // Publish crc out to object
91      crc = localCrc;
92    }
93  
94    final public void update(int b) {
95      crc = (crc >>> 8) ^ T[T8_0_start + ((crc ^ b) & 0xff)];
96    }
