46  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.auth;
47  
48  import java.util.Collection;
49  import java.util.Iterator;
50  import java.util.Map;
51  
52  
53  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.params.HttpParams;
54  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.logging.Log;
55  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.logging.LogFactory;
56  
57  /**
58   * This class provides utility methods for processing HTTP www and proxy authentication 
59   * challenges.
60   * 
61   * @author <a href="mailto:oleg@ural.ru">Oleg Kalnichevski</a>
62   * 
63   * @since 3.0
64   */
65  public final class AuthChallengeProcessor {
66  
67      private static final Log LOG = LogFactory.getLog(AuthChallengeProcessor.class);
68  
69      private HttpParams params = null;
70      
71      /**
72       * Creates an authentication challenge processor with the given {@link HttpParams HTTP 
73       * parameters}
74       * 
75       * @param params the {@link HttpParams HTTP parameters} used by this processor
76       */
77      public AuthChallengeProcessor(final HttpParams params) {
78          super();
79          if (params == null) {
80              throw new IllegalArgumentException("Parameter collection may not be null");
81          }
82          this.params = params;
83      }
84  
85      /**
86       * Determines the preferred {@link AuthScheme authentication scheme} that can be used 
87       * to respond to the given collection of challenges.
88       * 
89       * @param challenges the collection of authentication challenges
90       * 
91       * @return the preferred {@link AuthScheme authentication scheme}
92       * 
93       * @throws AuthChallengeException if the preferred authentication scheme 
94       * cannot be determined or is not supported
95       */
96      public AuthScheme selectAuthScheme(final Map challenges) throws AuthChallengeException {
97          if (challenges == null) {
98              throw new IllegalArgumentException("Challenge map may not be null"); 
99          }
100         Collection authPrefs = (Collection) this.params.getParameter(
101             AuthPolicy.AUTH_SCHEME_PRIORITY);
102         if (authPrefs == null || authPrefs.isEmpty()) {
103             authPrefs = AuthPolicy.getDefaultAuthPrefs();    
104         }
105         if (LOG.isDebugEnabled()) {
106             LOG.debug("Supported authentication schemes in the order of preference: " 
107                 + authPrefs);
108         }
109         AuthScheme authscheme = null;
110         String challenge = null;
111         Iterator item = authPrefs.iterator();
112         while (item.hasNext()) {
113             String id = (String) item.next();
114             challenge = (String) challenges.get(id.toLowerCase()); 
115             if (challenge != null) {
116                 if (LOG.isInfoEnabled()) {
117                     LOG.info(id + " authentication scheme selected");
118                 }
119                 try {
120                     authscheme = AuthPolicy.getAuthScheme(id);
121                 } catch (IllegalStateException e) {
122                     throw new AuthChallengeException(e.getMessage());
123                 }
124                 break;
125             } else {
126                 if (LOG.isDebugEnabled()) {
127                     LOG.debug("Challenge for " + id + " authentication scheme not available");
128                     // Try again
129                 }
130             }
131         }
132         if (authscheme == null) {
133             // If none selected, something is wrong
134             throw new AuthChallengeException(
135                 "Unable to respond to any of these challenges: "
136                     + challenges);
137         }
138         return authscheme;
139     }
140 
141     /**
142      * Processes the given collection of challenges and updates the 
143      * {@link AuthState state} of the authentication process.
144      * 
145      * @param challenges the collection of authentication challenges
146      * 
147      * @return the {@link AuthScheme authentication scheme} used to 
148      * process the challenge
149      * 
150      * @throws AuthChallengeException if authentication challenges cannot be 
151      * successfully processed or the preferred authentication scheme cannot
152      * be determined
153      */
154     public AuthScheme processChallenge(final AuthState state, final Map challenges)
155         throws MalformedChallengeException, AuthenticationException
156     {    
157         if (state == null) {
158             throw new IllegalArgumentException("Authentication state may not be null"); 
159         }
160         if (challenges == null) {
161             throw new IllegalArgumentException("Challenge map may not be null"); 
162         }
163         
164         if (state.isPreemptive() || state.getAuthScheme() == null) {
165             // Authentication not attempted before
166             state.setAuthScheme(selectAuthScheme(challenges));
167         }
168         AuthScheme authscheme = state.getAuthScheme();
169         String id = authscheme.getSchemeName();
170         if (LOG.isDebugEnabled()) {
171             LOG.debug("Using authentication scheme: " + id);
172         }
173         String challenge = (String) challenges.get(id.toLowerCase());
174         if (challenge == null) {
175             throw new AuthenticationException(id + 
176                 " authorization challenge expected, but not found");
177         }
178         authscheme.processChallenge(challenge);
179         LOG.debug("Authorization challenge processed");
180         return authscheme;
181     }
