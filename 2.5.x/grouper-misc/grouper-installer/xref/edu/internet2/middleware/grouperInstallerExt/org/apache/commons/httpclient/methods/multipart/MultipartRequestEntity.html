45  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.methods.multipart;
46  
47  import java.io.IOException;
48  import java.io.OutputStream;
49  import java.util.Random;
50  
51  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.methods.RequestEntity;
52  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.params.HttpMethodParams;
53  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.util.EncodingUtil;
54  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.logging.Log;
55  import edu.internet2.middleware.grouperInstallerExt.org.apache.commons.logging.LogFactory;
56  
57  /**
58   * Implements a request entity suitable for an HTTP multipart POST method.
59   * <p>
60   * The HTTP multipart POST method is defined in section 3.3 of
61   * <a href="http://www.ietf.org/rfc/rfc1867.txt">RFC1867</a>:
62   * <blockquote>
63   * The media-type multipart/form-data follows the rules of all multipart
64   * MIME data streams as outlined in RFC 1521. The multipart/form-data contains 
65   * a series of parts. Each part is expected to contain a content-disposition 
66   * header where the value is "form-data" and a name attribute specifies 
67   * the field name within the form, e.g., 'content-disposition: form-data; 
68   * name="xxxxx"', where xxxxx is the field name corresponding to that field.
69   * Field names originally in non-ASCII character sets may be encoded using 
70   * the method outlined in RFC 1522.
71   * </blockquote>
72   * </p>
73   * <p>This entity is designed to be used in conjunction with the 
74   * {@link edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.methods.PostMethod post method} to provide
75   * multipart posts.  Example usage:</p>
76   * <pre>
77   *  File f = new File("/path/fileToUpload.txt");
78   *  PostMethod filePost = new PostMethod("http://host/some_path");
79   *  Part[] parts = {
80   *      new StringPart("param_name", "value"),
81   *      new FilePart(f.getName(), f)
82   *  };
83   *  filePost.setRequestEntity(
84   *      new MultipartRequestEntity(parts, filePost.getParams())
85   *      );
86   *  HttpClient client = new HttpClient();
87   *  int status = client.executeMethod(filePost);
88   * </pre>
89   * 
90   * @since 3.0
91   */
92  public class MultipartRequestEntity implements RequestEntity {
93  
94      private static final Log log = LogFactory.getLog(MultipartRequestEntity.class);
95      
96      /** The Content-Type for multipart/form-data. */
97      private static final String MULTIPART_FORM_CONTENT_TYPE = "multipart/form-data";
98      
99      /**
100      * The pool of ASCII chars to be used for generating a multipart boundary.
101      */
102     private static byte[] MULTIPART_CHARS = EncodingUtil.getAsciiBytes(
103         "-_1234567890abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ");
104     
105     /**
106      * Generates a random multipart boundary string.
107      * @return
108      */
109     private static byte[] generateMultipartBoundary() {
110         Random rand = new Random();
111         byte[] bytes = new byte[rand.nextInt(11) + 30]; // a random size from 30 to 40
112         for (int i = 0; i < bytes.length; i++) {
113             bytes[i] = MULTIPART_CHARS[rand.nextInt(MULTIPART_CHARS.length)];
114         }
115         return bytes;
116     }
117     
118     /** The MIME parts as set by the constructor */
119     protected Part[] parts;
120     
121     private byte[] multipartBoundary;
122     
123     private HttpMethodParams params;
124     
125     /**
126      * Creates a new multipart entity containing the given parts.
127      * @param parts The parts to include.
128      * @param params The params of the HttpMethod using this entity.
129      */
130     public MultipartRequestEntity(Part[] parts, HttpMethodParams params) {
131         if (parts == null) {
132             throw new IllegalArgumentException("parts cannot be null");
133         }
134         if (params == null) {
135             throw new IllegalArgumentException("params cannot be null");
136         }
137         this.parts = parts;
138         this.params = params;
139     }
140 
141     /**
142      * Returns the MIME boundary string that is used to demarcate boundaries of
143      * this part. The first call to this method will implicitly create a new
144      * boundary string. To create a boundary string first the 
145      * HttpMethodParams.MULTIPART_BOUNDARY parameter is considered. Otherwise 
146      * a random one is generated.
147      * 
148      * @return The boundary string of this entity in ASCII encoding.
149      */
150     protected byte[] getMultipartBoundary() {
151         if (multipartBoundary == null) {
152             String temp = (String) params.getParameter(HttpMethodParams.MULTIPART_BOUNDARY);
153             if (temp != null) {
154                 multipartBoundary = EncodingUtil.getAsciiBytes(temp);
155             } else {
156                 multipartBoundary = generateMultipartBoundary();
157             }
158         }
159         return multipartBoundary;
160     }
161 
162     /**
163      * Returns <code>true</code> if all parts are repeatable, <code>false</code> otherwise.
164      * @see edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.methods.RequestEntity#isRepeatable()
165      */
166     public boolean isRepeatable() {
167         for (int i = 0; i < parts.length; i++) {
168             if (!parts[i].isRepeatable()) {
169                 return false;
170             }
171         }
172         return true;
173     }
174 
175     /* (non-Javadoc)
176      * @see edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.methods.RequestEntity#writeRequest(java.io.OutputStream)
177      */
178     public void writeRequest(OutputStream out) throws IOException {
179         Part.sendParts(out, parts, getMultipartBoundary());
180     }
181 
182     /* (non-Javadoc)
183      * @see edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.methods.RequestEntity#getContentLength()
184      */
185     public long getContentLength() {
186         try {
187             return Part.getLengthOfParts(parts, getMultipartBoundary());            
188         } catch (Exception e) {
189             log.error("An exception occurred while getting the length of the parts", e);
190             return 0;
191         }
192     }
193 
194     /* (non-Javadoc)
195      * @see edu.internet2.middleware.grouperInstallerExt.org.apache.commons.httpclient.methods.RequestEntity#getContentType()
196      */
197     public String getContentType() {
198         StringBuffer buffer = new StringBuffer(MULTIPART_FORM_CONTENT_TYPE);
199         buffer.append("; boundary=");
200         buffer.append(EncodingUtil.getAsciiString(getMultipartBoundary()));
201         return buffer.toString();
202     }
