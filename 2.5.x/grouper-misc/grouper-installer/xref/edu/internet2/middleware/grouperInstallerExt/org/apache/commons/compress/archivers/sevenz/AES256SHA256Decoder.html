18  package edu.internet2.middleware.grouperInstallerExt.org.apache.commons.compress.archivers.sevenz;
19  
20  import java.io.IOException;
21  import java.io.InputStream;
22  import java.security.GeneralSecurityException;
23  import java.security.MessageDigest;
24  import java.security.NoSuchAlgorithmException;
25  import javax.crypto.Cipher;
26  import javax.crypto.CipherInputStream;
27  import javax.crypto.SecretKey;
28  import javax.crypto.spec.IvParameterSpec;
29  import javax.crypto.spec.SecretKeySpec;
30  
31  class AES256SHA256Decoder extends CoderBase {
32      @Override
33      InputStream decode(final InputStream in, long uncompressedLength,
34              final Coder coder, final byte[] passwordBytes) throws IOException {
35          return new InputStream() {
36              private boolean isInitialized = false;
37              private CipherInputStream cipherInputStream = null;
38                  
39              private CipherInputStream init() throws IOException {
40                  if (isInitialized) {
41                      return cipherInputStream;
42                  }
43                  final int byte0 = 0xff & coder.properties[0];
44                  final int numCyclesPower = byte0 & 0x3f;
45                  final int byte1 = 0xff & coder.properties[1];
46                  final int ivSize = ((byte0 >> 6) & 1) + (byte1 & 0x0f);
47                  final int saltSize = ((byte0 >> 7) & 1) + (byte1 >> 4);
48                  if (2 + saltSize + ivSize > coder.properties.length) {
49                      throw new IOException("Salt size + IV size too long");
50                  }
51                  final byte[] salt = new byte[saltSize];
52                  System.arraycopy(coder.properties, 2, salt, 0, saltSize);
53                  final byte[] iv = new byte[16];
54                  System.arraycopy(coder.properties, 2 + saltSize, iv, 0, ivSize);
55  
56                  if (passwordBytes == null) {
57                      throw new IOException("Cannot read encrypted files without a password");
58                  }
59                  final byte[] aesKeyBytes;
60                  if (numCyclesPower == 0x3f) {
61                      aesKeyBytes = new byte[32];
62                      System.arraycopy(salt, 0, aesKeyBytes, 0, saltSize);
63                      System.arraycopy(passwordBytes, 0, aesKeyBytes, saltSize,
64                                       Math.min(passwordBytes.length, aesKeyBytes.length - saltSize));
65                  } else {
66                      final MessageDigest digest;
67                      try {
68                          digest = MessageDigest.getInstance("SHA-256");
69                      } catch (NoSuchAlgorithmException noSuchAlgorithmException) {
70                          IOException ioe = new IOException("SHA-256 is unsupported by your Java implementation");
71                          ioe.initCause(noSuchAlgorithmException);
72                          throw ioe;
73                          // TODO: simplify when Compress requires Java 1.6                
74  //                      throw new IOException("SHA-256 is unsupported by your Java implementation",
75  //                              noSuchAlgorithmException);
76                      }
77                      final byte[] extra = new byte[8];
78                      for (long j = 0; j < (1L << numCyclesPower); j++) {
79                          digest.update(salt);
80                          digest.update(passwordBytes);
81                          digest.update(extra);
82                          for (int k = 0; k < extra.length; k++) {
83                              ++extra[k];
84                              if (extra[k] != 0) {
85                                  break;
86                              }
87                          }
88                      }
89                      aesKeyBytes = digest.digest();
90                  }
91  
92                  final SecretKey aesKey = new SecretKeySpec(aesKeyBytes, "AES");
93                  try {
94                      final Cipher cipher = Cipher.getInstance("AES/CBC/NoPadding");
95                      cipher.init(Cipher.DECRYPT_MODE, aesKey, new IvParameterSpec(iv));
96                      cipherInputStream = new CipherInputStream(in, cipher);
97                      isInitialized = true;
98                      return cipherInputStream;
99                  } catch (GeneralSecurityException generalSecurityException) {
100                     IOException ioe = new IOException("Decryption error " +
101                                                       "(do you have the JCE Unlimited Strength Jurisdiction Policy Files installed?)");
102                     ioe.initCause(generalSecurityException);
103                     throw ioe;
104                     // TODO: simplify when Compress requires Java 1.6                
105 //                  throw new IOException("Decryption error " +
106 //                          "(do you have the JCE Unlimited Strength Jurisdiction Policy Files installed?)",
107 //                          generalSecurityException);
108                     }
109             }
110                 
111             @Override
112             public int read() throws IOException {
113                 return init().read();
114             }
115                 
116             @Override
117             public int read(byte[] b, int off, int len) throws IOException {
118                 return init().read(b, off, len);
119             }
120                 
121             @Override
122             public void close() {
123             }
124         };
125     }
126 }
