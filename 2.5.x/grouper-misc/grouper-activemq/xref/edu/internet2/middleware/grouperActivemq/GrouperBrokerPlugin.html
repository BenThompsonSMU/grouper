15  package edu.internet2.middleware.grouperActivemq;
16  
17  import org.apache.activemq.broker.Broker;
18  import org.apache.activemq.broker.BrokerPlugin;
19  
20  
21  /**
22   * An authorization ActiveMQ plugin where each operation on a destination is checked
23   * against Grouper permissions
24   * 
25   */
26  public class GrouperBrokerPlugin implements BrokerPlugin {
27  
28    /**
29     * constructor
30     */
31    public GrouperBrokerPlugin() {
32    }
33  
34    /**
35     * 
36     * @see org.apache.activemq.broker.BrokerPlugin#installPlugin(org.apache.activemq.broker.Broker)
37     */
38    public Broker installPlugin(Broker broker) {
39      return new GrouperBroker(broker);
40    }
41  
42  }
