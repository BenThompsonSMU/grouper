48     * retrieve a config from the config file or from cache
49     * @param <T> class which is the return type of config class
50     * @param configClass 
51     * @return the config object never null
52     */
53    @SuppressWarnings("unchecked")
54    protected static <T extends ConfigPropertiesCascadeBase> T retrieveConfig(Class<T> configClass) {
55      
56      if (configSingletonFromClass == null) {
57        configSingletonFromClass = 
58          new HashMap<Class<? extends ConfigPropertiesCascadeBase>, ConfigPropertiesCascadeBase>();
59      }
60      
61      ConfigPropertiesCascadeBase configPropertiesCascadeBase = configSingletonFromClass.get(configClass);
62      if (configPropertiesCascadeBase == null) {
63        configPropertiesCascadeBase = GrouperActivemqUtils.newInstance(configClass, true);
64        configSingletonFromClass.put(configClass, configPropertiesCascadeBase);
65        
66      }
67      //from the singleton, get the real config class
68      return (T)configPropertiesCascadeBase.retrieveFromConfigFileOrCache();
69    }

101   public Map<String, String> propertiesThreadLocalOverrideMap() {
102     if (propertiesThreadLocalOverrideMap == null) {
103       propertiesThreadLocalOverrideMap = new ThreadLocal<Map<Class<? extends ConfigPropertiesCascadeBase>, Map<String, String>>>();
104     }
105 
106     Map<Class<? extends ConfigPropertiesCascadeBase>, Map<String, String>> overrideMap = propertiesThreadLocalOverrideMap.get();
107     if (overrideMap == null) {
108       overrideMap = new HashMap<Class<? extends ConfigPropertiesCascadeBase>, Map<String, String>>();
109       propertiesThreadLocalOverrideMap.set(overrideMap);
110     }
111     Map<String, String> propertiesOverrideMapLocal = overrideMap.get(this.getClass());
112     if (propertiesOverrideMapLocal == null) {
113       propertiesOverrideMapLocal = new HashMap<String, String>();
114       overrideMap.put(this.getClass(), propertiesOverrideMapLocal);
115     }
116     return propertiesOverrideMapLocal;
117   }

162   /**
163    * get the properties object for this config file
164    * @param setValues if we should set the values for the properties.  
165    * if not, the values might not be correct, but this will be more performant
166    * depending on how many EL properties there are
167    * @return the properties
168    */
169   @SuppressWarnings("unchecked")
170   protected Properties propertiesHelper(boolean setValues) {
171     Properties tempResult = new Properties();
172     
173     tempResult.putAll(this.properties);
174     
175     Map<String, String> localPropertiesOverrideMap = propertiesOverrideMap();
176     
177     for (String key: localPropertiesOverrideMap.keySet()) {
178       tempResult.put(key, GrouperClientUtils.defaultString(localPropertiesOverrideMap.get(key)));
179     }
180     
181     localPropertiesOverrideMap = propertiesThreadLocalOverrideMap();
182     
183     for (String key: localPropertiesOverrideMap.keySet()) {
184       tempResult.put(key, GrouperClientUtils.defaultString(localPropertiesOverrideMap.get(key)));
185     }
186 
187     Properties result = new Properties();
188 
189     for (String key : (Set<String>)(Object)tempResult.keySet()) {
190       
191       String value = setValues ? this.properties.getProperty(key) : "";
192       
193       //lets look for EL
194       if (key.endsWith(EL_CONFIG_SUFFIX)) {
195         
196         if (setValues) {
197           //process the EL
198           value = GrouperActivemqUtils.substituteExpressionLanguage(value, null, true, true, true, false);
199         }
200         
201         //change the key name
202         key = key.substring(0, key.length() - EL_CONFIG_SUFFIX.length());
203       }
204       //cant be null, or hashtable exception
205       result.put(key, GrouperClientUtils.defaultString(value));
206     }
207     
208     return result;
209     
210   }

313     if (key.endsWith(EL_CONFIG_SUFFIX)) {

315     }
316     return propertyValueStringHelper(key, defaultValue, required);

319   /**
320    * get the property value as a string
321    * @param key
322    * @param defaultValue
323    * @param required true if required, if doesnt exist, throw exception
324    * @return the property value
325    */
326   protected PropertyValueResult propertyValueStringHelper(String key, String defaultValue, boolean required) {
327     
328     //lets look for EL
329     if (!key.endsWith(EL_CONFIG_SUFFIX)) {
330       
331       PropertyValueResult elPropertyValueResult = propertyValueStringHelper(key + EL_CONFIG_SUFFIX, null, false);
332       
333       if (elPropertyValueResult.isHasKey()) {
334         
335         //process the EL
336         String result = GrouperActivemqUtils.substituteExpressionLanguage(elPropertyValueResult.getTheValue(), null, true, true, true, false);
337         PropertyValueResult propertyValueResult = new PropertyValueResult(result, true);
338         return propertyValueResult;
339       }
340       
341     }
342     
343     //first check threadlocal map
344     boolean hasKey = false;
345     Map<String, String> overrideMap = propertiesThreadLocalOverrideMap();
346     
347     hasKey = overrideMap == null ? false : overrideMap.containsKey(key);
348     String value = hasKey ? overrideMap.get(key) : null;
349     if (!hasKey) {
350       
351       overrideMap = propertiesOverrideMap();
352       
353       hasKey = overrideMap == null ? null : overrideMap.containsKey(key);
354       value = hasKey ? overrideMap.get(key) : null;
355     }
356     if (!hasKey) {
357       hasKey = this.properties.containsKey(key);
358       value = hasKey ? this.properties.getProperty(key) : null;
359     }
360     if (!required && !hasKey) {
361       return new PropertyValueResult(defaultValue, false);
362     }
363     if (required && !hasKey) {
364       String error = "Cant find property: " + key + " in properties file: " + this.getMainConfigClasspath() + ", it is required";
365       
366       throw new RuntimeException(error);
367     }
368     value = GrouperClientUtils.trim(value);
369     value = substituteCommonVars(value);
370 
371     if (!required && GrouperClientUtils.isBlank(value)) {
372       return new PropertyValueResult(null, true);
373     }
374 
375     //do the validation if this is required
376     if (required && GrouperClientUtils.isBlank(value)) {
377       String error = "Property " + key + " in properties file: " + this.getMainConfigClasspath() + ", has a blank value, it is required";
378       
379       throw new RuntimeException(error);
380     }
381     
382     return new PropertyValueResult(value, true);
383   }

385   /**
386    * substitute common vars like $space$ and $newline$
387    * @param string
388    * @return the string
389    */
390   protected static String substituteCommonVars(String string) {
391     if (string == null) {
392       return string;
393     }
394     //short circuit
395     if (string.indexOf('$') < 0) {
396       return string;
397     }
398     //might have $space$
399     string = GrouperClientUtils.replace(string, "$space$", " ");
400     
401     //note, at some point we could be OS specific
402     string = GrouperClientUtils.replace(string, "$newline$", "\n"); 
403     return string;
404   }

453   protected static enum ConfigFileType {
454     
455     /**
456      * get a config file from the filesystem
457      */
458     FILE {
459 
460       @Override
461       public InputStream inputStream(String configFileTypeConfig,
462           ConfigPropertiesCascadeBase configPropertiesCascadeBase) {
463         File file = new File(configFileTypeConfig);
464         if (!file.exists() || !file.isFile()) {
465           throw new RuntimeException("Cant find config file from filesystem path: " + configFileTypeConfig);
466         }
467         try {
468           return new FileInputStream(file);
469         } catch (Exception e) {
470           throw new RuntimeException("Problem reading config file from filesystem path: " + file.getAbsolutePath(), e);
471         }
472       }
473     },
474     
475     /**
476      * get a config file from the classpath
477      */
478     CLASSPATH {
479 
480       /**
481        * 
482        */
483       @Override
484       public InputStream inputStream(String configFileTypeConfig,
485           ConfigPropertiesCascadeBase configPropertiesCascadeBase) {
486         URL url = GrouperClientUtils.computeUrl(configFileTypeConfig, true);
487         Exception exception = null;
488         if (url != null) {
489           try {
490             return url.openStream();
491           } catch (Exception e) {
492             exception = e;
493           }
494         }
495         
496         //if we didnt get there yet, lets look for a companion jar
497         Class<?> classInJar = configPropertiesCascadeBase.getClassInSiblingJar();
498         if (classInJar != null) {
499           File jarFile = classInJar == null ? null : GrouperClientUtils.jarFile(classInJar);
500           File parentDir = jarFile == null ? null : jarFile.getParentFile();
501           String fileName = parentDir == null ? null 
502               : (GrouperClientUtils.stripLastSlashIfExists(GrouperClientUtils.fileCanonicalPath(parentDir)) + File.separator + configFileTypeConfig);
503           File configFile = fileName == null ? null 
504               : new File(fileName);
505           
506           //looks like we have a match
507           if (configFile != null && configFile.exists() && configFile.isFile()) {
508             try {
509               return new FileInputStream(configFile);
510             } catch (Exception e) {
511               logError("Cant read config file: " + configFile.getAbsolutePath(), e);
512             }
513           }
514         }
515         //see if it is next to the jar
516         throw new RuntimeException("Cant find config file from classpath: " + configFileTypeConfig, exception);
517       }
518     };

592     /**
593      * 
594      * @param configFileFullConfig
595      */
596     public ConfigFile(String configFileFullConfig) {
597       
598       this.originalConfig = configFileFullConfig;
599       
600       int colonIndex = configFileFullConfig.indexOf(':');
601       
602       if (colonIndex == -1) {
603         throw new RuntimeException("Config file spec needs the type of config and a colon, e.g. file:/some/path/config.properties  '" + configFileFullConfig + "'");
604       }
605       
606       //lets get the type
607       String configFileTypeString = GrouperClientUtils.trim(GrouperClientUtils.prefixOrSuffix(configFileFullConfig, ":", true));
608       
609       if (GrouperClientUtils.isBlank(configFileTypeString)) {
610         throw new RuntimeException("Config file spec needs the type of config and a colon, e.g. file:/some/path/config.properties  '" + configFileFullConfig + "'");
611       }
612       
613       try {
614         this.configFileType = ConfigFileType.valueOfIgnoreCase(configFileTypeString);
615       } catch (Exception e) {
616         throw new RuntimeException("Config file spec needs the type of config and a colon, e.g. file:/some/path/config.properties  '" + configFileFullConfig + "', " + e.getMessage(), e);
617       }
618       
619       this.configFileTypeConfig = GrouperClientUtils.trim(GrouperClientUtils.prefixOrSuffix(configFileFullConfig, ":", false));
620       
621     }

660   /**
661    * get the config object from config files
662    * @return the config object
663    */
664   protected ConfigPropertiesCascadeBase retrieveFromConfigFiles() {
665     
666     //lets get the config hierarchy...
667     //properties from override first
668     Properties mainConfigFile = propertiesFromResourceName(this.getMainConfigClasspath(), false);
669 
670     String secondsToCheckConfigString = null;
671     
672     String overrideFullConfig = null;
673     
674     if (mainConfigFile != null) {
675       overrideFullConfig = mainConfigFile.getProperty(this.getHierarchyConfigKey());
676       secondsToCheckConfigString = mainConfigFile.getProperty(this.getSecondsToCheckConfigKey());
677     }
678     
679     //if couldnt find it from the override, get from example
680     if (GrouperClientUtils.isBlank(overrideFullConfig) || GrouperClientUtils.isBlank(secondsToCheckConfigString)) {
681       
682       Properties mainExampleConfigFile = propertiesFromResourceName(this.getMainExampleConfigClasspath(), false);
683       
684       if (mainExampleConfigFile != null) {
685         
686         if (GrouperClientUtils.isBlank(overrideFullConfig)) {
687           overrideFullConfig = mainExampleConfigFile.getProperty(this.getHierarchyConfigKey());
688         }
689         if (GrouperClientUtils.isBlank(secondsToCheckConfigString)) {
690           secondsToCheckConfigString = mainExampleConfigFile.getProperty(this.getSecondsToCheckConfigKey());
691         }
692 
693       }
694       
695     }
696 
697     //if hasnt found yet, there is a problem
698     if (GrouperClientUtils.isBlank(overrideFullConfig)) {
699       throw new RuntimeException("Cant find the hierarchy config key: " + this.getHierarchyConfigKey() 
700           + " in config files: " + this.getMainConfigClasspath()
701           + " or " + this.getMainExampleConfigClasspath());
702     }
703     
704     //if hasnt found yet, there is a problem
705     if (GrouperClientUtils.isBlank(secondsToCheckConfigString)) {
706       throw new RuntimeException("Cant find the seconds to check config key: " + this.getSecondsToCheckConfigKey() 
707           + " in config files: " + this.getMainConfigClasspath()
708           + " or " + this.getMainExampleConfigClasspath());
709     }
710 
711     //make a new return object based on this class
712     ConfigPropertiesCascadeBase result = GrouperActivemqUtils.newInstance(this.getClass(), true);
713 
714     try {
715       result.timeToCheckConfigSeconds = GrouperClientUtils.intValue(secondsToCheckConfigString);
716     } catch (Exception e) {
717       throw new RuntimeException("Invalid integer seconds to check config config value: " + secondsToCheckConfigString
718           + ", key: " + this.getSecondsToCheckConfigKey() 
719           + " in config files: " + this.getMainConfigClasspath()
720           + " or " + this.getMainExampleConfigClasspath());
721       
722     }
723     
724     //ok, we have the config file list...
725     //lets get this into a comma separated list
726     List<String> overrideConfigStringList = GrouperClientUtils.splitTrimToList(overrideFullConfig, ",");
727 
728     result.configFiles = new ArrayList<ConfigFile>();
729 
730     for (String overrideConfigString : overrideConfigStringList) {
731       
732       ConfigFile configFile = new ConfigFile(overrideConfigString);
733       result.configFiles.add(configFile);
734       
735       //lets append the properties
736       InputStream inputStream = configFile.getConfigFileType().inputStream(configFile.getConfigFileTypeConfig(), this);
737       
738       try {
739         result.properties.load(inputStream);
740       } catch (Exception e) {
741         throw new RuntimeException("Problem loading properties: " + overrideConfigString, e);
742       }
743     }
744     
745     return result;
746     
747   }

784   /**
785    * see if there is one in cache, if so, use it, if not, get from config files
786    * @return the config from file or cache
787    */
788   protected ConfigPropertiesCascadeBase retrieveFromConfigFileOrCache() {
789     
790     if (configFileCache == null) {
791       configFileCache = 
792         new HashMap<Class<? extends ConfigPropertiesCascadeBase>, ConfigPropertiesCascadeBase>();
793     }
794     
795     ConfigPropertiesCascadeBase configObject = configFileCache.get(this.getClass());
796     
797     if (configObject == null) {
798       
799       logDebug("Config file has not be created yet, will create now: " + this.getMainConfigClasspath());
800       
801       configObject = retrieveFromConfigFiles();
802       configFileCache.put(this.getClass(), configObject);
803       
804     } else {
805       
806       //see if that much time has passed
807       if (configObject.needToCheckIfFilesNeedReloading()) {
808         
809         synchronized (configObject) {
810           
811           configObject = configFileCache.get(this.getClass());
812           
813           //check again in case another thread did it
814           if (configObject.needToCheckIfFilesNeedReloading()) {
815             
816             if (configObject.filesNeedReloadingBasedOnContents()) {
817               configObject = retrieveFromConfigFiles();
818               configFileCache.put(this.getClass(), configObject);
819             }
820           }
821         }
822       }
823     }
824     
825     return configObject;
826   }

828   /**
829    * 
830    * @return true if need to reload this config, false if not
831    */
832   protected boolean needToCheckIfFilesNeedReloading() {
833     
834     //get the time that this was created
835     long lastCheckedTimeLocal = this.getLastCheckedTime();
836     
837     //get the timeToCheckSeconds if different
838     int timeToCheckSeconds = this.getTimeToCheckConfigSeconds();
839     
840     //never reload.  0 means reload all the time?
841     if (timeToCheckSeconds < 0) {
842       return false;
843     }
844     
845     //see if that much time has passed
846     if (System.currentTimeMillis() - lastCheckedTimeLocal > timeToCheckSeconds * 1000) {
847       return true;
848     }
849     return false;
850 
851   }
852 
853   /**
854    * 
855    * @return true if need to reload this config, false if not
856    */
857   protected boolean filesNeedReloadingBasedOnContents() {
858     try {
859       //lets look at all the files and see if they have changed...
860       for (ConfigFile configFile : this.configFiles) {
861         if (!GrouperClientUtils.equals(configFile.getContents(), configFile.retrieveContents(this))) {
862           logInfo("Contents changed for config file, reloading: " + configFile.getOriginalConfig(), null);
863           return true;
864         }
865       }
866     } catch (Exception e) {
867       //lets log and return the old one
868       logError("Error checking for changes in configs (will use previous version): " + this.getMainConfigClasspath(), e);
869     } finally {
870       //reset the time so we dont have to check again for a while
871       this.lastCheckedTime = System.currentTimeMillis();
872     }
873     return false;
874   }

1143       return false;
1144     }

1146

1150     }
1151     
1152     String extraError = "";
1153     try {
1154       
1155       
1156       Class<?> theClass = GrouperClientUtils.forName(value);
1157       if (classType.isAssignableFrom(theClass)) {
1158         return true;
1159       }
1160       extraError = " does not derive from class: " + classType.getSimpleName();
1161       
1162     } catch (Exception e) {
1163       extraError = ", " + GrouperClientUtils.getFullStackTrace(e);
1164     }
1165     String error = "Cant process property " + key + " in resource: " + this.getMainConfigClasspath() + ", the current" +
1166         " value is '" + value + "', which should be of type: " 
1167         + classType.getName() + extraError;
1168     System.err.println("Grouper error: " + error);
1169     LOG.error(error);
1170     return false;
