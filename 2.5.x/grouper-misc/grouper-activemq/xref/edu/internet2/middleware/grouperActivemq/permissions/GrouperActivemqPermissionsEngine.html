44  /**
45   * keep track of all permissions.  It retrieves permissions from Grouper, 
46   * caches them in memory, and periodically stores them to disk too.  There are two files 
47   * on disk, in case one doesnt get written fully.  At the end of the file is a success
48   * flag so we know it is complete.  It is failsafe, if Grouper has a problem, and there
49   * is a copy in memory or disk, then it will still work.  I believe the disk store is
50   * only read on startup, but Im not sure.
51   */
52  public class GrouperActivemqPermissionsEngine implements Job, StatefulJob {

65

67      
68      GrouperActivemqPermissionsEngine grouperActivemqPermissionsEngineLocal = grouperActivemqPermissionsEngine; 
69      
70      if (grouperActivemqPermissionsEngineLocal == null) {
71        synchronized (GrouperActivemqPermissionsEngine.class) {
72          grouperActivemqPermissionsEngineLocal = grouperActivemqPermissionsEngine;
73          if (grouperActivemqPermissionsEngineLocal == null) {
74  
75            performFullRefresh();
76            
77            grouperActivemqPermissionsEngineLocal = grouperActivemqPermissionsEngine;

80              throw new RuntimeException("No permissions from Grouper or local cache????");
81            }
82          }
83          
84        }
85      }    
86      return grouperActivemqPermissionsEngineLocal;
87

127   /**
128    * all permissions for activemq.  See if the user has the action and destination.
129    * Note, if there is an internal permissions in ActiveMQ for the user, it is allowed (if supposed to be),
130    * and if there is a hierarchical "inherit" permission for an ancestor, it is permitted
131    * @param user loginid to activemq
132    * @param action action to check (note, if you just have sendMessage,
133    * it will check the inherit destinations too
134    * @param destination to check
135    * @return the permissions
136    */
137   public static boolean hasPermission(String user, GrouperActivemqPermissionAction action,
138       String destination) {
139     
140     if (destination.startsWith("ActiveMQ.Advisory.Consumer.Queue.")) {
141       return hasPermission(user, action, destination.substring("ActiveMQ.Advisory.Consumer.Queue.".length()));
142     }
143     
144     //lets see if someone has a permission from cache
145     GrouperActivemqPermissionsEngine engine = permissions();
146     
147     boolean useCache = cachePermissionsMinutes() != 0;
148     
149     GrouperActivemqPermissionivemqPermission.html#GrouperActivemqPermission">GrouperActivemqPermission grouperActivemqPermission = new GrouperActivemqPermission(action, 
150         destination);
151     
152     if (whitelist.contains(grouperActivemqPermission)) {
153       return true;
154     }
155     
156     if (useCache) {
157       
158       Map<GrouperActivemqPermission, Boolean> userPermissions = engine.getCachedPermissionDecisions().get(user);
159       
160       if (userPermissions != null) {
161         
162         Boolean result = userPermissions.get(grouperActivemqPermission);
163         if (result != null) {
164           return result;
165         }
166       }
167     }
168 
169     //not in cache, get the real answer
170     boolean result = engine.hasPermissionNoCache(user, action, destination);
171     
172     if (useCache) {
173       
174       ExpirableCache<String, Map<GrouperActivemqPermission, Boolean>> cachedPermissionDecisions2 = engine.getCachedPermissionDecisions();
175       
176       if (cachedPermissionDecisions2 != null) {
177         Map<GrouperActivemqPermission, Boolean> userPermissions = cachedPermissionDecisions2.get(user);
178         
179         if (userPermissions == null) {
180           userPermissions = new HashMap<GrouperActivemqPermission, Boolean>();
181           cachedPermissionDecisions2.put(user, userPermissions);
182         }
183         
184         userPermissions.put(grouperActivemqPermission, result);
185         
186       }      
187     }
188     
189     return result;
190   }

212   /**
213    * all permissions for activemq, can easily be testable
214    * @param allPermissions 
215    * @param user loginid to activemq
216    * @param action action to check (note, if you just have sendMessage, it will check the inherit destinations too
217    * @param destination to check
218    * @return the permissions
219    */
220   static boolean hasPermissionHelper(Map<String, Set<GrouperActivemqPermission>> allPermissions, String user, GrouperActivemqPermissionAction action, String destination) {
221     
222     //get the permissions for a user
223     Set<GrouperActivemqPermission> permissionsForUser = allPermissions.get(user);
224     
225     if (permissionsForUser == null) {
226       return false;
227     }
228     
229     for (GrouperActivemqPermission permission : permissionsForUser) {
230       
231       //see if that is the exact permission
232       if (permission.getAction() == action && GrouperClientUtils.equals(permission.getDestination(), destination)) {
233         return true;
234       }
235 
236       //lets see if this is a inherited action
237       if (GrouperClientUtils.equals(permission.getDestination(), destination)
238           //if the permission destination is further up the hierarchy then the 
239           //requested destination, then we are all good
240           || destination.startsWith(permission.getDestination()+":")) {
241         if (action.actionAllowedIfArgumentAllowed(permission.getAction())) {
242           return true;
243         }
244       }
245       
246     }
247     
248     return false;
249     
250   }

279 
280     if (started) {
281       return;
282     }
283     
284     //do not synchronize the entire method so that all calls are not synchronized
285     synchronized(GrouperActivemqPermissionsEngine.class) {
286       
287       if (started) {
288         return;
289       }
290 
291       started = true;
292       
293       log.debug("Starting grouper activemq permissions daemon");
294 
295       //schedule full refresh job on startup
296       try {
297         performFullRefresh();
298       } catch (RuntimeException e) {
299         //do not throw so this is always available if Grouper is down
300         log.error("Error on startup", e);
301       }
302 
303       try {
304         scheduleQuartzJob();
305       } catch (RuntimeException e) {
306         //do not throw so this is always available if Grouper is down.  Though not sure why
307         //quartz schedule would fail... hmm
308         log.error("Error on startup", e);
309       }
310       Thread thread = new Thread(new Runnable() {
311 
312         @Override
313         public void run() {
314           try {
315             //do xmpp loop
316             xmppLoop();
317           } catch (Exception e) {
318             log.error("Error with xmpp", e);
319           }
320         }
321         
322       });
323       thread.start();
324     }

358    * do a full refresh in one minute (batch subsequent requests)
359    */
360   static void scheduleFullRefresh() {
361 
362     //if it is already scheduled, then we are all good
363     if (timer != null) {
364 
365       log.debug("Job is already scheduled at " + new Date(timerScheduledFor).toString()
366           + ", exiting");
367 
368       return;
369     }
370     timer = new Timer(true);
371 
372     int timeInFuture = 1000 * 60;
373 
374     timerScheduledFor = System.currentTimeMillis() + timeInFuture;
375 
376     //schedule in 60 seconds
377     timer.schedule(new TimerTask() {
378 
379       @Override
380       public void run() {
381         try {
382           performFullRefresh();
383         } catch (RuntimeException e) {
384           log.error("Error performing refresh", e);
385           throw e;
386         }
387       }
388     }, timeInFuture);
389   }

445   static synchronized void performFullRefresh() {
446     
447     if (log.isDebugEnabled()) {
448       log.debug("Performing full refresh of Grouper ActiveMQ permissions");
449     }
450     
451     Map<String, Set<GrouperActivemqPermission>> userToPermissionMap = null;
452     
453     try {
454       userToPermissionMap = retrievePermissionsFromGrouper();
455       
456     } catch (Exception e) {
457       log.error("Error getting objects from grouper", e);
458     }
459 
460     //got it, save to cache and file
461     if (userToPermissionMap != null) {
462       
463       //rewrite the cache
464       grouperActivemqPermissionsEngine = new GrouperActivemqPermissionsEngine(userToPermissionMap);
465       writePermissionsToFile(userToPermissionMap);
466 
467     } else {
468       
469       //couldnt get from grouper, read again from file
470       userToPermissionMap = readPermissionsFromFileCache();
471       grouperActivemqPermissionsEngine = new GrouperActivemqPermissionsEngine(userToPermissionMap);
472       
473     }
474     
475   }

481   static synchronized Map<String, Set<GrouperActivemqPermission>> retrievePermissionsFromGrouper() {
482 
483     //let another timer be scheduled, whether from schedule or xmpp
484     timer = null;
485 
486     GcGetPermissionAssignments gcGetPermissionAssignments = new GcGetPermissionAssignments();
487     
488     {
489       String nameOfAttributeDefsString = GrouperActivemqConfig.retrieveConfig().propertyValueStringRequired("grouperActivemq.permissionDefinitions");
490       
491       List<String> namesOfAttributeDefs = GrouperClientUtils.splitTrimToList(nameOfAttributeDefsString, ","); 
492 
493       for (String nameOfAttributeDef : namesOfAttributeDefs) {
494         gcGetPermissionAssignments.addAttributeDefName(nameOfAttributeDef);
495       }
496       
497     }
498     
499     {
500       
501       String theRoleNamesString = GrouperActivemqConfig.retrieveConfig().propertyValueStringRequired("grouperActivemq.roleNames");
502 
503       List<String> roleNames = GrouperClientUtils.splitTrimToList(theRoleNamesString, ","); 
504       
505       for (String roleName : roleNames) {
506         gcGetPermissionAssignments.addRoleName(roleName);
507       }
508       
509     }
510     Map<String, String> sourceToLoginIdAttributeName = GrouperActivemqConfig
511       .retrieveConfig().subjectSourceToLoginAttributeName();
512     
513     for (String loginAttributeName : sourceToLoginIdAttributeName.values()) {
514       if (!GrouperClientUtils.equals("id", loginAttributeName) && !GrouperClientUtils.equals("name", loginAttributeName)) {
515         gcGetPermissionAssignments.addSubjectAttributeName(loginAttributeName);
516       }
517       //hmm, cant add source
518     }
519     
520     //add actions, dont worry about rollups
521     for (GrouperActivemqPermissionAction grouperActivemqPermissionAction : GrouperActivemqPermissionAction.values()) {
522       gcGetPermissionAssignments.addAction(grouperActivemqPermissionAction.name());
523     }
524     
525     WsGetPermissionAssignmentsResults wsGetPermissionAssignmentsResults = gcGetPermissionAssignments.execute();
526 
527     if (log.isDebugEnabled()) {
528       log.debug("Received "
529             + GrouperClientUtils.length(wsGetPermissionAssignmentsResults
530                 .getWsPermissionAssigns())
531             + " permission entries from Grouper");
532     }
533 
534     WsSubject[] wsSubjects = wsGetPermissionAssignmentsResults.getWsSubjects();
535     
536     if (GrouperActivemqConfig.retrieveConfig().propertyValueBoolean("grouperActivemq.separateSubjectQuery")) {
537 
538       wsSubjects = retrieveSubjectsFromServer(wsGetPermissionAssignmentsResults, sourceToLoginIdAttributeName.values());
539       
540     }
541     
542     Map<String, Set<GrouperActivemqPermission>> userToPermissions = new TreeMap<String, Set<GrouperActivemqPermission>>();
543     
544     Map<String, String> loginidToSource = new HashMap<String, String>();
545     
546     String folderPrefix = GrouperActivemqConfig.retrieveConfig().propertyValueString("grouperActivemq.folderPrefix");
547     String requireBaseFolder = GrouperActivemqConfig.retrieveConfig().propertyValueString("grouperActivemq.requireBaseFolder");
548     
549     for (WsPermissionAssign wsPermissionAssign : GrouperClientUtils.nonNull(
550           wsGetPermissionAssignmentsResults.getWsPermissionAssigns(),
551         WsPermissionAssign.class)) {
552       
553       //get the subject
554       WsSubject wsSubject = retrieveSubject(
555             wsSubjects, wsPermissionAssign.getSourceId(),
556             wsPermissionAssign.getSubjectId());
557 
558       if (wsSubject == null) {
559         throw new RuntimeException("Why is wsSubject null??? "
560             + wsPermissionAssign.getSourceId()
561             + wsPermissionAssign.getSubjectId());
562       }
563 
564       String permissionName = wsPermissionAssign.getAttributeDefNameName();
565       
566       //lets throw out permissions that arent in the right namespace
567       if (!GrouperClientUtils.isBlank(requireBaseFolder)) {
568         if (!permissionName.startsWith(requireBaseFolder)) {
569           if (log.isDebugEnabled()) {
570             log.debug("Skipping " + permissionName + " since it is not in requireBaseFolder: " + requireBaseFolder);
571           }
572           continue;
573         }
574       }
575       
576       //strip out the prefix if configured
577       if (!GrouperClientUtils.isBlank(folderPrefix)) {
578         if (!permissionName.startsWith(folderPrefix)) {
579           if (log.isDebugEnabled()) {
580             log.debug("Skipping " + permissionName + " since it is not in folderPrefix: " + folderPrefix);
581           }
582           continue;
583         }
584         
585         permissionName = permissionName.substring(folderPrefix.length());
586         
587       }
588       
589       String loginIdAttributeName = sourceToLoginIdAttributeName.get(wsSubject.getSourceId());
590       
591       if (GrouperClientUtils.isBlank(loginIdAttributeName)) {
592         String error = "Why is there no grouperActivemq.subjectSource.X.subjectAttributeForLogin for the source: " + wsSubject.getSourceId();
593         log.error(error);
594         throw new RuntimeException(error);
595       }
596       
597       //lets get the loginid
598       String loginid = GrouperActivemqUtils.subjectAttributeValue(wsSubject,
599             wsGetPermissionAssignmentsResults.getSubjectAttributeNames(), loginIdAttributeName);

608       }

611       {
612         String expectedSource = loginidToSource.get(wsSubject.getSourceId());

614           loginidToSource.put(loginid, wsSubject.getSourceId());
615         } else {
616           if (!GrouperClientUtils.equals(expectedSource, wsSubject.getSourceId())) {
617             String error = "Error, there is a subject with login id: " + loginid + " in source: " 
618               + expectedSource + ", and in source: " + wsSubject.getSourceId();
619             log.error(error);
620             throw new RuntimeException(error);
621           }
622         }
623         
624       }
625       
626       //ok, we have login, permision, action, lets keep track
627       Set<GrouperActivemqPermission> permissionsForUser = userToPermissions.get(loginid);
628       
629       if (permissionsForUser == null) {
630         permissionsForUser = new TreeSet<GrouperActivemqPermission>();
631         userToPermissions.put(loginid, permissionsForUser);
632       }
633       
634       GrouperActivemqPermissionivemqPermission.html#GrouperActivemqPermission">GrouperActivemqPermission grouperActivemqPermission = new GrouperActivemqPermission();
635       GrouperActivemqPermissionAction grouperActivemqPermissionAction = GrouperActivemqPermissionAction.valueOfIgnoreCase(wsPermissionAssign.getAction(), true);
636       grouperActivemqPermission.setAction(grouperActivemqPermissionAction);
637       grouperActivemqPermission.setDestination(permissionName);
638       
639       permissionsForUser.add(grouperActivemqPermission);
640     }
641     
642     //update the memory cache
643     //cachedGrouperActivemqPermissions = userToPermissions;
644     
645     return userToPermissions;

649   /**
650    * write permissions to a file
651    * @param userToPermissions
652    */
653   static void writePermissionsToFile(Map<String, Set<GrouperActivemqPermission>> userToPermissions) {
654     
655     //lets make the permissions file
656     StringBuilder fileContents = new StringBuilder(
657         "# automatically generated from Grouper-ActiveMQ connector...\n\n");
658 
659     //  # automatically generated from Grouper-ActiveMQ connector
660     //
661     //  activemq.permission.0.user = mchyzer
662     //  activemq.permission.0.permission.0 = read__whatever
663     //  activemq.permission.0.permission.1 = write__whatever:whatever
664     //
665     //  # if this is here, the file is complete
666     //  activemq.permissionSuccess = true
667     
668     int userIndex = 0;
669     for (String userId : userToPermissions.keySet()) {
670       fileContents.append("activemq.permission.").append(userIndex).append(".user = ").append(userId).append("\n");
671       
672       Set<GrouperActivemqPermission> grouperActivemqPermissions = userToPermissions.get(userId);
673       
674       int permissionIndex = 0;
675       for (GrouperActivemqPermission grouperActivemqPermission : grouperActivemqPermissions) {
676         fileContents.append("activemq.permission.").append(userIndex).append(".permission.")
677           .append(permissionIndex).append(" = ").append(grouperActivemqPermission.getAction())
678           .append("__").append(grouperActivemqPermission.getDestination()).append("\n");
679         permissionIndex++;
680       }
681       
682       userIndex++;
683     }
684 
685     fileContents.append("\n# if this is here, the file is complete\nactivemq.permissionSuccess = true\n");
686     
687     if (log.isDebugEnabled()) {
688       log.debug("Generated file has "
689             + fileContents.length()
690             + " bytes");
691     }
692 
693     File fileToSave = fileToSave();
694 
695     //save this to file (try 3 times)
696     for (int i = 0; i < 3; i++) {
697 
698       try {
699 
700         boolean updated = GrouperClientUtils.saveStringIntoFile(fileToSave,
701             fileContents.toString(), true, true);
702         if (updated) {
703           log.debug("File: " + fileToSave.getAbsolutePath()
704               + " was saved since there were updates from Grouper");
705         } else {
706           log.debug("File: " + fileToSave.getAbsolutePath()
707               + " was not saved since there were no changes from Grouper");
708         }
709 
710         //we are done
711         break;
712       } catch (Exception e) {
713         log.error("Error saving file", e);
714       }
715       GrouperClientUtils.sleep(2000);
716 
717     }
718 
719   }

782   /**
783    * pattern to match a permission in the permissions file: write__whatever:whatever
784    */
785   private static Pattern permissionFilePattern = Pattern.compile("^(.*?)__(.*)$");
786   
787   /**
788    * read permissions from file cache.
789    * 
790    * @return the permissions or null if not found
791    */
792   static Map<String, Set<GrouperActivemqPermission>> readPermissionsFromFileCache() {
793     
794     Properties properties = fileToRead();
795     
796     if (properties == null) {
797       return null;
798     }
799     
800     Map<String, Set<GrouperActivemqPermission>> result = new HashMap<String, Set<GrouperActivemqPermission>>();
801     
802     //we have properties.  for loop so we dont have an endless loop somehow
803     for (int userIndex=0;userIndex<properties.size();userIndex++) {
804       
805       //  # automatically generated from Grouper-ActiveMQ connector
806       //
807       //  activemq.permission.0.user = mchyzer
808       //  activemq.permission.0.permission.0 = read__whatever
809       //  activemq.permission.0.permission.1 = write__whatever:whatever
810       //
811       //  # if this is here, the file is complete
812       //  activemq.permissionSuccess = true
813       
814       String user = GrouperClientUtils.propertiesValue(properties, "activemq.permission." + userIndex + ".user");
815       if (GrouperClientUtils.isBlank(user)) {
816         break;
817       }
818       
819       Set<GrouperActivemqPermission> permissions = new HashSet<GrouperActivemqPermission>();
820       
821       result.put(user, permissions);
822       
823       for (int permissionIndex=0;permissionIndex<10000;permissionIndex++) {
824         
825         String permissionString = GrouperClientUtils.propertiesValue(properties, 
826             "activemq.permission." + userIndex + ".permission." + permissionIndex);
827         
828         if (GrouperClientUtils.isBlank(permissionString)) {
829           break;
830         }
831         
832         Matcher matcher = permissionFilePattern.matcher(permissionString);
833         if (!matcher.matches()) {
834           //hmmm, what to do...
835           log.error("Error: invalid permission in file: " + permissionString);
836           //i guess keep going
837           continue;
838         }
839 
840         String actionString = matcher.group(1);
841         String permissionResource = matcher.group(2);
842         
843         GrouperActivemqPermissionAction grouperActivemqPermissionAction 
844           = GrouperActivemqPermissionAction.valueOfIgnoreCase(actionString, false);
845         
846         if (grouperActivemqPermissionAction == null) {
847           log.error("Error: invalid action in file: " + actionString + ", " + permissionString);
848           continue;
849         }
850 
851         //ok, we have everything
852         GrouperActivemqPermission grouperActivemqPermission = 
853           new GrouperActivemqPermission(grouperActivemqPermissionAction, permissionResource);
854         
855         permissions.add(grouperActivemqPermission);
856         
857       }
858       
859     }
860     
861     return result;
862   }

864   /**
865    * file to read, newest activemqGrouperPermissions.1.properties, 
866    * or activemqGrouperPermissions.2.properties
867    * @return the properties
868    */
869   static Properties fileToRead() {
870     
871     String permissionsDirString = GrouperActivemqConfig.retrieveConfig()
872       .propertyValueStringRequired("grouperActivemq.permissionsCacheDirectory");
873     
874     File permissionsDir = new File(permissionsDirString);
875     if (!permissionsDir.exists() || !permissionsDir.isDirectory()) {
876       throw new RuntimeException("Configured permissions cache dir doesnt exist: " +
877           "grouperActivemq.permissionsCacheDirectory: " + permissionsDir.getAbsolutePath());
878     }
879 
880     if (!permissionsDirString.endsWith("/") && !permissionsDirString.endsWith("\\")) {
881       permissionsDirString += File.separator;
882     }
883     
884     //two files:
885     File permissionsFile1 = new File(permissionsDirString + "activemqGrouperPermissions.1.properties");
886     File permissionsFile2 = new File(permissionsDirString + "activemqGrouperPermissions.2.properties");
887 
888     boolean permissionsFile1candidate = true;
889     boolean permissionsFile2candidate = true;
890     
891     //let's see if there is a #1 file
892     if (!permissionsFile1.exists()) {
893       permissionsFile1candidate = false;
894     }
895     
896     //let's see if there is a #2 file
897     if (!permissionsFile2.exists()) {
898       permissionsFile2candidate = false;
899     }
900     
901     Properties firstFileProperties = null;
902     
903     if (permissionsFile1candidate) {
904       //let's see if the first file is not complete
905       firstFileProperties = GrouperClientUtils.propertiesFromFile(permissionsFile1);
906       if (!GrouperClientUtils.propertiesValueBoolean(firstFileProperties, "activemq.permissionSuccess", false)) {
907         permissionsFile1candidate = false;
908       }
909       
910     }    
911     
912     Properties secondFileProperties = null;
913     if (permissionsFile2candidate) {
914       //let's see if the second file is not complete
915       secondFileProperties = GrouperClientUtils.propertiesFromFile(permissionsFile2);
916       if (!GrouperClientUtils.propertiesValueBoolean(secondFileProperties, "activemq.permissionSuccess", false)) {
917         permissionsFile2candidate = false;
918       }
919 
920     }
921 
922     if (!permissionsFile1candidate && !permissionsFile2candidate) {
923       return null;
924     }
925 
926     if (!permissionsFile1candidate) {
927       return firstFileProperties;
928     }
929 
930     if (!permissionsFile2candidate) {
931       return secondFileProperties;
932     }
933 
934     //lets see which file is newer, return that one
935     if (permissionsFile1.lastModified() > permissionsFile2.lastModified()) {
936       return firstFileProperties;
937     }
938 
939     return secondFileProperties;
940 
941   }

997    * @param wsGetPermissionAssignmentsResults

1001   static WsSubject[] retrieveSubjectsFromServer(
1002       WsGetPermissionAssignmentsResults wsGetPermissionAssignmentsResults, Collection<String> subjectAttributeNames) {
1003     
1004     //make sure no dupes
1005     Set<MultiKey> subjectSourceAndIds = new HashSet<MultiKey>();
1006     
1007     for (WsPermissionAssign wsPermissionAssign : wsGetPermissionAssignmentsResults.getWsPermissionAssigns()) {
1008       subjectSourceAndIds.add(new MultiKey(wsPermissionAssign.getSourceId(), wsPermissionAssign.getSubjectId()));
1009     }
1010     GcGetSubjects gcGetSubjects = new GcGetSubjects();
1011

1013       gcGetSubjects.addSubjectAttributeName(subjectAttributeName);
1014     }
1015     
1016     //add those to the get subjects request
1017     for (MultiKey subjectSourceAndId : subjectSourceAndIds) {
1018       gcGetSubjects.addWsSubjectLookup(new WsSubjectLookup((String)subjectSourceAndId.getKey(1),
1019           (String)subjectSourceAndId.getKey(0), null));
1020     }
1021     
1022     WsGetSubjectsResults wsGetSubjectsResults = gcGetSubjects.execute();
1023     
1024     return wsGetSubjectsResults.getWsSubjects();

1030    * @param subjects
1031    * @param sourceId
1032    * @param subjectId
1033    * @return probably shouldnt be null, but if cant be found, then will be null

1035   static WsSubject retrieveSubject(WsSubject[] subjects, String sourceId,
1036       String subjectId) {
1037

1040            && GrouperClientUtils.equals(subjectId, wsSubject.getId())) {

1043     }
