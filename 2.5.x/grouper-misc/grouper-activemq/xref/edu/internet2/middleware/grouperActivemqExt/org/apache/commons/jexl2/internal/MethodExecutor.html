39  /**
40   * Specialized executor to invoke a method on an object.
41   * @since 2.0
42   */
43  public final class MethodExecutor extends AbstractExecutor.Method {
44      /** Whether this method handles varargs. */
45      private final boolean isVarArgs;
46      /**
47       * Creates a new instance.
48       * @param is the introspector used to discover the method
49       * @param obj the object to find the method in
50       * @param name the method name
51       * @param args the method arguments
52       */
53      public MethodExecutor(Introspector is, Object obj, String name, Object[] args) {
54          super(obj.getClass(), discover(is, obj, name, args));
55          isVarArgs = method != null && isVarArgMethod(method);
56      }
57  
58      /**
59       * Invokes the method to be executed.
60       * @param o the object to invoke the method upon
61       * @param args the method arguments
62       * @return the result of the method invocation
63       * @throws IllegalAccessException Method is inaccessible.
64       * @throws InvocationTargetException Method body throws an exception.
65       */
66      @Override
67      public Object execute(Object o, Object[] args)
68          throws IllegalAccessException, InvocationTargetException  {
69          if (isVarArgs) {
70              Class<?>[] formal = method.getParameterTypes();
71              int index = formal.length - 1;
72              Class<?> type = formal[index].getComponentType();
73              if (args.length >= index) {
74                  args = handleVarArg(type, index, args);
75              }
76          }
77          if (method.getDeclaringClass() == ArrayListWrapper.class && o.getClass().isArray()) {
78              return method.invoke(new ArrayListWrapper(o), args);
79          } else {
80              return method.invoke(o, args);
81          }
82      }

134     /**
135      * Reassembles arguments if the method is a vararg method.
136      * @param type   The vararg class type (aka component type
137      *               of the expected array arg)
138      * @param index  The index of the vararg in the method declaration
139      *               (This will always be one less than the number of
140      *               expected arguments.)
141      * @param actual The actual parameters being passed to this method
142      * @return The actual parameters adjusted for the varargs in order
143      * to fit the method declaration.
144      */
145     protected Object[] handleVarArg(Class<?> type, int index, Object[] actual) {
146         final int size = actual.length - index;
147         // if no values are being passed into the vararg, size == 0
148         if (size == 1) {
149             // if one non-null value is being passed into the vararg,
150             // and that arg is not the sole argument and not an array of the expected type,
151             // make the last arg an array of the expected type
152             if (actual[index] != null) {
153                 Class<?> aclazz = actual[index].getClass();
154                 if (!aclazz.isArray() || !aclazz.getComponentType().equals(type)) {
155                     // create a 1-length array to hold and replace the last argument
156                     Object lastActual = Array.newInstance(type, 1);
157                     Array.set(lastActual, 0, actual[index]);
158                     actual[index] = lastActual;
159                 }
160             }
161             // else, the vararg is null and used as is, considered as T[]
162         } else {
163             // if no or multiple values are being passed into the vararg,
164             // put them in an array of the expected type
165             Object lastActual = Array.newInstance(type, size);
166             for (int i = 0; i < size; i++) {
167                 Array.set(lastActual, i, actual[index + i]);
168             }
169 
170             // put all arguments into a new actual array of the appropriate size
171             Object[] newActual = new Object[index + 1];
172             System.arraycopy(actual, 0, newActual, 0, index);
173             newActual[index] = lastActual;
174 
175             // replace the old actual array
176             actual = newActual;
177         }
178         return actual;
179     }

181    /**
182      * Determines if a method can accept a variable number of arguments.
183      * @param m a the method to check
184      * @return true if method is vararg, false otherwise
185      */
186     private static boolean isVarArgMethod(java.lang.reflect.Method m) {
187         Class<?>[] formal = m.getParameterTypes();
188         if (formal == null || formal.length == 0) {
189             return false;
190         } else {
191             Class<?> last = formal[formal.length - 1];
192             // if the last arg is an array, then
193             // we consider this a varargs method
194             return last.isArray();
195         }
196     }
