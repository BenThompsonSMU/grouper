166   /**
167    * construct based on attribute def name, assign all fields
168    * @param theAttributeDefName 
169    * @param wsAttributeDefNameLookup is the lookup to set looked up values
170    */
171   public WsAttributeDefName(AttributeDefName theAttributeDefName, WsAttributeDefNameLookup wsAttributeDefNameLookup) {
172     if (theAttributeDefName != null) {
173       this.setDescription(StringUtils.trimToNull(theAttributeDefName.getDescription()));
174       this.setDisplayName(theAttributeDefName.getDisplayName());
175       this.setName(theAttributeDefName.getName());
176       this.setUuid(theAttributeDefName.getId());
177       this.setExtension(theAttributeDefName.getExtension());
178       this.setDisplayExtension(theAttributeDefName.getDisplayExtension());
179       this.setAttributeDefId(theAttributeDefName.getAttributeDefId());
180       this.setAttributeDefName(theAttributeDefName.getAttributeDef().getName());
181       
182       //if greater then 2.2 then set id index
183       if (GrouperWsVersionUtils.retrieveCurrentClientVersion()
184           .greaterOrEqualToArg(GrouperVersion.valueOfIgnoreCase("v2_2_000"))) {
185         this.setIdIndex(theAttributeDefName.getIdIndex() == null ? null : theAttributeDefName.getIdIndex().toString());
186       }
187 
188     } else {
189       if (wsAttributeDefNameLookup != null) {
190         //no attributeDefName, set the look values so the caller can keep things in sync
191         this.setName(wsAttributeDefNameLookup.getName());
192         this.setUuid(wsAttributeDefNameLookup.getUuid());
193         this.setExtension(GrouperUtil.extensionFromName(wsAttributeDefNameLookup.getName()));
194         //if greater then 2.2 then set id index
195         if (GrouperWsVersionUtils.retrieveCurrentClientVersion()
196             .greaterOrEqualToArg(GrouperVersion.valueOfIgnoreCase("v2_2_000"))) {
197           this.setIdIndex(wsAttributeDefNameLookup.getIdIndex() == null ? null : wsAttributeDefNameLookup.getIdIndex().toString());
198         }
199 
200       }
201     }
202   }

204   /**
205    * construct based on pit attribute def name, assign all fields
206    * @param theAttributeDefName 
207    * @param wsAttributeDefNameLookup is the lookup to set looked up values
208    */
209   public WsAttributeDefName(PITAttributeDefName theAttributeDefName, WsAttributeDefNameLookup wsAttributeDefNameLookup) {
210     if (theAttributeDefName != null) {
211       this.setName(theAttributeDefName.getName());
212       this.setUuid(theAttributeDefName.getSourceId());
213       this.setExtension(GrouperUtil.extensionFromName(theAttributeDefName.getName()));
214       
215       PITAttributeDef theAttributeDef = PITAttributeDefFinder.findById(theAttributeDefName.getAttributeDefId(), false);
216       
217       if (theAttributeDef != null) {
218         this.setAttributeDefId(theAttributeDef.getSourceId());
219         this.setAttributeDefName(theAttributeDef.getName());
220       }
221       
222     } else {
223       if (wsAttributeDefNameLookup != null) {
224         //no attributeDefName, set the look values so the caller can keep things in sync
225         this.setName(wsAttributeDefNameLookup.getName());
226         this.setUuid(wsAttributeDefNameLookup.getUuid());
227         this.setExtension(GrouperUtil.extensionFromName(wsAttributeDefNameLookup.getName()));
228       }
229     }
230   }

333   public int compareTo(WsAttributeDefName o2) {
334     if (this == o2) {
335       return 0;
336     }
337     //lets by null safe here
338     if (this == null) {
339       return -1;
