233   /**
234    * prcess an exception, log, etc
235    * @param wsGetPermissionAssignmentsResultsCodeOverride
236    * @param theError
237    * @param e
238    */
239   public void assignResultCodeException(
240       WsGetPermissionAssignmentsResultsCode wsGetPermissionAssignmentsResultsCodeOverride, String theError,
241       Exception e) {
242 
243     if (e instanceof WsInvalidQueryException) {
244       wsGetPermissionAssignmentsResultsCodeOverride = GrouperUtil.defaultIfNull(
245           wsGetPermissionAssignmentsResultsCodeOverride, WsGetPermissionAssignmentsResultsCode.INVALID_QUERY);
246       //a helpful exception will probably be in the getMessage()
247       this.assignResultCode(wsGetPermissionAssignmentsResultsCodeOverride);
248       this.getResultMetadata().appendResultMessage(e.getMessage());
249       this.getResultMetadata().appendResultMessage(theError);
250       LOG.warn(e);
251 
252     } else {
253       wsGetPermissionAssignmentsResultsCodeOverride = GrouperUtil.defaultIfNull(
254           wsGetPermissionAssignmentsResultsCodeOverride, WsGetPermissionAssignmentsResultsCode.EXCEPTION);
255       LOG.error(theError, e);
256 
257       theError = StringUtils.isBlank(theError) ? "" : (theError + ", ");
258       this.getResultMetadata().appendResultMessage(
259           theError + ExceptionUtils.getFullStackTrace(e));
260       this.assignResultCode(wsGetPermissionAssignmentsResultsCodeOverride);
261 
262     }
263   }

406   /**
407    * pass in the attribute def ids that were found by inputs, and add the attribute
408    * def ids found by the attribute assign results
409    * @param usePIT 
410    * @param attributeDefIds
411    */
412   public void fillInAttributeDefs(boolean usePIT, Set<String> attributeDefIds) {
413     
414     Set<String> allAttributeDefIds = new HashSet<String>(GrouperUtil.nonNull(attributeDefIds));
415     
416     for (WsAttributeAssign wsAttributeAssign : GrouperUtil.nonNull(this.wsAttributeAssigns, WsAttributeAssign.class)) {
417       if (!StringUtils.isBlank(wsAttributeAssign.getAttributeDefId())) {
418         allAttributeDefIds.add(wsAttributeAssign.getAttributeDefId());
419       }
420       if (!StringUtils.isBlank(wsAttributeAssign.getOwnerAttributeDefId())) {
421         allAttributeDefIds.add(wsAttributeAssign.getOwnerAttributeDefId());
422       }
423     }
424     
425     //make sure all attr def names are there
426     for (WsAttributeDefName wsAttributeDefName : GrouperUtil.nonNull(this.wsAttributeDefNames, WsAttributeDefName.class) ) {
427       allAttributeDefIds.add(wsAttributeDefName.getAttributeDefId());
428     }
429     
430     //these should be there, but try again
431     for (WsPermissionAssign wsPermissionAssign : GrouperUtil.nonNull(this.wsPermissionAssigns, WsPermissionAssign.class)) {
432       if (!StringUtils.isBlank(wsPermissionAssign.getAttributeDefId())) {
433         allAttributeDefIds.add(wsPermissionAssign.getAttributeDefId());
434       }
435     }
436 
437     if (!usePIT) {
438       //security is already checked, lets pass these through...
439       this.wsAttributeDefs = new WsAttributeDef[allAttributeDefIds.size()];
440       
441       int i = 0;
442       for (String wsAttributeDefId : allAttributeDefIds) {
443         AttributeDef attributeDef = AttributeDefFinder.findByIdAsRoot(wsAttributeDefId, true);
444         this.wsAttributeDefs[i] = new WsAttributeDef(attributeDef, null);
445         i++;
446       }
447     } else {
448       List<PITAttributeDef> entries = new ArrayList<PITAttributeDef>();
449       
450       for (String wsAttributeDefId : allAttributeDefIds) {
451         entries.addAll(PITAttributeDefFinder.findBySourceId(wsAttributeDefId, true));
452       }
453       
454       this.wsAttributeDefs = new WsAttributeDef[entries.size()];
455       
456       int i = 0;
457       for (PITAttributeDef pitAttributeDef : entries) {
458         this.wsAttributeDefs[i] = new WsAttributeDef(pitAttributeDef, null);      
459         i++;
460       }
461     }
462   }

514   /**
515    * pass in the subject lookups that were found by inputs, and add the subject ids 
516    * found by the attribute assign results
517    * @param subjectLookups
518    * @param extraMemberIds 
519    * @param includeSubjectDetail 
520    * @param theSubjectAttributeNames 
521    */
522   public void fillInSubjects(WsSubjectLookup[] subjectLookups, Set<String> extraMemberIds, 
523       boolean includeSubjectDetail, String[] theSubjectAttributeNames) {
524         
525     Set<Subject> allSubjects = new HashSet<Subject>();
526     
527     for (WsSubjectLookup wsSubjectLookup : GrouperUtil.nonNull(subjectLookups, WsSubjectLookup.class)) {
528       if (wsSubjectLookup == null) {
529         continue;
530       }
531       Subject subject = wsSubjectLookup.retrieveSubject();
532       if (subject != null) {
533         if (!SubjectHelper.inList(allSubjects, subject)) {
534           allSubjects.add(subject);
535         }
536       }
537     }
538     
539     //process extra ones e.g. from list of any memberships passed in
540     for (String memberId : GrouperUtil.nonNull(extraMemberIds)) {
541       Member member = MemberFinder.findByUuid(GrouperSession.staticGrouperSession(), 
542           memberId, false);
543       if (member != null && !SubjectHelper.inList(allSubjects, 
544           member.getSubjectSourceId(), member.getSubjectId())) {
545         allSubjects.add(member.getSubject());
546       }
547     }
548     
549     for (WsAttributeAssign wsAttributeAssign : GrouperUtil.nonNull(this.wsAttributeAssigns, WsAttributeAssign.class)) {
550       if (!StringUtils.isBlank(wsAttributeAssign.getOwnerMemberSubjectId())) {
551         if (!SubjectHelper.inList(allSubjects, wsAttributeAssign.getOwnerMemberSourceId(), wsAttributeAssign.getOwnerMemberSubjectId())) {
552           Subject subject = SubjectFinder.findById(wsAttributeAssign.getOwnerMemberSubjectId(), null, wsAttributeAssign.getOwnerMemberSourceId(), false);
553           if (subject != null) {
554             allSubjects.add(subject);
555           }
556         }
557       }
558     }
559     
560     //security is already checked, lets pass these through...
561     this.wsSubjects = new WsSubject[allSubjects.size()];
562     
563     int i = 0;
564     for (Subject subject : allSubjects) {
565       this.wsSubjects[i] = new WsSubject(subject, theSubjectAttributeNames, null);
566       i++;
567     }
568   }

620   /**
621    * pass in the attribute def name ids that were found by inputs, and add the attribute
622    * def name ids found by the attribute assign results
623    * @param usePIT 
624    * @param pointInTimeFrom 
625    * @param pointInTimeTo 
626    * @param includeAssignmentsOnAssignments if assignments on assignments should be returned
627    * @param enabledBoolean 
628    * @param attributeDefNameIds
629    */
630   public void fillInAttributeAssigns(boolean usePIT, Timestamp pointInTimeFrom, Timestamp pointInTimeTo, 
631       boolean includeAssignmentsOnAssignments, Boolean enabledBoolean) {
632     
633     Set<String> allAttributeAssignIds = new HashSet<String>();
634     
635     for (WsPermissionAssign wsPermissionAssign : GrouperUtil.nonNull(this.wsPermissionAssigns, WsPermissionAssign.class)) {
636       if (!StringUtils.isBlank(wsPermissionAssign.getAttributeAssignId())) {
637         allAttributeAssignIds.add(wsPermissionAssign.getAttributeAssignId());
638       }
639     }
640     
641     if (!usePIT) {
642       List<AttributeAssign> attributeAssignList = new ArrayList<AttributeAssign>();
643       
644       for (String wsAttributeAssignId : allAttributeAssignIds) {
645         AttributeAssign attributeAssign = GrouperDAOFactory.getFactory().getAttributeAssign().findById(wsAttributeAssignId, true);
646         attributeAssignList.add(attributeAssign);
647       }
648   
649       if (includeAssignmentsOnAssignments) {
650         //security is already checked, lets pass these through...
651         Set<AttributeAssign> assignmentsOnAssignments = GrouperDAOFactory.getFactory().getAttributeAssign()
652           .findAssignmentsOnAssignments(attributeAssignList, null, enabledBoolean);
653         attributeAssignList.addAll(assignmentsOnAssignments);
654       }
655   
656       int i = 0;
657       this.wsAttributeAssigns = new WsAttributeAssign[attributeAssignList.size()];
658       for (AttributeAssign attributeAssign : attributeAssignList) {
659   
660         this.wsAttributeAssigns[i] = new WsAttributeAssign(attributeAssign);
661         i++;
662         
663       }
664     } else {
665       List<PITAttributeAssign> attributeAssignList = new ArrayList<PITAttributeAssign>();
666       
667       for (String wsAttributeAssignId : allAttributeAssignIds) {
668         attributeAssignList.addAll(GrouperDAOFactory.getFactory().getPITAttributeAssign().findBySourceId(wsAttributeAssignId, true));
669       }
670   
671       if (includeAssignmentsOnAssignments) {
