100   /**
101    * get a single parameter value for key.  If multiple exist, throw error
102    * @param paramMap is the map of params.  will get value from here if no request object
103    * @param httpServletRequest optional.  if there, will make sure no dupes
104    * @param key to lookup
105    * @return the value
106    */
107   public static String parameterValue(Map<String, String[]> paramMap,
108       HttpServletRequest httpServletRequest, String key) {
109     //if no servlet (probably just testing), get from map
110     if (httpServletRequest == null) {
111       if (paramMap == null || paramMap.isEmpty() || !paramMap.containsKey(key)) {
112         return null;
113       }
114       return paramMap.get(key)[0];
115     }
116     String[] values = httpServletRequest.getParameterValues(key);
117     if (values == null || values.length == 0) {
118       return null;
119     }
120     //there is probably something wrong if multiple values detected
121     if (values.length > 1) {
122       throw new RuntimeException(
123           "Multiple request parameter values where detected for key: " + key
124               + ", when only one is expected: " + GrouperUtil.toStringForLog(values));
125     }
126     return values[0];
127   }

129   /**
130    * retrieve the user principal (who is authenticated) from the (threadlocal)
131    * request object
132    * 
133    * @return the user principal name
134    */
135   public static String retrieveUserPrincipalNameFromRequest() {
136 
137     HttpServletRequest httpServletRequest = retrieveHttpServletRequest();
138     GrouperUtil
139         .assertion(httpServletRequest != null,
140             "HttpServletRequest is null, is the GrouperServiceServlet mapped in the web.xml?");
141     Principal principal = httpServletRequest.getUserPrincipal();
142     String principalName = null;
143     if (principal == null) {
144       principalName = httpServletRequest.getRemoteUser();
145       if (StringUtils.isBlank(principalName)) {
146         principalName = (String)httpServletRequest.getAttribute("REMOTE_USER");
147       }
148     } else {
149       principalName = principal.getName();
150     }
151     GrouperUtil.assertion(StringUtils.isNotBlank(principalName),
152         "There is no user logged in, make sure the container requires authentication");
153     return principalName;
154   }

156   /**
157    * retrieve the subject logged in to web service
158    * If there are four colons, then this is the source and subjectId since
159    * overlap in namespace
160    * 
161    * @return the subject
162    */
163   @SuppressWarnings({ "unchecked", "deprecation" })
164   public static Subject retrieveSubjectLoggedIn() {
165     
166     Map<String, Object> debugMap = GrouperServiceJ2ee.retrieveDebugMap();
167     
168     String authenticationClassName = GrouperWsConfig.getPropertyString(
169         GrouperWsConfig.WS_SECURITY_NON_RAMPART_AUTHENTICATION_CLASS,
170         WsGrouperDefaultAuthentication.class.getName());
171     String userIdLoggedIn = null;
172     if (wssecServlet()) {
173 
174       MessageContext msgCtx = MessageContext.getCurrentMessageContext();
175       Vector results = null;
176       if ((results = (Vector) msgCtx.getProperty(WSHandlerConstants.RECV_RESULTS)) == null) {
177         throw new RuntimeException("No Rampart security results!");
178       }
179       LOG.debug("Number of rampart results: " + results.size());
180       OUTER: for (int i = 0; i < results.size(); i++) {
181         WSHandlerResult rResult = (WSHandlerResult) results.get(i);
182         Vector wsSecEngineResults = rResult.getResults();
183 
184         for (int j = 0; j < wsSecEngineResults.size(); j++) {
185           WSSecurityEngineResult wser = (WSSecurityEngineResult) wsSecEngineResults
186               .get(j);
187           if (wser.getAction() == WSConstants.UT && wser.getPrincipal() != null) {
188 
189             //Extract the principal
190             WSUsernameTokenPrincipal principal = (WSUsernameTokenPrincipal) wser
191                 .getPrincipal();
192 
193             //Get user
194             userIdLoggedIn = principal.getName();
195             break OUTER;
196 
197           }
198         }
199       }
200       GrouperUtil.assertion(StringUtils.isNotBlank(userIdLoggedIn),
201           "There is no Rampart user logged in, make sure the container requires authentication");
202     } else {
203       //this is for container auth (or custom auth, non-rampart)
204       //get an instance
205       Class<? extends WsCustomAuthentication> theClass = GrouperUtil
206           .forName(authenticationClassName);
207 
208       WsCustomAuthentication wsAuthentication = GrouperUtil.newInstance(theClass);
209 
210       userIdLoggedIn = wsAuthentication
211           .retrieveLoggedInSubjectId(retrieveHttpServletRequest());
212     }
213 
214     // cant be blank!
215     if (StringUtils.isBlank(userIdLoggedIn)) {
216       //server is having trouble if got this far, but also the user's fault
217       throw new WsInvalidQueryException("No user is logged in");
218     }
219 
220     //null means dont look in a certain source
221     String sourceId = null;
222 
223     //see if we need to split with 4 colons in login name
224     if (StringUtils.contains(userIdLoggedIn, GrouperWsConfig.WS_SEPARATOR)) {
225       String[] sourceSubjectId = GrouperUtil.splitTrim(userIdLoggedIn,
226           GrouperWsConfig.WS_SEPARATOR);
227       sourceId = sourceSubjectId[0];
228       userIdLoggedIn = sourceSubjectId[1];
229     } else {
230       //see if there is a default source for all users to web service
231       sourceId = StringUtils.trimToNull(GrouperWsConfig.retrieveConfig().propertyValueString(GrouperWsConfig.WS_LOGGED_IN_SUBJECT_DEFAULT_SOURCE));
232     }
233 
234     String subjectIdPrefix = GrouperWsConfig
235         .getPropertyString("ws.security.prependToUserIdForSubjectLookup");
236     if (!StringUtils.isBlank(subjectIdPrefix)) {
237       userIdLoggedIn = subjectIdPrefix + userIdLoggedIn;
238     }
239     
240     //puts it in the log4j ndc context so userid is logged
241     if (NDC.getDepth() == 0) {
242       StringBuilder ndcBuilder = new StringBuilder("< ");
243       if (!StringUtils.isBlank(sourceId)) {
244         ndcBuilder.append(sourceId).append(" - ");
245       }
246       ndcBuilder.append(userIdLoggedIn).append(" - ");
247       HttpServletRequest request = retrieveHttpServletRequest();
248       if (request != null) {
249         ndcBuilder.append(request.getRemoteAddr());
250       }
251       ndcBuilder.append(" >");
252       NDC.push(ndcBuilder.toString());
253     }
254 
255     debugMap.put("userIdLoggedIn", userIdLoggedIn);
256     
257     Subject caller = null;
258     GrouperSession grouperSession = GrouperSession.startRootSession(false);
259     try {
260       final String SOURCE_ID = sourceId;
261       final String USER_ID_LOGGED_IN = userIdLoggedIn;
262       caller = (Subject)GrouperSession.callbackGrouperSession(grouperSession, new GrouperSessionHandler() {
263 
264         public Object callback(GrouperSession theGrouperSession)
265             throws GrouperSessionException {
266 
267           try {
268             //see if across all sources
269             if (SOURCE_ID == null) {
270               return SubjectFinder.findByIdOrIdentifier(USER_ID_LOGGED_IN, true);
271             }
272             //see if in specified sources
273             String[] sourceIds = GrouperUtil.splitTrim(SOURCE_ID, ",");
274             for (String curSource : sourceIds) {
275               Subject s = SubjectFinder.findByIdOrIdentifierAndSource(USER_ID_LOGGED_IN, curSource, false);
276               if (s != null) {
277                 return s;
278               }
279             }
280             throw new SubjectNotFoundException("Unable to find subject in source type(s): " + SOURCE_ID);
281           } catch (Exception e) {
282             //this is probably a system error...  not a user error
283             throw new RuntimeException("Cant find subject from login id: " + USER_ID_LOGGED_IN, e);
284           }
285         }
286       });
287 
288       Subject originalCaller = caller;
289       if (caller != null) {
290         debugMap.put("userIdLoggedInSource", caller.getSourceId());
291         if (!StringUtils.equals(caller.getId(), (String)debugMap.get("userIdLoggedIn"))) {
292           debugMap.put("userIdLoggedInSubjectId", caller.getId());
293         }
294       }
295 
296       caller = retrieveSubjectGrouperActAsHelper(caller);
297 
298       if (caller != null && originalCaller != null && !StringUtils.equals(caller.getId(), originalCaller.getId())) {
299         debugMap.put("userIdActAsSource", caller.getSourceId());
300         debugMap.put("userIdActAsSubjectId", caller.getId());
301       }
302       
303       //this is set in filter
304       GrouperContext grouperContext = GrouperContext.retrieveDefaultContext();
305       
306       Member member = MemberFinder.findBySubject(grouperSession, caller, true);
307       
308       grouperContext.setLoggedInMemberId(member.getUuid());
309     } finally {
310       GrouperSession.stopQuietly(grouperSession);
311     }
312 
313     
314     return caller;
315 
316   }

318   /**
319    * see if there is a grouper act as in play here
320    * @param loggedInSubject
321    * @return the subject
322    */
323   private static Subject retrieveSubjectGrouperActAsHelper(final Subject loggedInSubject) {
324     
325     //see if we are acting as someone else
326     String grouperActAsGroup = GrouperWsConfig.retrieveConfig().propertyValueString("ws.grouper.act.as.group");
327     if (StringUtils.isBlank(grouperActAsGroup)
328         || loggedInSubject == null) {
329       LOG.debug("No grouperActAs configured");
330       return loggedInSubject;
331     }
332 
333     GrouperSession session = null;
334     
335     HttpServletRequest httpServletRequest = retrieveHttpServletRequest();
336 
337     String grouperActAsSubjectId = httpServletRequest.getHeader("X-Grouper-actAsSubjectId");
338     String grouperActAsSubjectIdentifier = httpServletRequest.getHeader("X-Grouper-actAsSubjectIdentifier");
339     String grouperActAsSubjectSource = httpServletRequest.getHeader("X-Grouper-actAsSourceId");
340 
341     if (!StringUtils.isBlank(grouperActAsSubjectSource)) {
342 
343       if (!StringUtils.isBlank(grouperActAsSubjectId) 
344           && !StringUtils.isBlank(grouperActAsSubjectIdentifier)) {
345         throw new RuntimeException("You can only have one of X-Grouper-actAsSubjectId or X-Grouper-actAsSubjectIdentifier set!");
346       }
347 
348       if (StringUtils.isBlank(grouperActAsSubjectId) 
349           && StringUtils.isBlank(grouperActAsSubjectIdentifier)) {
350         throw new RuntimeException("You must have one of X-Grouper-actAsSubjectId or X-Grouper-actAsSubjectIdentifier set if X-Grouper-actAsSourceId is set!");
351       }
352 
353       try {
354         grouperActAsSubjectSource = new String(new Base64().decode(grouperActAsSubjectSource.getBytes("UTF-8")), "UTF-8");
355   
356         if (!StringUtils.isBlank(grouperActAsSubjectId)) {
357           grouperActAsSubjectId = new String(new Base64().decode(grouperActAsSubjectId.getBytes("UTF-8")), "UTF-8");
358         } else if (!StringUtils.isBlank(grouperActAsSubjectIdentifier)) {
359           grouperActAsSubjectIdentifier = new String(new Base64().decode(grouperActAsSubjectIdentifier.getBytes("UTF-8")), "UTF-8");
360         }
361       } catch (Exception e) {
362         throw new RuntimeException("Problem with: source: '" + grouperActAsSubjectSource + "', id: '" 
363             + grouperActAsSubjectId + "', or identifier: '" + grouperActAsSubjectIdentifier + "'", e);
364       }
365     } else {
366       if (!StringUtils.isBlank(grouperActAsSubjectId) 
367           || !StringUtils.isBlank(grouperActAsSubjectIdentifier)) {
368         throw new RuntimeException("If X-Grouper-actAsSubjectId or X-Grouper-actAsSubjectIdentifier is set, then you must have a X-Grouper-actAsSourceId!");
369       }
370       //there is nothing there, go back
371       return loggedInSubject;
372     }
373     
374     // get the all powerful user
375     Subject rootSubject = SubjectFinder.findRootSubject();
376 
377     try {
378       session = GrouperSession.start(rootSubject);
379 
380       Subject actAsSubject = null;
381 
382       if (!StringUtils.isBlank(grouperActAsSubjectId)) {
383         actAsSubject = SubjectFinder.findByIdAndSource(grouperActAsSubjectId, grouperActAsSubjectSource, true);
384       } else if (!StringUtils.isBlank(grouperActAsSubjectIdentifier)) {
385         actAsSubject = SubjectFinder.findByIdentifierAndSource(grouperActAsSubjectIdentifier, grouperActAsSubjectSource, true);
386       } else {
387         throw new RuntimeException("Why am I here?");
388       }
389 
390       final Subject ACT_AS_SUBJECT = actAsSubject;
391       
392       //cache key to get or set if a user can act as another
393       final MultiKey cacheKey = new MultiKey(loggedInSubject.getId(), loggedInSubject.getSource()
394           .getId(), actAsSubject.getId(), actAsSubject.getSource().getId());
395 
396       Boolean inCache = null;
397       
398       if (actAsCacheMinutes() > 0) {
399         inCache = grouperActAsCache().get(cacheKey);
400       } else {
401         inCache = false;
402       }
403 
404       if (inCache != null && Boolean.TRUE.equals(inCache)) {
405         LOG.debug("grouperActAs retrieved from cache");
406         //if in cache and true, then allow
407         return actAsSubject;
408       }
409       
410       {
411         //see if root or wheel group
412         Subject rootAllowedSubject = (Subject)GrouperSession.callbackGrouperSession(session, new GrouperSessionHandler() {
413 
414           public Object callback(GrouperSession grouperSession) throws GrouperSessionException {
415             if (PrivilegeHelper.isWheelOrRoot(loggedInSubject)) {
416               actAsCache().put(cacheKey, Boolean.TRUE);
417               if (LOG.isDebugEnabled()) {
418                 LOG.debug("grouperActAs allowed since logged in user is wheel or root: " + GrouperUtil.subjectToString(loggedInSubject));
419               }
420               return ACT_AS_SUBJECT;
421             }
422             return null;
423           }
424         });
425   
426         if (rootAllowedSubject != null) {
427           return rootAllowedSubject;
428         }
429       }
430       
431       //first separate by comma
432       String[] groupEntries = GrouperUtil.splitTrim(grouperActAsGroup, ",");
433 
434       //see if all throw exceptions
435       int countNoExceptions = 0;
436 
437       //we could also cache which entries the user is in...  not sure how many entries will be here
438       for (String groupEntry : groupEntries) {
439 
440         //each entry should be failsafe
441         try {
442           //now see if it is a multi input
443           if (StringUtils.contains(groupEntry, GrouperWsConfig.WS_SEPARATOR)) {
444 
445             //it is the group the user is in, and the group the act as has to be in
446             String[] groupEntryArray = GrouperUtil.splitTrim(groupEntry,
447                 GrouperWsConfig.WS_SEPARATOR);
448             String userMustBeInGroupName = groupEntryArray[0];
449             String actAsMustBeInGroupName = groupEntryArray[1];
450 
451             Group userMustBeInGroup = GroupFinder.findByName(session,
452                 userMustBeInGroupName, true);
453             Group actAsMustBeInGroup = GroupFinder.findByName(session,
454                 actAsMustBeInGroupName, true);
455 
456             if (userMustBeInGroup.hasMember(loggedInSubject)
457                 && actAsMustBeInGroup.hasMember(actAsSubject)) {
458               if (LOG.isDebugEnabled()) {
459                 LOG.debug("grouperActAs allowed since logged in user is in group: " + userMustBeInGroupName + ", and act as user is in group: " + actAsMustBeInGroupName);
460               }
461               //its ok, lets add to cache
462               actAsCache().put(cacheKey, Boolean.TRUE);
463               return actAsSubject;
464             }
465 
466           } else {
467             //else this is a straightforward rule where the logged in user just has to be in a group and
468             //can act as anyone
469             Group actAsGroup = GroupFinder.findByName(session, grouperActAsGroup, true);
470 
471             // if the logged in user is a member of the actAs group, then allow
472             // the actAs
473             if (actAsGroup.hasMember(loggedInSubject)) {
474               if (LOG.isDebugEnabled()) {
475                 LOG.debug("grouperActAs allowed since logged in user is in group: " + grouperActAsGroup);
476               }
477               //its ok, lets add to cache
478               actAsCache().put(cacheKey, Boolean.TRUE);
479               // this is the subject the web service wants to use
480               return actAsSubject;
481             }
482           }
483           countNoExceptions++;
484         } catch (Exception e) {
485           //just log and dont act since other entries could be fine
486           LOG.error("Problem with groupEntry: " + groupEntry + ", loggedInUser: "
487               + loggedInSubject + ", actAsSubject: " + actAsSubject, e);
488         }
489 
490       }
491 
492       if (countNoExceptions == 0) {
493         throw new RuntimeException("Problems seeing if web service user '"
494             + loggedInSubject + "' can actAs the other subject: '" + actAsSubject + "'");
495       }
496       // if not an effective member
497       throw new RuntimeException(
498           "A web service is specifying an actAsUser, but the groups specified in "
499               + GrouperWsConfig.WS_ACT_AS_GROUP + " in the grouper-ws.properties "
500               + " does not have a valid rule for member: '" + loggedInSubject
501               + "', and actAs: '" + actAsSubject + "'");
502     } catch (SessionException se) {
503       throw new RuntimeException(se);
504     } finally {
505       GrouperSession.stopQuietly(session);
506     }
507     
508   }

519   /**
520    * get the actAsCache, and init if not initted
521    * @return the actAsCache
522    */
523   private static GrouperCache<MultiKey, Boolean> actAsCache() {
524     if (actAsCache == null) {
525       int actAsTimeoutMinutes = actAsCacheMinutes();
526 
527       synchronized(GrouperServiceJ2ee.class) {
528         if (actAsCache == null) {
529           actAsCache = new GrouperCache<MultiKey, Boolean>(GrouperServiceJ2ee.class.getName() + "grouperWsActAsCache", 10000, false, 60*60*24, actAsTimeoutMinutes*60, false);
530         }
531       }
532     }
533     return actAsCache;
534   }

580   /**
581    * retrieve the subject to act as
582    * 
583    * @param actAsLookup that the caller wants to act as
584    * @return the subject
585    * @throws WsInvalidQueryException if there is a problem
586    */
587   public static Subject retrieveSubjectActAs(WsSubjectLookup actAsLookup)
588       throws WsInvalidQueryException {
589     Subject actAsSubject = retrieveSubjectActAsHelper(actAsLookup);
590     HooksContext.assignSubjectActAs(actAsSubject);
591 
592     //this is set in filter
593     GrouperContext grouperContext = GrouperContext.retrieveDefaultContext();
594 
595     GrouperSession grouperSession = GrouperSession.staticGrouperSession(false);
596     GrouperSession rootSession = grouperSession == null ? 
597         GrouperSession.startRootSession(false) : grouperSession.internal_getRootSession();
598 
599     
600     Member member = MemberFinder.findBySubject(rootSession, actAsSubject, true);
601     
602     grouperContext.setLoggedInMemberIdActAs(member.getUuid());
603     
604     return actAsSubject;
605   }

607   /**
608    * retrieve the subject to act as
609    * 
610    * @param actAsLookup that the caller wants to act as
611    * @return the subject
612    * @throws WsInvalidQueryException if there is a problem
613    */
614   private static Subject retrieveSubjectActAsHelper(WsSubjectLookup actAsLookup)
615       throws WsInvalidQueryException {
616 
617     final Subject loggedInSubject = retrieveSubjectLoggedIn();
618 
619     HooksContext.assignSubjectLoggedIn(loggedInSubject);
620     
621     //make sure allowed
622     final String userGroupName = GrouperWsConfig.retrieveConfig().propertyValueString(GrouperWsConfig.WS_CLIENT_USER_GROUP_NAME);
623     
624     final String loggedInSubjectId = loggedInSubject.getId();
625     if (!StringUtils.isBlank(userGroupName)) {
626       GrouperSession grouperSession = null;
627       
628       try {
629         //cache key to get or set if a user can act as another
630         final MultiKey cacheKey = new MultiKey(loggedInSubjectId, 
631             loggedInSubject.getSource().getId());
632 
633         Boolean allowedInCache = subjectAllowedCache().get(cacheKey);
634 
635         //if not in cache
636         if (allowedInCache == null) {
637           grouperSession = GrouperSession.startRootSession();
638           GrouperSession.internal_callbackRootGrouperSession(new GrouperSessionHandler() {
639             
640             public Object callback(GrouperSession rootGrouperSession) throws GrouperSessionException {
641               Group group = GroupFinder.findByName(rootGrouperSession, userGroupName, true);
642               if (!group.hasMember(loggedInSubject)) {
643                 //not allowed, cache it
644                 subjectAllowedCache().put(cacheKey, false);
645                 throw new RuntimeException("User is not authorized: " + loggedInSubject + ", " + group);
646               }
647               subjectAllowedCache().put(cacheKey, true);
648               return null;
649             }
650           });
651         } else {
652           //if in cache, reflect that
653           if (!allowedInCache) {
654             throw new RuntimeException("User is not authorized: " + loggedInSubject);
655           }
656         }
657       } catch (Exception e) {
658         LOG.error("user: '" + loggedInSubjectId + "' is not a member of group: '" + userGroupName 
659             + "', and therefore is not authorized to use the app (configured in local media.properties penn.uiGroup", e);
660         throw new RuntimeException("User is not authorized", e);
661       } finally {
662         GrouperSession.stopQuietly(grouperSession);
663       }
664     }
665 
666     
667     // if there is no actAs specified, then just use the logged in user
668     if (actAsLookup == null || actAsLookup.blank()) {
669       return loggedInSubject;
670     }
671     
672     GrouperSession grouperSession = GrouperSession.startRootSession();
673     Subject actAsSubject = null;
674     try {
675       actAsSubject = actAsLookup.retrieveSubject("actAsSubject");
676     } finally {
677       GrouperSession.stopQuietly(grouperSession);
678     }
679     
680     //see if same:
681     if (StringUtils.equals(loggedInSubjectId, actAsSubject.getId())
682         && StringUtils.equals(loggedInSubject.getSource().getId(), actAsSubject.getSource().getId())) {
683       return loggedInSubject;
684     }
685     
686     //lets see if in cache    
687 
688     //cache key to get or set if a user can act as another
689     MultiKey cacheKey = new MultiKey(loggedInSubjectId, loggedInSubject.getSource()
690         .getId(), actAsSubject.getId(), actAsSubject.getSource().getId());
691 
692     Boolean inCache = null;
693     
694     if (actAsCacheMinutes() > 0) {
695       inCache = actAsCache().get(cacheKey);
696     } else {
697       inCache = false;
698     }
699 
700     if (inCache != null && Boolean.TRUE.equals(inCache)) {
701       //if in cache and true, then allow
702       return actAsSubject;
703     }
704     
705     //see if root or wheel group
706     GrouperSession session = null;
707     try {
708       session = GrouperSession.start(loggedInSubject);
709       if (PrivilegeHelper.isRoot(session)) {
710         actAsCache().put(cacheKey, Boolean.TRUE);
711       return actAsSubject;
712     }
713     } catch (SessionException se) {
714       throw new RuntimeException(se);
715     } finally {
716       GrouperSession.stopQuietly(session);
717     }
718 
719     // so there is an actAs specified, lets see if we are allowed to use it
720     // first lets get the group you have to be in if you are going to
721     String actAsGroupName = GrouperWsConfig.retrieveConfig().propertyValueString(GrouperWsConfig.WS_ACT_AS_GROUP);
722 
723     // make sure there is one there
724     if (StringUtils.isBlank(actAsGroupName)) {
725 
726       //if none configured, then probably a caller problem
727       throw new WsInvalidQueryException(
728           "A web service is specifying an actAsUser, but there is no '"
729               + GrouperWsConfig.WS_ACT_AS_GROUP
730               + "' specified in the grouper-ws.properties");
731     }
732 
733     session = null;
734     // get the all powerful user
735     Subject rootSubject = SubjectFinder.findRootSubject();
736 
737     try {
738       session = GrouperSession.start(rootSubject);
739 
740       //first separate by comma
741       String[] groupEntries = GrouperUtil.splitTrim(actAsGroupName, ",");
742 
743       //see if all throw exceptions
744       int countNoExceptions = 0;
745 
746       //we could also cache which entries the user is in...  not sure how many entries will be here
747       for (String groupEntry : groupEntries) {
748 
749         //each entry should be failsafe
750         try {
751           //now see if it is a multi input
752           if (StringUtils.contains(groupEntry, GrouperWsConfig.WS_SEPARATOR)) {
753 
754             //it is the group the user is in, and the group the act as has to be in
755             String[] groupEntryArray = GrouperUtil.splitTrim(groupEntry,
756                 GrouperWsConfig.WS_SEPARATOR);
757             String userMustBeInGroupName = groupEntryArray[0];
758             String actAsMustBeInGroupName = groupEntryArray[1];
759 
760             Group userMustBeInGroup = GroupFinder.findByName(session,
761                 userMustBeInGroupName, true);
762             Group actAsMustBeInGroup = GroupFinder.findByName(session,
763                 actAsMustBeInGroupName, true);
764 
765             if (userMustBeInGroup.hasMember(loggedInSubject)
766                 && actAsMustBeInGroup.hasMember(actAsSubject)) {
767               //its ok, lets add to cache
768               actAsCache().put(cacheKey, Boolean.TRUE);
769               return actAsSubject;
770             }
771 
772           } else {
773             //else this is a straightforward rule where the logged in user just has to be in a group and
774             //can act as anyone
775             Group actAsGroup = GroupFinder.findByName(session, actAsGroupName, true);
776 
777             // if the logged in user is a member of the actAs group, then allow
778             // the actAs
779             if (actAsGroup.hasMember(loggedInSubject)) {
780               //its ok, lets add to cache
781               actAsCache().put(cacheKey, Boolean.TRUE);
782               // this is the subject the web service wants to use
783               return actAsSubject;
784             }
785           }
786           countNoExceptions++;
787         } catch (Exception e) {
788           //just log and dont act since other entries could be fine
789           LOG.error("Problem with groupEntry: " + groupEntry + ", loggedInUser: "
790               + loggedInSubject + ", actAsSubject: " + actAsSubject, e);
791         }
792 
793       }
794 
795       if (countNoExceptions == 0) {
796         throw new RuntimeException("Problems seeing if web service user '"
797             + loggedInSubject + "' can actAs the other subject: '" + actAsSubject + "'");
798       }
799       // if not an effective member
800       throw new RuntimeException(
801           "A web service is specifying an actAsUser, but the groups specified in "
802               + GrouperWsConfig.WS_ACT_AS_GROUP + " in the grouper-ws.properties "
803               + " does not have a valid rule for member: '" + loggedInSubject
804               + "', and actAs: '" + actAsSubject + "'");
805     } catch (SessionException se) {
806       throw new RuntimeException(se);
807     } finally {
808       GrouperSession.stopQuietly(session);
809     }
810 
811   }

921   public void doFilter(ServletRequest request, ServletResponse response,
922       FilterChain filterChain) throws IOException, ServletException {
923 
924     Map<String, Object> debugMap = null;
925     Long start = System.nanoTime();
926     try {
927 
928       request.setCharacterEncoding("UTF-8");
929       response.setCharacterEncoding("UTF-8");
930 
931       //make sure nulls are not returned for params for Axis bug where
932       //empty strings work, but nulls make things off a bit
933       request = new WsHttpServletRequest((HttpServletRequest) request);
934   
935       request.setAttribute("debugMap", debugMap);
936       
937       NDC.clear();
938       
939       //servlet will set this...
940       threadLocalServlet.remove();
941       threadLocalRequest.set((HttpServletRequest) request);
942       
943       debugMap = retrieveDebugMap();
944       
945       threadLocalResponse.set((HttpServletResponse) response);
946       threadLocalRequestStartMillis.set(System.currentTimeMillis());
947       
948       GrouperContextTypeBuiltIn.setDefaultContext(GrouperContextTypeBuiltIn.GROUPER_WS);
949   
950       //lets add the request, session, and response
951       HooksContext.setAttributeThreadLocal(HooksContext.KEY_HTTP_SERVLET_REQUEST, request, false);
952       HooksContext.setAttributeThreadLocal(HooksContext.KEY_HTTP_SESSION, 
953           ((HttpServletRequest)request).getSession(), false);
954       HooksContext.setAttributeThreadLocal(HooksContext.KEY_HTTP_SERVLET_RESPONSE, response, false);
955   
956       GrouperContext grouperContext = GrouperContext.createNewDefaultContext(
957           GrouperEngineBuiltin.WS, false, false);
958       
959       String xForwardedFor = ((HttpServletRequest)request).getHeader("X-Forwarded-For");
960       String remoteAddr = StringUtils.defaultIfBlank(xForwardedFor, request.getRemoteAddr());
961       grouperContext.setCallerIpAddress(remoteAddr);
962       
963       //get the proxy IP address
964       debugMap.put("start", timeFormat.format(new Date()));
965       debugMap.put("remoteAddr", remoteAddr);
966       debugMap.put("requestUrl", ((HttpServletRequest)request).getRequestURL());
967       
968       filterChain.doFilter(request, response);
969 
970     } catch (RuntimeException re) {
971       LOG.info("error in request", re);
972       debugMap.put("exception", ExceptionUtils.getFullStackTrace(re));
973       throw re;
974     } finally {
975       
976       GrouperWsLog.wsLog(debugMap, start);
977       GrouperWsLongRunningLog.wsLog(debugMap, start);
978       
979       threadLocalRequest.remove();
980       threadLocalResponse.remove();
981       threadLocalRequestStartMillis.remove();
982       threadLocalServlet.remove();
983       
984       HooksContext.clearThreadLocal();
985       ServletRequestUtils.requestEnd();
986 
987     }

1027   /**
1028    * if initted
1029    */
1030   private static boolean inittedOnce = false;
1031 
1032   /**
1033    * 
1034    */
1035   private static void initOnce() {
1036     
1037     if (!inittedOnce) {
1038       synchronized (GrouperServiceJ2ee.class) {
1039         if (!inittedOnce) {
1040 
1041           //  # Web service users who are in the following group can use the actAs field to act as someone else
1042           //  # You can put multiple groups separated by commas.  e.g. a:b:c, e:f:g
1043           //  # You can put a single entry as the group the calling user has to be in, and the grouper the actAs has to be in
1044           //  # separated by 4 colons
1045           //  # e.g. if the configured values is:       a:b:c, e:f:d :::: r:e:w, x:e:w
1046           //  # then if the calling user is in a:b:c or x:e:w, then the actAs can be anyone
1047           //  # if not, then if the calling user is in e:f:d, then the actAs must be in r:e:w.  If multiple rules, then 
1048           //  # if one passes, then it is a success, if they all fail, then fail.
1049           //  ws.act.as.group = etc:webServiceActAsGroup
1050           //
1051           //  # similar syntax as ws.act.as.group but for the grouper actas (e.g. for grouper messaging to WS bridge)
1052           //  ws.grouper.act.as.group = 
1053           
1054           String debugGroupConfig = GrouperWsConfig.retrieveConfig().propertyValueString("browser.debug.group");
1055           String grouperDebugGroupConfig = GrouperWsConfig.retrieveConfig().propertyValueString("ws.grouper.act.as.group");
1056           
1057           for (String config : new String[]{debugGroupConfig, grouperDebugGroupConfig}) {
1058             if (!StringUtils.isBlank(config)) {
1059               for (String groupNameConfig : GrouperUtil.splitTrim(config, ",")) {
1060                 //now see if it is a multi input
1061                 if (StringUtils.contains(groupNameConfig, GrouperWsConfig.WS_SEPARATOR)) {
1062   
1063                   //it is the group the user is in, and the group the act as has to be in
1064                   String[] groupEntryArray = GrouperUtil.splitTrim(groupNameConfig,
1065                       GrouperWsConfig.WS_SEPARATOR);
1066                   initGroup(groupEntryArray[0]);
1067                   initGroup(groupEntryArray[1]);
1068                 } else {
1069                   initGroup(groupNameConfig);
1070                 }
1071               }
1072             }
1073           }
1074           
1075           //  # If there is an entry here for group name, then all web service client users must be in this group (before the actAs)
1076           //  #ws.client.user.group.name = etc:webServiceClientUsers
1077           initGroup(GrouperWsConfig.retrieveConfig().propertyValueString("ws.client.user.group.name"));
1078           
1079           inittedOnce = true;
1080         }
1081       }
1082     }
