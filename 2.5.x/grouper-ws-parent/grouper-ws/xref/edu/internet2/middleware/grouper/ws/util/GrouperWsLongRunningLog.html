7   import java.util.Map;

9   import org.apache.commons.logging.Log;
10  
11  import edu.internet2.middleware.grouper.util.GrouperUtil;
12  import edu.internet2.middleware.grouper.ws.GrouperWsConfig;
13  import edu.internet2.middleware.grouperClient.util.GrouperClientUtils;
14  
15  
16  /**
17   * logger for ws events
18   */
19  public class GrouperWsLongRunningLog {
20    
21    /** logger */
22    private static final Log LOG = GrouperUtil.getLog(GrouperWsLongRunningLog.class);
23  
24    /**
25     * 
26     * @return true if debug enabled
27     */
28    static boolean isDebugEnabled() {
29      return LOG.isDebugEnabled();
30    }
31    
32    /**
33     * log something to the log file
34     * @param messageMap
35     * @param startTimeNanos nanos when the request started
36     */
37    @SuppressWarnings("static-access")
38    public static void wsLog(Map<String, Object> messageMap, Long startTimeNanos) {
39      if (LOG.isDebugEnabled()) {
40        try {
41          int logOverMillis = GrouperWsConfig.retrieveConfig().propertyValueInt("ws.longRunningRequestLogMillis", 30000);
42          long elapsedMillis = (System.nanoTime() - startTimeNanos) / 1000000;
43          if (logOverMillis >= 0 && elapsedMillis >= logOverMillis) {
44    
45            if (messageMap != null && startTimeNanos != null && !messageMap.containsKey("elapsedMillis")) {
46              messageMap.put("elapsedMillis", elapsedMillis);
47            }
48            LOG.debug(GrouperClientUtils.mapToString(messageMap));
49          }
50        } catch (RuntimeException re) {
51          LOG.error("error", re);
52        }
53      }
54    }
55  
56  }
