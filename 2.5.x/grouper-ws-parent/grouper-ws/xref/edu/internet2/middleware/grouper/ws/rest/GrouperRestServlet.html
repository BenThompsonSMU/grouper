298   /**
299    * for error messages, get a detailed report of the request
300    * @param request
301    * @return the string of descriptive result
302    */
303   public static String requestDebugInfo(HttpServletRequest request) {
304     StringBuilder result = new StringBuilder();
305     result.append(" uri: ").append(request.getRequestURI());
306     result.append(", method: ").append(request.getMethod());
307     result.append(", decoded url strings: ");
308     List<String> urlStrings = extractUrlStrings(request);
309     int urlStringsLength = GrouperUtil.length(urlStrings);
310     if (urlStringsLength == 0) {
311       result.append("[none]");
312     } else {
313       for (int i = 0; i < urlStringsLength; i++) {
314         result.append(i).append(": '").append(urlStrings.get(i)).append("'");
315         if (i != urlStringsLength - 1) {
316           result.append(", ");
317         }
318       }
319     }
320     return result.toString();
321   }

323   /**
324    * take a request and get the list of url strings for the rest web service
325    * @see #extractUrlStrings(String)
326    * @param request is the request to get the url strings out of
327    * @return the list of url strings
328    */
329   private static List<String> extractUrlStrings(HttpServletRequest request) {
330     String requestResourceFull = request.getRequestURI();
331     return extractUrlStrings(requestResourceFull);
332   }

334   /**
335    * <pre>
336    * take a request uri and break up the url strings not including the app name or servlet
337    * this does not include the url params (if applicable)
338    * if the input is: grouper-ws/servicesRest/xhtml/v1_3_000/groups/members
339    * then the result is a list of size 2: {"group", "members"}
340    * 
341    * </pre>
342    * @param requestResourceFull
343    * @return the url strings
344    */
345   private static List<String> extractUrlStrings(String requestResourceFull) {
346     String[] requestResources = StringUtils.split(requestResourceFull, '/');
347     List<String> urlStrings = new ArrayList<String>();
348     //loop through and decode
349     int index = 0;
350     for (String requestResource : requestResources) {
351       //skip the app name and lite servlet
352       if (index++ < 2) {
353         continue;
354       }
355       //unescape the url encoding
356       urlStrings.add(GrouperUtil.escapeUrlDecode(requestResource));
357     }
358     return urlStrings;
359   }
