143         return WsAttributeDefNameDeleteResultCode.ATTRIBUTE_DEF_NAME_UUID_DOESNT_MATCH_NAME;
144       }
145     },
146 
147     /** cant find the attributeDefName */
148     ATTRIBUTE_DEF_NAME_NOT_FOUND {

246   /**
247    * retrieve the attributeDefName for this lookup if not looked up yet.  pass in a grouper session
248    * @param grouperSession 
249    * @param invalidQueryReason is the text to go in the WsInvalidQueryException
250    * @return the attributeDefName
251    * @throws WsInvalidQueryException if there is a problem, and if the invalidQueryReason is set
252    */
253   public AttributeDefName retrieveAttributeDefNameIfNeeded(GrouperSession grouperSession,
254       String invalidQueryReason) throws WsInvalidQueryException {
255     
256     //see if we already retrieved
257     if (this.attributeDefNameFindResult != null) {
258       return this.attributeDefName;
259     }
260     try {
261       //assume success (set otherwise if there is a problem)
262       this.attributeDefNameFindResult = AttributeDefNameFindResult.SUCCESS;
263 
264       boolean hasUuid = !StringUtils.isBlank(this.uuid);
265 
266       boolean hasName = !StringUtils.isBlank(this.name);
267 
268       boolean hasIdIndex = !StringUtils.isBlank(this.idIndex);
269 
270       //must have a name or uuid
271       if (!hasUuid && !hasName && !hasIdIndex) {
272         this.attributeDefNameFindResult = AttributeDefNameFindResult.INVALID_QUERY;
273         if (!StringUtils.isEmpty(invalidQueryReason)) {
274           throw new WsInvalidQueryException("Invalid attributeDefName query for '"
275               + invalidQueryReason + "', " + this);
276         }
277         String logMessage = "Invalid query: " + this;
278         LOG.warn(logMessage);
279       }
280 
281       AttributeDefName theAttributeDefName = null;
282       
283       if (hasName) {
284         
285         theAttributeDefName = AttributeDefNameFinder.findByName(this.name, true);
286 
287       } else if (hasUuid) {
288         
289         theAttributeDefName = AttributeDefNameFinder.findById(this.uuid, true);
290 
291       } else if (hasIdIndex) {
292         
293         theAttributeDefName = AttributeDefNameFinder.findByIdIndexSecure(GrouperUtil.longValue(this.idIndex), true, 
294             new QueryOptions().secondLevelCache(false));
295 
296       }
297 
298       //make sure uuid matches 
299       if ((hasUuid && !StringUtils.equals(this.uuid, theAttributeDefName.getId()))
300           || (hasName && !StringUtils.equals(this.name, theAttributeDefName.getName()))
301           || (hasIdIndex && !GrouperUtil.equals(GrouperUtil.longValue(this.idIndex), theAttributeDefName.getIdIndex()))){
302         this.attributeDefNameFindResult = AttributeDefNameFindResult.ATTRIBUTE_DEF_NAME_UUID_DOESNT_MATCH_NAME;
303         String error = "AttributeDefName name '" + this.name + "' and uuid '" + this.uuid
304             + "', and idIndex: " + this.idIndex + " do not match";
305         if (!StringUtils.isEmpty(invalidQueryReason)) {
306           throw new WsInvalidQueryException(error + " for '" + invalidQueryReason
307               + "', " + this);
308         }
309         String logMessage = "Invalid query: " + this;
310         LOG.warn(logMessage);
311       }
312 
313       //success
314       this.attributeDefName = theAttributeDefName;
315 
316 
317     } catch (AttributeDefNameNotFoundException anf) {
318       this.attributeDefNameFindResult = AttributeDefNameFindResult.ATTRIBUTE_DEF_NAME_NOT_FOUND;
319       if (!StringUtils.isBlank(invalidQueryReason)) {
320         throw new WsInvalidQueryException("Invalid attributeDefName for '" + invalidQueryReason
321             + "', " + this, anf);
322       }
323     }
324     return this.attributeDefName;
325   }

327   /**
328    * retrieve the pit attribute def names for this lookup if not looked up yet.
329    * @param invalidQueryReason is the text to go in the WsInvalidQueryException
330    * @param pointInTimeFrom 
331    * @param pointInTimeTo 
332    * @return the pit attribute def name
333    * @throws WsInvalidQueryException if there is a problem, and if the invalidQueryReason is set
334    */
335   public Set<PITAttributeDefName> retrievePITAttributeDefNamesIfNeeded(String invalidQueryReason, Timestamp pointInTimeFrom, 
336       Timestamp pointInTimeTo) throws WsInvalidQueryException {
337 
338     //see if we already retrieved
339     if (this.attributeDefNameFindResult != null) {
340       return this.pitAttributeDefNames;
341     }
342 
343     //assume success (set otherwise if there is a problem)
344     this.attributeDefNameFindResult = AttributeDefNameFindResult.SUCCESS;
345     
346     try {
347       boolean hasUuid = !StringUtils.isBlank(this.uuid);
348 
349       boolean hasName = !StringUtils.isBlank(this.name);
350 
351       boolean hasIdIndex = !StringUtils.isBlank(this.idIndex);
352 
353       //must have a name or uuid
354       if (!hasUuid && !hasName) {
355         this.attributeDefNameFindResult = AttributeDefNameFindResult.INVALID_QUERY;
356 
357         String logMessage = "Invalid query: " + this;
358         
359         if (hasIdIndex) {
360           if (!StringUtils.isEmpty(invalidQueryReason)) {
361             throw new WsInvalidQueryException("Invalid attributeDefName query for '"
362                 + invalidQueryReason + "', cant lookup by idIndex: " + this);
363           }
364           logMessage = "Invalid attributeDefName query cant lookup by idIndex: " + this;
365         }
366         if (!StringUtils.isEmpty(invalidQueryReason)) {
367           throw new WsInvalidQueryException("Invalid attributeDefName query for '"
368               + invalidQueryReason + "', " + this);
369         }
370         LOG.warn(logMessage);
371       }
372 
373       if (hasUuid) {        
374         Set<PITAttributeDefName> theAttributeDefNames = PITAttributeDefNameFinder.findBySourceId(this.uuid, pointInTimeFrom, pointInTimeTo, true);
375 
376         //make sure name matches 
377         if (hasName && !StringUtils.equals(this.name, theAttributeDefNames.iterator().next().getName())) {
378           this.attributeDefNameFindResult = AttributeDefNameFindResult.ATTRIBUTE_DEF_NAME_UUID_DOESNT_MATCH_NAME;
379           String error = "AttributeDefName name '" + this.name + "' and uuid '" + this.uuid
380               + "' do not match";
381           if (!StringUtils.isEmpty(invalidQueryReason)) {
382             throw new WsInvalidQueryException(error + " for '" + invalidQueryReason
383                 + "', " + this);
384           }
385           String logMessage = "Invalid query: " + this;
386           LOG.warn(logMessage);
387         }
388 
389         //success
390         this.pitAttributeDefNames = new LinkedHashSet<PITAttributeDefName>(theAttributeDefNames);
391 
392       } else if (hasName) {
393         this.pitAttributeDefNames = PITAttributeDefNameFinder.findByName(this.name, pointInTimeFrom, pointInTimeTo, true, true);
394       }
395 
396     } catch (AttributeDefNameNotFoundException anf) {
397       this.attributeDefNameFindResult = AttributeDefNameFindResult.ATTRIBUTE_DEF_NAME_NOT_FOUND;
398       if (!StringUtils.isBlank(invalidQueryReason)) {
399         throw new WsInvalidQueryException("Invalid attributeDefName for '" + invalidQueryReason
400             + "', " + this, anf);
401       }
402     }
403     return this.pitAttributeDefNames;
404   }

468   public static Set<String> convertToAttributeDefNameIds(GrouperSession grouperSession, 
469       WsAttributeDefNameLookup[] wsAttributeDefNameLookups, StringBuilder errorMessage, AttributeDefType attributeDefType,
470       boolean usePIT, Timestamp pointInTimeFrom, Timestamp pointInTimeTo) {
471     return convertToAttributeDefNameIds(grouperSession, wsAttributeDefNameLookups, errorMessage, attributeDefType, 
472         usePIT, pointInTimeFrom, pointInTimeTo, new int[]{0});

475   /**
476    * convert attributeDefName lookups to attributeDefName ids
477    * @param grouperSession
478    * @param wsAttributeDefNameLookups
479    * @param errorMessage
480    * @param attributeDefType 
481    * @param usePIT 
482    * @param pointInTimeFrom 
483    * @param pointInTimeTo 
484    * @param lookupCount is an array of size one int where 1 will be added if there are records, and no change if not
485    * @return the attributeDefName ids
486    */
487   public static Set<String> convertToAttributeDefNameIds(GrouperSession grouperSession, 
488       WsAttributeDefNameLookup[] wsAttributeDefNameLookups, StringBuilder errorMessage, 
489       AttributeDefType attributeDefType, boolean usePIT, Timestamp pointInTimeFrom,
490       Timestamp pointInTimeTo, int[] lookupCount) {
491     //get all the attributeDefNames
492     //we could probably batch these to get better performance.
493     Set<String> attributeDefNameIds = null;
494     if (!GrouperServiceUtils.nullArray(wsAttributeDefNameLookups)) {
495       attributeDefNameIds = new LinkedHashSet<String>();
496       int i=0;
497       
498       boolean foundRecords = false;
499       
500       for (WsAttributeDefNameLookup wsAttributeDefNameLookup : wsAttributeDefNameLookups) {
501         
502         if (wsAttributeDefNameLookup == null || !wsAttributeDefNameLookup.hasData()) {
503           continue;
504         }
505         
506         if (!foundRecords) {
507           lookupCount[0]++;
508           foundRecords = true;
509         }
510         
511         if (!usePIT) {
512           wsAttributeDefNameLookup.retrieveAttributeDefNameIfNeeded(grouperSession);
513         } else {
514           wsAttributeDefNameLookup.retrievePITAttributeDefNamesIfNeeded(null, pointInTimeFrom, pointInTimeTo);
515         }
516         
517         AttributeDefName attributeDefName = wsAttributeDefNameLookup.retrieveAttributeDefName();
518         Set<PITAttributeDefName> pitAttributeDefNames = wsAttributeDefNameLookup.retrievePITAttributeDefNames();
519         
520         if (!usePIT && attributeDefName != null) {
521           if (attributeDefType != null) {
522             AttributeDef attributeDef = attributeDefName.getAttributeDef();
523             if (attributeDefType == attributeDef.getAttributeDefType()) {
524               attributeDefNameIds.add(attributeDefName.getId());
525             } else {
526               
527               if (errorMessage.length() > 0) {
528                 errorMessage.append(", ");
529               }
530               
531               errorMessage.append("Error on attributeDefName index: " + i + ", expecting attributeDefType: " 
532                   + attributeDefType + ", " 
533                   + wsAttributeDefNameLookup.toStringCompact());
534               
535             }
536           } else {
537             attributeDefNameIds.add(attributeDefName.getId());
538           }
539         } else if (usePIT && pitAttributeDefNames != null && pitAttributeDefNames.size() > 0) {
540           for (PITAttributeDefName pitAttributeDefName : pitAttributeDefNames) {
541             if (attributeDefType == null) {
542               attributeDefNameIds.add(pitAttributeDefName.getSourceId());
543             } else {
544               PITAttributeDef pitAttributeDef = PITAttributeDefFinder.findById(pitAttributeDefName.getAttributeDefId(), true);
545               if (attributeDefType.name().equals(pitAttributeDef.getAttributeDefTypeDb())) {
546                 attributeDefNameIds.add(pitAttributeDefName.getSourceId());
547               } else {
548                 if (errorMessage.length() > 0) {
549                   errorMessage.append(", ");
550                 }
551                 
552                 errorMessage.append("Error on attributeDefName index: " + i + ", expecting attributeDefType: " 
553                     + attributeDefType + ", " 
554                     + wsAttributeDefNameLookup.toStringCompact());
555                 
556               }
557             }
558           }
559         } else {
560           
561           if (errorMessage.length() > 0) {
562             errorMessage.append(", ");
563           }
564           
565           errorMessage.append("Error on attributeDefName index: " + i + ", " 
566               + wsAttributeDefNameLookup.retrieveAttributeDefNameFindResult() + ", " 
567               + wsAttributeDefNameLookup.toStringCompact());
568         }
569         
570         i++;
571       }
572       
573     }
574     return attributeDefNameIds;
575   }

578    * make sure this is an explicit toString
579    * @return return a compact to string
580    */
581   public String toStringCompact() {
582     if (!StringUtils.isBlank(this.name)) {
583       return "name: " + this.name;
584     }
585     if (!StringUtils.isBlank(this.uuid)) {
586       return "id: " + this.uuid;
587     }
588     if (!StringUtils.isBlank(this.idIndex)) {
589       return "idIndex: " + this.idIndex;
590     }
591     return "blank";
592   }
