43    /**
44     * assign attribute def name inheritance lite web service with REST
45     * @param wsSampleRestType is the type of rest (xml, xhtml, etc)
46     */
47    public static void assignAttributeDefNameInheritanceLite(WsSampleRestType wsSampleRestType) {
48  
49      try {
50        HttpClient httpClient = new HttpClient();
51        
52        DefaultHttpParams.getDefaultParams().setParameter(
53            HttpMethodParams.RETRY_HANDLER, new DefaultHttpMethodRetryHandler(0, false));
54  
55        //URL e.g. http://localhost:8093/grouper-ws/servicesRest/v1_3_000/...
56        //NOTE: aStem:aAttributeDefName urlencoded substitutes %3A for a colon
57        PostMethod method = new PostMethod(
58            RestClientSettings.URL + "/" + RestClientSettings.VERSION  
59              + "/attributeDefNames");
60        
61        httpClient.getParams().setAuthenticationPreemptive(true);
62        Credentials defaultcreds = new UsernamePasswordCredentials(RestClientSettings.USER, 
63            RestClientSettings.PASS);
64        
65        //no keep alive so response if easier to indent for tests
66        method.setRequestHeader("Connection", "close");
67        
68        //e.g. localhost and 8093
69        httpClient.getState()
70            .setCredentials(new AuthScope(RestClientSettings.HOST, RestClientSettings.PORT), defaultcreds);
71  
72        //Make the body of the request, in this case with beans and marshaling, but you can make
73        //your request document in whatever language or way you want
74        WsRestAssignAttributeDefNameInheritanceLiteRequest assignAttributeDefNameInheritanceLite = new WsRestAssignAttributeDefNameInheritanceLiteRequest();
75  
76        //parent of the inheritance
77        assignAttributeDefNameInheritanceLite.setAttributeDefNameName("aStem:permissionDefName");
78        
79        //we are doing an assignment
80        assignAttributeDefNameInheritanceLite.setAssign("T");
81        
82        //this is the child of the relation
83        assignAttributeDefNameInheritanceLite.setRelatedAttributeDefNameName("aStem:permissionDefName3");
84              
85        //get the xml / json / xhtml / paramString
86        String requestDocument = wsSampleRestType.getWsLiteRequestContentType().writeString(assignAttributeDefNameInheritanceLite);
87        
88        //make sure right content type is in request (e.g. application/xhtml+xml
89        String contentType = wsSampleRestType.getWsLiteRequestContentType().getContentType();
90        
91        method.setRequestEntity(new StringRequestEntity(requestDocument, contentType, "UTF-8"));
92        
93        httpClient.executeMethod(method);
94  
95        //make sure a request came back
96        Header successHeader = method.getResponseHeader("X-Grouper-success");
97        String successString = successHeader == null ? null : successHeader.getValue();
98        if (StringUtils.isBlank(successString)) {
99          throw new RuntimeException("Web service did not even respond!");
100       }
101       boolean success = "T".equals(successString);
102       String resultCode = method.getResponseHeader("X-Grouper-resultCode").getValue();
103       
104       String response = RestClientSettings.responseBodyAsString(method);
105 
106       Object resultObject = wsSampleRestType.getWsLiteResponseContentType().parseString(response);
107     
108       //see if problem
109       if (resultObject instanceof WsRestResultProblem) {
110         throw new RuntimeException(((WsRestResultProblem)resultObject).getResultMetadata().getResultMessage());
111       }
112 
113       //convert to object (from xhtml, xml, json, etc)
114       WsAssignAttributeDefNameInheritanceResults wsAssignAttributeDefNameInheritanceResults = (WsAssignAttributeDefNameInheritanceResults)resultObject;
115       
116       String resultMessage = wsAssignAttributeDefNameInheritanceResults.getResultMetadata().getResultMessage();
117 
118       // see if request worked or not
119       if (!success) {
120         throw new RuntimeException("Bad response from web service: resultCode: " + resultCode
121             + ", " + resultMessage);
122       }
123       
124       System.out.println("Server version: " + wsAssignAttributeDefNameInheritanceResults.getResponseMetadata().getServerVersion()
125           + ", result code: " + resultCode
126           + ", result message: " + resultMessage );
127 
128     } catch (Exception e) {
129       throw new RuntimeException(e);
130     }
131 
132   }

139   }
