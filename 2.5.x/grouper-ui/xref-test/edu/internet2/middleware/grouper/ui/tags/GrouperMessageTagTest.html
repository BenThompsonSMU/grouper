33  public class GrouperMessageTagTest extends TestCase {

35  	  /**
36  	   * Method main.

39  	  public static void main(String[] args) {
40  	    TestRunner.run(GrouperMessageTagTest.class);
41  	  }
42  	  
43  	  /**
44  	   * try a substitute call where a prefix happens later in the properties
45  	   */
46  	  public void testSubstitutePrefix() {
47  		  Map<String, String> map = new LinkedHashMap<String, String>();
48  		  map.put("b", "whatever4");
49  		  map.put("ab", "whatever3");
50  		  map.put("abc", "whatever");
51  		  map.put("abcd", "whatever2");
52  		  map = GrouperMessageTag.sortMapBySubstringFirst(map);
53  		  Iterator<String> iterator = map.keySet().iterator();
54  		  String key1 = iterator.next();
55  		  assertEquals(key1, "abcd");
56  		  String key2 = iterator.next();
57  		  assertEquals(key2, "abc");
58  		  String key3 = iterator.next();
59  		  assertEquals(key3, "ab");
60  		  String key4 = iterator.next();
61  		  assertEquals(key4, "b");
62  	  }
