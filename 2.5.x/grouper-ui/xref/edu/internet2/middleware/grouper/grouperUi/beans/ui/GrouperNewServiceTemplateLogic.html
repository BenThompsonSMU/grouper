22    /**
23    Do you want a "app:Wiki" folder created? (ID is "wiki", name is "Wiki")
24      Assign "app" type to the "app:Wiki" folder?
25    
26      Do you want a "app:Wiki:service" folder created?
27        Do you want a "app:Wiki:service:policy" folder created?
28          Do you want the policy folder assigned the "policy group" type?
29        Do you want a "app:Wiki:service:reference" folder created? (ID is "ref", name is "reference")
30          Do you want the ref folder assigned the "ref" type?
31        Do you want a "app:Wiki:service:attribute" folder created?
32    
33      Do you want a "app:Wiki:security" folder created?
34        Do you want the security folder assigned the "grouperSecurity" type?
35        Do you want a "app:Wiki:security:Wiki Admins" group created? (ID is "wikiAdmins", name is "Wiki Admins")
36          Do you want "app:Wiki:security:Wiki Admins" to have inherited ADMIN privileges on Groups on the "app:Wiki" folder?
37          Do you want "app:Wiki:security:Wiki Admins" to have inherited ADMIN privileges on Folders on the "app:Wiki" folder?
38          Do you want "app:Wiki:security:Wiki Admins" to have inherited ADMIN privileges on Attributes on the "app:Wiki" folder?
39      
40        Do you want a "app:Wiki:security:Wiki Readers" group created? (ID is "wikiReaders", name is "Wiki Readers")
41          Do you want "app:Wiki:security:Wiki Readers" to have inherited READ privileges on Groups on the "app:Wiki" folder?
42        Do you want a "app:Wiki:security:Wiki Updaters" group created? (ID is "wikiUpdaters", name is "Wiki Updaters")
43          Do you want "app:Wiki:security:Wiki Updaters" to have inherited UPDATE privileges on Groups on the "app:Wiki:service" folder?
44          Do you want "app:Wiki:security:Wiki Updaters" to be a member of "app:wiki:security:Wiki Readers"?
45     */
46    @Override
47    public List<ServiceAction> getServiceActions() {
48      
49      GrouperSession grouperSession = GrouperSession.staticGrouperSession();
50      Stem stem = StemFinder.findByUuid(grouperSession, this.getStemId(), true);
51      
52      StemTemplateContainer templateContainer = this.getStemTemplateContainer();
53      String baseStem = templateContainer.getTemplateKey();
54      String baseStemFriendlyName = templateContainer.getTemplateFriendlyName();
55      
56      if (StringUtils.isBlank(baseStemFriendlyName)) {
57        baseStemFriendlyName = baseStem;
58      }
59      
60      String baseStemDescription = StringUtils.isBlank(templateContainer.getTemplateDescription()) ?
61          TextContainer.retrieveFromRequest().getText().get("stemServiceBaseFolderDescription"): templateContainer.getTemplateDescription();
62      
63      String stemPrefix = "";
64      String stemPrefixDisplayName = "";
65      
66      boolean addFirstNode = false;
67      String optionalColon = "";
68      
69      if (StringUtils.isBlank(baseStem) && stem.isRootStem()) {
70        stemPrefix = "";
71        stemPrefixDisplayName = "";
72        baseStem = "";
73        baseStemFriendlyName = "";
74      } else if (StringUtils.isBlank(baseStem) && !stem.isRootStem()) {
75        stemPrefix = stem.getName();
76        stemPrefixDisplayName = stem.getDisplayName();
77        baseStem = "";
78        baseStemFriendlyName = "";
79        optionalColon = ":";
80      } else if (StringUtils.isNotBlank(baseStem) && stem.isRootStem()) {
81        stemPrefix = "";
82        stemPrefixDisplayName = "";
83        optionalColon = ":";
84        addFirstNode = true;
85      } else if (StringUtils.isNotBlank(baseStem) && !stem.isRootStem()) {
86        stemPrefix = stem.getName()+":";
87        stemPrefixDisplayName = stem.getDisplayName()+":";
88        optionalColon = ":";
89        addFirstNode = true;
90      }
91  
92      
93        List<ServiceAction> serviceActionsForStem = new ArrayList<ServiceAction>();
94        
95        List<ServiceActionArgument> args = new ArrayList<ServiceActionArgument>();
96      
97        //Do you want a "app:Wiki" folder created? (ID is "wiki", name is "Wiki")
98        args.add(new ServiceActionArgument("stemName", stemPrefix+baseStem));
99        args.add(new ServiceActionArgument("stemDisplayName", stemPrefixDisplayName+baseStemFriendlyName));
100       args.add(new ServiceActionArgument("stemDescription", baseStemDescription));
101       ServiceAction rootServiceAction = createNewServiceAction("newAppStem", true, 0, "stemServiceBaseFolderCreationConfirmation", ServiceActionType.stem, args, null);
102       
103       if (addFirstNode) {
104         serviceActionsForStem.add(rootServiceAction);
105       }
106       
107       //Do you want a "app:Wiki:service" folder created?
108       args = new ArrayList<ServiceActionArgument>();
109       args.add(new ServiceActionArgument("stemName", stemPrefix+baseStem+optionalColon+"service"));
110       args.add(new ServiceActionArgument("stemDisplayName", stemPrefixDisplayName+baseStemFriendlyName+optionalColon+"service"));
111       args.add(new ServiceActionArgument("stemDescription", TextContainer.retrieveFromRequest().getText().get("stemServiceServiceFolderDescription")));
112       ServiceAction levelOneServiceAction_One = createNewServiceAction("newAppServiceFolder", true, 1, "stemServiceBaseFolderCreationConfirmation", ServiceActionType.stem, args, 
113           addFirstNode ? rootServiceAction: null);
114       serviceActionsForStem.add(levelOneServiceAction_One);
115       if (addFirstNode) {        
116         rootServiceAction.addChildServiceAction(levelOneServiceAction_One);
117       }
118       
119       //Do you want a "app:Wiki:service:policy" folder created?
120       args = new ArrayList<ServiceActionArgument>();
121       final String stemNamePolicy = stemPrefix+baseStem+optionalColon+"service:policy";
122       args.add(new ServiceActionArgument("stemName", stemNamePolicy));
123       final String stemDisplayNamePolicy = stemPrefixDisplayName+baseStemFriendlyName+optionalColon+"service:policy";
124       args.add(new ServiceActionArgument("stemDisplayName", stemDisplayNamePolicy));
125       args.add(new ServiceActionArgument("stemDescription", TextContainer.retrieveFromRequest().getText().get("stemServicePolicyFolderDescription")));
126       ServiceAction levelTwoServiceAction_One = createNewServiceAction("newAppPolicyFolder", true, 2, "stemServiceBaseFolderCreationConfirmation", ServiceActionType.stem, args, levelOneServiceAction_One);
127       serviceActionsForStem.add(levelTwoServiceAction_One);
128       levelOneServiceAction_One.addChildServiceAction(levelTwoServiceAction_One);
129 
130       {
131         //Assign the "policy" type to the "app:Wiki:service:policy" folder?
132         args = new ArrayList<ServiceActionArgument>();
133         args.add(new ServiceActionArgument("stemName", stemNamePolicy));
134         args.add(new ServiceActionArgument("stemDisplayName", stemDisplayNamePolicy));
135         args.add(new ServiceActionArgument("type", GrouperObjectTypesSettings.POLICY));
136         ServiceAction policyTypeAction = createNewServiceAction("newAppPolicyType", true, 3, "stemServiceFolderTypeConfirmation", ServiceActionType.grouperType, args, null);
137         
138         serviceActionsForStem.add(policyTypeAction);
139         if (addFirstNode) {        
140           rootServiceAction.addChildServiceAction(policyTypeAction);
141         }
142       }
143       
144       //Do you want a "apps:wiki:service:reference" folder created? (id is "ref", name is "reference")
145       args = new ArrayList<ServiceActionArgument>();
146       final String stemNameRef = stemPrefix+baseStem+optionalColon+"service:ref";
147       args.add(new ServiceActionArgument("stemName", stemNameRef));
148       final String stemDisplayNameRef = stemPrefixDisplayName+baseStemFriendlyName+optionalColon+"service:ref";
149       args.add(new ServiceActionArgument("stemDisplayName", stemDisplayNameRef));
150       args.add(new ServiceActionArgument("stemDescription", TextContainer.retrieveFromRequest().getText().get("stemServiceRefFolderDescription")));
151       ServiceAction levelTwoServiceAction_Two = createNewServiceAction("newAppRefFolder", true, 2, "stemServiceBaseFolderCreationConfirmation",
152           ServiceActionType.stem, args, levelOneServiceAction_One);
153       serviceActionsForStem.add(levelTwoServiceAction_Two);
154       levelOneServiceAction_One.addChildServiceAction(levelTwoServiceAction_Two);
155 
156       {
157         //Assign the "ref" type to the "app:Wiki:service:ref" folder?
158         args = new ArrayList<ServiceActionArgument>();
159         args.add(new ServiceActionArgument("stemName", stemNameRef));
160         args.add(new ServiceActionArgument("stemDisplayName", stemDisplayNameRef));
161         args.add(new ServiceActionArgument("type", GrouperObjectTypesSettings.REF));
162         ServiceAction refTypeAction = createNewServiceAction("newAppRefType", true, 3, "stemServiceFolderTypeConfirmation", ServiceActionType.grouperType, args, null);
163         
164         serviceActionsForStem.add(refTypeAction);
165         if (addFirstNode) {        
166           rootServiceAction.addChildServiceAction(refTypeAction);
167         }
168       }
169       
170       //Do you want a "app:Wiki:service:attribute" folder created?
171       args = new ArrayList<ServiceActionArgument>();
172       args.add(new ServiceActionArgument("stemName", stemPrefix+baseStem+optionalColon+"service:attribute"));
173       args.add(new ServiceActionArgument("stemDisplayName", stemPrefixDisplayName+baseStemFriendlyName+optionalColon+"service:attribute"));
174       args.add(new ServiceActionArgument("stemDescription", TextContainer.retrieveFromRequest().getText().get("stemServiceAttributeFolderDescription")));
175       ServiceAction levelTwoServiceAction_Three = createNewServiceAction("newAppAttributeFolder", true, 2, "stemServiceBaseFolderCreationConfirmation",
176           ServiceActionType.stem, args, levelOneServiceAction_One);
177       serviceActionsForStem.add(levelTwoServiceAction_Three);
178       levelOneServiceAction_One.addChildServiceAction(levelTwoServiceAction_Three);
179       
180       //Do you want a "app:Wiki:security" folder created?
181       args = new ArrayList<ServiceActionArgument>();
182       final String stemNameSecurity = stemPrefix+baseStem+optionalColon+"security";
183       args.add(new ServiceActionArgument("stemName", stemNameSecurity));
184       final String stemDisplayNameSecurity = stemPrefixDisplayName+baseStemFriendlyName+optionalColon+"security";
185       args.add(new ServiceActionArgument("stemDisplayName", stemDisplayNameSecurity));
186       args.add(new ServiceActionArgument("stemDescription", TextContainer.retrieveFromRequest().getText().get("stemServiceSecurityFolderDescription")));
187       ServiceAction levelOneServiceAction_Two = createNewServiceAction("newAppSecurityFolder", true, 1, "stemServiceBaseFolderCreationConfirmation", 
188           ServiceActionType.stem, args, addFirstNode? rootServiceAction: null);
189       serviceActionsForStem.add(levelOneServiceAction_Two);
190       if (addFirstNode) {        
191         rootServiceAction.addChildServiceAction(levelOneServiceAction_Two);
192       }
193 
194       {
195         //Assign the "grouperSecurity" type to the "app:Wiki:security" folder?
196         args = new ArrayList<ServiceActionArgument>();
197         args.add(new ServiceActionArgument("stemName", stemNameSecurity));
198         args.add(new ServiceActionArgument("stemDisplayName", stemDisplayNameSecurity));
199         args.add(new ServiceActionArgument("type", GrouperObjectTypesSettings.GROUPER_SECURITY));
200         ServiceAction securityTypeAction = createNewServiceAction("newAppSecurityType", true, 2, "stemServiceFolderTypeConfirmation", ServiceActionType.grouperType, args, null);
201         
202         serviceActionsForStem.add(securityTypeAction);
203         if (addFirstNode) {        
204           rootServiceAction.addChildServiceAction(securityTypeAction);
205         }
206       }
207       
208       //Do you want a "app:Wiki:security:Wiki Admins" group created? (ID is "wikiAdmins", name is "Wiki Admins")
209       args = new ArrayList<ServiceActionArgument>();
210       String securityAdminsName = stemPrefix+baseStem+optionalColon+"security:"+baseStem+"Admins";
211       args.add(new ServiceActionArgument("groupName", securityAdminsName));
212       String securityAdminsDisplayName =  stemPrefixDisplayName+baseStemFriendlyName+optionalColon+"security:"+baseStemFriendlyName+(StringUtils.equals(baseStem, baseStemFriendlyName) ? "" : " ") + "Admins";
213       args.add(new ServiceActionArgument("groupDisplayName", securityAdminsDisplayName));
214       args.add(new ServiceActionArgument("groupDescription", TextContainer.retrieveFromRequest().getText().get("stemServiceSecurityAdminsGroupDescription")));
215       ServiceAction levelTwoServiceAction_Four = createNewServiceAction("newAppAdminsGroup", true, 2, "stemServiceBaseGroupCreationConfirmation", 
216           ServiceActionType.group, args, levelOneServiceAction_Two);
217       serviceActionsForStem.add(levelTwoServiceAction_Four);
218       levelOneServiceAction_Two.addChildServiceAction(levelTwoServiceAction_Four);
219       
220       //Do you want "app:Wiki:security:Wiki Admins" to have inherited ADMIN privileges on Groups on the "app:Wiki" folder?
221       args = new ArrayList<ServiceActionArgument>();
222       args.add(new ServiceActionArgument("groupName", securityAdminsName));
223       args.add(new ServiceActionArgument("groupDisplayName", securityAdminsDisplayName));
224       args.add(new ServiceActionArgument("parentStemName", stemPrefix+baseStem));
225       args.add(new ServiceActionArgument("parentStemDisplayName", stemPrefixDisplayName+baseStemFriendlyName));
226       args.add(new ServiceActionArgument("privilegeType", "ADMIN"));
227       args.add(new ServiceActionArgument("internalPrivilegeName", "admin"));
228       args.add(new ServiceActionArgument("templateItemType", "Groups"));
229       ServiceAction levelThreeServiceAction_One = createNewServiceAction("newAppAdminPrivilege", true, 3, "stemServiceBasePrivilegeCreationConfirmation", 
230           ServiceActionType.inheritedPrivilege, args, levelTwoServiceAction_Four);
231       serviceActionsForStem.add(levelThreeServiceAction_One);
232       levelTwoServiceAction_Four.addChildServiceAction(levelThreeServiceAction_One);
233       
234       //Do you want "app:Wiki:security:Wiki Admins" to have inherited ADMIN privileges on Folders on the "app:Wiki" folder?
235       args = new ArrayList<ServiceActionArgument>();
236       args.add(new ServiceActionArgument("groupName", securityAdminsName));
237       args.add(new ServiceActionArgument("groupDisplayName", securityAdminsDisplayName));
238       args.add(new ServiceActionArgument("parentStemName", stemPrefix+baseStem));
239       args.add(new ServiceActionArgument("parentStemDisplayName", stemPrefixDisplayName+baseStemFriendlyName));
240       args.add(new ServiceActionArgument("privilegeType", "ADMIN"));
241       args.add(new ServiceActionArgument("internalPrivilegeName", "stemAdmin"));
242       args.add(new ServiceActionArgument("templateItemType", "Folders"));
243       ServiceAction levelThreeServiceAction_Two = createNewServiceAction("newAppAdminPrivilege2", true, 3, "stemServiceBasePrivilegeCreationConfirmation", 
244           ServiceActionType.inheritedPrivilege, args, levelTwoServiceAction_Four);
245       serviceActionsForStem.add(levelThreeServiceAction_Two);
246       levelTwoServiceAction_Four.addChildServiceAction(levelThreeServiceAction_Two);
247       
248       //Do you want "app:Wiki:security:Wiki Admins" to have inherited ADMIN privileges on Attributes on the "app:Wiki" folder?
249       args = new ArrayList<ServiceActionArgument>();
250       args.add(new ServiceActionArgument("groupName", securityAdminsName));
251       args.add(new ServiceActionArgument("groupDisplayName", securityAdminsDisplayName));
252       args.add(new ServiceActionArgument("parentStemName", stemPrefix+baseStem));
253       args.add(new ServiceActionArgument("parentStemDisplayName", stemPrefixDisplayName+baseStemFriendlyName));
254       args.add(new ServiceActionArgument("privilegeType", "ADMIN"));
255       args.add(new ServiceActionArgument("internalPrivilegeName", "attrAdmin"));
256       args.add(new ServiceActionArgument("templateItemType", "Attributes"));
257       ServiceAction levelThreeServiceAction_Three = createNewServiceAction("newAppAdminPrivilege3", true, 3, "stemServiceBasePrivilegeCreationConfirmation",
258           ServiceActionType.inheritedPrivilege, args, levelTwoServiceAction_Four);
259       serviceActionsForStem.add(levelThreeServiceAction_Three);
260       levelTwoServiceAction_Four.addChildServiceAction(levelThreeServiceAction_Three);
261       
262       //Do you want a "app:Wiki:security:Wiki Readers" group created? (ID is "wikiReaders", name is "Wiki Readers")
263       args = new ArrayList<ServiceActionArgument>();
264       String securityReadersName = stemPrefix+baseStem+optionalColon+"security:"+baseStem+"Readers";
265       args.add(new ServiceActionArgument("groupName", securityReadersName));
266       String securityReadersDisplayName = stemPrefixDisplayName+baseStemFriendlyName+optionalColon+"security:"+baseStemFriendlyName+(StringUtils.equals(baseStem, baseStemFriendlyName) ? "" : " ") +"Readers";
267       args.add(new ServiceActionArgument("groupDisplayName", securityReadersDisplayName));
268       args.add(new ServiceActionArgument("groupDescription", TextContainer.retrieveFromRequest().getText().get("stemServiceSecurityReadersGroupDescription")));
269       ServiceAction levelTwoServiceAction_Five = createNewServiceAction("newAppReadersGroup", true, 2, "stemServiceBaseGroupCreationConfirmation",
270           ServiceActionType.group, args, levelOneServiceAction_Two);
271       serviceActionsForStem.add(levelTwoServiceAction_Five);
272       levelOneServiceAction_Two.addChildServiceAction(levelTwoServiceAction_Five);
273       
274       //Do you want "app:Wiki:security:Wiki Readers" to have inherited READ privileges on Groups on the "app:Wiki" folder?
275       args = new ArrayList<ServiceActionArgument>();
276       args.add(new ServiceActionArgument("groupName", securityReadersName));
277       args.add(new ServiceActionArgument("groupDisplayName", securityReadersDisplayName));
278       args.add(new ServiceActionArgument("parentStemName", stemPrefix+baseStem));
279       args.add(new ServiceActionArgument("parentStemDisplayName", stemPrefixDisplayName+baseStemFriendlyName));
280       args.add(new ServiceActionArgument("privilegeType", "READ"));
281       args.add(new ServiceActionArgument("internalPrivilegeName", "read"));
282       args.add(new ServiceActionArgument("templateItemType", "Groups"));
283       ServiceAction levelThreeServiceAction_Four = createNewServiceAction("newAppReadersPrivilege", true, 3, "stemServiceBasePrivilegeCreationConfirmation", 
284           ServiceActionType.inheritedPrivilege, args, levelTwoServiceAction_Five);
285       serviceActionsForStem.add(levelThreeServiceAction_Four);
286       levelTwoServiceAction_Five.addChildServiceAction(levelThreeServiceAction_Four);
287       
288       //Do you want a "app:Wiki:security:Wiki Updaters" group created? (ID is "wikiUpdaters", name is "Wiki Updaters")
289       args = new ArrayList<ServiceActionArgument>();
290       String securityUpdatersName = stemPrefix+baseStem+optionalColon+"security:"+baseStem+"Updaters";
291       args.add(new ServiceActionArgument("groupName", securityUpdatersName));
292       String securityUpdatersDisplayName = stemPrefixDisplayName+baseStemFriendlyName+optionalColon+"security:"+baseStemFriendlyName+(StringUtils.equals(baseStem, baseStemFriendlyName) ? "" : " ")+"Updaters";
293       args.add(new ServiceActionArgument("groupDisplayName", securityUpdatersDisplayName));      
294       args.add(new ServiceActionArgument("groupDescription", TextContainer.retrieveFromRequest().getText().get("stemServiceSecurityUpdatersGroupDescription")));
295       ServiceAction levelTwoServiceAction_Six = createNewServiceAction("newAppUpdatersPrivilege", true, 2, "stemServiceBaseGroupCreationConfirmation", 
296           ServiceActionType.group, args, levelOneServiceAction_Two);
297       serviceActionsForStem.add(levelTwoServiceAction_Six);
298       levelOneServiceAction_Two.addChildServiceAction(levelTwoServiceAction_Six);
299       
300       //Do you want "app:Wiki:security:Wiki Updaters" to have inherited UPDATE privileges on Groups on the "app:Wiki:service" folder?
301       args = new ArrayList<ServiceActionArgument>();
302       args.add(new ServiceActionArgument("groupName", securityUpdatersName));
303       args.add(new ServiceActionArgument("groupDisplayName", securityUpdatersDisplayName));
304       args.add(new ServiceActionArgument("parentStemName", stemPrefix+baseStem+optionalColon+"service"));
305       args.add(new ServiceActionArgument("parentStemDisplayName", stemPrefixDisplayName+baseStemFriendlyName+optionalColon+"service"));
306       args.add(new ServiceActionArgument("privilegeType", "UPDATE"));
307       args.add(new ServiceActionArgument("internalPrivilegeName", "update"));
308       args.add(new ServiceActionArgument("templateItemType", "Groups"));
309       ServiceAction levelThreeServiceAction_Five = createNewServiceAction("newAppUpdatersPrivilege2", true, 3, "stemServiceBasePrivilegeCreationConfirmation", 
310           ServiceActionType.inheritedPrivilege, args, levelTwoServiceAction_Five);
311       serviceActionsForStem.add(levelThreeServiceAction_Five);
312       levelTwoServiceAction_Six.addChildServiceAction(levelThreeServiceAction_Five);
313       
314       //Do you want "app:Wiki:security:Wiki Updaters" to be a member of "app:Wiki:security:Wiki Readers"?
315       args = new ArrayList<ServiceActionArgument>();
316       args.add(new ServiceActionArgument("groupNameMembership", securityUpdatersName));
317       args.add(new ServiceActionArgument("groupNameMembershipDisplayName", securityUpdatersDisplayName));
318       args.add(new ServiceActionArgument("groupNameMembershipOf", securityReadersName));
319       args.add(new ServiceActionArgument("groupNameMembershipOfDisplayName", securityReadersDisplayName));
320       ServiceAction levelThreeServiceAction_Six = createNewServiceAction("newAppReadersGroupMemberOfUpdaters", true, 3, "stemServiceBaseMemberAdditionConfirmation", 
321           ServiceActionType.membership, args, levelTwoServiceAction_Five);
322       serviceActionsForStem.add(levelThreeServiceAction_Six);
323       levelTwoServiceAction_Six.addChildServiceAction(levelThreeServiceAction_Six);
324       
325     return serviceActionsForStem;
326     
327   }
