1   package edu.internet2.middleware.grouper.grouperUi.beans.ui;
2   
3   import java.text.SimpleDateFormat;
4   import java.util.Date;
5   
6   import edu.internet2.middleware.grouper.app.reports.GrouperReportConfigurationBean;
7   import edu.internet2.middleware.grouper.app.reports.GrouperReportInstance;
8   import edu.internet2.middleware.grouper.ui.GrouperUiFilter;
9   import edu.internet2.middleware.subject.Subject;
10  import net.redhogs.cronparser.CronExpressionDescriptor;
11  
12  public class GuiReportConfig {
13    
14    /**
15     * config bean
16     */
17    private GrouperReportConfigurationBean reportConfigBean;
18    
19    /**
20     * most recent report instance run 
21     */
22    private GrouperReportInstance mostRecentReportInstance;
23    
24    public GuiReportConfig(GrouperReportConfigurationBean reportConfigBean, GrouperReportInstance mostRecentReportInstance) {
25      this.reportConfigBean = reportConfigBean;
26      this.mostRecentReportInstance = mostRecentReportInstance;
27    }
28    
29    public String getLastRunTime() {
30      if (mostRecentReportInstance == null) {
31        return null;
32      }
33      Date date = new Date(mostRecentReportInstance.getReportInstanceMillisSince1970());
34      String lastRunTime = new SimpleDateFormat("yyyy/MM/dd HH:mm:ss").format(date);
35      return lastRunTime;
36    }
37  
38    public String getUserFriendlyCron() {
39      try {      
40        return CronExpressionDescriptor.getDescription(reportConfigBean.getReportConfigQuartzCron());
41      } catch(Exception e) {
42        return "";
43      }
44    }
45    
46    public GrouperReportConfigurationBean getReportConfigBean() {
47      return reportConfigBean;
48    }
49  
50    public GrouperReportInstance getMostRecentReportInstance() {
51      return mostRecentReportInstance;
52    }
53    
54    public boolean isCanRead() {
55      Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
56      return reportConfigBean.isCanRead(loggedInSubject);
57    }
58  
59  }
