95    /**
96     * validate import list
97     * @param request
98     * @param response
99     */
100   public void groupImportValidateList(HttpServletRequest request, HttpServletResponse response) {
101     
102     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
103     
104     GrouperSession grouperSession = null;
105   
106     try {
107       
108       grouperSession = GrouperSession.start(loggedInSubject);
109 
110       GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
111 
112       String entityList = StringUtils.defaultString(request.getParameter("entityList"));
113       
114       //split trim by comma, semi, or whitespace
115       entityList = StringUtils.replace(entityList, ",", " ");
116       entityList = StringUtils.replace(entityList, ";", " ");
117       
118       String[] entityIdOrIdentifiers = GrouperUtil.splitTrim(entityList, null, true);
119 
120       if (GrouperUtil.length(entityIdOrIdentifiers) == 0) {
121 
122         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
123             "#entityListId",
124             TextContainer.retrieveFromRequest().getText().get("groupImportNoEntitiesSpecified")));
125         return;
126 
127       }
128       
129       if (GrouperUtil.length(entityIdOrIdentifiers) > 100) {
130 
131         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
132             TextContainer.retrieveFromRequest().getText().get("groupImportTooManyEntitiesToValidate")));
133         return;
134 
135       }
136       
137       //extra source ids and subjects ids
138       Set<GuiSubject> extraGuiSubjects = new LinkedHashSet<GuiSubject>();
139       
140       GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupImportContainer().setGroupImportExtraGuiSubjects(extraGuiSubjects);
141 
142       String source = request.getParameter("searchEntitySourceName");
143       
144       List<String> entityIdOrIdentifiersList = new ArrayList<String>(Arrays.asList(GrouperUtil.nonNull(
145           entityIdOrIdentifiers, String.class)));
146       
147       Map<String, Subject> entityIdOrIdentifierMap = null;
148       
149       if (StringUtils.equals("all", source)) {
150 
151         entityIdOrIdentifierMap = SubjectFinder.findByIdsOrIdentifiers(entityIdOrIdentifiersList);
152         
153       } else {
154 
155         entityIdOrIdentifierMap = SubjectFinder.findByIdsOrIdentifiers(entityIdOrIdentifiersList, source);
156 
157       }
158       
159       //lets add all the subjects
160       for (Subject subject : GrouperUtil.nonNull(entityIdOrIdentifierMap).values()) {
161         extraGuiSubjects.add(new GuiSubject(subject));
162       }
163 
164       //lets see which are missing
165       entityIdOrIdentifiersList.removeAll(GrouperUtil.nonNull(entityIdOrIdentifierMap).keySet());
166       
167       if (entityIdOrIdentifiersList.size() > 0) {
168         GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupImportContainer().setEntityIdsNotFound(GrouperUtil.join(entityIdOrIdentifiersList.iterator(), ", "));
169         
170         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
171             TextContainer.retrieveFromRequest().getText().get("groupImportEntityIdsNotFound")));
172 
173       }
174 
175       //clear out combobox
176       guiResponseJs.addAction(GuiScreenAction.newScript(
177           "dijit.byId('groupAddMemberComboId').set('displayedValue', ''); " +
178           "dijit.byId('groupAddMemberComboId').set('value', '');"));
179 
180       //select the option for enter in list
181       guiResponseJs.addAction(GuiScreenAction.newFormFieldValue("bulkAddOptions", "input"));
182 
183       //fill in the extra subjects
184       guiResponseJs.addAction(GuiScreenAction.newInnerHtmlFromJsp("#groupImportExtraMembersDivId", 
185           "/WEB-INF/grouperUi2/groupImport/groupImportExtraSubjects.jsp"));
186 
187     } finally {
188       GrouperSession.stopQuietly(grouperSession);
189     }

251       
252       if (exportAll) {
253         String headersCommaSeparated = GrouperUiConfig.retrieveConfig().propertyValueString(
254             "uiV2.group.exportAllSubjectFields");

257             "uiV2.group.exportAllSortField");

259         SimpleMembershipUpdateImportExport.exportGroupAllFieldsToBrowser(group, headersCommaSeparated, exportAllSortField, false);

264       }

266       GrouperUserDataApi.recentlyUsedGroupAdd(GrouperUiUserData.grouperUiGroupNameForUserData(), 
267           loggedInSubject, group);
268 
269     } finally {
270       GrouperSession.stopQuietly(grouperSession);
271     }

354   /**
355    * setup the extra groups (other than combobox), and maybe move the combobox down
356    * @param loggedInSubject
357    * @param request
358    * @param removeGroupId if removing one
359    * @param includeCombobox
360    * @param allGroups, pass in a blank linked hash set, and all groups will be populated including combobox
361    * @param errorOnNullCombobox true if an error should appear if there is nothing in the combobox
362    * @return true if ok, false if not
363    */
364   private boolean groupImportSetupExtraGroups(Subject loggedInSubject, 
365       HttpServletRequest request, GuiResponseJs guiResponseJs, boolean considerRemoveGroupId, boolean includeCombobox,
366       Set<Group> allGroups, boolean errorOnNullCombobox) {
367 
368     Set<GuiGroup> extraGuiGroups = new LinkedHashSet<GuiGroup>();
369     
370     GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupImportContainer().setGroupImportExtraGuiGroups(extraGuiGroups);
371     
372     String removeGroupId = null;
373 
374     //if removing a group id
375     if (considerRemoveGroupId) {
376       removeGroupId = request.getParameter("removeGroupId");
377       if (StringUtils.isBlank(removeGroupId)) {
378         throw new RuntimeException("Why would removeGroupId be empty????");
379       }
380     }
381 
382     //if moving combobox down to extra list or getting all groups
383     String comboValue = request.getParameter("groupImportGroupComboName");
384     
385     if (StringUtils.isBlank(comboValue)) {
386       //if didnt pick one from results
387       comboValue = request.getParameter("groupImportGroupComboNameDisplay");
388     }
389     
390     Group theGroup = StringUtils.isBlank(comboValue) ? null : new GroupFinder()
391         .assignPrivileges(AccessPrivilege.UPDATE_PRIVILEGES)
392         .assignSubject(loggedInSubject)
393         .assignFindByUuidOrName(true).assignScope(comboValue).findGroup();
394 
395     boolean success = true;
396     
397     if (theGroup == null) {
398       if (includeCombobox && errorOnNullCombobox) {
399         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
400             "#groupImportGroupComboErrorId",
401             TextContainer.retrieveFromRequest().getText().get("groupImportGroupNotFound")));
402         success = false;
403       }
404       
405     } else {
406       if (includeCombobox) {
407         extraGuiGroups.add(new GuiGroup(theGroup));
408       }
409       //always add to all groups
410       allGroups.add(theGroup);
411     }
412 
413     //loop through all the hidden fields (max 100)
414     for (int i=0;i<100;i++) {
415       String extraGroupId = request.getParameter("extraGroupId_" + i);
416       
417       //we are at the end
418       if (StringUtils.isBlank(extraGroupId)) {
419         break;
420       }
421       
422       //might be removing this one
423       if (considerRemoveGroupId && StringUtils.equals(removeGroupId, extraGroupId)) {
424         continue;
425       }
426       
427       theGroup = new GroupFinder()
428         .assignPrivileges(AccessPrivilege.UPDATE_PRIVILEGES)
429         .assignSubject(loggedInSubject)
430         .assignFindByUuidOrName(true).assignScope(extraGroupId).findGroup();
431       
432       extraGuiGroups.add(new GuiGroup(theGroup));
433 
434       //always add to all groups
435       allGroups.add(theGroup);
436       
437     }
438     return success;
439   }

1131   /**
1132    * setup the extra members (other than combobox), and maybe move the combobox down
1133    * @param loggedInSubject
1134    * @param request
1135    * @param considerRemoveSubjectSourceAndId if removing one
1136    * @param includeCombobox
1137    * @param allSubjects is a LinkedHashSet of subjects
1138    * @param errorOnNullCombobox is true if an error should appear if there is nothing in the combobox
1139    * @return true if ok, false if not
1140    */
1141   private boolean groupImportSetupExtraSubjects(Subject loggedInSubject, 
1142       HttpServletRequest request, GuiResponseJs guiResponseJs, boolean considerRemoveSubjectSourceAndId, 
1143       boolean includeCombobox, Set<Subject> allSubjects, boolean errorOnNullCombobox) {
1144 
1145     //extra source ids and subjects ids
1146     Set<MultiKey> extraSubjectSourceAndIds = new HashSet<MultiKey>();
1147     Set<GuiSubject> extraGuiSubjects = new LinkedHashSet<GuiSubject>();
1148     
1149     GrouperRequestContainer.retrieveFromRequestOrCreate().getGroupImportContainer().setGroupImportExtraGuiSubjects(extraGuiSubjects);
1150     
1151     Set<MultiKey> allSubjectsSourceAndIds = new HashSet<MultiKey>();
1152   
1153     String removeSubjectSourceAndId = null;
1154   
1155     //if removing a group id
1156     if (considerRemoveSubjectSourceAndId) {
1157       removeSubjectSourceAndId = request.getParameter("removeSubjectSourceAndId");
1158       if (StringUtils.isBlank(removeSubjectSourceAndId)) {
1159         throw new RuntimeException("Why would removeSubjectSourceAndId be empty????");
1160       }
1161     }
1162 
1163     Subject theSubject = null;
1164     
1165     {
1166       //if moving combobox down to extra list or getting all groups
1167       String comboValue = request.getParameter("groupAddMemberComboName");
1168       
1169       if (StringUtils.isBlank(comboValue)) {
1170         //if didnt pick one from results
1171         comboValue = request.getParameter("groupAddMemberComboDisplay");
1172       }
1173       
1174       try {
1175         GrouperSourceAdapter.searchForGroupsWithReadPrivilege(true);
1176         if (comboValue != null && comboValue.contains("||")) {
1177           String sourceId = GrouperUtil.prefixOrSuffix(comboValue, "||", true);
1178           String subjectId = GrouperUtil.prefixOrSuffix(comboValue, "||", false);
1179           theSubject =  SubjectFinder.findByIdOrIdentifierAndSource(subjectId, sourceId, false);
1180         } else {
1181           try {
1182             theSubject = StringUtils.isBlank(comboValue) ? null : SubjectFinder.findByIdOrIdentifier(comboValue, false);
1183           } catch (SubjectNotUniqueException snue) {
1184             //ignore
1185           }
1186         }
1187       } finally {
1188         GrouperSourceAdapter.clearSearchForGroupsWithReadPrivilege();
1189       }
1190     }
1191     boolean success = true;
1192     if (theSubject == null) {
1193       if (includeCombobox && errorOnNullCombobox) {
1194         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
1195             "#groupAddMemberComboErrorId", 
1196             TextContainer.retrieveFromRequest().getText().get("groupImportSubjectNotFound")));
1197         success = false;
1198       }
1199       
1200     } else {
1201       MultiKey multiKey = new MultiKey(theSubject.getSourceId(), theSubject.getId());
1202       if (includeCombobox) {
1203         if (!extraSubjectSourceAndIds.contains(multiKey)) {
1204           extraGuiSubjects.add(new GuiSubject(theSubject));
1205           extraSubjectSourceAndIds.add(multiKey);
1206         }
1207       }
1208       //always add to all groups
1209       if (!allSubjectsSourceAndIds.contains(multiKey)) {
1210         allSubjects.add(theSubject);
1211         allSubjectsSourceAndIds.add(multiKey);
1212       }
1213     }
1214   
1215     //loop through all the hidden fields (max 100)
1216     for (int i=0;i<100;i++) {
1217       String extraSourceIdSubjectId = request.getParameter("extraSourceIdSubjectId_" + i);
1218       
1219       //we are at the end
1220       if (StringUtils.isBlank(extraSourceIdSubjectId)) {
1221         break;
1222       }
1223       
1224       //might be removing this one
1225       if (considerRemoveSubjectSourceAndId && StringUtils.equals(removeSubjectSourceAndId, extraSourceIdSubjectId)) {
1226         continue;
1227       }
1228       
1229       theSubject = null;
1230 
1231       try {
1232         GrouperSourceAdapter.searchForGroupsWithReadPrivilege(true);
1233         if (extraSourceIdSubjectId != null && extraSourceIdSubjectId.contains("||")) {
1234           String sourceId = GrouperUtil.prefixOrSuffix(extraSourceIdSubjectId, "||", true);
1235           String subjectId = GrouperUtil.prefixOrSuffix(extraSourceIdSubjectId, "||", false);
1236           theSubject =  SubjectFinder.findByIdOrIdentifierAndSource(subjectId, sourceId, false);
1237         }
1238       } finally {
1239         GrouperSourceAdapter.clearSearchForGroupsWithReadPrivilege();
1240       }
1241       
1242       if (theSubject != null) {
1243         MultiKey multiKey = new MultiKey(theSubject.getSourceId(), theSubject.getId());
1244         if (!extraSubjectSourceAndIds.contains(multiKey)) {
1245           extraGuiSubjects.add(new GuiSubject(theSubject));
1246           extraSubjectSourceAndIds.add(multiKey);
1247         }
1248   
1249         if (!allSubjectsSourceAndIds.contains(multiKey)) {
1250           allSubjects.add(theSubject);
1251           allSubjectsSourceAndIds.add(multiKey);
1252         }
1253       }      
1254     }
1255     return success;
1256   }
