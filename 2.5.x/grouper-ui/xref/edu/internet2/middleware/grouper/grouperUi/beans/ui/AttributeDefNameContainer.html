16  /**
17   * attribute definition name container in new ui
18   * @author vsachdeva
19   */
20  public class AttributeDefNameContainer {
21    
22    private GuiAttributeDefName guiAttributeDefName;
23  
24    public GuiAttributeDefName getGuiAttributeDefName() {
25      return guiAttributeDefName;
26    }
27  
28    public void setGuiAttributeDefName(GuiAttributeDefName guiAttributeDefName) {
29      this.guiAttributeDefName = guiAttributeDefName;
30    }
31    
32    /**
33     * if the attribute def name is a favorite for the logged in user
34     */
35    private Boolean favorite;
36  
37    /**
38     * if the attribute def name is a favorite for the logged in user
39     * @return if favorite
40     */
41    public boolean isFavorite() {
42      
43      if (this.favorite == null) {
44        
45        final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
46    
47        this.favorite = (Boolean)GrouperSession.callbackGrouperSession(
48            GrouperSession.staticGrouperSession().internal_getRootSession(), new GrouperSessionHandler() {
49              
50              @Override
51              public Object callback(GrouperSession grouperSession) throws GrouperSessionException {
52                
53                Set<AttributeDefName> favorites = GrouperUtil.nonNull(
54                    GrouperUserDataApi.favoriteAttributeDefNames(GrouperUiUserData.grouperUiGroupNameForUserData(), loggedInSubject));
55                return favorites.contains(AttributeDefNameContainer.this.getGuiAttributeDefName().getAttributeDefName());
56                    
57              }
58            });
59      }
60      
61      return this.favorite;
62    }
