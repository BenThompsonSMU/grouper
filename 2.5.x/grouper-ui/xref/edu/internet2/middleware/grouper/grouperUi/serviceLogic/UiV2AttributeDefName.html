136   /**
137    * delete attribute def names
138    * @param request
139    * @param response
140    */
141   public void deleteAttributeDefNamesSubmit(HttpServletRequest request, HttpServletResponse response) {
142   
143     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
144 
145     GrouperSession grouperSession = null;
146   
147     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
148   
149     AttributeDef attributeDef = null;
150   
151     try {
152       grouperSession = GrouperSession.start(loggedInSubject);
153   
154       attributeDef = UiV2AttributeDef.retrieveAttributeDefHelper(request, AttributeDefPrivilege.ATTR_ADMIN, true).getAttributeDef();
155       
156       if (attributeDef == null) {
157         return;
158       }
159   
160   
161       Set<String> attributeDefNameIds = new HashSet<String>();
162       
163       for (int i=0;i<1000;i++) {
164         String attributeDefNameId = request.getParameter("attributeDefName_" + i + "[]");
165         if (!StringUtils.isBlank(attributeDefNameId)) {
166           attributeDefNameIds.add(attributeDefNameId);
167         }
168       }
169   
170       if (attributeDefNameIds.size() == 0) {
171         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
172             TextContainer.retrieveFromRequest().getText().get("attributeDefRemoveNoAttributeDefNameSelects")));
173         return;
174       }
175       int successes = 0;
176       int failures = 0;
177       
178       for (String attributeDefNameId : attributeDefNameIds) {
179         try {
180 
181           AttributeDefName attributeDefName = AttributeDefNameFinder.findById(attributeDefNameId, false);
182           
183           //not sure why this would happen
184           if (attributeDefName == null) {
185             
186             if (LOG.isDebugEnabled()) {
187               LOG.debug("Cant find attribute def name for id: " + attributeDefNameId + ", maybe it was already deleted");
188             }
189 
190             failures++;
191             
192           } else {
193       
194             attributeDefName.delete();
195       
196             successes++;
197           }
198           
199           
200         } catch (Exception e) {
201           LOG.warn("Error deleting attributeDefName id: " + attributeDefNameId + ", user: " + loggedInSubject, e);
202           failures++;
203         }
204       }
205   
206       GrouperRequestContainer.retrieveFromRequestOrCreate().getAttributeDefContainer().setSuccessCount(successes);
207       GrouperRequestContainer.retrieveFromRequestOrCreate().getAttributeDefContainer().setFailureCount(failures);
208   
209       if (failures > 0) {
210         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
211             TextContainer.retrieveFromRequest().getText().get("attributeDefDeleteAttributeDefNamesErrors")));
212       } else {
213         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.success, 
214             TextContainer.retrieveFromRequest().getText().get("attributeDefDeleteAttributeDefNamesSuccesses")));
215       }
216       
217       
218       //filterHelper(request, response, attributeDef);
219   
220       GrouperUserDataApi.recentlyUsedAttributeDefAdd(GrouperUiUserData.grouperUiGroupNameForUserData(), 
221           loggedInSubject, attributeDef);
222   
223     } finally {
224       GrouperSession.stopQuietly(grouperSession);
225     }
226

342   public void newAttributeDefName(HttpServletRequest request, HttpServletResponse response) {
343     
344     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
345     
346     GrouperSession grouperSession = null;
347   
348     try {
349   
350       grouperSession = GrouperSession.start(loggedInSubject);
351 
352       //see if there is a stem id for this
353       String objectStemId = request.getParameter("objectStemId");
354       
355       Pattern pattern = Pattern.compile("^[a-zA-Z0-9-_]+$");
356       
357       if (!StringUtils.isBlank(objectStemId) && pattern.matcher(objectStemId).matches()) {
358         GrouperRequestContainer.retrieveFromRequestOrCreate().getStemContainer().setObjectStemId(objectStemId);
359       }
360       
361       GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
362       
363       UiV2Stem.retrieveStemHelper(request, false, false, false).getStem();
364       
365       if (objectStemId != null) {
366         QueryOptions queryOptions = QueryOptions.create(null, null, 1, 2);
367         Set<AttributeDef> attributes = new AttributeDefFinder().assignPrivileges(NamingPrivilege.CREATE_PRIVILEGES)
368             .assignStemScope(Scope.ONE).assignSubject(loggedInSubject)
369             .assignQueryOptions(queryOptions).assignParentStemId(objectStemId).findAttributes();
370         
371         if (attributes.size() == 1) {
372           GrouperRequestContainer.retrieveFromRequestOrCreate().getAttributeDefContainer()
373           .setObjectAttributeDefId(attributes.iterator().next().getId());
374         }
375       }
376       
377       String attributeDefId = request.getParameter("attributeDefId");
378       
379       if (StringUtils.isNotBlank(attributeDefId)) {
380         AttributeDef attributeDef = AttributeDefFinder.findById(attributeDefId, false);
381         if (attributeDef == null) {
382           guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
383               TextContainer.retrieveFromRequest().getText().get("attributeDefCantFindAttributeDef")));
384           return;
385         }
386         
387         GrouperRequestContainer.retrieveFromRequestOrCreate().getAttributeDefContainer()
388         .setObjectAttributeDefId(attributeDef.getId());
389         
390       }
391       
392       guiResponseJs.addAction(GuiScreenAction.newInnerHtmlFromJsp("#grouperMainContentDivId", 
393           "/WEB-INF/grouperUi2/attributeDefName/newAttributeDefName.jsp"));
394   
395     } finally {
396       GrouperSession.stopQuietly(grouperSession);
397     }

406   public void newAttributeDefNameSubmit(HttpServletRequest request, HttpServletResponse response) {
407     
408     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
409   
410     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
411   
412     GrouperSession grouperSession = null;
413   
414     AttributeDefName attributeDefName = null;
415     
416     try {
417   
418       grouperSession = GrouperSession.start(loggedInSubject);
419       
420       String parentFolderId = request.getParameter("parentFolderComboName");
421       String attributeDefId = request.getParameter("attributeDefComboName");
422       String displayExtension = request.getParameter("attributeDefNameToEditDisplayExtension");
423       String extension = request.getParameter("attributeDefNameToEditExtension");
424       String description = request.getParameter("attributeDefNameToEditDescription");
425       
426       if (StringUtils.isBlank(attributeDefId)) {
427         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error,
428             "#attributeDefComboErrorId",
429             TextContainer.retrieveFromRequest().getText().get("attributeDefNameCreateRequiredAttributeDef")));
430         return;
431       }
432       
433       if (StringUtils.isBlank(parentFolderId)) {
434         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error,
435             "#parentFolderComboErrorId",
436             TextContainer.retrieveFromRequest().getText().get("attributeDefNameCreateRequiredParentStemId")));
437         return;
438       }
439       
440       if (StringUtils.isBlank(displayExtension)) {
441         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
442             "#name",
443             TextContainer.retrieveFromRequest().getText().get("attributeDefNameCreateErrorDisplayExtensionRequired")));
444         return;
445       }
446 
447       if (StringUtils.isBlank(extension)) {
448         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error, 
449             "#attributeDefNameId",
450             TextContainer.retrieveFromRequest().getText().get("attributeDefNameCreateErrorExtensionRequired")));
451         return;
452       }
453       
454       final Stem parentFolder = new StemFinder().assignPrivileges(NamingPrivilege.CREATE_PRIVILEGES)
455           .assignSubject(loggedInSubject)
456           .assignScope(parentFolderId).assignFindByUuidOrName(true).findStem();
457 
458       if (parentFolder == null) {
459         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error,
460             "#parentFolderComboErrorId",
461             TextContainer.retrieveFromRequest().getText().get("attributeDefNameCreateCantFindStemId")));
462         return;
463       }
464       
465       String stemName = parentFolder.getName();
466       String attributeDefNameName = stemName + ":" + extension;
467       if (AttributeDefNameFinder.findByName(attributeDefNameName, false) != null) {
468         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error,
469             TextContainer.retrieveFromRequest().getText().get("attributeDefNameExistsError")));
470         return;
471       }
472       
473       AttributeDef attributeDef = AttributeDefFinder.findById(attributeDefId, false);
474       if (attributeDef == null) {
475         guiResponseJs.addAction(GuiScreenAction.newValidationMessage(GuiMessageType.error,
476             "#attributeDefComboErrorId",
477             TextContainer.retrieveFromRequest().getText().get("attributeDefNameCreateRequiredAttributeDef")));
478         return;
479       }
480       
481       if (!attributeDef.getPrivilegeDelegate().canAttrAdmin(loggedInSubject)) {
482         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error,
483             TextContainer.retrieveFromRequest().getText().get("simpleAttributeNameUpdate.errorCantEditAttributeDef")));
484         return;
485       }
486          
487       attributeDefName = parentFolder.addChildAttributeDefName(attributeDef, extension, displayExtension);
488       attributeDefName.setDescription(description);
489       
490       attributeDefName.setDisplayExtensionDb(displayExtension);
491       
492       attributeDefName.store();
493       
494       //go to the view attribute def screen
495       guiResponseJs.addAction(GuiScreenAction.newScript("guiV2link('operation=UiV2AttributeDefName.viewAttributeDefName?attributeDefNameId=" + attributeDefName.getId() + "')"));
496 
497       //lets show a success message on the new screen
498       guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.success, 
499           TextContainer.retrieveFromRequest().getText().get("attributeDefNameCreateSuccess")));
500 
501     } catch (Exception e) {
502       LOG.warn("Error creating attribute def name: " + SubjectHelper.getPretty(loggedInSubject) + ", " + attributeDefName, e);
503       if (GrouperUiUtils.vetoHandle(guiResponseJs, e)) {
504         return;
505       }
506       guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
507           TextContainer.retrieveFromRequest().getText().get("attributeDefNameCreateError") 
508           + ": " + GrouperUtil.xmlEscape(e.getMessage(), true)));
509       return;
510     } finally {
511       GrouperSession.stopQuietly(grouperSession);
512     }

568   /**
569    * get the attribute def from the request
570    * @param request
571    * @param requirePrivilege
572    * @param requireAttributeDefName
573    * @return the stem finder result
574    */
575   public static RetrieveAttributeDefNameHelperResult retrieveAttributeDefNameHelper(HttpServletRequest request, 
576       Privilege requirePrivilege, boolean requireAttributeDefName) {
577 
578     //initialize the bean
579     GrouperRequestContainer grouperRequestContainer = GrouperRequestContainer.retrieveFromRequestOrCreate();
580     
581     AttributeDefContainer attributeDefContainer = grouperRequestContainer.getAttributeDefContainer();
582     AttributeDefNameContainer attributeDefNameContainer = grouperRequestContainer.getAttributeDefNameContainer();
583     
584     RetrieveAttributeDefNameHelperResult result = new RetrieveAttributeDefNameHelperResult();
585 
586     AttributeDefName attributeDefName = null;
587 
588     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
589 
590     String attributeDefNameId = request.getParameter("attributeDefNameId");
591     String attributeDefNameIndex = request.getParameter("attributeDefNameIndex");
592     String nameOfAttributeDefName = request.getParameter("nameOfAttributeDefName");
593     
594     boolean addedError = false;
595     
596     if (!StringUtils.isBlank(attributeDefNameId)) {
597       attributeDefName = AttributeDefNameFinder.findById(attributeDefNameId, false);
598     } else if (!StringUtils.isBlank(nameOfAttributeDefName)) {
599       attributeDefName = AttributeDefNameFinder.findByName(nameOfAttributeDefName, false);
600     } else if (!StringUtils.isBlank(attributeDefNameIndex)) {
601       long idIndex = GrouperUtil.longValue(attributeDefNameIndex);
602       attributeDefName = AttributeDefNameFinder.findByIdIndexSecure(idIndex, false, null);
603     } else {
604       
605       if (!requireAttributeDefName) {
606         return result;
607       }
608       
609       guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
610           TextContainer.retrieveFromRequest().getText().get("attributeDefNameCantFindAttributeDefNameId")));
611       addedError = true;
612     }
613 
614     
615     if (attributeDefName != null) {
616       attributeDefNameContainer.setGuiAttributeDefName(new GuiAttributeDefName(attributeDefName));
617       
618       AttributeDef attributeDef = attributeDefName.getAttributeDef();
619       attributeDefContainer.setGuiAttributeDef(new GuiAttributeDef(attributeDef));      
620       
621       boolean privsOk = true;
622 
623       if (requirePrivilege != null) {
624         if (requirePrivilege.equals(AttributeDefPrivilege.ATTR_ADMIN)) {
625           if (!attributeDefContainer.isCanAdmin()) {
626             guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
627                 TextContainer.retrieveFromRequest().getText().get("attributeDefNotAllowedToAdminAttributeDef")));
628             addedError = true;
629             privsOk = false;
630           }
631         } else if (requirePrivilege.equals(AttributeDefPrivilege.ATTR_VIEW)) {
632           if (!attributeDefContainer.isCanView()) {
633             guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
634                 TextContainer.retrieveFromRequest().getText().get("attributeDefNotAllowedToViewAttributeDef")));
635             addedError = true;
636             privsOk = false;
637           }
638         } else if (requirePrivilege.equals(AttributeDefPrivilege.ATTR_READ)) {
639           if (!attributeDefContainer.isCanRead()) {
640             guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
641                 TextContainer.retrieveFromRequest().getText().get("attributeDefNotAllowedToReadAttributeDef")));
642             addedError = true;
643             privsOk = false;
644           }
645         } else if (requirePrivilege.equals(AttributeDefPrivilege.ATTR_UPDATE)) {
646           if (!attributeDefContainer.isCanUpdate()) {
647             guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
648                 TextContainer.retrieveFromRequest().getText().get("attributeDefNotAllowedToUpdateAttributeDef")));
649             addedError = true;
650             privsOk = false;
651           }
652         }  
653       }
654       
655       if (privsOk) {
656         result.setAttributeDefName(attributeDefName);
657       }
658 
659     } else {
660       
661       if (!addedError && (!StringUtils.isBlank(attributeDefNameId) || !StringUtils.isBlank(nameOfAttributeDefName) || !StringUtils.isBlank(attributeDefNameIndex))) {
662         result.setAddedError(true);
663         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
664             TextContainer.retrieveFromRequest().getText().get("attributeDefCantFindAttributeDefName")));
665         addedError = true;
666       }
667       
668     }
669     result.setAddedError(addedError);
670   
671     //go back to the main screen, cant find group
672     if (addedError) {
673       guiResponseJs.addAction(GuiScreenAction.newInnerHtmlFromJsp("#grouperMainContentDivId", 
674           "/WEB-INF/grouperUi2/index/indexMain.jsp"));
675     }
676 
677     return result;
678     
679   }

732       @Override
733       public AttributeDefName lookup(HttpServletRequest request, GrouperSession grouperSession, String query) {
734         Subject loggedInSubject = grouperSession.getSubject();
735         
736         String attributeDefId = request.getParameter("attributeDefComboName");
737         
738         String attributeAssignTypeString = request.getParameter("attributeAssignType");
739         
740         AttributeAssignType attributeAssignType = AttributeAssignType.valueOfIgnoreCase(attributeAssignTypeString, false);
741         QueryOptions queryOptions = QueryOptions.create(null, null, 1, 2);
742         
743         if (StringUtils.isNotBlank(attributeDefId)) {
744           AttributeDef attributeDef = AttributeDefFinder.findById(attributeDefId, false);
745           if (attributeDef == null) {
746             throw new RuntimeException("given attribute def id "+attributeDefId+" is not valid.");
747           }
748           
749           Set<AttributeDefName> attributeDefNames = GrouperDAOFactory.getFactory().getAttributeDefName().findAllAttributeNamesSplitScopeSecure(
750               query, grouperSession, attributeDefId, loggedInSubject, 
751               AttributeDefPrivilege.ATTR_DEF_ATTR_READ_PRIVILEGES, queryOptions, attributeAssignType, null);
752           
753           if (attributeDefNames.size() > 1) {
754             throw new RuntimeException("shouldn't have found more than one attribute def name.");
755           }
756           if (attributeDefNames.size() == 1) {
757             return attributeDefNames.iterator().next();
758           }
759           return null;
760         } else {
761           
762           Set<AttributeDefName> attributeDefNames = GrouperDAOFactory.getFactory().getAttributeDefName().findAllAttributeNamesSplitScopeSecure(
763               query, grouperSession, null, loggedInSubject, 
764               AttributeDefPrivilege.ATTR_DEF_ATTR_READ_PRIVILEGES, queryOptions, attributeAssignType, null);
765           
766           if (attributeDefNames.size() > 1) {
767             throw new RuntimeException("shouldn't have found more than one attribute def name: '" + query + "'");
768           }
769           if (attributeDefNames.size() == 1) {
770             return attributeDefNames.iterator().next();
771           }
772           return null;
773         }

1034    * @param request

1037   public void editAttributeDefNameInheritanceSubmit(HttpServletRequest request, HttpServletResponse response) {
1038     
1039     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
1040   
1041     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
1042   
1043     GrouperSession grouperSession = null;
1044   
1045     try {
1046   
1047       grouperSession = GrouperSession.start(loggedInSubject);
1048       
1049       AttributeDefName attributeDefName = retrieveAttributeDefNameHelper(request, AttributeDefPrivilege.ATTR_VIEW, true).getAttributeDefName();
1050       
1051       if (attributeDefName == null) {
1052         return;
1053       }
1054       
1055       AttributeDef attributeDef = attributeDefName.getAttributeDef();
1056       
1057       if (!attributeDef.getPrivilegeDelegate().canAttrAdmin(loggedInSubject)) {
1058         guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
1059             TextContainer.retrieveFromRequest().getText().get("simpleAttributeUpdate.errorCantEditAttributeDef")));
1060         return;
1061       }
1062       
1063       String[] defNamesThatImply = request.getParameterValues("defNamesThatImmediatelyImply[]");
1064       String[] defNamesImpliedBy = request.getParameterValues("defNamesImpliedByImmediate[]");
1065       
1066       if (defNamesThatImply != null) {
1067         for (AttributeDefName attributeDefNameThatImply:  attributeDefName.getAttributeDefNameSetDelegate().getAttributeDefNamesThatImplyThis()) {   
1068           attributeDefNameThatImply.getAttributeDefNameSetDelegate().removeFromAttributeDefNameSet(attributeDefName);
1069         }
1070         
1071         for (String attributeDefNameIdAddImply: defNamesThatImply) {
1072           AttributeDefName attributeDefNameThatImply = AttributeDefNameFinder.findById(attributeDefNameIdAddImply, true);
1073           attributeDefNameThatImply.getAttributeDefNameSetDelegate().addToAttributeDefNameSet(attributeDefName);
1074         }
1075       }
1076       
1077       if (defNamesImpliedBy != null) {
1078         
1079         for (AttributeDefName attributeDefNameImpliedBy:  attributeDefName.getAttributeDefNameSetDelegate().getAttributeDefNamesImpliedByThis()) {   
1080           attributeDefName.getAttributeDefNameSetDelegate().removeFromAttributeDefNameSet(attributeDefNameImpliedBy);
1081         }
1082         
1083         for (String attributeDefNameImpliedByAdd: defNamesImpliedBy) {        
1084           AttributeDefName attributeDefNameImpliedBy = AttributeDefNameFinder.findById(attributeDefNameImpliedByAdd, true);
1085           attributeDefName.getAttributeDefNameSetDelegate().addToAttributeDefNameSet(attributeDefNameImpliedBy);
1086         }
1087       }
1088       
1089       //go to the view attributeDef screen
1090       guiResponseJs.addAction(GuiScreenAction.newScript("guiV2link('operation=UiV2AttributeDef.viewAttributeDef&attributeDefId=" + attributeDefName.getAttributeDefId() + "')"));
1091       
1092       //lets show a success message on the new screen
1093       guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.success, 
1094           TextContainer.retrieveFromRequest().getText().get("attributeDefNameInheritanceEditSuccess")));
1095 
1096     } catch (Exception e) {

1098         return;
1099       }
1100       guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
1101           TextContainer.retrieveFromRequest().getText().get("attributeDefNameInheritanceEditError") 
1102           + ": " + GrouperUtil.xmlEscape(e.getMessage(), true)));
1103       return;
1104     } finally {
1105       GrouperSession.stopQuietly(grouperSession);
1106     }
