42        return guiMembershipContainer;
43      }

49      
50      return this.getGuiMembershipContainers().values().iterator().next();

54    /**
55     * get all the privileges comma separated
56     * @return the privileges comma separated
57     */
58    public String getPrivilegesCommaSeparated() {
59      
60      StringBuilder result = new StringBuilder();
61      
62      boolean first = true;
63      
64      //go through privs
65      for (String listName : GrouperUtil.nonNull(this.guiMembershipContainers).keySet()) {
66        
67        if (!first) {
68          result.append(", ");
69        }
70        
71        //this is not a priv
72        if (StringUtils.equals("members", listName)) {
73          continue;
74        }
75        Privilege privilege = null;
76        if (this.getGuiGroup() != null) {
77          privilege = AccessPrivilege.listToPriv(listName);
78        } else if (this.getGuiStem() != null) {
79          privilege = NamingPrivilege.listToPriv(listName);
80        } else if (this.getGuiAttributeDef() != null) {
81          privilege = AttributeDefPrivilege.listToPriv(listName);
82        }
83        String privName = privilege.getName();
84        result.append(TextContainer.retrieveFromRequest().getText().get("priv." + privName));
85        
86        first = false;
87      }
88      
89      return result.toString();
90    }

216   /**
217    * construct
218    * @param membershipSubjectContainer1
219    */
220   public GuiMembershipSubjectContainer(
221       MembershipSubjectContainer membershipSubjectContainer1) {
222     super();
223     this.membershipSubjectContainer = membershipSubjectContainer1;
224     this.guiMember = new GuiMember(membershipSubjectContainer1.getMember());
225     this.guiSubject = new GuiSubject(membershipSubjectContainer1.getSubject());
226     this.guiMembershipContainers = new LinkedHashMap<String, GuiMembershipContainer>();
227     this.allGuiMemberships = new LinkedHashMap<String, List<GuiMembership>>();
228     if (membershipSubjectContainer1.getGroupOwner() != null) {
229       this.guiGroup = new GuiGroup(membershipSubjectContainer1.getGroupOwner());
230     }
231     if (membershipSubjectContainer1.getStemOwner() != null) {
232       this.guiStem = new GuiStem(membershipSubjectContainer1.getStemOwner());
233     }
234     if (membershipSubjectContainer1.getAttributeDefOwner() != null) {
235       this.guiAttributeDef = new GuiAttributeDef(membershipSubjectContainer1.getAttributeDefOwner());
236     }
237     
238     for (String fieldName : GrouperUtil.nonNull(membershipSubjectContainer1.getMembershipContainers()).keySet()) {
239       this.guiMembershipContainers.put(fieldName, new GuiMembershipContainer(
240           membershipSubjectContainer1.getMembershipContainers().get(fieldName)));
241     }
242     
243     for (String fieldName : GrouperUtil.nonNull(membershipSubjectContainer1.getAllMemberships().keySet())) {
244       this.allGuiMemberships.put(fieldName, new ArrayList<GuiMembership>());
245       for (Membership membership : membershipSubjectContainer1.getAllMemberships().get(fieldName)) {
246         this.allGuiMemberships.get(fieldName).add(new GuiMembership(membership));
247       }
248     }
249   }
