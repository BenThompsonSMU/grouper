63    /**
64     * @return all the report config types (eg: SQL)
65     */
66    public List<String> getAllReportConfigTypes() {
67      
68      List<String> reportConfigTypes = new ArrayList<String>();
69      
70      for (ReportConfigType type: ReportConfigType.values()) {
71        reportConfigTypes.add(type.name());
72      }
73      
74      return reportConfigTypes;
75    }
76    
77    /**
78     * @return all the report config formats (eg: CSV)
79     */
80    public List<String> getAllReportConfigFormats() {
81      
82      List<String> reportConfigFormats = new ArrayList<String>();
83      
84      for (ReportConfigFormat format: ReportConfigFormat.values()) {
85        reportConfigFormats.add(format.name());
86      }
87      return reportConfigFormats;
88    }
