30  package edu.internet2.middleware.grouper.ui.tags;
31  
32  import java.io.IOException;
33  
34  import javax.servlet.jsp.JspException;
35  import javax.servlet.jsp.tagext.SimpleTagSupport;
36  
37  import org.apache.commons.lang.StringUtils;
38  
39  /**
40   * <pre>
41   * This will generate a menu.  You need something to attach the context click to, e.g.
42   * 
43   * &lt;a id="advancedLink" href="#" class="smallLink" onclick="this.oncontextmenu(event); return false"&gt;Advanced&lt;/a&gt;
44   * 
45   * </pre>
46   * @author mchyzer
47   *
48   */
49  public class GrouperMenuTag extends SimpleTagSupport {
50  
51    /**
52     * the jquery handle (e.g. #someId) which this menu should be attached to.  note
53     * that any element you are attaching to must have an id attribute defined
54     */
55    private String contextZoneJqueryHandle;
56    
57    /**
58     * the jquery handle (e.g. #someId) which this menu should be attached to.  note
59     * that any element you are attaching to must have an id attribute defined
60     * @param contextZoneJqueryHandle1 the contextZoneJqueryHandle to set
61     */
62    public void setContextZoneJqueryHandle(String contextZoneJqueryHandle1) {
63      this.contextZoneJqueryHandle = contextZoneJqueryHandle1;
64    }
65  
66    /** true if context menu, false if not */
67    private boolean contextMenu;
68    
69    /**
70     * true if context menu, false if not
71     * @param contextMenu1 the contextMenu to set
72     */
73    public void setContextMenu(boolean contextMenu1) {
74      this.contextMenu = contextMenu1;
75    }
76  
77    /** the operation called to define the structure of the menu */
78    private String structureOperation;
79    
80    /**
81     * the operation called to define the structure of the menu
82     * @param structureOperation1 the structureOperation to set
83     */
84    public void setStructureOperation(String structureOperation1) {
85      this.structureOperation = structureOperation1;
86    }
87  
88    /** when events occur (onclick), then that operation is called via ajax */
89    private String operation;
90    
91    /**
92     * when events occur (onclick), then that operation is called via ajax
93     * @param operation1 the operation to set
94     */
95    public void setOperation(String operation1) {
96      this.operation = operation1;
97    }
98    /** the id of the HTML element of the menu */
99    private String menuId;
100   
101   /**
102    * the id of the HTML element of the menu
103    * @param menuId1 the menuId to set
104    */
105   public void setMenuId(String menuId1) {
106     this.menuId = menuId1;
107   }
108   /**
109    * init fields on construct
110    */
111   public GrouperMenuTag() {
112   }
113 
114   /**
115    * @see javax.servlet.jsp.tagext.SimpleTagSupport#doTag()
116    */
117   @Override
118   public void doTag() throws JspException, IOException {
119 
120     StringBuilder result = new StringBuilder();
121     
122 //    <span id="advancedMenu" ></span>
123 //    <script type="text/javascript">
124 //    guiInitDhtmlxMenu("advancedMenu", "SimpleMembershipUpdate.advancedMenu", 
125 //        "SimpleMembershipUpdate.advancedMenuStructure", true, "#advancedLink");
126 //    </script>
127     
128     //if it was shorthand, prefix with the full path
129     if (!StringUtils.contains(this.operation, "/")) {
130       this.operation = "../app/" + this.operation;
131     }
132     if (!StringUtils.contains(this.structureOperation, "/")) {
133       this.structureOperation = "../app/" + this.structureOperation;
134     }
135     
136     result.append("<span id=\"").append(this.menuId).append("\"");
137     result.append("></span>\n");
138     result.append("<script type=\"text/javascript\"> \n");
139     result.append("guiInitDhtmlxMenu('").append(this.menuId).append("', '")
140       .append(this.operation).append("', '").append(this.structureOperation)
141       .append("', ").append(this.contextMenu).append(", '");
142     result.append(this.contextZoneJqueryHandle).append("' );\n");
143     result.append("</script>\n");
144 
145     this.getJspContext().getOut().print(result.toString());
146   }
147 
148 }
