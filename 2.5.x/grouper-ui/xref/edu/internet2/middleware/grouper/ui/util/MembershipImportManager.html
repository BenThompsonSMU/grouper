55  /**
56   * Class that reads an XML configuration file and hands off actual data import
57   * to an implementation associated with the selected import format. The configuration 
58   * file is determined by looking up the media.properties key 'membership-import.config'.
59   * The base Grouper UI distribution does not have a value set. If you want to allow membership
60   * import you must configure one or more formats appropriate to your site. 
61   * <pre>
62  &lt;membership-import&gt;
63      &lt;format name=&quot;Tab separated&quot; separator=&quot;\t&quot; id-field=&quot;1&quot;  field-type=&quot;id&quot;
64                 ignore-existing=&quot;true&quot; 
65                 class=&quot;edu.internet2.middleware.grouper.ui.util.DefaultMembershipImporter&quot;/&gt;
66      &lt;format name=&quot;Comma separated&quot; separator=&quot;,&quot; id-field=&quot;1&quot;  field-type=&quot;id&quot;
67                 ignore-existing=&quot;true&quot; 
68                 class=&quot;edu.internet2.middleware.grouper.ui.util.DefaultMembershipImporter&quot;/&gt;
69  &lt;/membership-import&gt;  </pre>
70  <p><strong>ignore-existing=true</strong> indicates that no error should be reported if an imported Subject is already a member of the group<br/>
71  <strong>class</strong> specifies the implementation for this format<br/>
72  <strong>name</strong> is the text that users will see and select - unless there is only one format, in
73  which case it will be used as a default.</p>
74  <p>Other attributes are interpreted by the implementation class. There is no formal DTD/Schema.</p> 
75   * <p />
76   * 
77   * @author Gary Brown.
78   * @version $Id: MembershipImportManager.java,v 1.6 2009-10-16 10:30:08 isgwb Exp $
79   */
80  public class MembershipImportManager implements Serializable{
81  	private boolean active=false;
82  	//MCH 20090811 XXX take this out of session it is not serializable
83  	private transient Document configXml;
84  	private transient Map formatCache = new HashMap();
85  
86  	
87  	/**
88  	 * @param config - media.properties. The key 'membership-import.config' defines XML configuratoin file
89  	 * @param nav - nav.properties. Provides localized messages
90  	 * @throws Exception
91  	 */
92  	public MembershipImportManager() throws Exception{
93  		super();
94  		init();
95  		
96  	}
97  	
98  	private void init() throws Exception{
99  		if(configXml != null) {
100 			return;
101 		}
102 		String configResource = null;
103 		try {
104 			configResource=GrouperUiFilter.retrieveSessionMediaResourceBundle().getString("membership-import.config");
105 		}catch(MissingResourceException e){
106 			return;
107 		}
108 		try {
109 			configXml = DOMHelper.getDomFromResourceOnClassPath(configResource);
110 		}catch(Exception e){}
111 		List available = getAvailableFormats();
112 		if(!available.isEmpty()) active=true;
113 	}
114 	
115 	/**
116 	 * Returns true if the XML configuration file exists and there is at least
117 	 * one import format defined. 
118 	 * @return whether the UI should present import controls
119 	 */
120 	public boolean isActive() {
121 		return active;
122 	}
123 	
124 	/**
125 	 * Returns the configured import formats
126 	 * @return list of import format names
127 	 */
128 	public List getAvailableFormats() throws Exception{
129 		List res = new ArrayList();
130 		init();
131 		NodeList nl = configXml.getElementsByTagName("format");
132 		Element el =null;
133 		for (int i=0; i<nl.getLength();i++) {
134 			el = (Element) nl.item(i);
135 			res.add(el.getAttribute("name"));
136 			formatCache.put(el.getAttribute("name"),el);
137 		}
138 		return res;
139 	}
140 	
141 	/**
142 	 * Looks up the configuration for the supplied format, instantiates the
143 	 * configured implemetation class and calls its 'load' method
144 	 * @param format
145 	 * @param group
146 	 * @param input
147 	 * @param output
148 	 * @param field
149 	 * @return count of errors encountered during the load
150 	 * @throws IOException
151 	 * @throws SchemaException
152 	 */
153 	public int load(String format,Group group, Reader input, PrintWriter output,Field field) throws Exception,IOException,SchemaException{
154 		init();
155 		MembershipImporter importer=null;
156 		Element fe = getFormat(format);
157 		try {
158 			importer = (MembershipImporter)Class.forName(fe.getAttribute("class")).newInstance();
159 		}catch(Exception e) {
160 			throw new IllegalArgumentException("Could not instantiate importer class [" + fe.getAttribute("class") + "]");
161 		}
162 		return importer.load(group,input,output,fe,field);
163 	}
164 	
165 	private Element getFormat(String name) throws Exception{
166 		init();
167 		return (Element)formatCache.get(name);
168 	}
169 
170 }
