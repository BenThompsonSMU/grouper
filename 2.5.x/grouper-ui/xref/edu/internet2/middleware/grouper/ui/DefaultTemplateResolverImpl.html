81  	 */
82  	public String getObjectType(Object obj) {
83  		Class claz = obj.getClass();
84  		Method method = null;
85  		String objType = null;
86  		try {
87  			//
88  			method = claz.getMethod("getObjectType", new Class[] {});
89  			objType = (String) method.invoke(obj, new Object[] {});
90  		} catch (NoSuchMethodException e) {
91  			if (obj instanceof Collection)
92  				return "Collection";
93  			if (obj instanceof Field)
94  				return "Field";
95  			objType = claz.getName().substring(
96  					claz.getName().lastIndexOf(".") + 1);
97  		} catch (IllegalAccessException e) {
98  			objType = "Unknown";
99  		} catch (InvocationTargetException e) {
100 			objType = "Unknown";
101 		}
102 		return objType;
103 	}

106 	 * (non-Javadoc)
107 	 * 
108 	 * @see edu.internet2.middleware.grouper.ui.TemplateResolver#getTemplateName(java.lang.Object,
109 	 *      java.lang.String, java.util.ResourceBundle,
110 	 *      javax.servlet.http.HttpServletRequest)
111 	 */
112 	public String getTemplateName(Object obj, String view,
113 			ResourceBundle bundle, HttpServletRequest request) {
114 		String templateName = null;
115 		String objectType = getObjectType(obj);
116 
117 		Class claz = this.getClass();
118 		Method method = (Method) cache.get(objectType);
119 		if (method == null) {
120 			try {
121 				method = claz
122 						.getMethod("get" + objectType + "TemplateName",
123 								new Class[] { Object.class, view.getClass(),
124 										ResourceBundle.class,
125 										HttpServletRequest.class });
126 				cache.put(objectType, method);
127 			} catch (Exception e) {
128 				throw new IllegalArgumentException(claz.getName()
129 						+ " does not know how to find templates for "
130 						+ obj.getClass().getName());
131 			}
132 		}
133 		try {
134 			templateName = (String) method.invoke(this, new Object[] { obj,
135 					view, bundle, request });
136 		} catch (Exception e) {
137 			throw new IllegalArgumentException(
138 					"Problem executing code to find template name for "
139 							+ obj.getClass().getName() + ":" + e.getMessage());
140 		}
141 		return templateName;
142 	}

145 	// of null or a value
146 	protected String getResource(ResourceBundle bundle, String key) {
147 		String val = null;
148 		try {
149 			val = bundle.getString(key);
150 			if ("".equals(val))
151 				val = null;
152 
153 		} catch (Exception e) {
154 
155 		}
156 		if (val != null)
157 			UIThreadLocal.put("lastDynamicTemplateKey", key);
158 		return val;
159 	}

504 	 * Depending on the subject type may want to have a different template. The
505 	 * same subject type may be retrieved by different Sources, and may have
506 	 * different attributes available.
507 	 * 
508 	 * Look for more specific keys first. If a value=* then don`t use the specific 
509 	 * default - try the next key instead/ 
510 	 * <ul><li>&lt;source&gt;.subjectType.&lt;type&gt;.view.&lt;view&gt;</li>
511 	 *     <li>&lt;source&gt;.subjectType.&lt;type&gt;.view.default</li>
512 	 *     <li>&lt;source&gt;.view.&lt;view&gt;</li>
513 	 *     <li>&lt;source&gt;.view.default</li>
514 	 *     <li>subjectType.&lt;type&gt;.view.&lt;view&gt;</li>
515 	 *     <li>subjectType.&lt;type&lt;.view.default</li>
516 	 *     <li>subject.view.&lt;view&gt;</li>
517 	 *     <li>subject.view.default</li>
518 	 * </ul>
519 	 * 
520 	 * @param object to find template for 
521 	 * @param view name of template to find
522 	 * @param mediaResources ResourceBundle containing template names and values
523 	 * @param request HttpServletRequest 
524 	 * @return name of template
525 	 *  
526 	 */
527 	public String getI2miSubjectTemplateName(Object object, String view,
528 			ResourceBundle mediaResources, HttpServletRequest request) {
529 		SubjectAsMap subject = (SubjectAsMap) object;
530 		String tName = null;
531 		Source so = (Source)subject.get("source");
532 		String source = so.getId();
533 		String subjectType = (String) subject.get("subjectType");
534 
535 		tName = getResource(mediaResources, source + ".subjectType."
536 				+ subjectType + ".view." + view);
537 
538 		if (tName == null) {
539 			tName = getResource(mediaResources, source + ".subjectType."
540 					+ subjectType + ".view.default");
541 		}
542 
543 		if (tName == null) {
544 			tName = getResource(mediaResources, source + ".view." + view);
545 		}
546 		if (tName == null) {
547 			tName = getResource(mediaResources, source + ".view.default");
548 		}
549 
550 		if (tName == null || tName.equals("*")) {//ignore Source
551 			tName = getResource(mediaResources, "subjectType." + subjectType
552 					+ ".view." + view);
553 		}
554 		if (tName == null || tName.equals("*")) {//use default for type
555 			tName = getResource(mediaResources, "subjectType." + subjectType
556 					+ ".view.default");
557 		}
558 		if (tName == null || tName.equals("*")) {//Ignore type
559 			tName = getResource(mediaResources, "subject.view." + view);
560 		}
561 		if (tName == null) {
562 			tName = getResource(mediaResources, "subject.view.default");
563 		}
564 		return tName;
565 	}
