58    public static String nodeToString(Node node) throws Exception 
59    {
60      TransformerFactory tf  = TransformerFactory.newInstance();
61      Transformer t = tf.newTransformer();
62      t.setOutputProperty(OutputKeys.METHOD,"xml");
63      
64      StringWriter w = new StringWriter();
65      t.transform(new DOMSource(node),new StreamResult(w));
66      return w.toString();
67    }

94     /**
95      * Serializes an Element to a String - with/without an Xml declaration
96      */
97    public static String elementToString(Element element,boolean withDeclaration) throws Exception 
98    { 
99      OutputFormat format = new OutputFormat(element.getOwnerDocument());
100     NodeList nl = element.getElementsByTagName("*");
101     HashMap hm = new HashMap();
102     for (int i=0;i<nl.getLength();i++) {
103     	hm.put(((Element)nl.item(i)).getNodeName(),"");	
104     }
105     Iterator i = hm.keySet().iterator();
106     String[] nonEscapingElements = new String[hm.size()];
107     int j=0;
108     while(i.hasNext()) {
109     	nonEscapingElements[j++] = (String)i.next();	
110     }
111     format.setNonEscapingElements(nonEscapingElements);
112     format.setPreserveSpace(true);
113     format.setIndenting(true);
114     StringWriter w = new StringWriter();
115     XMLSerializer serializer = new XMLSerializer(w,format);
116     
117     serializer.asDOMSerializer();
118     serializer.serialize(element);
119     String result = w.toString();
120     if(withDeclaration) return result;
121     
122     int endDec = result.indexOf("?>");
123     int startElement = result.indexOf("<",endDec);
124     return result.substring(startElement);
125    
126   }

134   /**
135     * looks for file on classpath and parses it into a Document
136     */
137     public static Document getDomFromResourceOnClassPath(String resource) throws Exception {
138     	InputStream is=null;
139     	if(!resource.startsWith("jar:")) {
140     		is = DOMHelper.class.getClassLoader().getResourceAsStream(resource);
141     	}else{
142 			String jar = resource.replaceAll("jar:file:(.*?)!.*","$1");
143 			String path = resource.replaceAll("jar:file:/.*?!/(.*)","$1");
144 			JarFile jarFile = new JarFile(jar);
145 			is = jarFile.getInputStream(jarFile.getEntry(path));		
146     		
147     	}
148     	
149     	if (is == null) {
150     		throw new IllegalArgumentException("Resource on classpath not found: '" + resource + "'");
151     	}
152     	
153 		DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
154     	DocumentBuilder db = dbf.newDocumentBuilder();
155     	Document doc = db.parse(is);
156     	return doc;
157     	
158     }

160   /**
161     * Assumes tag only occurs once and contains only text / CDATA
162     * If tag does not exist 'nullable' determines if an Exception is thrown
163     * 
164     */
165     public static String getText(String elementName,Document doc, boolean nullable) throws Exception {
166     		NodeList nl = doc.getElementsByTagName(elementName);
167 			if(nl.getLength()<1) {
168 				if(nullable) {
169 					return null;
170 				}else {
171 					throw(new Exception("Cannot find " + elementName + " tag"));	
172 				}
173 			}			
174 			Element stmt = (Element)nl.item(0);
175 			stmt.normalize();
176 			nl = stmt.getChildNodes();
177 			if(nl.getLength() !=1 ) {
178 				throw(new Exception("Cannot process " + elementName + " tag"));
179 			}
180 			Node n = nl.item(0);
181 			if(n.getNodeType()!=Node.TEXT_NODE && n.getNodeType()!=Node.CDATA_SECTION_NODE) {
182 				throw(new Exception("Cannot process " + elementName + " tag"));	
183 			}
184 			return ((CharacterData)n).getData().trim();	
185     }

214     /**
215     * Returns immediate child elements with given name
216     */
217     public static Collection getImmediateElements(Element element,String elementName) throws Exception {
218     		NodeList nl = element.getElementsByTagName(elementName);
219 			Collection elements = new Vector();
220 			if(nl.getLength()<1) {
221 				return elements;
222 			}			
223 			Element child;
224 			for(int i=0;i<nl.getLength();i++) {
225 				child = (Element) nl.item(i);
226 				if(child.getParentNode().equals(element)) {
227 					elements.add(child);	
228 				}	
229 			}
230 			
231 			
232 			return elements;	
233     }

235   /**
236     * Returns immediate child element with given name - first only if > 1
237     */ 
238     public static Element getImmediateElement(Element element,String elementName) throws Exception {
239     		
240 			Collection elements = getImmediateElements(element,elementName);
241 			if(elements.size() < 1) return null;
242 			Iterator i = elements.iterator();			
243 			return (Element)i.next();
244     }

285 	/**
286 		* Convenience method for transformations
287 		*/
288 		public static void transform(String data,File out,String xsl,Map parameters) throws Exception{
289 			TransformerFactory tf  = TransformerFactory.newInstance();
290 			Transformer t = tf.newTransformer(new StreamSource(xsl));
291 			if(parameters != null) {
292 							String key;
293 							Iterator it = parameters.keySet().iterator();
294 							while(it.hasNext()) {
295 								key = (String)it.next();
296 								t.setParameter(key,parameters.get(key));
297 							}
298 						}
299 			t.transform(new StreamSource(new StringReader(data)),new StreamResult(out));
300     	
301     		
302 		}
303 	public static Document transform(Document doc,String xsl) throws Exception{
304 			return transform(doc,xsl,null);
305     		
306 		}
307 	public static Document transform(Document doc,String xsl,Map parameters) throws Exception{
308 			TransformerFactory tf  = TransformerFactory.newInstance();
309 			Transformer t = tf.newTransformer(new StreamSource(xsl));
310 			if(parameters != null) {
311 				String key;
312 				Iterator it = parameters.keySet().iterator();
313 				while(it.hasNext()) {
314 					key = (String)it.next();
315 					t.setParameter(key,parameters.get(key));
316 				}
317 			}
318 			DOMResult res = new DOMResult();
319 			t.transform(new DOMSource(doc),res);
320 			return (Document)res.getNode();
321     		
322 		}
