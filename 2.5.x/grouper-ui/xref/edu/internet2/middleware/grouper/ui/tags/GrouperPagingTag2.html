56    /**
57     * page size from http request from paging gui object
58     * @param request
59     * @return the page size
60     */
61    public static int pageSize(HttpServletRequest request) {
62      String pageSizeString = request.getParameter("pagingTagPageSize");
63      int pageSize = GrouperUtil.intValue(pageSizeString);
64      return pageSize;
65    }

67    /**
68     * process a request from a paging tag
69     * @param request
70     * @param guiPaging
71     * @param queryOptions 
72     * @param pagesizeDefaultProperty
73     */
74    public static void processRequest(HttpServletRequest request, GuiPaging guiPaging, 
75        QueryOptions queryOptions, String pagesizeDefaultProperty) {
76  
77      //how many per page
78      String pageSizeString = request.getParameter("pagingTagPageSize");
79      int pageSize = -1;
80      if (!StringUtils.isBlank(pageSizeString)) {
81        pageSize = GrouperUtil.intValue(pageSizeString);
82      } else {
83        pagesizeDefaultProperty = StringUtils.defaultIfEmpty(pagesizeDefaultProperty, "pager.pagesize.default");
84        pageSize = GrouperUiConfig.retrieveConfig().propertyValueInt(pagesizeDefaultProperty, 50);
85      }
86      guiPaging.setPageSize(pageSize);
87      
88      //1 indexed
89      String pageNumberString = request.getParameter("pagingTagPageNumber");
90      
91      int pageNumber = 1;
92      if (!StringUtils.isBlank(pageNumberString)) {
93        pageNumber = GrouperUtil.intValue(pageNumberString);
94      }
95      
96      guiPaging.setPageNumber(pageNumber);
97  
98      if (queryOptions != null) {
99        queryOptions.paging(pageSize, pageNumber, true);
100     }
101   }

196   /**
197    * tag logic
198    */
199   @Override
200   public void doTag() throws JspException, IOException {
201     
202     if (this.guiPaging == null) {
203       throw new NullPointerException("guiPaging cannot be null, does it exist not null in the objec model?");
204     }
205     
206     StringBuilder result = new StringBuilder();
207     
208     //  <div class="pull-right">Showing 1-10 of 25 &middot; <a href="#">First</a> | <a href="#">Prev</a> | <a href="#">Next</a> | <a href="#">Last</a></div>
209     //  <form class="form-inline form-small">
210     //    <label for="show-entries">Show:&nbsp;</label>
211     //    <select id="show-entries" class="span2">
212     //      <option>10</option>
213     //      <option>25</option>
214     //      <option>50</option>
215     //      <option>100</option>
216     //    </select>
217     //  </form>
218     //  <form style="display:none;">
219     //    <input type="hidden" name="pagingTagPageNumber" value="" />
220     //  </form>
221 
222     QueryPaging queryPaging = this.guiPaging.queryPaging();
223     
224     result.append("  <div class=\"pull-right\">" + GrouperUiUtils.message("paging2.showing") + " " + (queryPaging.getFirstIndexOnPage()+1) 
225         + "-" + (queryPaging.getLastIndexOnPage()+1) + " " + GrouperUiUtils.message("paging2.of") + " " + this.guiPaging.getTotalRecordCount() + " &middot; ");
226 
227     String javascriptEventPrefix = "ajax('";
228     String javascriptEventSuffix = "', {formIds: '" + this.formName + "Id" + (StringUtils.isBlank(this.ajaxFormIds) ? "" : ("," + this.ajaxFormIds)) + "'}); return false;";
229 
230     if (queryPaging.isFirstPage()) {
231       result.append(GrouperUiUtils.message("paging2.first") + " | " + GrouperUiUtils.message("paging2.prev"));
232     } else {
233       result.append("<a href=\"#\" onclick=\"" + javascriptEventPrefix + this.refreshOperation + javascriptEventSuffix + "\">" 
234           + GrouperUiUtils.message("paging2.first") + "</a> | ");
235       result.append("<a href=\"#\" onclick=\"" + javascriptEventPrefix 
236           + appendToUrl(this.refreshOperation, "pagingTagPageNumber=" + (queryPaging.getPageNumber()-1) ) + javascriptEventSuffix + "\">" 
237           + GrouperUiUtils.message("paging2.prev") + "</a>");
238     }
239     result.append(" | ");
240     if (queryPaging.isLastPage()) {
241       result.append(GrouperUiUtils.message("paging2.next") + " | " + GrouperUiUtils.message("paging2.last"));
242     } else {
243       result.append("<a href=\"#\" onclick=\"" + javascriptEventPrefix 
244           + appendToUrl(this.refreshOperation, "pagingTagPageNumber=" + (queryPaging.getPageNumber()+1) ) + javascriptEventSuffix + "\">" + GrouperUiUtils.message("paging2.next") + "</a>");
245       result.append(" | ");
246       result.append("<a href=\"#\" onclick=\"" + javascriptEventPrefix 
247           + appendToUrl(this.refreshOperation, "pagingTagPageNumber=" + queryPaging.getNumberOfPages() ) + javascriptEventSuffix + "\">" + GrouperUiUtils.message("paging2.last") + "</a>");
248     }
249     result.append("</div>\n");
250     result.append("  <form class=\"form-inline form-small\" name=\"" + this.formName + "\" id=\"" + this.formName + "Id\">\n");
251     result.append("    <label for=\"show-entries\">" + GrouperUiUtils.message("paging2.show") + "&nbsp;</label>\n");
252     result.append("    <select name=\"pagingTagPageSize\" id=\"show-entries\" class=\"span2\" onchange=\"" + javascriptEventPrefix + this.refreshOperation + javascriptEventSuffix + "\">\n");
253 
254     this.guiPaging.getPageSize();
255 
256     String pageSizesString = GrouperUiConfig.retrieveConfig().propertyValueString("pager.pagesize.selection", "10 25 50 100");
257 
258     List<String> pageSizesList = GrouperUtil.splitTrimToList(pageSizesString, " ");
259     int defaultPageSize = GrouperUiConfig.retrieveConfig().propertyValueInt("pager.pagesize.default", 50);
260 
261     if (!pageSizesList.contains(defaultPageSize + "")) {
262       LOG.error("Why is default page size not in page size list? " + defaultPageSize + ", " + pageSizesString);
263     }
264     
265     for (String pageSizeString : pageSizesList) {
266 
267       boolean selected = StringUtils.equals(pageSizeString, Integer.toString(this.guiPaging.getPageSize()));
268 
269       //result.append("      <option>25</option>\n");
270       result.append("      <option" + (selected ? " selected=\"selected\" " : "") + ">" + pageSizeString + "</option>\n");
271 
272     }
273 
274     result.append("    </select>\n");
275     result.append("  </form>\n");
276     //this is a form if something needs to send the current page number to the server...
277     result.append("  <form style=\"display:none;\" name=\"" + this.formName + "PageNumber\" id=\"" + this.formName + "PageNumberId\">\n");
278     result.append("    <input type=\"hidden\" name=\"pagingTagPageNumber\" value=\"" + queryPaging.getPageNumber() + "\" />\n");
279     result.append("  </form>\n");
280     
281     this.getJspContext().getOut().print(result.toString());
282   }

284   /**
285    * append to url with question of ampersand
286    * @param url
287    * @param append
288    * @return the url
289    */
290   private static String appendToUrl(String url, String append) {
291     if (url.contains("?") ) {
292       return url + "&" + append;
293     }
294     return url + "?" + append;
295   }
