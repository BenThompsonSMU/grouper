43  import edu.internet2.middleware.grouper.Field;
44  import edu.internet2.middleware.grouper.Group;

52  /**
53   * Default implementation of the MembershipImporter interface. This class
54   * expects single line records e.g. tab/comma separated fields where one
55   * of the fields can be used to lookup a Subject by id or identifier.
56   * <p>This class expects to find the following attributes in the config Element:<br/>
57   * separator e.g. ',' or '\t', used to determine fields<br/>
58   * id-field - an integer used to determine which field position to use to lookup a Subject>br/>
59   * field-type - 'id' or 'identifier' specifies the type of lookup to do<br/>
60   * ignore-existing - 'false' or 'true' determies whether trying to add an existing member is considered an error</p>
61   * <p>Other implementations
62   * can be as complex as necessary  
63   * <p />
64   * 
65   * @author Gary Brown.
66   * @version $Id: DefaultMembershipImporter.java,v 1.6 2009-08-12 04:52:14 mchyzer Exp $
67   */
68  
69  public class DefaultMembershipImporter implements MembershipImporter{
70  
71  	
72  	/* (non-Javadoc)
73  	 * @see edu.internet2.middleware.grouper.ui.util.MembershipImporter#load(edu.internet2.middleware.grouper.Group, java.io.Reader, java.io.PrintWriter, org.w3c.dom.Element, edu.internet2.middleware.grouper.Field, java.util.ResourceBundle)
74  	<p>Recognises the following config Element attributes:</p>
75  	  <table width="100%" border="1" cellspacing="5px">
76    <tr> 
77      <td><strong><font face="Arial, Helvetica, sans-serif">separator</font></strong></td>
78      <td><font face="Arial, Helvetica, sans-serif">used to split input lines into 
79        fields </font></td>
80    </tr>
81    <tr> 
82      <td><strong><font face="Arial, Helvetica, sans-serif">id-field</font></strong></td>
83      <td><font face="Arial, Helvetica, sans-serif">integer, starting at 1, which 
84        indicates which field should be used to 'lookup' a subject</font></td>
85    </tr>
86    <tr> 
87      <td><strong><font face="Arial, Helvetica, sans-serif">field-type</font></strong></td>
88      <td><font face="Arial, Helvetica, sans-serif">=id or identifier. Specifies 
89        how to lookup a subject</font></td>
90    </tr>
91  </table>
92  	 *
93  	 **/
94  	public int load(Group group, Reader input, PrintWriter output,Element config,Field field)
95  			throws IOException,SchemaException {
96  	  ResourceBundle nav = GrouperUiFilter.retrieveSessionNavResourceBundle();
97  		int errorCount=0;
98  		String separator=config.getAttribute("separator");
99  		if(separator==null || separator.equals("")) throw new IllegalArgumentException(nav.getString("groups.import.message.no-separator"));
100 		if(separator.equals("\\t")) separator="\t";
101 		int idField = Integer.parseInt(config.getAttribute("id-field"))-1;
102 		String fieldType=config.getAttribute("field-type");
103 		boolean useId=false;
104 		if("id".equals(fieldType)) {
105 			useId=true;
106 		}else if(!"identifier".equals(fieldType)) {
107 			throw new IllegalArgumentException(nav.getString("groups.import.message.bad-field-type"));
108 		}
109 		boolean ignoreExisting = Boolean.parseBoolean(config.getAttribute("ignore-existing"));
110 		String line;
111 		String[] parts;
112 		String id;
113 		Subject subject;
114 		LineNumberReader lineInput = new LineNumberReader(input);
115 		int lineCount=0;
116 		while ((line=lineInput.readLine())!=null) {
117 			lineCount++;
118 			parts=line.split(separator);
119 			if(parts.length<idField) {
120 				output.println(nav.getString("groups.import.message.insufficient-fields") + separator + line);
121 				errorCount++;
122 			}else{
123 				id=parts[idField].trim();
124 				if(id.indexOf("\"")==0) {
125 					id = id.replaceAll("^\"(.*?)\"$","$1");
126 				}
127 				try {
128 					if(useId) {
129 						subject=SubjectFinder.findById(id, true);
130 					}else{
131 						subject=SubjectFinder.findByIdentifier(id, true);
132 					}
133 					if(group.hasImmediateMember(subject,field)) {
134 						if(!ignoreExisting) {
135 							printHtmlError(nav.getString("groups.import.message.existing-member") + separator + line,output);
136 							errorCount++;
137 						}
138 					}else{
139 						try {
140 							group.addMember(subject,field);
141 							output.println(nav.getString("groups.import.message.successful") + separator+ line);
142 						}catch(Exception ex) {
143 							printHtmlError(nav.getString("groups.import.message.error")+" " + ex.getMessage() + separator + line,output);
144 							errorCount++;
145 						}
146 					}
147 				}catch(SubjectNotFoundException e){
148 					printHtmlError(nav.getString("groups.import.message.no-subject") + separator + line,output);
149 					errorCount++;
150 				}catch(SubjectNotUniqueException e) {
151 					printHtmlError(nav.getString("groups.import.message.subject-not-unique") + separator + line,output);
152 					errorCount++;
153 				}
154 				output.flush();
155 			}
156 			
157 		}
158 		if(lineCount==0) {
159 			printHtmlError(nav.getString("groups.import.message.no-data"),output);
160 		}
161 		
162 		return errorCount;
163 	}
164 	
165 	private void printHtmlError(String error,PrintWriter writer) {
166 		writer.print("<span style='color:#ff0000'>" + error + "</span>\n");
167 	}
168 }
