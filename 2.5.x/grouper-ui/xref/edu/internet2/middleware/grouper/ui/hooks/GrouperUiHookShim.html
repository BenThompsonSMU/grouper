18   */
19  public class GrouperUiHookShim {
20   
21    /**
22     *
23     */
24    public GrouperUiHookShim() {
25    }
26   
27    /**
28     * https://getbootstrap.com/docs/4.0/components/alerts/
29     * bootstrap message types
30     */
31    private static Set<String> messageTypes = GrouperUtil.toSet("primary", "secondary", "success", "danger", "warning", "info", "light", "dark");
32    
33    /**
34     * https://getbootstrap.com/docs/4.0/components/alerts/
35     * @param bootstrapAlertType "primary", "secondary", "success", "danger", "warning", "info", "light", "dark"
36     * @param escapedMessage note single quotes will be escaped, maybe dont include single quotes
37     */
38    public static void addMessageToScreen(String bootstrapAlertType, String escapedMessage) {
39      
40      if (messageTypes.contains(bootstrapAlertType)) {
41        throw new RuntimeException("bootstrapAlertType '" + bootstrapAlertType + "' must be one of: " + StringUtils.join(messageTypes, ','));
42      }
43      
44      addScript("$('#messaging').before('<div class=\"messaging row-fluid\""
45          + " style=\"display: block;\"><div role=\"alert\" class=\"alert alert-" + bootstrapAlertType + "\"><button type=\"button\""
46          + " class=\"close\" data-dismiss=\"alert\">x</button>"
47          + GrouperUtil.escapeSingleQuotes(escapedMessage) + "</div></div>');");
48       
49    }
50  
51    /**
52     * @param script
53     */
54    public static void addScript(String script) {
55      GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
56      guiResponseJs.addAction(GuiScreenAction.newScript(script));
57       
58    }
59  }
