116     this.searchDelay = searchDelay1;

119   /**
120    * 
121    * @return pro
122    */
123   public int searchDelayProcessed() {
124     if (this.searchDelay == null || this.searchDelay < 0) {
125       return 500;
126     }
127     return this.searchDelay;
128   }
129   
130   /**
131    * default downarrow to false
132    * @return if has down arrow
133    */
134   public boolean hasDownArrowProcessed() {
135     if (this.hasDownArrow == null) {
136       return false;
137     }
138     return this.hasDownArrow;
139   }
