232    * find the request parameter names by prefix
233    * @param prefix
234    * @return the set, never null
235    */
236   @SuppressWarnings("unchecked")
237   public static Set<String> requestParameterNamesByPrefix(String prefix) {
238     HttpServletRequest httpServletRequest = retrieveHttpServletRequest();
239     Set<String> result = new LinkedHashSet<String>();
240     Enumeration<String> paramNames = httpServletRequest.getParameterNames();
241     
242     //cycle through all
243     while(paramNames.hasMoreElements()) {
244       String paramName = paramNames.nextElement();
245       
246       //see if starts with
247       if (paramName.startsWith(prefix)) {
248         result.add(paramName);
249       }
250     }
251     
252     
253     return result;
254   }

320   /**
321    * retrieve the subject logged in
322    * @param allowNoUserLoggedIn true if allowed to have no user, false if expecting a user
323    * @return the subject
324    */
325   private static Subject retrieveSubjectLoggedInHelper(boolean allowNoUserLoggedIn) {
326 
327     //currently assumes user is in getUserPrincipal
328     HttpServletRequest request = retrieveHttpServletRequest();
329 
330     String userIdLoggedIn = remoteUser(request);
331 
332     //this cant change!
333     String previousRemoteUser = (String)request.getSession().getAttribute("authUser");
334     
335     if (!StringUtils.isBlank(previousRemoteUser) && !StringUtils.equals(previousRemoteUser, userIdLoggedIn)) {
336       request.getSession().invalidate();
337       throw new RuntimeException("New user logged in!  was: '" + previousRemoteUser + "', and now is: '" + userIdLoggedIn + "'" );
338     }
339 
340     GrouperSession grouperSession = SessionInitialiser.getGrouperSession(request.getSession());
341     if (grouperSession != null && grouperSession.getSubject() != null) {
342       
343       return grouperSession.getSubject();
344     }
345     
346     SessionContainer sessionContainer = SessionContainer.retrieveFromSession();
347     
348     Subject subjectLoggedIn = sessionContainer.getSubjectLoggedIn();
349     
350     if (subjectLoggedIn != null) {
351       return subjectLoggedIn;
352     }
353   
354     if (StringUtils.isBlank(userIdLoggedIn)) {
355       if (allowNoUserLoggedIn) {
356         return null;
357       }
358       throw new NoUserAuthenticatedException("Cant find logged in user");
359     }
360     
361     GrouperSession rootSession = GrouperSession.startRootSession();
362     try {
363       String sourceIds = GrouperUiConfig.retrieveConfig().propertyValueString("grouper.ui.authentication.sourceIds");
364       if (StringUtils.isBlank(sourceIds)) {
365         subjectLoggedIn = SubjectFinder.findByIdOrIdentifier(userIdLoggedIn, true);
366       } else {
367         Subject theSubjectLoggedIn = null;
368         for (String sourceId : GrouperUtil.splitTrim(sourceIds, ",")) {
369           Subject tempSubject = SubjectFinder.findByIdOrIdentifierAndSource(userIdLoggedIn, sourceId, false);
370           if (tempSubject != null) {
371             if (theSubjectLoggedIn == null) {
372               theSubjectLoggedIn = tempSubject;
373             } else {
374               throw new SubjectNotUniqueException("Found multiple matching subjects: '" + userIdLoggedIn + "'");
375             }
376           }
377         }
378         subjectLoggedIn = theSubjectLoggedIn;
379         if (subjectLoggedIn == null) {
380           throw new SubjectNotFoundException("Cannot find subject by id or identifier: '" + userIdLoggedIn + "'");
381         }
382       }
383     } catch (RuntimeException re) {
384       if (re instanceof SubjectNotFoundException && allowNoUserLoggedIn) {
385         return null;
386       }
387       //this is probably a system error...  not a user error
388       GrouperUtil.injectInException(re, "Cant find subject from login id: " + userIdLoggedIn);
389       throw re;
390     } finally {
391       GrouperSession.stopQuietly(rootSession);
392     }
393         
394     return subjectLoggedIn;
395 
396   }

398   /**
399    * make sure user is allowed in this section
400    * @param uiSection
401    * @param subjectLoggedIn
402    * @param response
403    */
404   private static void ensureUserAllowedInSection(UiSection uiSection, Subject subjectLoggedIn, HttpServletResponse response) {
405 
406     if (subjectLoggedIn == null && uiSection.isAnonymous()) {
407       return;
408     }
409 
410     //if the user is allowed in the admin ui, we are all good
411     Set<UiSection> uiSectionsThatAllowThisSection = GrouperUtil.nonNull(uiSection.getUiSectionsThatAllowThisSection());
412     for (UiSection currentSection : uiSectionsThatAllowThisSection) {
413       if (SessionContainer.retrieveFromSession().getAllowedUiSections().contains(currentSection)) {
414         return;
415       }
416     }
417     StringBuilder groups = new StringBuilder();
418     for (UiSection currentSection : uiSectionsThatAllowThisSection) {
419       String mediaKey = currentSection.getMediaKey();
420       if (!StringUtils.isBlank(mediaKey)) {
421         
422         if (subjectLoggedIn == null) {
423           try {
424             response.sendRedirect(GrouperUiFilter.retrieveServletContext() + "/grouperExternal/public/UiV2Public.index?operation=UiV2Public.postIndex&function=UiV2Public.error&code=anonymousSessionNotAllowed");
425           } catch (IOException ioe) {
426             throw new RuntimeException("Error", ioe);
427           }
428           throw new ControllerDone();
429         }
430         
431         String thisError = requireUiGroup(mediaKey, subjectLoggedIn, true);
432         if (!StringUtils.isBlank(thisError)) {
433           groups.append(thisError).append(", ");
434         } else {
435           //this means allowed, we are all good
436           groups = new StringBuilder();
437           break;
438         }
439       }
440     }
441     
442     //if there is an error or more, that is bad
443     if (groups.length() > 0) {
444       //strip last comma
445       String groupsString = groups.substring(0,groups.length()-2);
446       String errorsString = GrouperUiUtils.message("ui.error.not.in.required.group", false, true, 
447           GrouperUtil.subjectToString(subjectLoggedIn), groupsString);
448       LOG.error(errorsString);
449       GrouperUiUtils.appendErrorToRequest(errorsString);
450       
451       if (RequestContainer.retrieveFromRequest().isAjaxRequest()) {
452         GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
453         guiResponseJs.addAction(GuiScreenAction.newAlert(
454             GrouperUiUtils.message("simpleMembershipUpdate.notAllowedInUi")));
455         throw new ControllerDone();
456         
457       }
458       //ui request, just throw exception
459       throw new RuntimeException(errorsString);
460       
461     }
462     
463     //keep this in session so we dont have to keep checking
464     SessionContainer.retrieveFromSession().getAllowedUiSections().add(uiSection);
465   }

467   /**
468    * use the media properties key to see if a group is required, then make sure the user is in that group
469    * @param mediaKeyOfGroup
470    * @param subjectLoggedIn
471    * @param allowUserIfGroupNotConfigured is true if user is allowed if group is not configured, default to true
472    * for previous behavior
473    * @return the error message group name
474    */
475   public static String requireUiGroup(String mediaKeyOfGroup, Subject subjectLoggedIn, boolean allowUserIfGroupNotConfigured) {
476     
477     //see if member of login group
478     String groupToRequire = GrouperUiConfig.retrieveConfig().propertyValueString(mediaKeyOfGroup);
479     if (LOG.isDebugEnabled()) {
480       LOG.debug("mediaKeyOfGroup: " + mediaKeyOfGroup + ", groupToRequire: " + groupToRequire + ", subject: " + GrouperUtil.subjectToString(subjectLoggedIn));
481     }
482     if (!StringUtils.isBlank(groupToRequire)) {
483       
484       GrouperSession grouperSession = null;
485       
486       //get a session, close it if you started it
487       boolean startedSession = false;
488       try {
489         grouperSession = GrouperSession.staticGrouperSession(false);
490         if (grouperSession == null) {
491           grouperSession = GrouperSession.startRootSession();
492           startedSession = true;
493         }
494         if (!PrivilegeHelper.isWheelOrRoot(grouperSession.getSubject())) {
495           grouperSession = grouperSession.internal_getRootSession();
496         }
497         Group group = GroupFinder.findByName(grouperSession, groupToRequire, true);
498         if (subjectLoggedIn == null || !group.hasMember(subjectLoggedIn)) {
499           
500           String error = groupToRequire;
501           return error;
502         }
503         return null;
504       } catch (Exception e) {
505         throw new RuntimeException("Problem with user: " + GrouperUtil.subjectToString(subjectLoggedIn) + ", " + groupToRequire, e);
506       } finally {
507         if (startedSession) {
508           GrouperSession.stopQuietly(grouperSession);
509         }
510       }
511     }
512     if (!allowUserIfGroupNotConfigured) {
513       return mediaKeyOfGroup;
514     }
515     return null;
516   }

607   /**
608    * 
609    * @param httpServletRequest
610    * @return user name
611    */
612   public static String remoteUser(HttpServletRequest httpServletRequest) {
613     
614     Map<String, Object> debugLog = LOG.isDebugEnabled() ? new LinkedHashMap<String, Object>() : null;
615     try {
616       String remoteUser = httpServletRequest.getRemoteUser();
617       
618       if (LOG.isDebugEnabled()) {
619         debugLog.put("httpServletRequest.getRemoteUser()", remoteUser);
620       }
621       
622       if (StringUtils.isBlank(remoteUser)) {
623         String userHeaderName = GrouperUiConfig.retrieveConfig().propertyValueString(
624             "grouper.ui.authentication.http.header");
625         if (StringUtils.isNotBlank(userHeaderName)) {
626           remoteUser = httpServletRequest.getHeader(userHeaderName);
627           if (LOG.isDebugEnabled()) {
628             debugLog.put(userHeaderName + " header", remoteUser);
629           }
630         }
631       }
632       
633       if (StringUtils.isBlank(remoteUser)) {
634         //this is how mod_jk passes env vars
635         remoteUser = (String)httpServletRequest.getAttribute("REMOTE_USER");
636         if (LOG.isDebugEnabled()) {
637           debugLog.put("REMOTE_USER attribute", remoteUser);
638         }
639       }
640       
641       if (StringUtils.isBlank(remoteUser) && httpServletRequest.getUserPrincipal() != null) {
642         //this is how mod_jk passes env vars
643         remoteUser = httpServletRequest.getUserPrincipal().getName();
644         if (LOG.isDebugEnabled()) {
645           debugLog.put("httpServletRequest.getUserPrincipal().getName()", remoteUser);
646         }
647       }
648       if (StringUtils.isBlank(remoteUser)) {
649         HttpSession session = httpServletRequest.getSession(false);
650         remoteUser = (String)(session == null ? null : session.getAttribute("authUser"));
651         if (LOG.isDebugEnabled()) {

713     
714     String uri = httpServletRequest.getRequestURI();

716     //TODO might want to check the servlet param name from web.xml: ignore

734       return UiSection.ANONYMOUS;
735     }
736     
737     boolean externalServlet = uri.matches("^/[^/]+/grouperExternal/app/[^/]+$");
738 
739     String operation = httpServletRequest.getParameter("operation");
740 
741     String theClass = null;

747     if (externalServlet) {

750         if (theClass.startsWith(UiV2ExternalSubjectSelfRegister.class.getSimpleName())) {
751           return UiSection.EXTERNAL;

755         return UiSection.ANONYMOUS;

758     }

760     if (!StringUtils.isBlank(operation)) {
761       
762       //anyone can see the menu...
763       if (theClass.equals("Misc") || theClass.equals("MiscMenu")) {
764         return UiSection.ANONYMOUS;
765       }
766     }
767 
768     //must be admin UI
769     return UiSection.ADMIN_UI;

1337    */
1338   public void sendErrorEmailIfNeeded() {
1339     try {
1340       HttpServletRequest httpServletRequest = retrieveHttpServletRequest();
1341 
1342       String error = httpServletRequest == null ? null : (String)httpServletRequest.getAttribute("error");
1343       if (!StringUtils.isBlank(error)) {
1344         
1345         String errorMailAddresses = GrouperUiConfig.retrieveConfig().propertyValueString("errorMailAddresses");
1346 
1347         if (!StringUtils.isBlank(errorMailAddresses)) {
1348           
1349           String loggedInSubjectString = "dont know";
1350           try {
1351             Subject loggedInSubject = retrieveSubjectLoggedIn();
1352             if (loggedInSubject == null) {
1353               loggedInSubjectString = "none";
1354             } else {
1355               loggedInSubjectString = loggedInSubject.getSource().getId() + " - " + loggedInSubject.getId();
1356             }
1357           } catch (RuntimeException re) {
1358             LOG.error(re);
1359           }
1360           
1361           String requestParams = GrouperUiUtils.requestParams();
1362           error = "Server name: " + GrouperUtil.hostname() + "\n" 
1363             + "IP Address: " + httpServletRequest.getRemoteAddr() + "\n"
1364             + "User: " + loggedInSubjectString + "\n"
1365             + "URL: " + httpServletRequest.getRequestURL() + "\n"
1366             + "Request params: " + requestParams + "\n"
1367             + "\n\nError: " + error; 
1368           
1369           new GrouperEmail().setTo(errorMailAddresses).setSubject("grouperUi error").setBody(error).send();
1370         }
1371       }
1372       
1373     } catch (Exception e) {
1374       LOG.error("Error sending email", e);
1375     }
1376   }
