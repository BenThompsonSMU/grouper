32  package edu.internet2.middleware.grouper.ui;
33  
34  import java.util.ResourceBundle;
35  import edu.internet2.middleware.grouper.Group;
36  import edu.internet2.middleware.grouper.Stem;
37  import edu.internet2.middleware.grouper.ui.util.ObjectAsMap;
38  import edu.internet2.middleware.grouper.ui.util.SubjectPrivilegeAsMap;
39  
40  /**
41   * Implementation of ComparatorHelper used to sort GroupOrStems. Will also
42   * sort SubjectPrivilegeAsMaps as these can be shown as groups or stems.
43   * <p>Implementation determines whether underlying Object is a Group or Stem,
44   * and calls appropriate Helper implementation</p>
45   * <p />
46   * 
47   * @author Gary Brown.
48   * @version $Id: GroupOrStemComparatorHelper.java,v 1.1 2007-03-15 15:30:16 isgwb Exp $
49   */
50  
51  public class GroupOrStemComparatorHelper implements GrouperComparatorHelper{
52  	private GroupComparatorHelperaratorHelper.html#GroupComparatorHelper">GroupComparatorHelper groupHelper = new GroupComparatorHelper();
53  	private StemComparatorHelperparatorHelper.html#StemComparatorHelper">StemComparatorHelper stemHelper = new StemComparatorHelper();
54  	/**
55  	 * 
56  	 */
57  	public GroupOrStemComparatorHelper() {
58  		super();
59  		// TODO Auto-generated constructor stub
60  	}
61  	
62  	
63  	/* (non-Javadoc)
64  	 * @see edu.internet2.middleware.grouper.ui.GrouperComparatorHelper#getComparisonString(java.lang.Object, java.util.ResourceBundle, java.lang.String)
65  	 */
66  	public String getComparisonString(Object obj, ResourceBundle config,
67  			String context) {
68  		if(obj instanceof GroupOrStem) {
69  			if(((GroupOrStem)obj).isGroup()) {
70  				return groupHelper.getComparisonString(((GroupOrStem)obj).getGroup(),config,context);
71  			}else{
72  				return stemHelper.getComparisonString(((GroupOrStem)obj).getStem(),config,context);
73  			}
74  		}
75  	
76  		if(! (obj instanceof SubjectPrivilegeAsMap)) {
77  			throw new IllegalArgumentException(obj + "is not a group or stem: " + (obj == null ? null : obj.getClass()));
78  		}
79  		SubjectPrivilegeAsMap spam = (SubjectPrivilegeAsMap)obj;
80  		ObjectAsMap groupOrStem = (ObjectAsMap)spam.get("groupOrStem");
81  		Object wrapped = groupOrStem.getWrappedObject();
82  		if(wrapped instanceof Group) return groupHelper.getComparisonString(wrapped,config,context);
83  		if(wrapped instanceof Stem) return stemHelper.getComparisonString(wrapped,config,context);
84  		throw new IllegalStateException(wrapped + " is not a group or stem");
85  		
86  	}
87  }
