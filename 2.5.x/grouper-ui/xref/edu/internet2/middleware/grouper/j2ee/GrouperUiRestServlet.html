213     //initialize the bean

312       
313       try {
314         instance = GrouperUtil.newInstance(GrouperUtil.forName(className));
315       } catch (RuntimeException re) {
316         GrouperUtil.injectInException(re, "Problem calling class and method: " + className + "." + methodName);
317         throw re;
318       }
319       
320       try {
321         
322         GrouperUtil.callMethod(instance.getClass(), instance, methodName,

349         }
350       } catch (ControllerDone cd) {
351         printToScreen = cd.isPrintGuiReponseJs();
352         //do nothing, this is ok
353       } catch (RuntimeException re) {
354         String error = "Problem calling reflection from URL: " + className + "." + methodName + "\n\n" + ExceptionUtils.getFullStackTrace(re);
355         LOG.error(error);
356         GrouperUiUtils.appendErrorToRequest(error);
357         
358         if (uiv2) {
359           guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
360               TextContainer.retrieveFromRequest().getText().get("guiMiscErrorPrefix")
361               + " " + GrouperUiUtils.escapeHtml(re.getMessage(), true)));
362         } else {
363           //if adding text area for form file submits, make sure to do that in errors too

365           
366           //print out error message for user, a new one

368           guiResponseJs.addAction(GuiScreenAction.newCloseModal());
369           guiResponseJs.setAddTextAreaTag(addTextArea);
370           guiResponseJs.addAction(GuiScreenAction.newAlert("Error: " + GrouperUiUtils.escapeHtml(re.getMessage(), true)));
371         }        
372       }

418   /**
419    * for error messages, get a detailed report of the request
420    * @param request
421    * @return the string of descriptive result
422    */
423   public static String requestDebugInfo(HttpServletRequest request) {
424     StringBuilder result = new StringBuilder();
425     result.append(" uri: ").append(request.getRequestURI());
426     result.append(", method: ").append(request.getMethod());
427     result.append(", decoded url strings: ");
428     List<String> urlStrings = extractUrlStrings(request);
429     int urlStringsLength = GrouperUtil.length(urlStrings);
430     if (urlStringsLength == 0) {
431       result.append("[none]");
432     } else {
433       for (int i = 0; i < urlStringsLength; i++) {
434         result.append(i).append(": '").append(urlStrings.get(i)).append("'");
435         if (i != urlStringsLength - 1) {
436           result.append(", ");
437         }
438       }
439     }
440     return result.toString();
441   }

455   /**
456    * TODO change to GrouperRestServlet in next release
457    * <pre>
458    * take a request uri and break up the url strings not including the app name or servlet
459    * this does not include the url params (if applicable)
460    * if the input is: grouper-ws/servicesRest/xhtml/v1_3_000/groups/members
461    * then the result is a list of size 2: {"group", "members"}
462    * 
463    * </pre>
464    * @param requestResourceFull
465    * @return the url strings
466    */
467   private static List<String> extractUrlStrings(String requestResourceFull) {
468     String[] requestResources = StringUtils.split(requestResourceFull, '/');
469     List<String> urlStrings = new ArrayList<String>();
470     //loop through and decode
471     int index = 0;
472     for (String requestResource : requestResources) {
473       //skip the app name and lite servlet
474       if (index++ < 2) {
475         continue;
476       }
477       //unescape the url encoding
478       urlStrings.add(GrouperUtil.escapeUrlDecode(requestResource));
479     }
480     return urlStrings;
481   }
