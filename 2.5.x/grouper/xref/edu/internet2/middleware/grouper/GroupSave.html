351   /**
352    * <pre>
353    * create or update a group.  Do not throw checked exceptions, wrap in unchecked
354    * 
355    * Note this will not rename a group at this time (might in future)
356    * 
357    * This is a static method since setters to Group objects persist to the DB
358    * 
359    * Steps:
360    * 
361    * 1. Find the group by groupNameToEdit
362    * 2. Internally set all the fields of the stem (no need to reset if already the same)
363    * 3. Store the group (insert or update) if needed
364    * 4. Return the group object
365    * 
366    * This runs in a tx so that if part of it fails the whole thing fails, and potentially the outer
367    * transaction too
368    * </pre>
369    * @return the group
370    * @deprecated use save() instead
371    */
372   @Deprecated
373   public Group saveUnchecked() {
374     return this.save();
375   }

378   /**
379    * <pre>
380    * create or update a group.  Note this will not rename a group at this time (might in future)
381    * 
382    * This is a static method since setters to Group objects persist to the DB
383    * 
384    * Steps:
385    * 
386    * 1. Find the group by groupNameToEdit
387    * 2. Internally set all the fields of the stem (no need to reset if already the same)
388    * 3. Store the group (insert or update) if needed
389    * 4. Return the group object
390    * 
391    * This runs in a tx so that if part of it fails the whole thing fails, and potentially the outer
392    * transaction too
393    * </pre>
394    * @return the group
395    * @throws StemNotFoundException
396    * @throws InsufficientPrivilegeException
397    * @throws StemAddException
398    * @throws GroupModifyException
399    * @throws GroupNotFoundException
400    * @throws GroupAddException
401    */
402   public Group save() 
403         throws StemNotFoundException, InsufficientPrivilegeException, StemAddException, 
404         GroupModifyException, GroupNotFoundException, GroupAddException {
405 
406     //help with incomplete entries
407     if (StringUtils.isBlank(this.name)) {
408       this.name = this.groupNameToEdit;
409     }
410 
411     //get from uuid since could be a rename
412     if (StringUtils.isBlank(this.groupNameToEdit) && !StringUtils.isBlank(this.uuid)) {
413       Group group = GroupFinder.findByUuid(GrouperSession.staticGrouperSession(), this.uuid, false, new QueryOptions().secondLevelCache(false));
414       if (group != null) {
415         this.groupNameToEdit = group.getName();
416       }
417     }
418     
419     if (StringUtils.isBlank(this.groupNameToEdit)) {
420       this.groupNameToEdit = this.name;
421     }
422     
423     //validate
424     //get the stem name
425     if (!StringUtils.contains(GroupSave.this.name, ":")) {
426       throw new RuntimeException("Group name must exist and must contain at least one stem name (separated by colons): '" + GroupSave.this.name + "'" );
427     }
428 
429     //default to insert or update
430     GroupSave.this.saveMode = (SaveMode)ObjectUtils.defaultIfNull(GroupSave.this.saveMode, SaveMode.INSERT_OR_UPDATE);
431     final SaveMode SAVE_MODE = GroupSave.this.saveMode;
432 
433     try {
434       //do this in a transaction
435       Group="../../../../edu/internet2/middleware/grouper/Group.html#Group">Group group = (Group)GrouperTransaction.callbackGrouperTransaction(new GrouperTransactionHandler() {
436   
437         @SuppressWarnings("cast")
438         @Override
439         public Object callback(GrouperTransaction grouperTransaction)
440             throws GrouperDAOException {
441           
442           grouperTransaction.setCachingEnabled(false);
443           
444           return (Group)GrouperSession.callbackGrouperSession(GroupSave.this.grouperSession, new GrouperSessionHandler() {
445 
446               @Override
447               public Object callback(GrouperSession theGrouperSession)
448                   throws GrouperSessionException {
449                 
450                 String groupNameForError = GrouperUtil.defaultIfBlank(GroupSave.this.groupNameToEdit, GroupSave.this.name);
451                 
452                 int lastColonIndex = GroupSave.this.name.lastIndexOf(':');
453                 boolean topLevelGroup = lastColonIndex < 0;
454         
455                 //empty is root stem
456                 String parentStemNameNew = GrouperUtil.parentStemNameFromName(GroupSave.this.name);
457                 
458                 //note, this might be blank
459                 String parentStemDisplayNameNew = GrouperUtil.parentStemNameFromName(GroupSave.this.displayName);
460                 String extensionNew = GrouperUtil.extensionFromName(GroupSave.this.name);
461                 
462                 String displayExtensionFromDisplayNameNew = GrouperUtil.extensionFromName(GroupSave.this.displayName);
463 
464                 //figure out the display extension from the extension or the name (or both!)
465                 String theDisplayExtension = null;
466                 
467                 //if blank, and display name blank, then, use extension
468                 if (StringUtils.isBlank(GroupSave.this.displayExtension) && StringUtils.isBlank(GroupSave.this.displayName)) {
469                   theDisplayExtension = extensionNew;
470                 } else if (!StringUtils.isBlank(GroupSave.this.displayExtension) && !StringUtils.isBlank(GroupSave.this.displayName)) {
471                   //if neither blank
472                   if (!StringUtils.equals(displayExtensionFromDisplayNameNew, GroupSave.this.displayExtension)) {
473                     throw new RuntimeException("The display extension '" + GroupSave.this.displayExtension 
474                         + "' is not consistent with the last part of the group name '" 
475                         + displayExtensionFromDisplayNameNew + "', display name: " + GroupSave.this.displayName);
476                   }
477                   theDisplayExtension = displayExtensionFromDisplayNameNew;
478                 } else if (!StringUtils.isBlank(GroupSave.this.displayExtension)) {
479                   theDisplayExtension = GroupSave.this.displayExtension;
480                 } else if (!StringUtils.isBlank(GroupSave.this.displayName)) {
481                   theDisplayExtension = displayExtensionFromDisplayNameNew;
482                 } else {
483                   throw new RuntimeException("Shouldnt get here");
484                 }
485 
486                 
487                 //lets find the stem
488                 Stem parentStem = null;
489                 
490                 try {
491                   parentStem = topLevelGroup ? StemFinder.findRootStem(theGrouperSession) 
492                       : StemFinder.findByName(theGrouperSession, parentStemNameNew, true);
493                 } catch (StemNotFoundException snfe) {
494                   
495                   //see if we should fix this problem
496                   if (GroupSave.this.createParentStemsIfNotExist) {
497                     
498                     //at this point the stem should be there (and is equal to currentStem), 
499                     //just to be sure, query again
500                     if (GrouperLoader.isDryRun()) {
501                       parentStem = StemFinder.findByName(theGrouperSession, parentStemNameNew, false);
502                       if (parentStem == null) {
503                         parentStem = new Stem();
504                         parentStem.setNameDb(parentStemNameNew);
505                         parentStem.setExtensionDb(GrouperUtil.extensionFromName(parentStemNameNew));
506                       }
507                     } else {
508                       parentStem = Stem._createStemAndParentStemsIfNotExist(theGrouperSession, parentStemNameNew, parentStemDisplayNameNew);
509                     }
510                   } else {
511                     throw new GrouperSessionException(new StemNotFoundException("Cant find stem: '" + parentStemNameNew 
512                         + "' (from update on stem name: '" + groupNameForError + "')"));
513                   }
514                 }
515                 
516                 Group theGroup = null;
517                 //see if update
518                 boolean isUpdate = SAVE_MODE.isUpdate(GroupSave.this.groupNameToEdit, GroupSave.this.name);
519         
520                 if (isUpdate) {
521                   String parentStemNameLookup = GrouperUtil.parentStemNameFromName(GroupSave.this.groupNameToEdit);
522                   if (!StringUtils.equals(parentStemNameLookup, parentStemNameNew)) {
523                     throw new GrouperSessionException(new GroupModifyException("Can't move a group.  Existing parentStem: '"
524                         + parentStemNameLookup + "', new stem: '" + parentStemNameNew + "'"));
525                 }    
526                 try {
527                     theGroup = GroupFinder.findByName(theGrouperSession, GroupSave.this.groupNameToEdit, true);
528                     
529                     //while we are here, make sure uuid's match if passed in
530                     if (!StringUtils.isBlank(GroupSave.this.uuid) && !StringUtils.equals(GroupSave.this.uuid, theGroup.getUuid())) {
531                       throw new RuntimeException("UUID group changes are not supported: new: " + GroupSave.this.uuid + ", old: " 
532                           + theGroup.getUuid() + ", " + groupNameForError);
533                     }
534                     
535                   } catch (GroupNotFoundException gnfe) {
536                     if (SAVE_MODE.equals(SaveMode.INSERT_OR_UPDATE) || SAVE_MODE.equals(SaveMode.INSERT)) {
537                       isUpdate = false;
538                     } else {
539                         throw new GrouperSessionException(gnfe);
540                     }
541                   }
542                 }
543                 //default
544                 GroupSave.this.saveResultType = SaveResultType.NO_CHANGE;
545                 boolean needsSave = false;
546                 
547                 boolean isRename = false;
548                 
549                 //if inserting
550                 if (!isUpdate) {
551                   GroupSave.this.saveResultType = SaveResultType.INSERT;
552                   if (GroupSave.this.typeOfGroup == null || GroupSave.this.typeOfGroup == TypeOfGroup.group) {
553                     theGroup = parentStem.internal_addChildGroup(extensionNew, theDisplayExtension, GroupSave.this.uuid);
554                   } else if (GroupSave.this.typeOfGroup == TypeOfGroup.role) {
555                     theGroup = (Group)parentStem.internal_addChildRole(extensionNew, theDisplayExtension, GroupSave.this.uuid);
556                   } else if (GroupSave.this.typeOfGroup == TypeOfGroup.entity) {
557                     theGroup = (Group)parentStem.internal_addChildEntity(extensionNew, theDisplayExtension, GroupSave.this.uuid);
558                   } else {
559                     throw new RuntimeException("Not expecting type of group: " + GroupSave.this.typeOfGroup);
560                   }
561                   
562                 } else {
563                   //check if different so it doesnt make unneeded queries
564                   if (!StringUtils.equals(theGroup.getExtension(), extensionNew)) {
565                       
566                       //lets just confirm that one doesnt exist
567                       String newName = GrouperUtil.parentStemNameFromName(theGroup.getName()) + ":" + extensionNew;
568                       
569                       Group existingGroup = GroupFinder.findByName(theGrouperSession.internal_getRootSession(), newName, false);
570                       
571                       if (existingGroup != null && !StringUtils.equals(theGroup.getUuid(), existingGroup.getUuid())) {
572                         throw new GroupModifyAlreadyExistsException("Group already exists: " + newName);
573                       }
574                       
575                       theGroup.setExtension(extensionNew, GroupSave.this.setAlternateNameIfRename);
576                     GroupSave.this.saveResultType = SaveResultType.UPDATE;
577                     needsSave = true;
578                     isRename = true;
579                   }
580                   if (!StringUtils.equals(theGroup.getDisplayExtension(), theDisplayExtension)) {
581                     GroupSave.this.saveResultType = SaveResultType.UPDATE;
582                     theGroup.setDisplayExtension(theDisplayExtension);
583                     needsSave = true;
584                   }
585                 }
586 
587                 if (GroupSave.this.idIndex != null) {
588                   if (GroupSave.this.saveResultType == SaveResultType.INSERT) {
589                     
590                     if (theGroup.assignIdIndex(GroupSave.this.idIndex)) {
591                       needsSave = true;
592                     }
593                     
594                   } else {
595                     //maybe they are equal...
596                     throw new RuntimeException("Cannot update idIndex for an already created group: " + GroupSave.this.idIndex + ", " + theGroup.getName());
597                   }
598                 }
599 
600                 //now compare and put all attributes (then store if needed)
601                 //null throws exception? hmmm.  remove attribute if blank
602                 if (!StringUtils.equals(StringUtils.defaultString(StringUtils.trim(theGroup.getDescription())), 
603                     StringUtils.defaultString(StringUtils.trim(GroupSave.this.description)))) {
604                   needsSave = true;
605                   if (GroupSave.this.saveResultType == SaveResultType.NO_CHANGE) {
606                     GroupSave.this.saveResultType = SaveResultType.UPDATE;
607                   }
608                   theGroup.setDescription(GroupSave.this.description);
609                 }
610                 
611                 if (!isRename) {
612                   if (!StringUtils.equals(StringUtils.defaultString(StringUtils.trim(theGroup.getAlternateName())), 
613                       StringUtils.defaultString(StringUtils.trim(GroupSave.this.alternateName)))) {
614                     needsSave = true;
615                     if (GroupSave.this.saveResultType == SaveResultType.NO_CHANGE) {
616                       GroupSave.this.saveResultType = SaveResultType.UPDATE;
617                     }
618                     if (StringUtils.isBlank(GroupSave.this.alternateName)) {
619                       theGroup.setAlternateNameDb(null);
620                     } else {
621                       theGroup.addAlternateName(StringUtils.trim(GroupSave.this.alternateName));
622                     }
623                   }
624                 }
625 
626                 //compare type of group
627                 if (GroupSave.this.typeOfGroup != null && GroupSave.this.typeOfGroup != theGroup.getTypeOfGroup()) {
628                   needsSave = true;
629                   if (GroupSave.this.saveResultType == SaveResultType.NO_CHANGE) {
630                     GroupSave.this.saveResultType = SaveResultType.UPDATE;
631                   }
632                   theGroup.setTypeOfGroup(GroupSave.this.typeOfGroup);
633                 }
634 
635                 //only store once
636                 if (needsSave) {
637                   theGroup.store();
638                 }
639 
640                 boolean changedPrivs = false;
641                                                 
642                 boolean readDefaultChecked = theGroup.hasRead(SubjectFinder.findAllSubject());
643       
644                 boolean viewDefaultChecked = theGroup.hasView(SubjectFinder.findAllSubject());
645       
646                 boolean optinDefaultChecked = theGroup.hasOptin(SubjectFinder.findAllSubject());
647       
648                 boolean optoutDefaultChecked = theGroup.hasOptout(SubjectFinder.findAllSubject());
649       
650                 boolean attrReadDefaultChecked = theGroup.hasGroupAttrRead(SubjectFinder.findAllSubject());
651                             
652                 if (GroupSave.this.privAllView != null && GroupSave.this.privAllView != viewDefaultChecked) {
653                   if (GroupSave.this.privAllView) {
654                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.VIEW, false);
655                   } else {
656                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.VIEW, false);
657                   }
658                 }
659       
660                 if (GroupSave.this.privAllRead != null && GroupSave.this.privAllRead != readDefaultChecked) {
661                   if (GroupSave.this.privAllRead) {
662                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.READ, false);
663                   } else {
664                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.READ, false);
665                   }
666                 }
667                 if (GroupSave.this.privAllOptin != null && GroupSave.this.privAllOptin != optinDefaultChecked) {
668                   if (GroupSave.this.privAllOptin) {
669                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.OPTIN, false);
670                   } else {
671                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.OPTIN, false);
672                   }
673                 }
674                 if (GroupSave.this.privAllOptout != null && GroupSave.this.privAllOptout != optoutDefaultChecked) {
675                   if (GroupSave.this.privAllOptout) {
676                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.OPTOUT, false);
677                   } else {
678                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.OPTOUT, false);
679                   }
680                 }
681                 if (GroupSave.this.privAllAttrRead != null && GroupSave.this.privAllAttrRead != attrReadDefaultChecked) {
682                   if (GroupSave.this.privAllAttrRead) {
683                     changedPrivs = changedPrivs | theGroup.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.GROUP_ATTR_READ, false);
684                   } else {
685                     changedPrivs = changedPrivs | theGroup.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.GROUP_ATTR_READ, false);
686                   }
687                 }
688                 
689                 if (changedPrivs) {
690                   if (GroupSave.this.saveResultType == SaveResultType.NO_CHANGE) {
691                     GroupSave.this.saveResultType = SaveResultType.UPDATE;
692                   }
693                 }
694                 
695 			          return theGroup;
696               }
697           });
698         }
699       });
700       return group;
701     } catch (RuntimeException re) {
702       
703       GrouperUtil.injectInException(re, "Problem saving group: " + this.name + ", thread: " + Integer.toHexString(Thread.currentThread().hashCode()));
704       
705       Throwable throwable = re.getCause();
706       if (throwable instanceof StemNotFoundException) {
707         throw (StemNotFoundException)throwable;
708       }
709       if (throwable instanceof InsufficientPrivilegeException) {
710         throw (InsufficientPrivilegeException)throwable;
711       }
712       if (throwable instanceof StemAddException) {
713         throw (StemAddException)throwable;
714       }
715       if (throwable instanceof GroupModifyException) {
716         throw (GroupModifyException)throwable;
717       }
718       if (throwable instanceof GroupNotFoundException) {
719         throw (GroupNotFoundException)throwable;
720       }
721       if (throwable instanceof GroupAddException) {
722         throw (GroupAddException)throwable;
723       }
724       //must just be runtime
725       throw re;
726     }
