60    /** logger */
61    private static final Log LOG = GrouperUtil.getLog(CompositeFinder.class);

63    /**
64     * Find where the specified {@link Group} is a {@link Composite}
65     * factor.
66     * <pre class="eg">
67     * Set composites = CompositeFinder.findAsFactor(g);
68     * </pre>
69     * @param   g   Search on this {@link Group}.
70     * @return  A set of {@link Composite} objects.
71     * @since   1.0
72     */
73    public static Set<Composite> findAsFactor(Group g) {
74      Set<Composite>            where = new LinkedHashSet<Composite>();
75      GrouperSession  s     = GrouperSession.staticGrouperSession();
76      Member          m     = s.getMember();
77      Composite       c;
78      Iterator        it    = GrouperDAOFactory.getFactory().getComposite().findAsFactor( g ).iterator();
79      while (it.hasNext()) {
80        c = (Composite) it.next();
81        try {
82          if ( m.canView( c.getOwnerGroup() ) ) {
83            where.add(c);
84          }
85        }
86        catch (GroupNotFoundException eGNF) {
87          LOG.error(E.COMPF_FINDASFACTOR + eGNF.getMessage() );
88        }
89      } 
90      return where;
91    }
92  
93    /**
94     * Find {@link Composite} owned by this {@link Group}.
95     * <pre class="eg">
96     * Composite c = CompositeFinder.findAsOwner(g);
97     * </pre>
98     * @deprecated use findAsOwner(Group, boolean) instead.
99     * @param   g   Search on this {@link Group}.
100    * @return  c   {@link Composite} owned by this {@link Group}.
101    * @throws  CompositeNotFoundException
102    * @since   1.0
103    */
104   @Deprecated
105   public static Composite findAsOwner(Group g) 
106     throws  CompositeNotFoundException, GroupNotFoundException {
107     return findAsOwner(g, true);
108   }
109 
110   /**
111    * Find {@link Composite} owned by this {@link Group}.
112    * <pre class="eg">
113    * Composite c = CompositeFinder.findAsOwner(g, false);
114    * </pre>
115    * @param   g   Search on this {@link Group}.
116    * @param throwExceptionIfNotFound true to throw exception if not found
117    * @return  c   {@link Composite} owned by this {@link Group}.
118    * @throws  CompositeNotFoundException if throwExceptionIfNotFound is true, and composite is not found
119    * @throws GroupNotFoundException if the group owner of composite cant be found (this is a problem)
120    * @since   1.0
121    */
122   public static Composite findAsOwner(Group g, boolean throwExceptionIfNotFound) 
123     throws  CompositeNotFoundException, GroupNotFoundException {
124     GrouperSession  s = GrouperSession.staticGrouperSession();
125     Member          m = s.getMember();
126     
127     Composite       c = null;
128     
129     c =GrouperDAOFactory.getFactory().getComposite().findAsOwner(g, throwExceptionIfNotFound);
130     if (c == null) {
131       return null;
132     }
133     if ( m.canView( c.getOwnerGroup() ) ) {
134       return c;
135     }
136     if (throwExceptionIfNotFound) {
137       throw new CompositeNotFoundException();
138     }
139     return null;
140   } // public static Composite findAsOwner(g)
141 
142 } // public class CompositeFinder
