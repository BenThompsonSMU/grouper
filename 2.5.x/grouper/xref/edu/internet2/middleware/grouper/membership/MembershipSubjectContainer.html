167   /**
168    * add effective memberships for inheritance of privileges or 
169    * GrouperAll for stem
170    * @param membershipSubjectContainers
171    */
172   public static void considerNamingPrivilegeInheritance(final Set<MembershipSubjectContainer> membershipSubjectContainers) {
173 
174     Set<Stem> stems = new HashSet<Stem>();
175     
176     //get the list of groups
177     for (MembershipSubjectContainer membershipSubjectContainer : membershipSubjectContainers) {
178       Stem stem = membershipSubjectContainer.getStemOwner();
179       
180       //note not sure why this would be null, these should be stem owned memberships
181       if (stem != null) {
182         stems.add(stem);
183       }
184     }
185     
186     Map<MultiKey, Boolean> stemIdPermissionNameAllowedForGrouperAll = new HashMap<MultiKey, Boolean>();
187 
188     for (Stem stem : stems) {
189 
190       boolean grouperAllHasStemAdmin = stem.hasStemAdmin(SubjectFinder.findAllSubject());
191       stemIdPermissionNameAllowedForGrouperAll.put(new MultiKey(stem.getId(), NamingPrivilege.STEM_ADMIN.getName()), grouperAllHasStemAdmin);
192       
193       boolean grouperAllHasCreate = grouperAllHasStemAdmin || stem.hasCreate(SubjectFinder.findAllSubject());
194       stemIdPermissionNameAllowedForGrouperAll.put(new MultiKey(stem.getId(), NamingPrivilege.CREATE.getName()), grouperAllHasCreate);
195 
196       boolean grouperAllHasAttrRead = grouperAllHasStemAdmin || grouperAllHasCreate || stem.hasStemAttrRead(SubjectFinder.findAllSubject());
197       stemIdPermissionNameAllowedForGrouperAll.put(new MultiKey(stem.getId(), NamingPrivilege.STEM_ATTR_READ.getName()), grouperAllHasAttrRead);
198 
199       boolean grouperAllHasAttrUpdate = grouperAllHasStemAdmin || grouperAllHasCreate || stem.hasStemAttrUpdate(SubjectFinder.findAllSubject());
200       stemIdPermissionNameAllowedForGrouperAll.put(new MultiKey(stem.getId(), NamingPrivilege.STEM_ATTR_UPDATE.getName()), grouperAllHasAttrUpdate);
201     }
202     
203     Set<String> stemFieldNames = GrouperUtil.toSet(Field.FIELD_NAME_STEM_ADMINS, Field.FIELD_NAME_CREATORS, 
204         Field.FIELD_NAME_STEM_ATTR_READERS, Field.FIELD_NAME_STEM_ATTR_UPDATERS);
205     
206     Subject rootSubject = SubjectFinder.findRootSubject();
207     Subject everyEntitySubject = SubjectFinder.findAllSubject();
208     
209     for (MembershipSubjectContainer membershipSubjectContainer : membershipSubjectContainers) {
210       
211       Stem stem = membershipSubjectContainer.getStemOwner();
212       
213       boolean grouperAllHasStemAdmin = stemIdPermissionNameAllowedForGrouperAll.get(new MultiKey(stem.getId(), NamingPrivilege.STEM_ADMIN.getName()));
214       boolean grouperAllHasCreate = stemIdPermissionNameAllowedForGrouperAll.get(new MultiKey(stem.getId(), NamingPrivilege.CREATE.getName()));
215       boolean grouperAllHasAttrRead = stemIdPermissionNameAllowedForGrouperAll.get(new MultiKey(stem.getId(), NamingPrivilege.STEM_ATTR_READ.getName()));
216       boolean grouperAllHasAttrUpdate = stemIdPermissionNameAllowedForGrouperAll.get(new MultiKey(stem.getId(), NamingPrivilege.STEM_ATTR_UPDATE.getName()));
217       
218       Subject subject = membershipSubjectContainer.getSubject();
219       
220       //if we are on grouper system
221       if (SubjectHelper.eq(subject, rootSubject)) {
222         
223         for (String fieldName : stemFieldNames) {
224           //it is also effective, merge that with whatever was there
225           membershipSubjectContainer.addMembership(fieldName, MembershipAssignType.EFFECTIVE);
226         }
227         
228       } else {
229         //else
230         boolean isEveryEntity = SubjectHelper.eq(everyEntitySubject, membershipSubjectContainer.getSubject());
231         
232         //see what the subject has
233         boolean subjectHasStemAdminEffective = membershipSubjectContainer.getMembershipContainers().get(Field.FIELD_NAME_STEM_ADMINS) != null
234             && membershipSubjectContainer.getMembershipContainers().get(Field.FIELD_NAME_STEM_ADMINS).getMembershipAssignType().isNonImmediate()
235             || (isEveryEntity ? false : grouperAllHasStemAdmin);
236 
237         boolean subjectHasStemAdmin = membershipSubjectContainer.getMembershipContainers().get(Field.FIELD_NAME_STEM_ADMINS) != null 
238             || subjectHasStemAdminEffective || grouperAllHasStemAdmin;
239         
240         boolean subjectHasCreateEffective = membershipSubjectContainer.getMembershipContainers().get(Field.FIELD_NAME_CREATORS) != null
241             && membershipSubjectContainer.getMembershipContainers().get(Field.FIELD_NAME_CREATORS).getMembershipAssignType().isNonImmediate()
242             || subjectHasStemAdmin || (isEveryEntity ? false : grouperAllHasCreate);
243         
244         boolean subjectHasCreate = membershipSubjectContainer.getMembershipContainers().get(Field.FIELD_NAME_CREATORS) != null 
245             || subjectHasCreateEffective || grouperAllHasCreate;
246         
247         boolean subjectHasAttrReadEffective = membershipSubjectContainer.getMembershipContainers().get(Field.FIELD_NAME_STEM_ATTR_READERS) != null
248             && membershipSubjectContainer.getMembershipContainers().get(Field.FIELD_NAME_STEM_ATTR_READERS).getMembershipAssignType().isNonImmediate()
249             || subjectHasCreate || subjectHasStemAdmin || (isEveryEntity ? false : grouperAllHasAttrRead);
250         
251         boolean subjectHasAttrUpdateEffective = membershipSubjectContainer.getMembershipContainers().get(Field.FIELD_NAME_STEM_ATTR_UPDATERS) != null
252             && membershipSubjectContainer.getMembershipContainers().get(Field.FIELD_NAME_STEM_ATTR_UPDATERS).getMembershipAssignType().isNonImmediate()
253             || subjectHasCreate || subjectHasStemAdmin  || (isEveryEntity ? false : grouperAllHasAttrUpdate);
254 
255         //if the subject has an effective stem priv, add it in
256         if (subjectHasStemAdminEffective) {
257           membershipSubjectContainer.addMembership(Field.FIELD_NAME_STEM_ADMINS, MembershipAssignType.EFFECTIVE);
258         }
259         if (subjectHasCreateEffective) {
260           membershipSubjectContainer.addMembership(Field.FIELD_NAME_CREATORS, MembershipAssignType.EFFECTIVE);
261         }
262         if (subjectHasAttrReadEffective) {
263           membershipSubjectContainer.addMembership(Field.FIELD_NAME_STEM_ATTR_READERS, MembershipAssignType.EFFECTIVE);
264         }
265         if (subjectHasAttrUpdateEffective) {
266           membershipSubjectContainer.addMembership(Field.FIELD_NAME_STEM_ATTR_UPDATERS, MembershipAssignType.EFFECTIVE);
267         }
268         
269       }
270     }
271   }

273   /**
274    * consider a new membership
275    * @param fieldName
276    * @param membershipAssignType
277    */
278   public void addMembership(String fieldName, MembershipAssignType newMembershipAssignType) {
279     MembershipContainer membershipContainer = this.getMembershipContainers().get(fieldName);
280 
281     if(membershipContainer != null) {
282       //maybe we need to change or set the type
283       newMembershipAssignType = MembershipAssignType.convert(membershipContainer.getMembershipAssignType(),newMembershipAssignType);
284       membershipContainer.setMembershipAssignType(newMembershipAssignType);
285     } else {
286       //create a new one where one didnt exist before
287       membershipContainer = new MembershipContainer(fieldName, newMembershipAssignType);
288       this.getMembershipContainers().put(fieldName, membershipContainer);
289     }
290   }

367   /**
368    * @see Object#toString()
369    */
370   @Override
371   public String toString() {
372     StringBuilder result = new StringBuilder();
373     
374     if (this.groupOwner != null) {
375       result.append("Group: ");
376       result.append(this.groupOwner.getName());
377       result.append(", ");
378     }
379     
380     if (this.stemOwner != null) {
381       result.append("Stem: ");
382       result.append(this.stemOwner.getName());
383       result.append(", ");
384     }
385     
386     if (this.attributeDefOwner != null) {
387       result.append("Attribute def: ");
388       result.append(this.attributeDefOwner.getName());
389       result.append(", ");
390     }
391     
392     if (this.subject == null) {
393       result.append("Subject: null");
394     } else {
395       result.append(GrouperUtil.subjectToString(this.subject));
396     }
397     result.append(": ");
398     if (GrouperUtil.length(this.membershipContainers) == 0) {
399       result.append(" no memberships");
400     } else {
401       
402       Set<String> fieldNameSet = this.membershipContainers.keySet();
403       int index = 0;
404       for (String fieldName: fieldNameSet) {
405         result.append(this.membershipContainers.get(fieldName));
406         if (index < fieldNameSet.size()-1) {
407           result.append(", ");
408         }
409         index++;
410       }
411     }
412     return result.toString();
413   }

417    * GrouperAll for attributeDef
418    * @param membershipSubjectContainers
419    */
420   public static void considerAttributeDefPrivilegeInheritance(final Set<MembershipSubjectContainer> membershipSubjectContainers) {
421 
422     Set<AttributeDef> attributeDefs = new HashSet<AttributeDef>();
423     
424     //get the list of attribute defs
425     for (MembershipSubjectContainer membershipSubjectContainer : membershipSubjectContainers) {
426       AttributeDef attributeDef = membershipSubjectContainer.getAttributeDefOwner();
427       
428       //note, not sure why this would be null, these should be attribute def owned memberships
429       if (attributeDef != null) {
430         attributeDefs.add(attributeDef);
431       }
432     }
433     
434     Map<MultiKey, Boolean> attributeDefIdPermissionNameAllowedForGrouperAll = new HashMap<MultiKey, Boolean>();
435 
436     for (AttributeDef attributeDef : attributeDefs) {
437       
438       boolean grouperAllHasAdmin = attributeDef.getPrivilegeDelegate().hasAttrAdmin(SubjectFinder.findAllSubject());
439       attributeDefIdPermissionNameAllowedForGrouperAll.put(new MultiKey(attributeDef.getId(), AttributeDefPrivilege.ATTR_ADMIN.getName()), grouperAllHasAdmin);
440       
441       boolean grouperAllHasUpdate = grouperAllHasAdmin || attributeDef.getPrivilegeDelegate().hasAttrUpdate(SubjectFinder.findAllSubject());
442       attributeDefIdPermissionNameAllowedForGrouperAll.put(new MultiKey(attributeDef.getId(), AttributeDefPrivilege.ATTR_UPDATE.getName()), grouperAllHasUpdate);
443 
444       boolean grouperAllHasRead = grouperAllHasAdmin || attributeDef.getPrivilegeDelegate().hasAttrRead(SubjectFinder.findAllSubject());
445       attributeDefIdPermissionNameAllowedForGrouperAll.put(new MultiKey(attributeDef.getId(), AttributeDefPrivilege.ATTR_READ.getName()), grouperAllHasRead);

448       attributeDefIdPermissionNameAllowedForGrouperAll.put(new MultiKey(attributeDef.getId(), AttributeDefPrivilege.ATTR_OPTIN.getName()), grouperAllHasOptin);
