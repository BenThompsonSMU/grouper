1   package edu.internet2.middleware.grouper.app.grouperTypes;
2   
3   import java.util.Arrays;
4   import java.util.Collections;
5   import java.util.HashMap;
6   import java.util.List;
7   import java.util.Map;
8   
9   import edu.internet2.middleware.grouper.cfg.GrouperConfig;
10  import edu.internet2.middleware.grouper.util.GrouperUtil;
11  
12  public class GrouperObjectTypesSettings {
13    
14    public static final String REF = "ref";
15    public static final String BASIS = "basis";
16    public static final String POLICY = "policy";
17    public static final String ETC = "etc";
18    public static final String ORG = "org";
19    public static final String TEST = "test";
20    public static final String SERVICE = "service";
21    public static final String APP = "app";
22    public static final String READ_ONLY = "readOnly";
23    public static final String GROUPER_SECURITY = "grouperSecurity";
24    
25    // bundle is only being used in auto assign types to suggest folder by assigned type ref 
26    // even though the folder extension matched "bundle" 
27    public static final String BUNDLE = "bundle";
28  
29    /**
30     * if object types are enabled
31     * @return if object types are enabled
32     */
33    public static boolean objectTypesEnabled() {
34      return GrouperConfig.retrieveConfig().propertyValueBoolean("objectTypes.enable", true);
35    }
36    
37    /**
38     * 
39     * @return the stem name with no last colon
40     */
41    public static String objectTypesStemName() {
42      return GrouperUtil.stripEnd(GrouperConfig.retrieveConfig().propertyValueString("objectTypes.systemFolder", 
43          GrouperConfig.retrieveConfig().propertyValueString("grouper.rootStemForBuiltinObjects") + ":objectTypes"), ":");
44    }
45    
46    /**
47     * all the object types
48     * @return
49     */
50    public static List<String> getObjectTypeNames() {
51      return Collections.unmodifiableList(Arrays.asList(BASIS, REF, POLICY, ETC, GROUPER_SECURITY, ORG, APP, SERVICE, READ_ONLY, TEST));
52    }
53    
54    /**
55     * object types that require data owner and member description 
56     * @return
57     */
58    public static List<String> getDataOwnerMemberDescriptionRequiringObjectTypeNames() {
59      return Collections.unmodifiableList(Arrays.asList(REF, BASIS, POLICY, ORG));
60    }
61  
62    /**
63     * object types that require service
64     * @return
65     */
66    public static List<String> getServiceRequiringObjectTypeNames() {
67      return Collections.unmodifiableList(Arrays.asList(APP));
68    }
69    
70    /**
71     * map showing which folder extension should go with with type 
72     * @return map of folder extension to object type
73     */
74    public static Map<String, String> getFolderExtensionToTypeSuggestion() {
75      
76      Map<String, String> extensionToType = new HashMap<String, String>();
77      
78      for (String objectType: getObjectTypeNames()) {
79        extensionToType.put(objectType, objectType);
80      }
81      // for folders that end with bundle we want to suggest that ref be assigned to them
82      extensionToType.put(BUNDLE, REF);
83      
84      return extensionToType;
85    }
86    
87  }
