23  package edu.internet2.middleware.grouper.app.gsh;
24  import bsh.CallStack;
25  import bsh.Interpreter;
26  import edu.internet2.middleware.grouper.GroupType;
27  import edu.internet2.middleware.grouper.GrouperSession;
28  import edu.internet2.middleware.grouper.exception.InsufficientPrivilegeException;
29  import edu.internet2.middleware.grouper.exception.SchemaException;
30  
31  /**
32   * Add a {@link GroupType}.
33   * <p/>
34   * @author  blair christensen.
35   * @version $Id: typeAdd.java,v 1.2 2008-09-29 03:38:28 mchyzer Exp $
36   * @since   0.1.0
37   */
38  public class typeAdd {
39  
40    // PUBLIC CLASS METHODS //
41  
42    /**
43     * Add a {@link GroupType}.
44     * <p/>
45     * @param   i           BeanShell interpreter.
46     * @param   stack       BeanShell call stack.
47     * @param   name        Name of {@link GroupType} to add.
48     * @return  {@link GroupType}
49     * @throws  GrouperShellException
50     * @since   0.1.0
51     */
52    public static GroupType invoke(Interpreter i, CallStack stack, String name) 
53      throws  GrouperShellException
54    {
55      GrouperShell.setOurCommand(i, true);
56      try {
57        return invoke(GrouperShell.getSession(i), name);
58      }
59      catch (InsufficientPrivilegeException eIP)  {
60        GrouperShell.error(i, eIP);
61      }
62      catch (SchemaException eS)                  {
63        GrouperShell.error(i, eS);
64      }
65      throw new GrouperShellException(GshErrorMessages.TYPE_ADD + name);
66    }
67    
68    /**
69     * Add a {@link GroupType}.
70     * <p/>
71     * @param   grouperSession
72     * @param   name        Name of {@link GroupType} to add.
73     * @return  {@link GroupType}
74     */
75    @SuppressWarnings("deprecation")
76    public static GroupType invoke(GrouperSession grouperSession, String name) {
77      return GroupType.createType(grouperSession, name);
78    }
79  
80  } // public class typeAdd
