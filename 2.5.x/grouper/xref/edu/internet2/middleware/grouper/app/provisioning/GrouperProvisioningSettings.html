22    private static void populateTargets() {
23      
24      Map<String, String> propertiesMap = GrouperConfig.retrieveConfig().propertiesMap(grouperProvisioningTargetKey);
25      
26      for (Entry<String, String> entry: propertiesMap.entrySet()) {
27            
28        String property = entry.getKey();
29        
30        // key is the configured key, the value of .key
31        String key = entry.getValue();
32        
33        Matcher matcher = grouperProvisioningTargetKey.matcher(property);
34        
35        if (matcher.matches()) {
36          // name is the part of the property key
37          String name = matcher.group(1);
38          
39          String groupAllowedToAssign = GrouperConfig.retrieveConfig().propertyValueString("provisioning.target."+name+".groupAllowedToAssign", null);
40          boolean allowAssignmentsOnlyOnOneStem = GrouperConfig.retrieveConfig().propertyValueBoolean("provisioning.target."+name+".allowAssignmentsOnlyOnOneStem", false);
41          boolean readOnly = GrouperConfig.retrieveConfig().propertyValueBoolean("provisioning.target."+name+".readOnly", false);
42

44          target.setGroupAllowedToAssign(groupAllowedToAssign);
45          target.setAllowAssignmentsOnlyOnOneStem(allowAssignmentsOnlyOnOneStem);
46          target.setReadOnly(readOnly);
47          targets.put(name, target);
