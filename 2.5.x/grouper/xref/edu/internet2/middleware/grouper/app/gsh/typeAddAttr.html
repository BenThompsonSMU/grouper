23  package edu.internet2.middleware.grouper.app.gsh;
24  import bsh.CallStack;
25  import bsh.Interpreter;
26  import edu.internet2.middleware.grouper.GroupType;
27  import edu.internet2.middleware.grouper.GroupTypeFinder;
28  import edu.internet2.middleware.grouper.GrouperSession;
29  import edu.internet2.middleware.grouper.attr.AttributeDefName;
30  import edu.internet2.middleware.grouper.exception.InsufficientPrivilegeException;
31  import edu.internet2.middleware.grouper.exception.SchemaException;
32  
33  /**
34   * Add an <code>Attribute</code> to a <code>GroupType</code>.
35   * <p/>
36   * @author  blair christensen.
37   * @version $Id: typeAddAttr.java,v 1.3 2009-03-15 06:37:23 mchyzer Exp $
38   * @since   0.1.0
39   */
40  public class typeAddAttr {
41  
42    // PUBLIC CLASS METHODS //
43  
44    /**
45     * Add an <code>Attribute</code> to a <code>GroupType</code>.
46     * <p/>
47     * @param   i           BeanShell interpreter.
48     * @param   stack       BeanShell call stack.
49     * @param   type        Add to this <code>GroupType</code>.
50     * @param   name        Name of <code>Attribute</code>.
51     * @return  <code>Field</code>
52     * @throws  GrouperShellException
53     * @since   0.1.0
54     */
55    public static AttributeDefName invoke(
56      Interpreter i, CallStack stack, String type, String name
57    ) 
58      throws  GrouperShellException
59    {
60      GrouperShell.setOurCommand(i, true);
61      try {
62        GrouperSession  s = GrouperShell.getSession(i);
63        return invoke(s, type, name);
64      }
65      catch (InsufficientPrivilegeException eIP)  {
66        GrouperShell.error(i, eIP);
67      }
68      catch (SchemaException eS)                  {
69        GrouperShell.error(i, eS);
70      }
71      throw new GrouperShellException(GshErrorMessages.TYPE_ADDATTR + type);
72    }
73    
74    /**
75     * Add an <code>Attribute</code> to a <code>GroupType</code>.
76     * <p/>
77     * @param   grouperSession
78     * @param   type        Add to this <code>GroupType</code>.
79     * @param   name        Name of <code>Attribute</code>.
80     * @return  <code>Field</code>
81     */
82    @SuppressWarnings("deprecation")
83    public static AttributeDefName invoke(GrouperSession grouperSession, String type, String name) {
84      GroupType       t = GroupTypeFinder.find(type, true);
85      return t.addAttribute(grouperSession, name);
86    }
87  
88  } // public class typeAddAttr
