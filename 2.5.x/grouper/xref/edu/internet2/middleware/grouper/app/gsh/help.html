29  /**

31   * <p/>
32   * @author  blair christensen.
33   * @version $Id: help.java,v 1.20 2009-08-12 12:44:45 shilen Exp $

36  public class help {
37  
38    /**
39     * get help on a specific command
40     * @param interpreter
41     * @param callStack
42     * @param helpOn
43     */
44    public static void invoke(Interpreter interpreter, CallStack callStack,
45        String helpOn) {
46      interpreter.print(getHelp(helpOn));
47    }
48    
49    /**
50     * Display usage information.
51     * <p/>
52     * @param   i           BeanShell interpreter.
53     * @param   stack       BeanShell call stack.
54     * @since   0.0.1
55     */
56    public static void invoke(Interpreter i, CallStack stack) {
57      GrouperShell.setOurCommand(i, true);
58      i.print(getHelp(null));
59    }
60    
61    /**
62     * get help
63     * @param helpOn if specific command
64     */
65    public static void invoke(String helpOn) {
66      String help = getHelp(helpOn);
67      System.out.print(help);
68    }
