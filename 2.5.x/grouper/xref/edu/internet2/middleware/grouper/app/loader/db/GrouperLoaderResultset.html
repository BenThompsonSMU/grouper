69     * if we should bulk lookup subjects to add/remove
70     */
71    public void bulkLookupSubjects() {
72  
73      //do not bulk retrieve subjects if not configured to do that
74      if (!GrouperLoaderConfig.retrieveConfig().propertyValueBoolean("loader.bulkLookupSubjects", true)) {
75        return;
76      }
77  
78      //dont do this twice if has already done on the outer list
79      if (this.hasBulkLookupedSubjects) {
80        return;
81      }
82      
83      this.hasBulkLookupedSubjects = true;
84      
85      //lets assume all rows are the same
86      if (GrouperUtil.length(this.data) == 0) {
87        return;
88      }
89  
90      //if we are looking by subject id or identifier and dont have the source
91      Set<String> subjectIdsOrIdentifiers = new HashSet<String>();
92  
93      //if we have the source and are looking by source and id or identifier
94      Map<String, Set<String>> sourceToSubjectIdsOrIdentifiers = new LinkedHashMap<String, Set<String>>();
95  
96      String subjectIdCol = null;
97      String sourceIdCol = null;
98  
99      String defaultSubjectSourceId = GrouperLoaderConfig.retrieveConfig().propertyValueString(
100         GrouperLoaderConfig.DEFAULT_SUBJECT_SOURCE_ID);
101     
102     for (Row row : this.data) {
103       
104       if (subjectIdCol == null) {
105         
106         String subjectId = (String) row.getCell(
107             GrouperLoaderResultset.SUBJECT_ID_COL, false);
108         
109         if (!StringUtils.isBlank(subjectId)) {
110           subjectIdCol = GrouperLoaderResultset.SUBJECT_ID_COL;
111         } else {
112           String subjectIdentifier = (String) row.getCell(
113               GrouperLoaderResultset.SUBJECT_IDENTIFIER_COL, false);
114           if (!StringUtils.isBlank(subjectIdentifier)) {
115             subjectIdCol = GrouperLoaderResultset.SUBJECT_IDENTIFIER_COL;
116           } else {
117             String subjectIdOrIdentifier = (String) row.getCell(
118                 GrouperLoaderResultset.SUBJECT_ID_OR_IDENTIFIER_COL, false);
119             if (!StringUtils.isBlank(subjectIdOrIdentifier)) {
120               subjectIdCol = GrouperLoaderResultset.SUBJECT_ID_OR_IDENTIFIER_COL;
121             } else {
122               //wtf?  dont do anthing, cant find subject id col, let the normal error message prevail
123               return;
124             }
125           }
126         }
127         //see if there is a source id col
128         String subjectSourceId = (String) row.getCell(
129             GrouperLoaderResultset.SUBJECT_SOURCE_ID_COL, false);
130         if (!StringUtils.isBlank(subjectSourceId)) {
131           sourceIdCol = GrouperLoaderResultset.SUBJECT_SOURCE_ID_COL;
132         }
133       }
134       String sourceId = null;
135       String subjectIdOrIdentifer = null;
136       if (!StringUtils.isBlank(sourceIdCol)) {
137         sourceId = (String) row.getCell(
138             GrouperLoaderResultset.SUBJECT_SOURCE_ID_COL, false);
139       }
140       //default this to what is in the config file
141       sourceId = StringUtils.defaultString(sourceId, defaultSubjectSourceId);
142       subjectIdOrIdentifer = (String) row.getCell(subjectIdCol, false);
143 
144       if (!StringUtils.isBlank(sourceId)) {
145         Set<String> subjectIdsOrIdentifiersForSource = sourceToSubjectIdsOrIdentifiers.get(sourceId);
146         //lazy load for source
147         if (subjectIdsOrIdentifiersForSource == null) {
148           subjectIdsOrIdentifiersForSource = new HashSet<String>();
149           sourceToSubjectIdsOrIdentifiers.put(sourceId, subjectIdsOrIdentifiersForSource);
150         }
151         subjectIdsOrIdentifiersForSource.add(subjectIdOrIdentifer);
152       } else {
153         //no source, just keep track of identifier
154         subjectIdsOrIdentifiers.add(subjectIdOrIdentifer);
155       }
156 
157     }
158     
159     Map<String, Map<String, Subject>> sourceToSubjectIdOrIdentifierToSubject = new HashMap<String, Map<String, Subject>>();
160     
161     //lets resolve the subjects
162     //see if there are any where we know the source
163     if (GrouperUtil.length(sourceToSubjectIdsOrIdentifiers) > 0) {
164       
165       Map<String, Subject> localSubjectIdOrIdentifierToSubject = null;
166       
167       for (String sourceId : sourceToSubjectIdsOrIdentifiers.keySet()) {
168         Set<String> subjectIdsOrIdentifiersForSource = sourceToSubjectIdsOrIdentifiers.get(sourceId);
169         //what are they?
170         if (StringUtils.equals(subjectIdCol, GrouperLoaderResultset.SUBJECT_ID_COL)) {
171 
172           localSubjectIdOrIdentifierToSubject = SubjectFinder.findByIds(subjectIdsOrIdentifiersForSource, sourceId);
173         
174         } else if (StringUtils.equals(subjectIdCol, GrouperLoaderResultset.SUBJECT_IDENTIFIER_COL)) {
175         
176           localSubjectIdOrIdentifierToSubject = SubjectFinder.findByIdentifiers(subjectIdsOrIdentifiersForSource, sourceId);
177           
178         } else if (StringUtils.equals(subjectIdCol, GrouperLoaderResultset.SUBJECT_ID_OR_IDENTIFIER_COL)) {
179           
180           localSubjectIdOrIdentifierToSubject = SubjectFinder.findByIdsOrIdentifiers(subjectIdsOrIdentifiersForSource, sourceId);
181           
182         } else {
183           throw new RuntimeException("Not expecting subjectIdCol: " + subjectIdCol);
184         }
185         
186         //keep track in map
187         sourceToSubjectIdOrIdentifierToSubject.put(sourceId, localSubjectIdOrIdentifierToSubject);
188         
189       }
190     }
191 
192     Map<String, Subject> subjectIdOrIdentifierToSubject = null;
193 
194     if (GrouperUtil.length(subjectIdsOrIdentifiers) > 0) {
195       //what are they?
196       if (StringUtils.equals(subjectIdCol, GrouperLoaderResultset.SUBJECT_ID_COL)) {
197         subjectIdOrIdentifierToSubject = SubjectFinder.findByIds(subjectIdsOrIdentifiers);
198       } else if (StringUtils.equals(subjectIdCol, GrouperLoaderResultset.SUBJECT_IDENTIFIER_COL)) {
199         subjectIdOrIdentifierToSubject = SubjectFinder.findByIdentifiers(subjectIdsOrIdentifiers);
200       } else if (StringUtils.equals(subjectIdCol, GrouperLoaderResultset.SUBJECT_ID_OR_IDENTIFIER_COL)) {
201         subjectIdOrIdentifierToSubject = SubjectFinder.findByIdsOrIdentifiers(subjectIdsOrIdentifiers);
202       } else {
203         throw new RuntimeException("Not expecting subjectIdCol: " + subjectIdCol);
204       }
205     }
206     
207     for (Row row : this.data) {
208 
209       String subjectIdOrIdentifier = (String)row.getCell(subjectIdCol, false);
210       Subject subject = null;
211       
212       //get the subject from the source id map
213       if (!StringUtils.isBlank(sourceIdCol) || !StringUtils.isBlank(defaultSubjectSourceId)) {
214         String sourceId = StringUtils.defaultString((String)row.getCell(sourceIdCol, false), defaultSubjectSourceId);
215         if (!StringUtils.isBlank(sourceId)) {
216           Map<String, Subject> localSubjectIdOrIdentifierToSubject = sourceToSubjectIdOrIdentifierToSubject.get(sourceId);
217           if (localSubjectIdOrIdentifierToSubject != null) {
218             subject = localSubjectIdOrIdentifierToSubject.get(subjectIdOrIdentifier);
219           }
220         }
221       }
222       if (subject == null && subjectIdOrIdentifierToSubject != null) {
223         //get the subject from the subjectId map
224         subject = subjectIdOrIdentifierToSubject.get(subjectIdOrIdentifier);
225       }
226 
227       //if we found it, set it
228       if (subject != null) {
229         row.subject = subject;
230       }
231       
232     }
233 
234     
235 
236   }

397   /**
398    * get a resultset based on a db and query
399    * @param grouperLoaderDb
400    * @param query
401    * @param jobName 
402    * @param hib3GrouperLoaderLog 
403    */
404   public GrouperLoaderResultset(GrouperLoaderDb grouperLoaderDb, String query,
405       String jobName, Hib3GrouperLoaderLog hib3GrouperLoaderLog) {
406 
407     this.dataIndex = null;
408     
409     //small security check (not failsafe, but better than nothing)
410     if (!query.toLowerCase().trim().startsWith("select")) {
411       throw new RuntimeException("Invalid query, must start with select: " + query);
412     }
413     Connection connection = null;
414     Statement statement = null;
415     ResultSet resultSet = null;
416     try {
417       connection = grouperLoaderDb.connection();
418       try {
419         // create and execute a SELECT
420         statement = connection.createStatement();
421         resultSet = statement.executeQuery(query);
422 
423         //lets get some column info and stuff
424         ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
425 
426         int columnCount = resultSetMetaData.getColumnCount();
427         for (int i = 0; i < columnCount; i++) {
428           this.columnNames.add(resultSetMetaData.getColumnLabel(i + 1));
429           this.columnNameToIndex.put(resultSetMetaData.getColumnLabel(i + 1).toUpperCase(), i);
430           this.columnTypes.add(resultSetMetaData.getColumnType(i + 1));
431         }
432 
433         while (resultSet.next()) {
434           Row row = new Row();
435           Object[] rowData = new Object[columnCount];
436           row.setRowData(rowData);
437           this.data.add(row);
438           //at this point, assume everything is either a string or a timestamp
439           for (int i = 0; i < columnCount; i++) {
440             if (this.columnTypes.get(i) == Types.TIMESTAMP) {
441               rowData[i] = resultSet.getTimestamp(i + 1);
442             } else {
443               rowData[i] = resultSet.getString(i + 1);
444             }
445           }
446 
447         }
448       } finally {
449         //this is important so no one sneaks some delete statement in there
450         GrouperUtil.rollbackQuietly(connection);
451       }
452     } catch (SQLException se) {
453       throw new RuntimeException("Problem with query: " + query + ",  on db: "
454           + grouperLoaderDb, se);
455     } finally {
456       GrouperUtil.closeQuietly(resultSet);
457       GrouperUtil.closeQuietly(statement);
458       GrouperUtil.closeQuietly(connection);
459     }
460     this.convertToSubjectIdIfNeeded(jobName, hib3GrouperLoaderLog, null);
461 
462   }

469   private String subjectIdCol(String subjectIdType) {
470     String subjectIdCol = "SUBJECT_ID";
471 
472     if (!StringUtils.isBlank(subjectIdType)) {
473 
474       if (StringUtils.equalsIgnoreCase(subjectIdType, "SUBJECT_ID")
475           || StringUtils.equalsIgnoreCase(subjectIdType, "subjectId")) {

483       } else {
484         throw new RuntimeException("Not expecting subjectIdType: '" + subjectIdType
485             + "', should be subjectId, subjectIdentifier, or subjectIdOrIdentifier");
486       }
487     }
488     return subjectIdCol;

1708         newMap.put(key, row);
