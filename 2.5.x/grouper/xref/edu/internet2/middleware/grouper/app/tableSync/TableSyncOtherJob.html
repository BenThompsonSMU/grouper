18   *
19   */
20  @DisallowConcurrentExecution
21  public class TableSyncOtherJob extends OtherJobBase {
22  
23    /**
24     * 
25     */
26    public TableSyncOtherJob() {
27    }
28  
29    /**
30     * @see edu.internet2.middleware.grouper.app.loader.OtherJobBase#run(edu.internet2.middleware.grouper.app.loader.OtherJobBase.OtherJobInput)
31     */
32    @Override
33    public OtherJobOutput run(OtherJobInput otherJobInput) {
34      
35      String jobName = otherJobInput.getJobName();
36      
37      // grouperClientTableSyncConfigKey = OTHER_JOB_personSourceSync
38      jobName = jobName.substring("OTHER_JOB_".length(), jobName.length());
39      final String grouperClientConfigKeyName = "otherJob." + jobName + ".grouperClientTableSyncConfigKey";
40      
41      String grouperClientTableSyncConfigKey = GrouperLoaderConfig.retrieveConfig().propertyValueString(grouperClientConfigKeyName);
42      
43      if (StringUtils.isBlank(grouperClientTableSyncConfigKey)) {
44        grouperClientTableSyncConfigKey = jobName;
45      }
46      
47      final String grouperClientTableFromName = "grouperClient.syncTable." + grouperClientTableSyncConfigKey + ".tableFrom";
48      String tableFrom = GrouperClientConfig.retrieveConfig().propertyValueString(grouperClientTableFromName);
49      if (StringUtils.isBlank(tableFrom)) {
50        throw new RuntimeException("Cant find grouperClient config for " + grouperClientTableSyncConfigKey 
51            + " tableSync, checking property: " + grouperClientTableFromName);
52      }
53      
54      GcTableSync gcTableSync = new GcTableSync();
55      gcTableSync.setKey(grouperClientTableSyncConfigKey);
56      
57      String tableSyncType = GrouperLoaderConfig.retrieveConfig().propertyValueStringRequired("otherJob." + jobName + ".syncType");
58      GcTableSyncOutput gcTableSyncOutput = null;
59      if (StringUtils.equalsIgnoreCase(tableSyncType, "full")) {
60        gcTableSyncOutput = gcTableSync.fullSync();
61      } else if (StringUtils.equalsIgnoreCase(tableSyncType, "incremental")) {
62        gcTableSyncOutput = gcTableSync.incrementalSync();
63      } else {
64        throw new RuntimeException("Wrong tableSync type: '" + tableSyncType + "' for job: '" + jobName + "'");
65      }
66      
67      otherJobInput.getHib3GrouperLoaderLog().setDeleteCount(gcTableSyncOutput.getDelete());
68      otherJobInput.getHib3GrouperLoaderLog().setInsertCount(gcTableSyncOutput.getInsert());
69      otherJobInput.getHib3GrouperLoaderLog().setUpdateCount(gcTableSyncOutput.getUpdate());
70      otherJobInput.getHib3GrouperLoaderLog().setTotalCount(gcTableSyncOutput.getTotal());
71      otherJobInput.getHib3GrouperLoaderLog().setJobMessage(gcTableSyncOutput.getMessage());
72      
73      return null;
74    }
