43    @Override
44    public OtherJobOutput run(OtherJobInput otherJobInput) {
45      
46      GrouperSession.startRootSession();
47      
48      updateMetadataOnDirectStemsChildren();
49      updateMetadataOnIndirectGrouperObjects();
50      
51      return null;
52    }

65      hib3GrouperLoaderLog.setJobName(jobName);
66      hib3GrouperLoaderLog.setJobType(GrouperLoaderType.OTHER_JOB.name());

74      new GrouperObjectTypesJob().run(otherJobInput);

78    protected static List<Stem> updateMetadataOnDirectStemsChildren() {
79      
80      if (!GrouperObjectTypesSettings.objectTypesEnabled()) {
81        return new ArrayList<Stem>();
82      }
83      
84      List<Stem> stems = new ArrayList<Stem>(new StemFinder().assignAttributeCheckReadOnAttributeDef(false)
85          .assignNameOfAttributeDefName(objectTypesStemName()+":"+GROUPER_OBJECT_TYPE_DIRECT_ASSIGNMENT).addAttributeValuesOnAssignment("true").findStems());
86      
87      
88      for (Stem stem: stems) {
89        List<GrouperObjectTypesAttributeValue> attributeValues = GrouperObjectTypesConfiguration.getGrouperObjectTypesAttributeValues(stem);
90        
91        for (GrouperObjectTypesAttributeValue attributeValue: attributeValues) {
92          GrouperObjectTypesConfiguration.saveOrUpdateTypeAttributes(attributeValue, stem);        
93        }
94        
95      }
96      
97      return stems;
98      
99    }
100   
101   
102   protected static void updateMetadataOnIndirectGrouperObjects() {
103     
104     if (!GrouperObjectTypesSettings.objectTypesEnabled()) {
105       return;
106     }
107     
108     Set<GrouperObject> indirectGrouperObjects = new HashSet<GrouperObject>();
109     
110     List<Stem> stems = new ArrayList<Stem>(new StemFinder().assignAttributeCheckReadOnAttributeDef(false)
111         .assignNameOfAttributeDefName(objectTypesStemName()+":"+GROUPER_OBJECT_TYPE_DIRECT_ASSIGNMENT).addAttributeValuesOnAssignment("false").findStems());
112     
113     List<Group> groups = new ArrayList<Group>(new GroupFinder().assignAttributeCheckReadOnAttributeDef(false)
114         .assignNameOfAttributeDefName(objectTypesStemName()+":"+GROUPER_OBJECT_TYPE_DIRECT_ASSIGNMENT).addAttributeValuesOnAssignment("false").findGroups());
115     
116     indirectGrouperObjects.addAll(stems);
117     indirectGrouperObjects.addAll(groups);
118     
119     for (GrouperObject grouperObject: indirectGrouperObjects) {
120      GrouperObjectTypesConfiguration.copyConfigFromParent(grouperObject);
121     }
122     
123   }
