63    /**
64     * run report
65     * @param reportConfigBean
66     * @param reportInstance
67     * @param owner
68     */
69    public static void runReport(GrouperReportConfigurationBean reportConfigBean,
70        GrouperReportInstance reportInstance, GrouperObject owner) {
71      
72      long startTime = System.currentTimeMillis();
73      File fileToDelete = null;
74      try {
75        // get report data
76        GrouperReportData grouperReportData = reportConfigBean.getReportConfigType().retrieveReportDataByConfig(reportConfigBean);
77    
78        reportInstance.setGrouperReportConfigurationBean(reportConfigBean);
79        reportInstance.setReportInstanceRows(Long.valueOf(grouperReportData.getData().size()));
80    
81        // now the file is in the report instance
82        reportConfigBean.getReportConfigFormat().formatReport(grouperReportData, reportInstance);
83          
84        long endTime = System.currentTimeMillis();
85        reportInstance.setReportElapsedMillis(endTime - startTime);
86    
87        String randomEncryptionKey = RandomStringUtils.random(16, true, true);
88        
89        reportInstance.setReportInstanceEncryptionKey(Morph.encrypt(randomEncryptionKey));
90  
91        fileToDelete = reportInstance.getReportFileUnencrypted();
92        
93        String reportDestination = GrouperConfig.retrieveConfig().propertyValueString("reporting.storage.option");
94        if (StringUtils.isBlank(reportDestination)) {
95          throw new RuntimeException("reporting.storage.option cannot be blank. Valid values are S3 and fileSystem");
96        }
97        
98        if (!reportDestination.equals("S3") && !reportDestination.equals("fileSystem")) {
99          throw new RuntimeException("reporting.storage.option is not valid. Use S3 or fileSystem");
100       }
101         
102       if (reportDestination.equals("S3")) {
103         String s3Url = uploadFileToS3(reportInstance.getReportFileUnencrypted(), randomEncryptionKey);
104         reportInstance.setReportInstanceFilePointer(s3Url);
105       } else {
106         String reportOutputDirectory = GrouperConfig.retrieveConfig().propertyValueString("reporting.file.system.path");
107         if (StringUtils.isBlank(reportOutputDirectory)) {          
108           throw new RuntimeException("reporting.file.system.path cannot be blank");
109         }
110         String filePath = saveFileToFileSystem(reportInstance.getReportFileUnencrypted(), randomEncryptionKey, reportOutputDirectory);
111         reportInstance.setReportInstanceFilePointer(filePath);
112       }
113       
114       reportInstance.setReportInstanceFileName(reportInstance.getReportFileUnencrypted().getName());
115       reportInstance.setReportInstanceStatus(GrouperReportInstance.STATUS_SUCCESS);
116       
117     } catch(Exception e) {
118       LOG.error("Error occurred generating report for config name "+reportConfigBean.getReportConfigName(), e);
119       reportInstance.setReportInstanceStatus(GrouperReportInstance.STATUS_ERROR);
120     } finally {
121       if (fileToDelete != null && fileToDelete.exists()) {        
122         FileUtils.deleteQuietly(fileToDelete);
123       }
124     }
125     
126     try {
127       GrouperReportInstanceService.saveReportInstanceAttributes(reportInstance, owner);
128     } catch(Exception e) {
129       LOG.error("Error saving report instance. Config name is "+reportConfigBean.getReportConfigName());
130       return;
131     }
132       
133     if (reportInstance.getReportInstanceStatus().equals(GrouperReportInstance.STATUS_SUCCESS)) {
134       try {
135         //now the attribute assign id on report instance is populated and we can use it to generate link in the email
136         sendReportLinkViaEmail(reportInstance);
137       } catch(Exception e) {
138         LOG.error("Error sending report email. Config name is "+reportConfigBean.getReportConfigName());
139         return;
140       }
141       
142       // now the email attributes are populated on the report instance, let's save them as well.
143       try {
144         GrouperReportInstanceService.saveReportInstanceAttributes(reportInstance, owner);
145       } catch(Exception e) {
146         LOG.error("Error saving report instance. Config name is "+reportConfigBean.getReportConfigName());
147         return;
148       }
149     }
150     
151   }

153   /**
154    * send report link in email
155    * @param reportInstance
156    */
157   private static void sendReportLinkViaEmail(GrouperReportInstance reportInstance) {
158     GrouperReportConfigurationBean configBean = reportInstance.getGrouperReportConfigurationBean();
159     
160     if (!configBean.isReportConfigSendEmail()) {
161       LOG.info("Config send email is set to false. not going to send any emails");
162       return;
163     }
164     
165     String uiUrl = GrouperConfig.getGrouperUiUrl(false);
166     
167     if (StringUtils.isBlank(uiUrl)) {
168       LOG.error("grouper.properties grouper.ui.url is blank/null. Please fix that first. No emails have been sent.");
169       return;
170     }
171     
172     boolean sendToViewers = configBean.isReportConfigSendEmailToViewers();
173     
174     Group emailGroup = null;
175     
176     if (sendToViewers) {
177       String groupId = configBean.getReportConfigViewersGroupId();
178       emailGroup = GroupFinder.findByUuid(GrouperSession.startRootSession(), groupId, false);
179     } else {
180       emailGroup = GroupFinder.findByUuid(GrouperSession.startRootSession(), configBean.getReportConfigSendEmailToGroupId(), false);
181     }
182     
183     if (emailGroup == null) {
184       LOG.error("group to send email to for config: "+configBean.getReportConfigName()+ " is null. not sending any emails");
185       return;
186     }
187     
188     String templateSubject = configBean.getReportConfigEmailSubject();
189     if (StringUtils.isBlank(templateSubject)) {
190       templateSubject = GrouperConfig.retrieveConfig().propertyValueString("reporting.email.subject");
191     }
192     
193     if (StringUtils.isBlank(templateSubject)) {
194       templateSubject = "Report $$reportConfigName$$ generated";
195     }
196     
197     String subject = StringUtils.replace(templateSubject, "$$reportConfigName$$", configBean.getReportConfigName());
198     
199     String templateBody = configBean.getReportConfigEmailBody();
200     if (StringUtils.isBlank(templateBody)) {
201       templateBody = GrouperConfig.retrieveConfig().propertyValueString("reporting.email.body");
202     }
203     if (StringUtils.isBlank(templateBody)) {
204       templateBody = "Hello $$subjectName$$, \n\n Report $$reportConfigName$$ has been generated. Download the report: $$reportLink$$ \n\n Thanks";
205     }
206     
207     List<String> emailSuccessSubjects = new ArrayList<String>();
208     List<String> emailFailureSubjects = new ArrayList<String>();
209     
210     Set<Member> members = emailGroup.getMembers();
211     for (Member member: members) {
212       if (StringUtils.equals(member.getSubject().getType().getName(), SubjectTypeEnum.PERSON.getName())) {
213         String emailAddress = GrouperEmailUtils.getEmail(member.getSubject());
214         if (StringUtils.isBlank(emailAddress)) {
215           LOG.info("For subject: "+member.getSubjectId()+" no email address found.");
216           emailFailureSubjects.add(member.getSubject().getSourceId()+"::::"+member.getSubjectId());
217           continue;
218         }
219 
220         try {
221           String emailBody = buildEmailBody(member.getSubject(), templateBody, reportInstance, uiUrl, configBean);
222           new GrouperEmail().setBody(emailBody).setSubject(subject).setTo(emailAddress).send();
223           emailSuccessSubjects.add(member.getSubject().getSourceId()+"::::"+member.getSubjectId());
224         } catch (Exception e) {
225           emailFailureSubjects.add(member.getSubject().getSourceId()+"::::"+member.getSubjectId());
226           LOG.error("Error sending report email to "+emailAddress+ " Config name is "+configBean.getReportConfigName());
227         }
228       }
229     }
230     
231     reportInstance.setReportInstanceEmailToSubjects(StringUtils.join(emailSuccessSubjects, ","));
232     reportInstance.setReportInstanceEmailToSubjectsError(StringUtils.join(emailFailureSubjects, ","));
233     
234   }

236   /**
237    * build personalized email content for recipient
238    * @param recipient
239    * @param templateBody
240    * @param reportInstance
241    * @param uiUrl
242    * @param configBean
243    * @return
244    */
245   private static String buildEmailBody(Subject recipient, String templateBody, 
246       GrouperReportInstance reportInstance, String uiUrl, GrouperReportConfigurationBean configBean) {
247     
248     String link = null;
249     
250     AttributeAssign attributeAssign = AttributeAssignFinder.findById(configBean.getAttributeAssignmentMarkerId(), true);
251     Stem stem = attributeAssign.getOwnerStem();
252     if (stem != null) {
253       link = "grouperUi/app/UiV2Main.index?operation=UiV2GrouperReport.viewReportInstanceDetailsForFolder&attributeAssignId="+reportInstance.getAttributeAssignId();
254       link = link + "&stemId="+stem.getId();
255     } else {
256       Group group = attributeAssign.getOwnerGroup();
257       link = "grouperUi/app/UiV2Main.index?operation=UiV2GrouperReport.viewReportInstanceDetailsForGroup&attributeAssignId="+reportInstance.getAttributeAssignId();
258       link = link + "&groupId="+group.getId();
259     }
260     
261     GrouperReportConfigurationBean reportConfigBean = reportInstance.getGrouperReportConfigurationBean();
262     
263     String emailBody = StringUtils.replace(templateBody, "$$reportConfigName$$", reportConfigBean.getReportConfigName());
264     
265     emailBody = StringUtils.replace(emailBody, "$$reportConfigDescription$$", reportConfigBean.getReportConfigDescription());
266     
267     emailBody = StringUtils.replace(emailBody, "$$reportLink$$", uiUrl+link);
268     
269     emailBody = StringUtils.replace(emailBody, "$$subjectName$$", recipient.getName());
270     
271     return emailBody;
272   }

286   /**
287    * @param reportInstance
288    * @return report content from s3 for a given report instance
289    */
290   private static String getReportContentFromS3(GrouperReportInstance reportInstance) {
291     
292     String accessKey = GrouperConfig.retrieveConfig().propertyValueString("reporting.s3.access.key");
293     String secretKey = GrouperConfig.retrieveConfig().propertyValueString("reporting.s3.secret.key");
294     
295     String encryptionKey = Morph.decrypt(reportInstance.getReportInstanceEncryptionKey());
296     SecretKey encryptionKeySecret = new SecretKeySpec(encryptionKey.getBytes(), "AES");
297     EncryptionMaterials encryptionMaterials = new EncryptionMaterials(encryptionKeySecret);
298     
299     try {
300       
301       AWSCredentials credentials = new BasicAWSCredentials(accessKey, secretKey);
302       AWSCredentialsProvider credentialsProvider = new AWSStaticCredentialsProvider(credentials);
303       
304       AmazonS3URI s3Uri = new AmazonS3URI(reportInstance.getReportInstanceFilePointer());
305       AmazonS3 s3Client = AmazonS3EncryptionClientBuilder.standard()
306               .withRegion(s3Uri.getRegion())
307               .withCredentials(credentialsProvider)
308               .withEncryptionMaterials(new StaticEncryptionMaterialsProvider(encryptionMaterials))
309               .build();
310       
311       // Download and decrypt the object.
312       S3Object downloadedObject = s3Client.getObject(s3Uri.getBucket(), s3Uri.getKey());     
313       byte[] decrypted = com.amazonaws.util.IOUtils.toByteArray(downloadedObject.getObjectContent());
314       return new String(decrypted);
315     } catch (Exception e) {
316       throw new RuntimeException("Error getting report content from S3", e);
317     }
318   }

320   /**
321    * @param reportInstance
322    * @return report content from file system for a given report instance
323    */
324   private static String getReportContentFromFileSystem(GrouperReportInstance reportInstance) {
325     
326     String encryptionKey = Morph.decrypt(reportInstance.getReportInstanceEncryptionKey());
327     SecretKey encryptionKeySecret = new SecretKeySpec(encryptionKey.getBytes(), "AES");
328     
329     Cipher cipher;
330     try {
331       cipher = Cipher.getInstance("AES");
332       cipher.init(Cipher.DECRYPT_MODE, encryptionKeySecret);
333       
334       FileInputStream inputStream = new FileInputStream(reportInstance.getReportInstanceFilePointer());
335       File file = new File(reportInstance.getReportInstanceFilePointer());
336       byte[] inputBytes = new byte[(int) file.length()];
337       inputStream.read(inputBytes);
338        
339       byte[] outputBytes = cipher.doFinal(inputBytes);
340       inputStream.close();
341       return new String(outputBytes);
342       
343     } catch (Exception e) {
344       throw new RuntimeException("Error in getting report content from file system", e);
345     }
346     
347   }

349   /**
350    * save report file to file system
351    * @param file
352    * @param encryptionKey
353    * @param reportOutputDirectory
354    * @return pointer to the file location
355    */
356   private static String saveFileToFileSystem(File file, String encryptionKey, String reportOutputDirectory) {
357     
358     Calendar calendar = new GregorianCalendar();
359     calendar.setTimeInMillis(System.currentTimeMillis());
360     
361     String baseDirectory = reportOutputDirectory.endsWith(File.separator) ? reportOutputDirectory: reportOutputDirectory + File.separator;
362     
363     String reportDirectoryPath = baseDirectory + "reports" + File.separator + calendar.get(Calendar.YEAR) + File.separator 
364         + StringUtils.leftPad(""+(calendar.get(Calendar.MONTH)+1), 2, '0') + File.separator + StringUtils.leftPad(""+calendar.get(Calendar.DAY_OF_MONTH), 2, '0')
365         + File.separator + GrouperUtil.uniqueId();
366     
367     SecretKey encryptionKeySecret = new SecretKeySpec(encryptionKey.getBytes(), "AES");
368     
369     try {
370       
371       Cipher cipher = Cipher.getInstance("AES");
372       cipher.init(Cipher.ENCRYPT_MODE, encryptionKeySecret);
373       
374       FileInputStream inputStream = new FileInputStream(file);
375       byte[] inputBytes = new byte[(int) file.length()];
376       inputStream.read(inputBytes);
377        
378       byte[] outputBytes = cipher.doFinal(inputBytes);
379       
380       GrouperUtil.mkdirs(new File(reportDirectoryPath));
381        
382       File outputFile = new File(reportDirectoryPath+File.separator+file.getName());
383       GrouperUtil.fileCreateNewFile(outputFile);
384        
385       FileOutputStream outputStream = new FileOutputStream(outputFile);
386       outputStream.write(outputBytes);
387        
388       inputStream.close();
389       outputStream.close();
390       
391     } catch (Exception e) {
392       throw new RuntimeException("Error in saving report to file system", e);
393     }
394     
395     return reportDirectoryPath+File.separator+file.getName();
396   }

444   /**
445    * upload given file to file system
446    * @param file
447    * @param encryptionKey
448    * @return s3 url of file 
449    */
450   private static String uploadFileToS3(File file, String encryptionKey) {
451     
452     String bucketName = GrouperConfig.retrieveConfig().propertyValueString("reporting.s3.bucket.name");
453     String region = GrouperConfig.retrieveConfig().propertyValueString("reporting.s3.region");
454     
455     String accessKey = GrouperConfig.retrieveConfig().propertyValueString("reporting.s3.access.key");
456     String secretKey = GrouperConfig.retrieveConfig().propertyValueString("reporting.s3.secret.key");
457     
458     String fileObjKeyName = file.getName();
459     
460     SecretKey encryptionKeySecret = new SecretKeySpec(encryptionKey.getBytes(), "AES");
461     EncryptionMaterials encryptionMaterials = new EncryptionMaterials(encryptionKeySecret);
462     
463     try {
464       
465       AWSCredentials credentials = new BasicAWSCredentials(accessKey, secretKey);
466       AWSCredentialsProvider credentialsProvider = new AWSStaticCredentialsProvider(credentials);
467       
468       AmazonS3 s3Client = AmazonS3EncryptionClientBuilder.standard()
469               .withRegion(region)
470               .withCredentials(credentialsProvider)
471               .withEncryptionMaterials(new StaticEncryptionMaterialsProvider(encryptionMaterials))
472               .build();
473       
474       // Upload a file as a new object with ContentType and title specified.
475       PutObjectRequest request = new PutObjectRequest(bucketName, fileObjKeyName, file);
476       ObjectMetadata metadata = new ObjectMetadata();
477       metadata.setContentType("plain/text");
478       request.setMetadata(metadata);
479       s3Client.putObject(request);
480       
481       return s3Client.getUrl(bucketName, fileObjKeyName).toString();
482     } catch(AmazonServiceException e) {
483       LOG.error("Error uploading report file to S3. file name: "+file.getName());
484       throw new RuntimeException("Error uploading file to S3");
485     } catch(SdkClientException e) {
486       LOG.error("Error uploading report file to S3. file name: "+file.getName());
487       throw new RuntimeException("Error uploading file to S3");
488     }
