147    * @param throwException 
148    * @param startTime
149    */
150   private static void storeLogInDb(Hib3GrouperLoaderLog hib3GrouploaderLog,
151       boolean throwException, long startTime) {
152     //store this safely
153     try {
154       
155       long endTime = System.currentTimeMillis();
156       hib3GrouploaderLog.setEndedTime(new Timestamp(endTime));
157       hib3GrouploaderLog.setMillis((int)(endTime-startTime));
158       
159       hib3GrouploaderLog.store();
160       
161     } catch (RuntimeException e) {
162       LOG.error("Problem storing final log", e);
163       //dont preempt an existing exception
164       if (throwException) {
165         throw e;
166       }
167     }
