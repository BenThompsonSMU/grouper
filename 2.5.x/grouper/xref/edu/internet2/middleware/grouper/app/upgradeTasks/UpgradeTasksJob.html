35  @DisallowConcurrentExecution

37    
38    /**
39     * logger 
40     */
41    private static final Log LOG = GrouperUtil.getLog(UpgradeTasksJob.class);
42    
43    /**
44     * attribute definition for upgrade tasks
45     */
46    public static final String UPGRADE_TASKS_DEF = "upgradeTasksDef";
47  
48    /**
49     * version
50     */
51    public static final String UPGRADE_TASKS_VERSION_ATTR = "upgradeTasksVersion";
52    
53    /**
54     * group holding metadata
55     */
56    public static final String UPGRADE_TASKS_METADATA_GROUP = "upgradeTasksMetadataGroup";
57    
58    /**
59     * run the daemon
60     * @param args
61     */
62    public static void main(String[] args) {
63      runDaemonStandalone();
64    }
65  
66    /**
67     * run standalone
68     */
69    public static void runDaemonStandalone() {
70      GrouperSession grouperSession = GrouperSession.startRootSession();
71      Hib3GrouperLoaderLog3GrouperLoaderLog.html#Hib3GrouperLoaderLog">Hib3GrouperLoaderLog hib3GrouperLoaderLog = new Hib3GrouperLoaderLog();
72      
73      hib3GrouperLoaderLog.setHost(GrouperUtil.hostname());
74      String jobName = "OTHER_JOB_upgradeTasks";
75  
76      hib3GrouperLoaderLog.setJobName(jobName);
77      hib3GrouperLoaderLog.setJobType(GrouperLoaderType.OTHER_JOB.name());
78      hib3GrouperLoaderLog.setStatus(GrouperLoaderStatus.STARTED.name());
79      hib3GrouperLoaderLog.store();
80      
81      OtherJobInput otherJobInput = new OtherJobInput();
82      otherJobInput.setJobName(jobName);
83      otherJobInput.setHib3GrouperLoaderLog(hib3GrouperLoaderLog);
84      otherJobInput.setGrouperSession(grouperSession);
85      new UpgradeTasksJob().run(otherJobInput);
86    }
87    
88    /**
89     * @see edu.internet2.middleware.grouper.app.loader.OtherJobBase#run(edu.internet2.middleware.grouper.app.loader.OtherJobBase.OtherJobInput)
90     */
91    @Override
92    public OtherJobOutput run(OtherJobInput otherJobInput) {
93      
94      String groupName = grouperUpgradeTasksStemName() + ":" + UpgradeTasksJob.UPGRADE_TASKS_METADATA_GROUP;
95      Group group = GroupFinder.findByName(GrouperSession.staticGrouperSession(), groupName, true);
96      String upgradeTasksVersionName = grouperUpgradeTasksStemName() + ":" + UpgradeTasksJob.UPGRADE_TASKS_VERSION_ATTR;
97      String versionString = group.getAttributeValueDelegate().retrieveValueString(upgradeTasksVersionName);
98      
99      int oldDBVersion = Integer.parseInt(versionString);    
100     int newDBVersion = UpgradeTasks.currentVersion();
101     
102     for (int version = oldDBVersion + 1; version <= newDBVersion; version++) {
103       String enumName = "V" + version;
104       UpgradeTasksInterface task = Enum.valueOf(UpgradeTasks.class, enumName);
105       task.updateVersionFromPrevious();
106       group.getAttributeValueDelegate().assignValue(upgradeTasksVersionName, "" + version);
107       
108       LOG.info("Upgraded to version " + enumName);
109     }
110     
111     otherJobInput.getHib3GrouperLoaderLog().setJobMessage("Finished running job, previousVersion=" + oldDBVersion + ", currentVersion=" + newDBVersion);
112     otherJobInput.getHib3GrouperLoaderLog().store();
113 
114     LOG.info("UpgradeTasksJob finished successfully.");
115     return null;
116   }
117   
118   /**
119    * stem name for upgrade tasks
120    * @return stem name
121    */
122   public static String grouperUpgradeTasksStemName() {
123     return GrouperCheckConfig.attributeRootStemName() + ":upgradeTasks";
124   }
