54     * @since   @HEAD@
55     */
56    public void create(RegistrySubject _subj)
57      throws  GrouperDAOException {
58      HibernateSession.byObjectStatic().save(_subj);
59      
60      for (String key : GrouperUtil.nonNull(_subj.getAttributes()).keySet()) {
61        RegistrySubjectAttributete.html#RegistrySubjectAttribute">RegistrySubjectAttribute registrySubjectAttribute = new RegistrySubjectAttribute();
62        registrySubjectAttribute.setName(key);
63        String value = _subj.getAttributeValue(key);
64        registrySubjectAttribute.setValue(value);
65        registrySubjectAttribute.setSearchValue(value == null ? null : value.toLowerCase());
66        registrySubjectAttribute.setSubjectId(_subj.getId());
67        GrouperDAOFactory.getFactory().getRegistrySubjectAttribute().create(registrySubjectAttribute);
68      }
69      
70    }

74     */

78    }

91    /**
92     * @since
93     */
94    public RegistrySubject find(String id, boolean exceptionIfNotFound) 
95      throws  GrouperDAOException,
96              SubjectNotFoundException {
97      RegistrySubject subj = HibernateSession.byHqlStatic()
98        .createQuery(
99          "from RegistrySubject as rs where " 
100         + "     rs.id   = :id  ")
101       .setCacheable(false) 
102       .setString( "id",   id   )
103       .uniqueResult(RegistrySubject.class);
104     if (subj == null && exceptionIfNotFound) {
105       throw new SubjectNotFoundException("subject not found"); 
106     }
107     return subj;
108   }
109 
110   /**
111    * @since
112    */
113   public RegistrySubject find(String id, String type, boolean exceptionIfNotFound) 
114     throws  GrouperDAOException,
115             SubjectNotFoundException {
116     RegistrySubject subj = HibernateSession.byHqlStatic()
117       .createQuery(
118         "from RegistrySubject as rs where " 
119         + "     rs.id   = :id    "
120         + " and rs.typeString = :type  ")
121       .setCacheable(false) 
122       .setString( "id",   id   )
123       .setString( "type", type )
124       .uniqueResult(RegistrySubject.class);
125     if (subj == null && exceptionIfNotFound) {
126       throw new SubjectNotFoundException("subject not found"); 
127     }
128     return subj;
129   }

134   {
135     hibernateSession.byHql().createQuery("delete from RegistrySubject").executeUpdate();
136   }
