20  package edu.internet2.middleware.grouper.internal.dao.hib3;
21  
22  import java.util.Set;
23  
24  import org.hibernate.HibernateException;
25  
26  import edu.internet2.middleware.grouper.RegistrySubjectAttribute;
27  import edu.internet2.middleware.grouper.hibernate.HibernateSession;
28  import edu.internet2.middleware.grouper.internal.dao.RegistrySubjectAttributeDAO;
29  
30  
31  /**
32   * marker class for hbm loading
33   */
34  public class Hib3RegistrySubjectAttributeDAO extends Hib3DAO implements RegistrySubjectAttributeDAO {
35  
36    /**
37     * @see edu.internet2.middleware.grouper.internal.dao.RegistrySubjectAttributeDAO#create(edu.internet2.middleware.grouper.RegistrySubjectAttribute)
38     */
39    public void create(RegistrySubjectAttribute _subjAttribute) {
40      HibernateSession.byObjectStatic().save(_subjAttribute);
41    }
42  
43    /**
44     * @see edu.internet2.middleware.grouper.internal.dao.RegistrySubjectAttributeDAO#delete(edu.internet2.middleware.grouper.RegistrySubjectAttribute)
45     */
46    public void delete(RegistrySubjectAttribute _subjAttr) {
47      HibernateSession.byObjectStatic().delete(_subjAttr);
48    }
49  
50    /**
51     * @see edu.internet2.middleware.grouper.internal.dao.RegistrySubjectAttributeDAO#find(java.lang.String, java.lang.String, boolean)
52     */
53    public RegistrySubjectAttribute find(String subjectId, String attributeName, boolean exceptionIfNotFound) {
54      RegistrySubjectAttribute subjAttribute = HibernateSession.byHqlStatic()
55          .createQuery(
56            "from RegistrySubjectAttribute as rsa where " 
57            + "     rsa.subjectId   = :subjectId and rsa.name = :attributeName   ")
58          .setCacheable(false) 
59          .setString( "subjectId",   subjectId   )
60          .setString( "attributeName",   attributeName   )
61          .uniqueResult(RegistrySubjectAttribute.class);
62      if (subjAttribute == null && exceptionIfNotFound) {
63        throw new RuntimeException("subject attribute not found"); 
64      }
65      return subjAttribute;
66    }
67  
68    /**
69     * @see edu.internet2.middleware.grouper.internal.dao.RegistrySubjectAttributeDAO#findByRegistrySubjectId(java.lang.String)
70     */
71    public Set<RegistrySubjectAttribute> findByRegistrySubjectId(String subjectId) {
72      Set<RegistrySubjectAttribute> subjAttributes = HibernateSession.byHqlStatic()
73          .createQuery(
74            "from RegistrySubjectAttribute as rsa where " 
75            + "     rsa.subjectId   = :subjectId    ")
76          .setCacheable(false) 
77          .setString( "subjectId",   subjectId   ).listSet(RegistrySubjectAttribute.class);
78      
79      return subjAttributes;
80    }
81  
82    // @since   @HEAD@
83    protected static void reset(HibernateSession hibernateSession) 
84      throws  HibernateException
85    {
86      hibernateSession.byHql().createQuery("delete from RegistrySubjectAttribute").executeUpdate();
87    }
88  
89    /**
90     * @see edu.internet2.middleware.grouper.internal.dao.RegistrySubjectAttributeDAO#update(edu.internet2.middleware.grouper.RegistrySubjectAttribute)
91     */
92    public void update(RegistrySubjectAttribute _subjAttr) {
93      HibernateSession.byObjectStatic().update(_subjAttr);
94    } 
95  
96  
97  }
