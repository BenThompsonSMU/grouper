18  import java.util.List;
19  import java.util.Set;
20  
21  import org.apache.commons.logging.Log;
22  
23  import edu.internet2.middleware.grouper.hibernate.HibernateSession;
24  import edu.internet2.middleware.grouper.internal.dao.MessageDAO;
25  import edu.internet2.middleware.grouper.internal.dao.QueryOptions;
26  import edu.internet2.middleware.grouper.messaging.GrouperMessageHibernate;
27  import edu.internet2.middleware.grouper.util.GrouperUtil;
28  
29  /**
30   * Data Access Object for messaging index
31   * @author  mchyzer
32   * @version $Id$
33   */
34  public class Hib3MessageDAO extends Hib3DAO implements MessageDAO {
35    
36    /**
37     * 
38     */
39    @SuppressWarnings("unused")
40    private static final String KLASS = Hib3MessageDAO.class.getName();
41  
42    /**
43     * @param hibernateSession
44     */
45    static void reset(HibernateSession hibernateSession) {
46      
47      hibernateSession.byHql().createQuery("delete from GrouperMessageHibernate").executeUpdate();
48      
49    }
50  
51    /**
52     * @see edu.internet2.middleware.grouper.internal.dao.MessageDAO#findById(java.lang.String, boolean)
53     */
54    public GrouperMessageHibernate findById(String id, boolean exceptionIfNotFound) {
55      GrouperMessageHibernate message = HibernateSession.byHqlStatic()
56        .createQuery("from GrouperMessageHibernate where id = :theId")
57        .setString("theId", id).uniqueResult(GrouperMessageHibernate.class);
58      
59      if (message == null && exceptionIfNotFound) {
60        throw new RuntimeException("Cant find message by id: " + id);
61      }
62      
63      return message;
64    }
65  
66  
67    /**
68     * @see edu.internet2.middleware.grouper.internal.dao.MessageDAO#findByFromMemberId(java.lang.String)
69     */
70    public Set<GrouperMessageHibernate> findByFromMemberId(String fromMemberId) {
71      return HibernateSession.byHqlStatic()
72        .createQuery("from GrouperMessageHibernate where fromMemberId = :theFromMemberId")
73        .setString("theFromMemberId", fromMemberId).listSet(GrouperMessageHibernate.class);
74      
75    }
76  
77    /**
78     * @see edu.internet2.middleware.grouper.internal.dao.MessageDAO#findByQueue(String, int)
79     */
80    public List<GrouperMessageHibernate> findByQueue(String queue, int pageSize) {
81      
82      List<GrouperMessageHibernate> messages = HibernateSession.byHqlStatic()
83        .createQuery("from GrouperMessageHibernate gmh "
84            + " where gmh.queueName = :theQueueName"
85            + " and (gmh.state = 'IN_QUEUE' or (gmh.state = 'GET_ATTEMPTED'"
86            + " and gmh.attemptTimeExpiresMillis < :attemptTimeExpired )) "
87            + " order by gmh.sentTimeMicros, gmh.id ")
88        .setString("theQueueName", queue).setLong("attemptTimeExpired", System.currentTimeMillis())
89        .options(QueryOptions.create(null, null, 1, pageSize))
90        .list(GrouperMessageHibernate.class);
91      return messages;
92    }
93  
94    /**
95     * @see edu.internet2.middleware.grouper.internal.dao.MessageDAO#saveOrUpdate(GrouperMessageHibernate)
96     */
97    public void saveOrUpdate(GrouperMessageHibernate grouperMessageHibernate) {
98      HibernateSession.byObjectStatic().saveOrUpdate(grouperMessageHibernate);
99    }
100   
101   /**
102    * @see edu.internet2.middleware.grouper.internal.dao.MessageDAO#delete(GrouperMessageHibernate)
103    */
104   public void delete(final GrouperMessageHibernate grouperMessageHibernate) {
105     HibernateSession.byObjectStatic().delete(grouperMessageHibernate);
106   }
107 
108   /** logger */
109   @SuppressWarnings("unused")
110   private static final Log LOG = GrouperUtil.getLog(Hib3MessageDAO.class);

113 }
