76    /**
77     * 
78     * @see edu.internet2.middleware.grouper.internal.dao.AttributeAssignActionDAO#findById(java.lang.String, boolean)
79     */
80    public AttributeAssignAction findById(String id, boolean exceptionIfNotFound) {
81      
82      Set<AttributeAssignAction> attributeAssignActions = attributeAssignActionFlashCacheRetrieveById(id, null);
83      if (GrouperUtil.length(attributeAssignActions) > 0) {
84        return attributeAssignActions.iterator().next();
85      }      
86      
87      AttributeAssignAction attributeAssignAction = HibernateSession.byHqlStatic().createQuery(
88          "from AttributeAssignAction where id = :theId")
89        .setString("theId", id).uniqueResult(AttributeAssignAction.class);
90      if (attributeAssignAction == null && exceptionIfNotFound) {
91        throw new AttributeAssignActionNotFoundException("Cant find attribute assign action by id: " + id);
92      }
93  
94      if (attributeAssignAction != null) {
95        attributeAssignActionFlashCacheAddIfSupposedTo(attributeAssignAction);
96      }
97      
98      return attributeAssignAction;
99    }

133   /**
134    * @see edu.internet2.middleware.grouper.internal.dao.AttributeAssignActionDAO#findByAttributeDefId(java.lang.String)
135    */
136   public Set<AttributeAssignAction> findByAttributeDefId(String attributeDefId) {
137     
138     Set<AttributeAssignAction> attributeAssignActions = attributeAssignActionFlashCacheRetrieveByAttributeDefId(attributeDefId, null);
139     if (attributeAssignActions != null) {
140       return attributeAssignActions;
141     }      
142     
143     attributeAssignActions = HibernateSession.byHqlStatic().createQuery(
144       "from AttributeAssignAction where attributeDefId = :theAttributeDefId order by nameDb")
145       .setString("theAttributeDefId", attributeDefId)
146       .setCacheable(true)
147       .setCacheRegion(KLASS + ".FindByAttributeDefId")
148       .listSet(AttributeAssignAction.class);
149 
150     if (GrouperUtil.length(attributeAssignActions) > 0) {
151       attributeAssignActionFlashCacheAddIfSupposedToByAttributeDef(attributeDefId, attributeAssignActions);
152     }
153 
154     return attributeAssignActions;
155     
156   }

158   /**
159    * @see edu.internet2.middleware.grouper.internal.dao.AttributeAssignActionDAO#findByUuidOrKey(java.lang.String, java.lang.String, java.lang.String, boolean)
160    */
161   public AttributeAssignAction findByUuidOrKey(String id, String attributeDefId,
162       String name, boolean exceptionIfNull) {
163     
164     if (!StringUtils.isBlank(id)) {
165       Set<AttributeAssignAction> attributeAssignActions = attributeAssignActionFlashCacheRetrieveById(id, null);
166       if (GrouperUtil.length(attributeAssignActions) > 0) {
167         return attributeAssignActions.iterator().next();
168       }
169     }
170     
171     if (!StringUtils.isBlank(attributeDefId) && !StringUtils.isBlank(name)) {
172       Set<AttributeAssignAction> attributeAssignActions = attributeAssignActionFlashCacheRetrieveById(attributeDefId, null);
173       
174       for (AttributeAssignAction attributeAssignAction : GrouperUtil.nonNull(attributeAssignActions)) {
175         if (StringUtils.equals(name, attributeAssignAction.getName())) {
176           return attributeAssignAction;
177         }
178       }
179     }
180     
181     try {
182       AttributeAssignAction attributeAssignAction = HibernateSession.byHqlStatic()
183         .createQuery("from AttributeAssignAction as theAttributeAssignAction " +
184         		"where theAttributeAssignAction.id = :theId " +
185         		"or (theAttributeAssignAction.attributeDefId = :theAttributeDefId " +
186         		"and theAttributeAssignAction.nameDb = :theNameDb)")
187         .setCacheable(true)
188         .setCacheRegion(KLASS + ".FindByUuidOrName")
189         .setString("theId", id)
190         .setString("theAttributeDefId", attributeDefId)
191         .setString("theNameDb", name)
192         .uniqueResult(AttributeAssignAction.class);
193       if (attributeAssignAction == null && exceptionIfNull) {
194         throw new RuntimeException("Can't find attributeAssignAction by id: '" 
195             + id + "' or attributeDefId: " + attributeDefId + ", name: '" + name + "'");
196       }
197       if (attributeAssignAction != null) {
198         attributeAssignActionFlashCacheAddIfSupposedTo(attributeAssignAction);
199       }
200 
201       return attributeAssignAction;
202     }
203     catch (GrouperDAOException e) {
204       String error = "Problem find attributeAssignAction by id: '" 
205             + id + "' or attributeDefId: " + attributeDefId 
206             + ", name: '" + name + "', " + e.getMessage();
207       throw new GrouperDAOException( error, e );
208     }
209 
210   }

226         "contextId = :theContextId, " +
227         "createdOnDb = :theCreatedOnDb, " +
228         "lastUpdatedDb = :theLastUpdatedDb " +
229         "where id = :theId")
230         .setLong("theHibernateVersionNumber", attributeAssignAction.getHibernateVersionNumber())

232         .setLong("theLastUpdatedDb", attributeAssignAction.getLastUpdatedDb())
233         .setString("theContextId", attributeAssignAction.getContextId())
234         .setString("theId", attributeAssignAction.getId()).executeUpdate();
235     
236     attributeAssignActionFlashCacheAddIfSupposedTo(attributeAssignAction);

240   /**
241    * get a group fom flash cache
242    * @param id
243    * @param queryOptions
244    * @return the assignments
245    */
246   private static Set<AttributeAssignAction> attributeAssignActionFlashCacheRetrieveById(Object id, QueryOptions queryOptions) {
247     if (attributeAssignActionFlashCacheableById(id, queryOptions)) {
248       MultiKey attributeAssignActionFlashKey = attributeAssignActionFlashCacheMultikeyId(id);
249       //see if its already in the cache
250       Set<AttributeAssignAction> attributeAssignActions = attributeAssignActionFlashCache.get(attributeAssignActionFlashKey);
251       if (attributeAssignActions != null) {
252         if (LOG.isDebugEnabled()) {
253           LOG.debug("retrieving from attributeAssignAction flash cache by id: " + id);
254         }
255         return attributeAssignActions;
256       }
257     }
258     return null;
259   }

261   /**
262    * see if this is cacheable
263    * @param id
264    * @param queryOptions 
265    * @return if cacheable
266    */
267   private static boolean attributeAssignActionFlashCacheableById(Object id, QueryOptions queryOptions) {
268   
269     if (!GrouperConfig.retrieveConfig().propertyValueBoolean(GROUPER_FLASHCACHE_ATTRIBUTE_ASSIGN_ACTION, true)) {
270       return false;
271     }
272   
273     if (id == null || ((id instanceof String) && StringUtils.isBlank((String)id))) {
274       return false;
275     }
276   
277     if (!HibUtils.secondLevelCaching(true, queryOptions)) {
278       return false;
279     }
280     
281     return true;
282   }

307   /**
308    * add group to cache if not null
309    * @param attributeAssignAction
310    */
311   private static void attributeAssignActionFlashCacheAddIfSupposedTo(AttributeAssignAction attributeAssignAction) {
312     if (!GrouperConfig.retrieveConfig().propertyValueBoolean(GROUPER_FLASHCACHE_ATTRIBUTE_ASSIGN_ACTION, true)) {
313       return;
314     }
315     if (attributeAssignAction != null) {
316       attributeAssignActionFlashCache.put(attributeAssignActionFlashCacheMultikeyId(attributeAssignAction.getId()), GrouperUtil.toSet(attributeAssignAction));
317     }
318   }

320   /**
321    * add group to cache if not null
322    * @param attributeDefId 
323    * @param attributeAssignActions 
324    */
325   private static void attributeAssignActionFlashCacheAddIfSupposedToByAttributeDef(String attributeDefId, Set<AttributeAssignAction> attributeAssignActions) {
326     if (StringUtils.isBlank(attributeDefId) || attributeAssignActions == null || !GrouperConfig.retrieveConfig().propertyValueBoolean(GROUPER_FLASHCACHE_ATTRIBUTE_ASSIGN_ACTION, true)) {
327       return;
328     }
329     
330     for (AttributeAssignAction attributeAssignAction : attributeAssignActions ) {
331       attributeAssignActionFlashCache.put(attributeAssignActionFlashCacheMultikeyId(attributeAssignAction.getId()), GrouperUtil.toSet(attributeAssignAction));
332     }
333     attributeAssignActionFlashCache.put(attributeAssignActionFlashCacheMultikeyAttributeDefId(attributeDefId), 
334         attributeAssignActions);
335 
336   }
