290   private Set<Object[]> findAllByGroupOwnerOptionsHelper(Collection<String> totalGroupIds, Collection<String> totalMemberIds,
291       Collection<Field> fields,
292       Set<Source> sources, Stem stem, Scope stemScope, Boolean checkSecurity, FieldType fieldType,
293       QueryOptions queryOptionsForMember, String filterForMember, boolean splitScopeForMember, 
294       boolean hasFieldForMember, Timestamp pointInTimeFrom, Timestamp pointInTimeTo) {
295 
296     QueryOptions.initTotalCount(queryOptionsForMember);
297     
298     if (checkSecurity == null) {
299       checkSecurity = Boolean.TRUE;
300     }
301     
302     if (GrouperUtil.length(fields) != 0) {
303       throw new RuntimeException("Fields not supported for now.");
304     }
305     
306     if (fieldType != null) {
307       throw new RuntimeException("Field type not supported for now.");
308     }
309     
310     if ((stem == null) != (stemScope == null)) {
311       throw new RuntimeException("If stem is set, then stem scope must be set.  If stem isnt set, then stem scope must not be set: " + stem + ", " + stemScope);
312     }
313     
314     final List<String> totalGroupIdsList = GrouperUtil.listFromCollection(totalGroupIds);
315     List<String> totalMemberIdsList = GrouperUtil.listFromCollection(totalMemberIds);
316 
317     Set<Object[]> totalResults = new HashSet<Object[]>();
318     
319     GrouperSession grouperSession = GrouperSession.staticGrouperSession();
320     
321     final Subject grouperSessionSubject = grouperSession.getSubject();
322     
323     final Set<Privilege> privilegesTheUserHasFinal = new HashSet<Privilege>();
324     
325     if (GrouperUtil.length(fields) == 0 || !fields.iterator().next().isGroupAccessField()) {
326       privilegesTheUserHasFinal.addAll(AccessPrivilege.READ_PRIVILEGES);
327     } else {
328       privilegesTheUserHasFinal.addAll(AccessPrivilege.ADMIN_PRIVILEGES);
329     }
330     
331     //just check security on one stem to help performance
332     if (checkSecurity && GrouperUtil.length(totalGroupIds) == 1) {
333       boolean allowed = (Boolean)GrouperSession.callbackGrouperSession(
334           GrouperSession.staticGrouperSession().internal_getRootSession(), new GrouperSessionHandler() {
335 
336         /**
337          * 
338          */
339         @Override
340         public Object callback(GrouperSession theGrouperSession) throws GrouperSessionException {
341           Group theGroup = GroupFinder.findByUuid(theGrouperSession, totalGroupIdsList.get(0), false);
342           if (theGroup == null) {
343             return false;
344           }
345           for (Privilege privilegeTheUserHas : privilegesTheUserHasFinal) {
346              if (theGroup.canHavePrivilege(grouperSessionSubject, privilegeTheUserHas.getName(), false)) {
347                return true;
348              }
349           }
350           return false;
351              
352         }
353       });
354 
355       //if there is one stem, and checking security, and the user is not allowed to STEM it, then no results
356       if (!allowed) {
357         return totalResults;
358       }
359       
360       checkSecurity = false;
361     }
362     
363     int groupBatches = GrouperUtil.batchNumberOfBatches(totalGroupIds, 100);
364 
365     for (int groupIndex = 0; groupIndex < groupBatches; groupIndex++) {
366       
367       List<String> groupIds = GrouperUtil.batchList(totalGroupIdsList, 100, groupIndex);
368       
369       int memberBatches = GrouperUtil.batchNumberOfBatches(totalMemberIds, 100);
370 
371       for (int memberIndex = 0; memberIndex < memberBatches; memberIndex++) {
372         
373         List<String> memberIds = GrouperUtil.batchList(totalMemberIdsList, 100, memberIndex);          
374         
375         int groupIdsSize = GrouperUtil.length(groupIds);
376         int memberIdsSize = GrouperUtil.length(memberIds);
377         
378         if (groupIdsSize == 0 && memberIdsSize == 0 && stem == null) {
379           throw new RuntimeException("Must pass in group(s), member(s), and/or stem");
380         }
381 
382         ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
383 
384         String selectPrefix = "select distinct pitms, pitg, pitm, m ";
385         
386         //note: mysql wont let you do count distinct of multiple columns
387         String countPrefix = "select count(*) ";
388 
389         StringBuilder sql = new StringBuilder(" from Member m, PITMembershipView pitms, Group g, Field f, PITMember pitm, PITGroup pitg, PITField pitf ");
390         
391         boolean changedQuery = false;
392         
393         if (checkSecurity) { 
394           changedQuery = grouperSession.getAccessResolver().hqlFilterGroupsWhereClause(
395             grouperSessionSubject, byHqlStatic, 
396             sql, "g.uuid", privilegesTheUserHasFinal);
397         }
398 
399         if (changedQuery && sql.toString().contains(" where ")) {
400           sql.append(" and ");
401         } else {
402           sql.append(" where ");
403         }
404         
405         sql.append(" pitms.ownerGroupId = pitg.id "
406             + " and pitms.fieldId = pitf.id "
407             + " and pitms.memberId = pitm.id "
408             + " and pitg.sourceId = g.uuid "
409             + " and pitf.sourceId = f.uuid "
410             + " and pitm.sourceId = m.uuid ");
411         
412         if (sources != null && sources.size() > 0) {
413           sql.append(" and m.subjectSourceIdDb in ").append(HibUtils.convertSourcesToSqlInString(sources));
414         }
415         
416         if (stem != null) {
417           switch (stemScope) {
418             case ONE:
419               
420               sql.append(" and g.parentUuid = :stemId ");
421               byHqlStatic.setString("stemId", stem.getUuid());
422               break;
423             case SUB:
424               
425               sql.append(" and g.nameDb like :stemSub ");
426               byHqlStatic.setString("stemSub", stem.getName() + ":%");
427               
428               break;
429             default:
430               throw new RuntimeException("Not expecting scope: " + stemScope);
431           }
432         }
433         
434         if (GrouperUtil.length(fields) > 0) {
435 
436           sql.append(" and f.uuid in ( ");
437           Set<String> fieldStrings = new HashSet<String>();
438           for (Field field : fields) {
439             fieldStrings.add(field.getUuid());
440           }
441           sql.append(HibUtils.convertToInClause(fieldStrings, byHqlStatic));
442           sql.append(" ) ");
443         }
444 
445         if (GrouperUtil.length(fields) == 0 && (fieldType == null || fieldType == FieldType.LIST)) {
446           
447           //add on the column
448           sql.append(" and f.typeString = 'list' ");
449         }
450         if (fieldType == FieldType.ACCESS) {
451           //add on the column
452           sql.append(" and f.typeString = 'access' ");
453         }
454         if (groupIdsSize > 0) {
455           sql.append(" and g.uuid in (");
456           sql.append(HibUtils.convertToInClause(groupIds, byHqlStatic));
457           sql.append(") ");
458         }
459         if (memberIdsSize > 0) {
460           sql.append(" and m.uuid in (");
461           sql.append(HibUtils.convertToInClause(memberIds, byHqlStatic));
462           sql.append(") ");
463         }
464         
465         if (!StringUtils.isBlank(filterForMember)) {
466 
467           filterForMember = filterForMember.toLowerCase();
468 
469           String[] filtersForMember = splitScopeForMember ? GrouperUtil.splitTrim(filterForMember, " ") 
470               : new String[]{filterForMember};
471 
472           if (sql.length() > 0) {
473             sql.append(" and ");
474           }
475           sql.append(" ( ");
476 
477           int index = 0;
478           
479           String searchFieldName = SearchStringEnum.getDefaultSearchString().getFieldName();
480 
481           for (String theFilter : filtersForMember) {
482             if (index != 0) {
483               sql.append(" and ");
484             }
485             sql.append(" ( m." + searchFieldName + " like :filterString" + index + " ) ");
486 
487             if (!theFilter.endsWith("%")) {
488               theFilter += "%";
489             }
490             if (!theFilter.startsWith("%")) {
491               theFilter = "%" + theFilter;
492             }
493             byHqlStatic.setString("filterString" + index, theFilter);
494             index++;
495           }
496           sql.append(" ) ");
497         }
498
