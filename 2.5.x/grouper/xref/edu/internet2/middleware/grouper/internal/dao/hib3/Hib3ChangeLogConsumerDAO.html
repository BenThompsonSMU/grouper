16  package edu.internet2.middleware.grouper.internal.dao.hib3;
17  import java.util.Set;
18  
19  import edu.internet2.middleware.grouper.changeLog.ChangeLogConsumer;
20  import edu.internet2.middleware.grouper.hibernate.HibernateSession;
21  import edu.internet2.middleware.grouper.internal.dao.ChangeLogConsumerDAO;
22  
23  /**
24   * Data Access Object for changeLog consumer
25   * @author  mchyzer
26   * @version $Id: Hib3ChangeLogConsumerDAO.java,v 1.1 2009-06-09 17:24:13 mchyzer Exp $
27   */
28  public class Hib3ChangeLogConsumerDAO extends Hib3DAO implements ChangeLogConsumerDAO {
29    
30    /**
31     * 
32     */
33    @SuppressWarnings("unused")
34    private static final String KLASS = Hib3ChangeLogConsumerDAO.class.getName();
35  
36    /**
37     * 
38     * @see edu.internet2.middleware.grouper.internal.dao.ChangeLogConsumerDAO#saveOrUpdate(edu.internet2.middleware.grouper.changeLog.ChangeLogConsumer)
39     */
40    public void saveOrUpdate(ChangeLogConsumer changeLogConsumer) {
41      HibernateSession.byObjectStatic().saveOrUpdate(changeLogConsumer);
42    }
43  
44    /**
45     * @see edu.internet2.middleware.grouper.internal.dao.ChangeLogConsumerDAO#findByName(String, boolean)
46     */
47    public ChangeLogConsumer findByName(String name, boolean exceptionIfNotFound) {
48      
49      ChangeLogConsumer changeLogConsumer = HibernateSession.byHqlStatic().createQuery("from ChangeLogConsumer where name = :theName")
50        .setString("theName", name).uniqueResult(ChangeLogConsumer.class);
51      
52      if (changeLogConsumer == null && exceptionIfNotFound) {
53        throw new RuntimeException("Couldnt find a change log consumer by name: '" + name + "'");
54      }
55      
56      return changeLogConsumer;
57    }
58    
59  
60    /**
61     * reset the changeLog types
62     * @param hibernateSession
63     */
64    static void reset(HibernateSession hibernateSession) {
65      hibernateSession.byHql().createQuery("delete from ChangeLogConsumer").executeUpdate();
66    }
67  
68    /**
69     * @see edu.internet2.middleware.grouper.internal.dao.ChangeLogConsumerDAO#findAll()
70     */
71    public Set<ChangeLogConsumer> findAll() {
72      return HibernateSession.byHqlStatic().createQuery("from ChangeLogConsumer").listSet(ChangeLogConsumer.class);
73    }
74    
75  }
