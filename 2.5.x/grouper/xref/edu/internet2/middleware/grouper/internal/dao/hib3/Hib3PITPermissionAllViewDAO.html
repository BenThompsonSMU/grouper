43    private static final String PERMISSION_ENTRY_TABLES = "PITGroup gr, PITMembershipView gmav, PITMember gm, PITField gf, PITRoleSet grs, PITAttributeDef gad, PITAttributeAssign gaa, PITAttributeDefName gadn, PITAttributeDefNameSet gadns, PITAttributeAssignAction gaaa, PITAttributeAssignActionSet gaaas";

45    private static final String PERMISSION_ENTRY_WHERE_CLAUSE = "gmav.ownerGroupId = gr.id and gmav.fieldId = gf.id and gf.typeDb = 'list' and gf.nameDb = 'members' and gmav.memberId = gm.id and gadn.attributeDefId = gad.id and gad.attributeDefTypeDb = 'perm' and gaa.attributeDefNameId = gadns.ifHasAttributeDefNameId and gadn.id = gadns.thenHasAttributeDefNameId and gaa.attributeAssignActionId = gaaas.ifHasAttrAssignActionId and gaaa.id = gaaas.thenHasAttrAssignActionId and ((grs.ifHasRoleId = gr.id and gaa.ownerGroupId = grs.thenHasRoleId and gaa.attributeAssignTypeDb = 'group') or (gmav.ownerGroupId = gaa.ownerGroupId  and gmav.memberId = gaa.ownerMemberId and gaa.attributeAssignTypeDb = 'any_mem' and grs.ifHasRoleId = gr.id and grs.depth='0'))";

50    private static final String KLASS = Hib3PITPermissionAllViewDAO.class.getName();

52    /**
53     * @see edu.internet2.middleware.grouper.internal.dao.PITPermissionAllViewDAO#findPermissions(java.util.Collection, java.util.Collection, java.util.Collection, java.util.Collection, java.util.Collection, java.sql.Timestamp, java.sql.Timestamp)
54     */
55    public Set<PermissionEntry> findPermissions(Collection<String> attributeDefSourceIds, Collection<String> attributeDefNameSourceIds, 
56        Collection<String> roleSourceIds, Collection<String> actions, Collection<String> memberSourceIds, Timestamp pointInTimeFrom, 
57        Timestamp pointInTimeTo) {
58      
59      int memberIdsSize = GrouperUtil.length(memberSourceIds);
60      int roleIdsSize = GrouperUtil.length(roleSourceIds);
61      int actionsSize = GrouperUtil.length(actions);
62      int attributeDefIdsSize = GrouperUtil.length(attributeDefSourceIds);
63      int attributeDefNameIdsSize = GrouperUtil.length(attributeDefNameSourceIds);
64      
65      if (memberIdsSize == 0 && roleIdsSize == 0 && attributeDefIdsSize == 0 && attributeDefNameIdsSize == 0) {
66        throw new RuntimeException("Illegal query, you need to pass in members and/or attributeDefId(s) and/or roleId(s) and/or attributeDefNameIds");
67      }
68      
69      //too many bind vars
70      if (memberIdsSize + roleIdsSize + attributeDefIdsSize + attributeDefNameIdsSize + actionsSize > 100) {
71        throw new RuntimeException("Too many memberIdsSize " + memberIdsSize 
72            + " roleIdsSize " + roleIdsSize + " or attributeDefIdsSize " 
73            + attributeDefIdsSize + " or attributeDefNameIds " + attributeDefNameIdsSize + " or actionsSize " + actionsSize );
74      }
75  
76      ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
77  
78      String selectPrefix = "select distinct " + PERMISSION_ENTRY_COLUMNS + " ";
79      
80      StringBuilder sqlTables = new StringBuilder(" from " + PERMISSION_ENTRY_TABLES + " ");
81      
82      StringBuilder sqlWhereClause = new StringBuilder(" " + PERMISSION_ENTRY_WHERE_CLAUSE + " ");
83      
84      GrouperSession grouperSession = GrouperSession.staticGrouperSession();
85      
86      Subject grouperSessionSubject = grouperSession.getSubject();
87      
88      grouperSession.getAttributeDefResolver().hqlFilterAttrDefsWhereClause(
89        grouperSessionSubject, byHqlStatic, 
90        sqlTables, sqlWhereClause, "gad.sourceId", AttributeDefPrivilege.ATTR_READ_PRIVILEGES);
91      
92      boolean changedQuery = grouperSession.getAccessResolver().hqlFilterGroupsWhereClause(
93          grouperSessionSubject, byHqlStatic, 
94          sqlTables, "gr.sourceId", AccessPrivilege.ATTRIBUTE_READ_PRIVILEGES);
95      
96      StringBuilder sql;
97      if (changedQuery && sqlTables.toString().contains(" where ")) {
98        if (sqlWhereClause.length() > 0) {
99          sql = sqlTables.append(" and ").append(sqlWhereClause);
100       } else {
101         throw new RuntimeException("Unexpected.");
102       }
103     } else {
104       sql = sqlTables.append(" where ").append(sqlWhereClause);
105     }
106     
107     if (actionsSize > 0) {
108       sql.append(" and gaaa.nameDb in (");
109       sql.append(HibUtils.convertToInClause(actions, byHqlStatic));
110       sql.append(") ");
111     }
112     if (roleIdsSize > 0) {
113       sql.append(" and gr.sourceId in (");
114       sql.append(HibUtils.convertToInClause(roleSourceIds, byHqlStatic));
115       sql.append(") ");
116     }
117     if (attributeDefIdsSize > 0) {
118       sql.append(" and gad.sourceId in (");
119       sql.append(HibUtils.convertToInClause(attributeDefSourceIds, byHqlStatic));
120       sql.append(") ");
121     }
122     if (attributeDefNameIdsSize > 0) {
123       sql.append(" and gadn.sourceId in (");
124       sql.append(HibUtils.convertToInClause(attributeDefNameSourceIds, byHqlStatic));
125       sql.append(") ");
126     }
127     if (memberIdsSize > 0) {
128       sql.append(" and gm.sourceId in (");
129       sql.append(HibUtils.convertToInClause(memberSourceIds, byHqlStatic));
130       sql.append(") ");
131     }
132     
133     if (pointInTimeFrom != null) {
134       Long endDateAfter = pointInTimeFrom.getTime() * 1000;
135       sql.append(" and (gmav.membershipEndTimeDb is null or gmav.membershipEndTimeDb > '" + endDateAfter + "')");
136       sql.append(" and (gmav.groupSetEndTimeDb is null or gmav.groupSetEndTimeDb > '" + endDateAfter + "')");
137       sql.append(" and (gaaas.endTimeDb is null or gaaas.endTimeDb > '" + endDateAfter + "')");
138       sql.append(" and (gadns.endTimeDb is null or gadns.endTimeDb > '" + endDateAfter + "')");
139       sql.append(" and (grs.endTimeDb is null or grs.endTimeDb > '" + endDateAfter + "')");
140       sql.append(" and (gaa.endTimeDb is null or gaa.endTimeDb > '" + endDateAfter + "')");
141     }
142     
143     if (pointInTimeTo != null) {
144       Long startDateBefore = pointInTimeTo.getTime() * 1000;
145       sql.append(" and gmav.membershipStartTimeDb < '" + startDateBefore + "'");
146       sql.append(" and gmav.groupSetStartTimeDb < '" + startDateBefore + "'");
147       sql.append(" and gaaas.startTimeDb < '" + startDateBefore + "'");
148       sql.append(" and gadns.startTimeDb < '" + startDateBefore + "'");
149       sql.append(" and grs.startTimeDb < '" + startDateBefore + "'");
150       sql.append(" and gaa.startTimeDb < '" + startDateBefore + "'");
151     }
152     
153     byHqlStatic
154       .setCacheable(false)
155       .setCacheRegion(KLASS + ".findPermissions");
156 
157     int maxAssignments = GrouperConfig.retrieveConfig().propertyValueInt("ws.findPermissions.maxResultSize", 30000);
158     
159     String sqlString = sql.toString();
160     
161     //if we did where and, then switch to where
162     sqlString = sqlString.replaceAll("where\\s+and", "where");
163     
164     Set<PITPermissionAllView> resultsTemp = byHqlStatic.createQuery(selectPrefix + sqlString)
165       .assignConvertHqlColumnsToObject(true)
166       .listSet(PITPermissionAllView.class);
167 
168     int size = GrouperUtil.length(resultsTemp);
169     if (maxAssignments >= 0) {
170       
171       //see if too many
172       if (size > maxAssignments) {
173         throw new RuntimeException("Too many results: " + size);
174       }
175     }
176     
177     Set<PermissionEntry> results = new LinkedHashSet<PermissionEntry>(resultsTemp);
178 
179     //nothing to filter
180     if (size == 0) {
181       return results;
182     }
183     
184     //if the hql didnt filter, we need to do that here
185     results = grouperSession.getAttributeDefResolver().postHqlFilterPermissions(grouperSessionSubject, results);
186     
187     //we should be down to the secure list
188     return results;
189   }
