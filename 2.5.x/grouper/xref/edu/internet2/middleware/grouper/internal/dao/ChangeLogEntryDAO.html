20  package edu.internet2.middleware.grouper.internal.dao;
21  
22  import java.util.List;
23  import java.util.Set;
24  
25  import edu.internet2.middleware.grouper.changeLog.ChangeLogEntry;
26  
27  
28  /**
29   * change log entry data access methods
30   */
31  public interface ChangeLogEntryDAO extends GrouperDAO {
32    
33    /** 
34     * insert a change log entry object 
35     * @param changeLogEntry 
36     */
37    public void save(ChangeLogEntry changeLogEntry);
38    
39    /**
40     * insert a batch of change log entry objects
41     * @param changeLogEntries
42     * @param isTempBatch
43     */
44    public void saveBatch(Set<ChangeLogEntry> changeLogEntries, boolean isTempBatch);
45    
46    /** 
47     * update a change log entry object 
48     * @param changeLogEntry 
49     */
50    public void update(ChangeLogEntry changeLogEntry);
51    
52    /** 
53     * delete change log entry object 
54     * @param changeLogEntry 
55     */
56    public void delete(ChangeLogEntry changeLogEntry);
57    
58    /**
59     * get the change log records after a sequence number, not including the sequence number
60     * @param afterSequenceNumber
61     * @param batchSize is the size of the batch
62     * @return the records
63     */
64    public List<ChangeLogEntry> retrieveBatch(long afterSequenceNumber, int batchSize);
65  
66    /**
67     * find by sequence number.  This is NOT a secure method, a grouperSession does not need to be open
68     * @param sequenceNumber uniquely identifies rows in change log, newer records have greater than older
69     * @param exceptionIfNotFound
70     * @return the change log entry or null if not there
71     */
72    public ChangeLogEntry findBySequenceNumber(long sequenceNumber, boolean exceptionIfNotFound);
73  }
