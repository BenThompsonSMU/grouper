229   /**
230    * 
231    * @return set of PITMembershipView objects
232    */
233   public Set<PITMembershipView> execute() {
234     
235     List<Criterion> criterionList = new ArrayList<Criterion>();
236     
237     if (this.ownerGroupId != null) {
238       Set<PITGroup> pitGroups = GrouperDAOFactory.getFactory().getPITGroup().findBySourceId(this.ownerGroupId, true);
239       Set<String> ids = new LinkedHashSet<String>();
240       for (PITGroup pitGroup : pitGroups) {
241         ids.add(pitGroup.getId());
242       }
243       
244       criterionList.add(Restrictions.in(PITMembershipView.FIELD_OWNER_ID, ids));
245     } else if (this.ownerStemId != null) {
246       Set<PITStem> pitStems = GrouperDAOFactory.getFactory().getPITStem().findBySourceId(this.ownerStemId, true);
247       Set<String> ids = new LinkedHashSet<String>();
248       for (PITStem pitStem : pitStems) {
249         ids.add(pitStem.getId());
250       }
251       
252       criterionList.add(Restrictions.in(PITMembershipView.FIELD_OWNER_ID, ids));
253     } else if (this.ownerAttrDefId != null) {
254       Set<PITAttributeDef> pitAttributeDefs = GrouperDAOFactory.getFactory().getPITAttributeDef().findBySourceId(this.ownerAttrDefId, true);
255       Set<String> ids = new LinkedHashSet<String>();
256       for (PITAttributeDef pitAttributeDef : pitAttributeDefs) {
257         ids.add(pitAttributeDef.getId());
258       }
259       
260       criterionList.add(Restrictions.in(PITMembershipView.FIELD_OWNER_ID, ids));
261     }
262     
263     if (this.memberId != null) {
264       Set<PITMember> pitMembers = GrouperDAOFactory.getFactory().getPITMember().findBySourceId(this.memberId, true);
265       Set<String> ids = new LinkedHashSet<String>();
266       for (PITMember pitMember : pitMembers) {
267         ids.add(pitMember.getId());
268       }
269       
270       criterionList.add(Restrictions.in(PITMembershipView.FIELD_MEMBER_ID, ids));
271     }
272     
273     if (this.fieldId != null) {
274       Set<PITField> pitFields = GrouperDAOFactory.getFactory().getPITField().findBySourceId(this.fieldId, true);
275       Set<String> ids = new LinkedHashSet<String>();
276       for (PITField pitField : pitFields) {
277         ids.add(pitField.getId());
278       }
279       
280       criterionList.add(Restrictions.in(PITMembershipView.FIELD_FIELD_ID, ids));
281     }
282     
283     if (this.startDateAfter != null) {
284       criterionList.add(Restrictions.or(
285           Restrictions.ge(PITMembershipView.FIELD_MEMBERSHIP_START_TIME_DB, this.startDateAfter.getTime() * 1000), 
286           Restrictions.ge(PITMembershipView.FIELD_GROUP_SET_START_TIME_DB, this.startDateAfter.getTime() * 1000)));
287     }
288     
289     if (this.startDateBefore != null) {
290       criterionList.add(Restrictions.le(PITMembershipView.FIELD_MEMBERSHIP_START_TIME_DB, this.startDateBefore.getTime() * 1000));
291       criterionList.add(Restrictions.le(PITMembershipView.FIELD_GROUP_SET_START_TIME_DB, this.startDateBefore.getTime() * 1000));
292     }
293     
294     if (this.endDateAfter != null) {
295       criterionList.add(Restrictions.or(
296           Restrictions.isNull(PITMembershipView.FIELD_MEMBERSHIP_END_TIME_DB),
297           Restrictions.ge(PITMembershipView.FIELD_MEMBERSHIP_END_TIME_DB, this.endDateAfter.getTime() * 1000)));
298       
299       criterionList.add(Restrictions.or(
300           Restrictions.isNull(PITMembershipView.FIELD_GROUP_SET_END_TIME_DB),
301           Restrictions.ge(PITMembershipView.FIELD_GROUP_SET_END_TIME_DB, this.endDateAfter.getTime() * 1000)));
302     }
303     
304     if (this.endDateBefore != null) {
305       criterionList.add(Restrictions.or(
306           Restrictions.le(PITMembershipView.FIELD_MEMBERSHIP_END_TIME_DB, this.endDateBefore.getTime() * 1000), 
307           Restrictions.le(PITMembershipView.FIELD_GROUP_SET_END_TIME_DB, this.endDateBefore.getTime() * 1000)));
308     }
309     
310     if (this.extraCriterion != null) {
311       criterionList.add(this.extraCriterion);
312     }
313     
314     Criterion allCriteria = HibUtils.listCrit(criterionList);
315     
316     Set<PITMembershipView> results = HibernateSession.byCriteriaStatic()
317       .options(this.queryOptions).listSet(PITMembershipView.class, allCriteria);
318     return results;
319   }
