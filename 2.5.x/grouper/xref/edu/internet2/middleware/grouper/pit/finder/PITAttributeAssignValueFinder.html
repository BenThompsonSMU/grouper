16  package edu.internet2.middleware.grouper.pit.finder;
17  
18  import java.sql.Timestamp;
19  import java.util.LinkedHashSet;
20  import java.util.Set;
21  
22  import edu.internet2.middleware.grouper.misc.GrouperDAOFactory;
23  import edu.internet2.middleware.grouper.pit.PITAttributeAssign;
24  import edu.internet2.middleware.grouper.pit.PITAttributeAssignValue;
25  
26  /**
27   * Find point in time attribute assign values.
28   * 
29   * @author shilen
30   * $Id$
31   */
32  public class PITAttributeAssignValueFinder {
33  
34    /**
35     * Find point in time attribute assign values by attribute assign and date ranges.
36     * This assumes if you have access to the attribute assign, then you have access to the values too.  
37     * So there are no security checks here.
38     * @param attributeAssign
39     * @param pointInTimeFrom
40     * @param pointInTimeTo
41     * @return set of pit attribute assign values
42     */
43    public static Set<PITAttributeAssignValue> findByPITAttributeAssign(PITAttributeAssign attributeAssign, Timestamp pointInTimeFrom, Timestamp pointInTimeTo) {
44      
45      Set<PITAttributeAssignValue> values = GrouperDAOFactory.getFactory().getPITAttributeAssignValue().findByPITAttributeAssignId(attributeAssign.getId(), null);
46      Set<PITAttributeAssignValue> valuesInRange = new LinkedHashSet<PITAttributeAssignValue>();
47  
48      for (PITAttributeAssignValue value : values) {
49        if (pointInTimeFrom != null) {
50          if (!value.isActive() && value.getEndTime().before(pointInTimeFrom)) {
51            continue;
52          }
53        }
54        
55        if (pointInTimeTo != null) {
56          if (value.getStartTime().after(pointInTimeTo)) {
57            continue;
58          }
59        }
60        
61        valuesInRange.add(value);
62      }
63      
64      return valuesInRange;
65    }
66  }
