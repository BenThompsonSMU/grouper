107   /**
108    * get the attribute def ids to ignore when sending to change log, and audit 
109    * @return the set of attribute definition ids
110    */
111   public Set<String> attributeDefIdsToIgnoreChangeLogAndAudit() {
112     
113     if (this.attributeDefIdsToIgnoreChangeLogAndAuditSet == null) {
114       
115       synchronized (this) {
116 
117         if (this.attributeDefIdsToIgnoreChangeLogAndAuditSet == null) {
118           Set<String> result = new HashSet<String>();
119           
120           if (GrouperStartup.isFinishedStartupSuccessfully()) {
121             String namesOfAttributeDefsCommaSeparated = this.propertyValueString("grouper.attribute.namesOfAttributeDefsToIgnoreAuditsChangeLogPit");
122               
123             Set<String> tempResult = attributeDefIdsToIgnoreChangeLogAndAuditSetCache.get(namesOfAttributeDefsCommaSeparated);
124             
125             if (tempResult == null) {
126               
127               Set<String> namesOfAttributeDefs = new HashSet<String>();
128               
129               // GRP-1695: hard code built in ignore attribute defs and names
130               // $$grouper.attribute.rootStem$$:userData:grouperUserDataValueDef
131               namesOfAttributeDefs.add(GrouperUserDataUtils.grouperUserDataStemName() + ":" + GrouperUserDataUtils.USER_DATA_VALUE_DEF);
132               
133               // $$grouper.attribute.rootStem$$:instrumentationData:instrumentationDataInstanceCountsDef
134               namesOfAttributeDefs.add(InstrumentationDataUtils.grouperInstrumentationDataStemName() + ":" + InstrumentationDataUtils.INSTRUMENTATION_DATA_INSTANCE_COUNTS_DEF);
135   
136               // $$grouper.attribute.rootStem$$:instrumentationData:instrumentationDataInstanceDetailsDef
137               namesOfAttributeDefs.add(InstrumentationDataUtils.grouperInstrumentationDataStemName() + ":" + InstrumentationDataUtils.INSTRUMENTATION_DATA_INSTANCE_DETAILS_DEF);
138   
139               // $$grouper.attribute.rootStem$$:instrumentationData:instrumentationDataCollectorDetailsDef
140               namesOfAttributeDefs.add(InstrumentationDataUtils.grouperInstrumentationDataStemName() + ":" + InstrumentationDataUtils.INSTRUMENTATION_DATA_COLLECTOR_DETAILS_DEF);
141               
142               // $$grouper.attribute.rootStem$$:attribute:loaderMetadata:loaderMetadataValueDef
143               namesOfAttributeDefs.add(GrouperCheckConfig.loaderMetadataStemName() + ":loaderMetadataValueDef");
144               
145               // deprovisioning
146               namesOfAttributeDefs.add(GrouperDeprovisioningAttributeNames.retrieveAttributeDefBaseDef().getName());
147               namesOfAttributeDefs.add(GrouperDeprovisioningAttributeNames.retrieveAttributeDefNameValueDef().getName());
148               
149               if (!StringUtils.isBlank(namesOfAttributeDefsCommaSeparated)) {
150                 namesOfAttributeDefs.addAll(GrouperUtil.splitTrimToSet(namesOfAttributeDefsCommaSeparated, ","));
151               }
152               
153               if (GrouperUtil.length(namesOfAttributeDefs) > 0) {
154                 //get a root session, or use existing
155                 GrouperSession grouperSession = GrouperSession.staticGrouperSession(false);
156                 boolean startedSession = grouperSession == null;
157                 try {
158                   if (startedSession) {
159                     grouperSession = GrouperSession.startRootSession();
160                   } else {
161                     grouperSession = grouperSession.internal_getRootSession();
162                   }
163                   
164                   for (String nameOfAttributeDef : namesOfAttributeDefs) {
165                     try {
166                       //if not there log it.  e.g. for UI you might ignore attributes, but wont be there if testing the API
167                       AttributeDef attributeDef = AttributeDefFinder.findByName(nameOfAttributeDef, false);
168                       
169                       if (attributeDef == null) {
170                         LOG().error("Attribute def not found: " + nameOfAttributeDef);
171                         continue;
172                       }
173                       
174                       result.add(attributeDef.getId());
175                       
176                       if (result.size() > 150) {
177                         throw new RuntimeException("Cant have a size of more than 150 for attributeDefs excluded from audits and PIT");
178                       }
179                       
180                     } catch (RuntimeException re) {
181                       GrouperUtil.injectInException(re, "name of attributeDef configured "
182                           + "in grouper properties file: grouper.attribute.namesOfAttributeDefsToIgnoreAuditsChangeLogPit, "
183                           + "that attribute cannot be found.  ");
184                       throw re;
185                     }
186                   }
187                   
188                 } finally {
189                   if (startedSession) {
190                     GrouperSession.stopQuietly(grouperSession);
191                   }
192                 }
193               }
194               
195               //you dont want callers modifying this
196               result = Collections.unmodifiableSet(result);
197   
198               //put back in cache
199               attributeDefIdsToIgnoreChangeLogAndAuditSetCache.put(namesOfAttributeDefsCommaSeparated, result);
200             } else {
201               result = GrouperUtil.nonNull(tempResult);
202             }
203             this.attributeDefIdsToIgnoreChangeLogAndAuditSet = result;
204               
205           }
206         }
207 
208       }
209       
210     }
211     return GrouperUtil.nonNull(this.attributeDefIdsToIgnoreChangeLogAndAuditSet);
212     
213   }

224   /**
225    * get the attribute def name ids to ignore when sending to change log, and audit 
226    * @return the set of attribute def name ids
227    */
228   public Set<String> attributeDefNameIdsToIgnoreChangeLogAndAudit() {
229     
230     if (this.attributeDefNameIdsToIgnoreChangeLogAndAuditSet == null) {
231       
232       synchronized (this) {
233 
234         if (this.attributeDefNameIdsToIgnoreChangeLogAndAuditSet == null) {
235           Set<String> result = new HashSet<String>();
236           
237           String namesOfAttributeDefNamesCommaSeparated = this.propertyValueString("grouper.attribute.namesOfAttributeDefNamesToIgnoreAuditsChangeLogPit");
238             
239           Set<String> tempResult = attributeDefNameIdsToIgnoreChangeLogAndAuditSetCache.get(namesOfAttributeDefNamesCommaSeparated);
240           
241           if (tempResult == null) {
242             
243             Set<String> namesOfAttributeDefNames = new HashSet<String>();
244             
245             // GRP-1695: hard code built in ignore attribute defs and names
246             // $$grouper.attribute.rootStem$$:attestation:attestationCalculatedDaysLeft
247             namesOfAttributeDefNames.add(GrouperAttestationJob.retrieveAttributeDefNameCalculatedDaysLeft().getName());
248             
249             // $$grouper.attribute.rootStem$$:attestation:attestationLastEmailedDate
250             namesOfAttributeDefNames.add(GrouperAttestationJob.retrieveAttributeDefNameEmailedDate().getName());
251             
252             if (!StringUtils.isBlank(namesOfAttributeDefNamesCommaSeparated)) {
253               namesOfAttributeDefNames.addAll(GrouperUtil.splitTrimToSet(namesOfAttributeDefNamesCommaSeparated, ","));
254             }
255 
256             if (GrouperUtil.length(namesOfAttributeDefNames) > 0) {
257               //get a root session, or use existing
258               GrouperSession grouperSession = GrouperSession.staticGrouperSession(false);
259               boolean startedSession = grouperSession == null;
260               try {
261                 if (startedSession) {
262                   grouperSession = GrouperSession.startRootSession();
263                 } else {
264                   grouperSession = grouperSession.internal_getRootSession();
265                 }
266                 
267                 for (String nameOfAttributeDefName : namesOfAttributeDefNames) {
268                   try {
269                     //if not there log it.  e.g. for UI you might ignore attributes, but wont be there if testing the API
270                     AttributeDefName attributeDefName = AttributeDefNameFinder.findByName(nameOfAttributeDefName, false);
271                     
272                     if (attributeDefName == null) {
273                       LOG().error("Attribute def name not found: " + nameOfAttributeDefName);
274                       continue;
275                     }
276                     
277                     result.add(attributeDefName.getId());
278                     
279                     if (result.size() > 150) {
280                       throw new RuntimeException("Cant have a size of more than 150 for attributeDefNames excluded from audits and PIT");
281                     }
282                     
283                   } catch (RuntimeException re) {
284                     GrouperUtil.injectInException(re, "name of attributeDefName configured "
285                         + "in grouper properties file: grouper.attribute.namesOfAttributeDefNamesToIgnoreAuditsChangeLogPit, "
286                         + "that attribute cannot be found.  ");
287                     throw re;
288                   }
289                 }
290                 
291               } finally {
292                 if (startedSession) {
293                   GrouperSession.stopQuietly(grouperSession);
294                 }
295               }
296             }
297             
298             //you dont want callers modifying this
299             result = Collections.unmodifiableSet(result);
300 
301             //put back in cache
302             attributeDefNameIdsToIgnoreChangeLogAndAuditSetCache.put(namesOfAttributeDefNamesCommaSeparated, result);
303           } else {
304             result = GrouperUtil.nonNull(tempResult);
305           }
306           this.attributeDefNameIdsToIgnoreChangeLogAndAuditSet = result;
307             
308         }
309 
310       }
311       
312     }
313 
314     return GrouperUtil.nonNull(this.attributeDefNameIdsToIgnoreChangeLogAndAuditSet);
315     
316   }

542   /**
543    * get the UI url with a slash on the end
544    * @param exceptionIfNull
545    * @return the UI URL
546    */
547   public static String getGrouperUiUrl(boolean exceptionIfNull) {
548     String url = getProperty("grouper.ui.url");
549     if (StringUtils.isBlank(url)) {
550       if (exceptionIfNull) {
551         throw new RuntimeException("grouper.ui.url is null in grouper.properties");
552       }
553       return null;
554     }
555     if (url.endsWith("/")) {
556       return url;
557     }
558     return url + "/";
559   }
