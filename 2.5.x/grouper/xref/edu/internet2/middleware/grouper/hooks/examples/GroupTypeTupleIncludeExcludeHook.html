87    /**
88     * see if this is configured in the grouper.properties, if so, register this hook
89     * @param tryAgainIfNotBefore 
90     */
91    public static void registerHookIfNecessary(boolean tryAgainIfNotBefore) {
92      
93      if (registered && !tryAgainIfNotBefore) {
94        return;
95      }
96      
97      //if trying again, but already registered, fine
98      if (tryAgainIfNotBefore && registeredSuccess) {
99        return;
100     }
101     
102     //see if there are config entries
103     Properties grouperProperties = GrouperConfig.retrieveConfig().properties();
104 
105     boolean useIncludeExcludeOverride = StringUtils.equals(GrouperConfig.retrieveConfig().propertiesOverrideMap().get("grouperIncludeExclude.use"), "true");
106     
107     boolean useGrouperIncludeExclude = useIncludeExcludeOverride || GrouperUtil.propertiesValueBoolean(
108         grouperProperties, "grouperIncludeExclude.use", false);
109 
110     boolean useRequireGroupsOverride = StringUtils.equals(GrouperConfig.retrieveConfig().propertiesOverrideMap().get("grouperIncludeExclude.requireGroups.use"), "true");
111     
112     boolean useRequireGroups = useRequireGroupsOverride || GrouperUtil.propertiesValueBoolean(
113         grouperProperties, "grouperIncludeExclude.requireGroups.use", false);
114 
115     //register the hook
116     if (useGrouperIncludeExclude || useRequireGroups) {
117       registeredSuccess = true;
118       //register this hooks
119       GrouperHooksUtils.addHookManual(GrouperHookType.GROUP_TYPE_TUPLE.getPropertyFileKey(), GroupTypeTupleIncludeExcludeHook.class);
120       GrouperHooksUtils.addHookManual(GrouperHookType.ATTRIBUTE.getPropertyFileKey(), AttributeIncludeExcludeHook.class);
121     }
122     
123     registered = true;
124 
125   }

128    * 
129    * @param attributeName
130    * @return the group name from the config file
131    */
132   public static String groupNameFromAndGroupAttributeName(String attributeName) {
133     int i=0;
134     while (true) {
135       String propertyName = "grouperIncludeExclude.requireGroup.name." + i;
136       String propertyValue = GrouperConfig.retrieveConfig().propertyValueString(propertyName);
137       if (StringUtils.isBlank(propertyValue)) {
138         break;
139       }
140       if (StringUtils.equals(attributeName, propertyValue)) {
141         //this is the right index, get the group name
142         return GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.group." + i);
143       }
144       i++;
145     }
146     throw new RuntimeException("Cant find config entry for andGroup attribute name: " + attributeName 
147         + ", e.g. config name: grouperIncludeExclude.requireGroup.name.{i}");  }

171   /**
172    * @param postInsertBean
173    * @param isInsert
174    */
175   private void groupTypeTupleHelper(HooksGroupTypeTupleBean postInsertBean, boolean isInsert) {
176     boolean useGrouperIncludeExclude = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.use", false);
177     boolean useRequireGroups = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.requireGroups.use", false);
178     
179     //dont do anything if not using this type
180     if (!useGrouperIncludeExclude && !useRequireGroups) {
181       return;
182     }
183 
184     GroupTypeTuple groupTypeTuple = postInsertBean.getGroupTypeTuple();
185 
186     String groupUuid = groupTypeTuple.getGroupUuid();
187 
188     //make sure this is the right type
189     String includeExcludeTypeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.type.name");
190     String requireGroupsTypeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroups.type.name");
191 
192     //there better be a session now!
193     GrouperSession grouperSession = GrouperSession.staticGrouperSession();
194 
195     try {
196       
197       GroupType includeExcludeType = useGrouperIncludeExclude ? GroupTypeFinder.find(includeExcludeTypeName, false) : null;
198       boolean fireHook = (useGrouperIncludeExclude && isInsert)
199         ? StringUtils.equals(groupTypeTuple.getTypeUuid(), 
200             includeExcludeType == null ? null : includeExcludeType.getUuid()) : false;
201 
202       GroupType requireGroupsType = useRequireGroups ? GroupTypeFinder.find(requireGroupsTypeName, false) : null;
203       fireHook = fireHook || (useRequireGroups 
204         ? StringUtils.equals(groupTypeTuple.getTypeUuid(), 
205             requireGroupsType == null ? null : requireGroupsType.getUuid()) : false);
206         
207       //see if a custom group type
208       if (!fireHook) {
209         
210         //#grouperIncludeExclude.requireGroup.name.0 = requireActiveEmployee
211         //#grouperIncludeExclude.requireGroup.attributeOrType.0 = type
212         int i = 0;
213         while (true) {
214           String name = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.name." + i);
215           if (StringUtils.isBlank(name)) {
216             break;
217           }
218           String attributeOrType = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.attributeOrType." + i);
219           if (StringUtils.equalsIgnoreCase("type", attributeOrType)) {
220             GroupType type = GroupTypeFinder.find(name, false);
221             if (type != null) {
222               fireHook = fireHook ||StringUtils.equals(groupTypeTuple.getTypeUuid(), type.getUuid());
223               if (fireHook) {
224                 break;
225               }
226             }
227           }
228           i++;
229         }
230         
231       }
232       if (!fireHook) {
233         return;
234       }
235         
236       //not sure why this would be null, could be a stale state problem if so
237       Group typedGroup = postInsertBean.getGroupTypeTuple().retrieveGroup(true);
238 
239       manageIncludesExcludesAndGroups(grouperSession, typedGroup, includeExcludeTypeName + " changed on group: " + typedGroup.getName());
240       
241     } catch (Exception e) {
242       throw new RuntimeException("Error doing include/exclude on group: " + groupUuid, e);
243     }
244   }

246   /**
247    * convert a system of record extension to an overall extension
248    * @param extension is overall extension or system of record extension
249    * @return the overall extension
250    */
251   public static String convertToOverall(String extension) {
252     //strip system of record suffix (i.e. we are putting the type on the system of record group)
253     if (extension.endsWith(systemOfRecordExtensionSuffix())) {
254       extension = extension.substring(0, 
255           extension.length() - systemOfRecordExtensionSuffix().length());
256     }
257     return extension;
258     
259   }

261   /**
262    * return a set of groups including the one passed in, related to this group.
263    * if the groups arent found, dont worry
264    * i.e. if include/exclude or requireGroups, find related groups
265    * @param group
266    * @return the set of groups
267    */
268   public static Set<Group> relatedGroups(Group group) {
269     Set<Group> groups = new LinkedHashSet<Group>();
270     
271     groups.add(group);
272     
273     String name = group.getName();
274     
275     //baseName is name without suffix if applicable
276     String baseName = name;
277     
278     boolean useGrouperIncludeExclude = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.use", false);
279     
280     if (useGrouperIncludeExclude) {
281       
282       if (name.endsWith(systemOfRecordExtensionSuffix())) {
283         baseName = GrouperUtil.stripSuffix(name, systemOfRecordExtensionSuffix());
284       } else if (name.endsWith(includeExtensionSuffix())) {
285         baseName = GrouperUtil.stripSuffix(name, includeExtensionSuffix());
286       } else if (name.endsWith(includesMinusExcludesExtensionSuffix())) {
287         baseName = GrouperUtil.stripSuffix(name, includesMinusExcludesExtensionSuffix());
288       } else if (name.endsWith(systemOfRecordAndIncludesExtensionSuffix())) {
289         baseName = GrouperUtil.stripSuffix(name, systemOfRecordAndIncludesExtensionSuffix());
290       }
291     }
292     boolean useRequireGroups = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.requireGroups.use", false);
293     
294     //strip off something like _requireGroups15
295     if (useRequireGroups) {
296       Matcher matcher = requireGroupsPattern.matcher(name);
297       if (matcher.matches()) {
298         baseName = matcher.group(1);
299       }
300     }
301 
302     if (LOG.isDebugEnabled()) {
303       LOG.debug("name: " + name + ", baseName: " + baseName);
304     }
305     
306     GrouperSession grouperSession = GrouperSession.staticGrouperSession().internal_getRootSession();
307     
308     Group groupFind = null;
309     
310     //now lets find all the groups
311     if (!StringUtils.equals(name, baseName)) {
312       groupFind = GroupFinder.findByName(grouperSession, baseName,false);
313       if (groupFind != null) {
314         groups.add(groupFind);
315       }
316     }
317 
318     if (useGrouperIncludeExclude) {
319       
320       //system of record
321       groupFind = GroupFinder.findByName(grouperSession, baseName + systemOfRecordExtensionSuffix(),false);
322       if (groupFind != null) {
323         groups.add(groupFind);
324       }
325     
326       //include
327       groupFind = GroupFinder.findByName(grouperSession, baseName + includeExtensionSuffix(),false);
328       
329       if (LOG.isDebugEnabled()) {
330         LOG.debug("Searching for group: " + (baseName + includeExtensionSuffix()) + ", found: " + (groupFind != null));
331       }
332       
333       if (groupFind != null) {
334         groups.add(groupFind);
335       }
336     
337       //exclude
338       groupFind = GroupFinder.findByName(grouperSession, baseName + excludeExtensionSuffix(),false);
339       if (groupFind != null) {
340         groups.add(groupFind);
341       }
342 
343       //include and sor
344       groupFind = GroupFinder.findByName(grouperSession, baseName + includesMinusExcludesExtensionSuffix(),false);
345       if (groupFind != null) {
346         groups.add(groupFind);
347       }
348       
349     }
350     
351     if (useRequireGroups) {
352       
353       //any require groups
354       for (int i=1;i<200;i++) {
355         groupFind = GroupFinder.findByName(grouperSession, baseName + requireGroupsExtensionSuffix(1),false);
356         if (groupFind == null) {
357           break;
358         }
359         groups.add(groupFind);
360       }
361     }
362     
363     return groups;
364   }

380   /**
381    * @param grouperSession
382    * @param typedGroup
383    * @param summaryForLog some string that will be logged to debug...
384    */
385   public static void manageIncludesExcludesAndGroups(GrouperSession grouperSession, Group typedGroup, String summaryForLog) {
386     
387     //make sure this is the right type
388     String includeExcludeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.type.name");
389     String groupTypeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroups.type.name");
390 
391     //make these sorted so that it is more easily testable
392     Set<Group> andGroups = new TreeSet<Group>();
393     
394     boolean includeExclude = false;
395     GroupType includesExcludesType = null;
396     GroupType requireGroupsType = null;
397     try {
398       
399       includesExcludesType = GroupTypeFinder.find(includeExcludeName, false);
400       
401       includeExclude = includesExcludesType != null && typedGroup.hasType(includesExcludesType, false);
402       
403       //if other groups are there, then this is include/exclude.  we dont remove when the checkbox is
404       //unchecked
405       String overallExtension = typedGroup.getExtension();
406       
407       overallExtension = convertToOverall(overallExtension);
408 
409       String stemName = GrouperUtil.parentStemNameFromName(typedGroup.getName());
410       String overallName = stemName + ":" + overallExtension;
411 
412       includeExclude = includeExclude || GroupFinder.findByName(grouperSession, 
413           overallName + includeExtensionSuffix(), false) != null;
414       includeExclude = includeExclude || GroupFinder.findByName(grouperSession, 
415           overallName + excludeExtensionSuffix(), false) != null;
416       
417       requireGroupsType = GroupTypeFinder.find(groupTypeName, false);
418   
419       boolean hasRequireGroupsType = requireGroupsType != null && typedGroup.hasType(requireGroupsType, false);
420       
421       String andGroupsAttributeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroups.attributeName");
422       boolean hasAndGroupsAttributeName = StringUtils.isNotBlank(andGroupsAttributeName);
423       
424       if (hasRequireGroupsType && hasAndGroupsAttributeName) {
425         String groupNames = typedGroup.getAttributeValue(andGroupsAttributeName, false, false);
426         if (!StringUtils.isBlank(groupNames)) {
427           String[] andGroupNames = GrouperUtil.splitTrim(groupNames, ",");
428           for (String andGroupName: andGroupNames) {
429             andGroups.add(GroupFinder.findByName(grouperSession, andGroupName, true));
430           }
431         }
432       }
433       
434       if (hasRequireGroupsType) {
435         Set<AttributeDefName> attrs = GrouperDAOFactory.getFactory().getAttributeDefName().findByAttributeDef(requireGroupsType.internal_getAttributeDefForAttributes().getId());
436         
437         for (AttributeDefName attr : GrouperUtil.nonNull(attrs)) {
438           
439           //see if this is not a custom require group
440           if (hasAndGroupsAttributeName && StringUtils.equals(andGroupsAttributeName, attr.getLegacyAttributeName(true))) {
441             continue;
442           }
443           
444           String valueString = typedGroup.getAttributeValue(attr.getLegacyAttributeName(true), false, false);
445     
446           boolean valueBoolean = GrouperUtil.booleanValue(valueString, false);
447           
448           if (valueBoolean) {
449             String groupName = groupNameFromAndGroupAttributeName(attr.getLegacyAttributeName(true));
450             andGroups.add(GroupFinder.findByName(grouperSession, groupName, true));
451           }
452           
453         }
454       }
455       //now try custom types
456       //#grouperIncludeExclude.requireGroup.name.0 = requireActiveEmployee
457       //#grouperIncludeExclude.requireGroup.attributeOrType.0 = type
458       //#grouperIncludeExclude.requireGroup.group.0 = school:community:activeEmployee
459       int i=0;
460       while (true) {
461         
462         String typeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.name." + i);
463         if (StringUtils.isBlank(typeName)) {
464           break;
465         }
466         String attributeOrType = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.attributeOrType." + i);
467         //just check types
468         if (StringUtils.equalsIgnoreCase("type", attributeOrType)) {
469           String groupName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroup.group." + i);
470           
471           GroupType type = GroupTypeFinder.find(typeName, true);
472           //if the hooked group has this type, then good to go
473           if (typedGroup.getTypesDb().contains(type)) {
474             Group group = GroupFinder.findByName(grouperSession, groupName, true);
475             andGroups.add(group);
476           }
477         }
478         i++;
479       }
480       
481     } catch (SchemaException se) {
482       throw new RuntimeException(se);
483     } catch (GroupNotFoundException gnfe) {
484       throw new RuntimeException(gnfe);
485     }
486     manageIncludesExcludesAndGroups(typedGroup, includeExclude, andGroups, summaryForLog);
487   }

536     if (useTransaction) {
537 
538       HibernateSession.callbackHibernateSession(GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, AuditControl.WILL_NOT_AUDIT, new HibernateHandler() {

541             throws GrouperDAOException {
542           manageIncludesExcludesAndGroupsHelper(typedGroup, isIncludeExclude, 
543               andGroups, calledFromForLog);
544           return null;
545         }
546       });

548       
549     } else {
550

552           andGroups, calledFromForLog);
553       
554     }
555

559   /**
560    * change a typed group into include and exclude group lists and andGroups also
561    * this should be called from manageIncludesExcludesAndGroups
562    * @param typedGroup 
563    * @param isIncludeExclude 
564    * @param andGroups 
565    * @param calledFromForLog summary of where this is coming from for debug log
566    */
567   private static void manageIncludesExcludesAndGroupsHelper(Group typedGroup, boolean isIncludeExclude, 
568       Set<Group> andGroups, String calledFromForLog) {
569 
570     if (Group.deleteOccuring()) {
571       return;
572     }
573     
574     String groupUuid = null;
575     
576     if (LOG.isDebugEnabled()) {
577       StringBuilder summary = new StringBuilder();
578       summary.append("manageIncludeExclude called for group: " + typedGroup.getName() 
579           + ", includeExclude? " + isIncludeExclude + " and " + GrouperUtil.length(andGroups) + " andGroups: ");
580       Iterator<Group> iterator = GrouperUtil.nonNull(andGroups).iterator();
581       boolean first = true;
582       while (iterator.hasNext()) {
583         if (!first) {
584           summary.append(", ");
585         }
586         first = false;
587         summary.append(iterator.next().getExtension());
588       }
589       summary.append(".  ").append(calledFromForLog);
590       LOG.debug(summary.toString());
591      }
592     
593     //there better be a session now!
594     GrouperSession grouperSession = GrouperSession.staticGrouperSession();
595     try {
596       
597       groupUuid = typedGroup.getUuid();
598       
599       String overallExtension = typedGroup.getExtension();
600       
601       String overallDisplayExtension = typedGroup.getDisplayExtension();
602 
603       //strip system of record suffix (i.e. we are putting the type on the system of record group)
604       if (overallExtension.endsWith(systemOfRecordExtensionSuffix())) {
605         overallExtension = overallExtension.substring(0, 
606             overallExtension.length() - systemOfRecordExtensionSuffix().length());
607 
608         //strip system of record suffixes
609         if (overallDisplayExtension.endsWith(systemOfRecordExtensionSuffix())) {
610           overallDisplayExtension = overallDisplayExtension.substring(0, 
611               overallDisplayExtension.length() - systemOfRecordExtensionSuffix().length());
612         }
613         if (overallDisplayExtension.endsWith(systemOfRecordDisplayExtensionSuffix())) {
614           overallDisplayExtension = overallDisplayExtension.substring(0, 
615               overallDisplayExtension.length() - systemOfRecordDisplayExtensionSuffix().length());
616         }
617         
618       }
619       
620       String stemName = GrouperUtil.parentStemNameFromName(typedGroup.getName());
621       String overallName = stemName + ":" + overallExtension;
622 
623       //create the overall group if not exist already
624       Group overallGroup = null;
625       
626       if (StringUtils.equals(typedGroup.getName(), overallName)) {
627         overallGroup = typedGroup;
628       } else {
629         overallGroup = GroupFinder.findByName(grouperSession, overallName, false);
630       }
631       
632       String overallDescription = overallDescription(overallExtension, overallDisplayExtension);
633       if (overallGroup == null) {
634         LOG.debug("Adding overall group: " + overallName);
635         overallGroup = Group.saveGroup(grouperSession, null, null, overallName, overallDisplayExtension, 
636             overallDescription, null, false);
637       } else if (StringUtils.isBlank(overallGroup.getDescription())) {
638         overallGroup.setDescription(overallDescription);
639         overallGroup.store();
640       }
641 
642       //create system of record if not exist
643       String systemOfRecordName = overallName + systemOfRecordExtensionSuffix();
644 
645       Group systemOfRecordGroup = GroupFinder.findByName(grouperSession, systemOfRecordName, false);
646       if (systemOfRecordGroup == null ) {
647         
648         String systemOfRecordDescription = systemOfRecordDescription(overallExtension, overallDisplayExtension);
649         String systemOfRecordDisplayExtension = overallDisplayExtension + systemOfRecordDisplayExtensionSuffix();
650         LOG.debug("Adding system of record group: " + systemOfRecordName);
651         systemOfRecordGroup = Group.saveGroup(grouperSession, null, null, systemOfRecordName, systemOfRecordDisplayExtension, 
652             systemOfRecordDescription, null, true);
653         
654       }
655       
656       //create includes group if not exist
657       String includesName = overallName + includeExtensionSuffix();
658 
659       Group includesGroup = GroupFinder.findByName(grouperSession, includesName, false);
660       if (isIncludeExclude && includesGroup == null ) {
661         
662         String includesDescription = includeDescription(overallExtension, overallDisplayExtension);
663         String includesDisplayExtension = overallDisplayExtension + includeDisplayExtensionSuffix();
664         LOG.debug("Adding includes group: " + includesName);
665         includesGroup = Group.saveGroup(grouperSession, null, null, includesName, includesDisplayExtension, 
666             includesDescription, null, true);
667         
668       }
669       
670       //create excludes group if not exist
671       String excludesName = overallName + excludeExtensionSuffix();
672 
673       Group excludesGroup = GroupFinder.findByName(grouperSession, excludesName, false);
674       if (isIncludeExclude && excludesGroup == null ) {
675         
676         String excludesDescription = excludeDescription(overallExtension, overallDisplayExtension);
677         String excludesDisplayExtension = overallDisplayExtension + excludeDisplayExtensionSuffix();
678         LOG.debug("Adding excludes group: " + excludesName);
679         excludesGroup = Group.saveGroup(grouperSession, null, null, excludesName, excludesDisplayExtension, 
680             excludesDescription, null, true);
681         
682       }
683       
684       //create excludes group if not exist
685       String systemOfRecordAndIncludesName = overallName + systemOfRecordAndIncludesExtensionSuffix();
686 
687       Group systemOfRecordAndIncludesGroup = GroupFinder.findByName(grouperSession, systemOfRecordAndIncludesName, false);
688       if (isIncludeExclude && systemOfRecordAndIncludesGroup == null ) {
689         
690         String systemOfRecordAndIncludesDescription = systemOfRecordAndIncludesDescription(overallExtension, overallDisplayExtension);
691         String systemOfRecordAndIncludesDisplayExtension = overallDisplayExtension + systemOfRecordAndIncludesDisplayExtensionSuffix();
692         LOG.debug("Adding system of record group: " + systemOfRecordAndIncludesName);
693         systemOfRecordAndIncludesGroup = Group.saveGroup(grouperSession, null, null, systemOfRecordAndIncludesName, systemOfRecordAndIncludesDisplayExtension, 
694             systemOfRecordAndIncludesDescription, null, true);
695         
696       }
697 
698       //dont need to do this if not include/exclude
699       if (isIncludeExclude) {
700         //lets make sure systemOfRecord and includes are members
701         systemOfRecordAndIncludesGroup.addMember(SubjectFinder.findById(systemOfRecordGroup.getUuid(), true), false);
702         systemOfRecordAndIncludesGroup.addMember(SubjectFinder.findById(includesGroup.getUuid(), true), false);
703       }
704       
705       //at this point, the main group must have no members
706       if (overallGroup.getComposite(false) == null) {
707         
708         //lets get all the existing members and add to the system of record group
709         //lets do this if the overall group has members, and if the system of record group
710         //doesnt have any members yet
711         Set<Member> existingOverallMembers = overallGroup.getImmediateMembers();
712         
713         if (existingOverallMembers.size() > 0) {
714           
715           //note: lists arent covered here...
716           for (Member member : existingOverallMembers) {
717             LOG.debug("Removing member from overall, and adding to system of record: " 
718                 + member.getSubjectSourceIdDb() + " - " + member.getSubjectIdDb());
719             //dont add system of record to itself
720             if (!StringUtils.equals(member.getSubjectIdDb(), systemOfRecordGroup.getUuid())) {
721               systemOfRecordGroup.addMember(member.getSubject(), false);
722             }
723             overallGroup.deleteMember(member, false);
724           }
725         }
726 
727       }
728       int andGroupsLength = GrouperUtil.length(andGroups);
729 
730       //################################################
731       //lets remove unneeded groups
732       int i=andGroupsLength;
733       int highestIndexToDelete = -1;
734       while(true) {
735         
736         String requireGroupsSuffixExtension = requireGroupsExtensionSuffix(i);
737         String requireGroupsName = overallName + requireGroupsSuffixExtension;
738         Group requireGroupsSuffix = GroupFinder.findByName(grouperSession, requireGroupsName, false);
739         if (requireGroupsSuffix != null) {
740           highestIndexToDelete = i;
741         }
742         if (requireGroupsSuffix == null && i > 10) {
743           break;
744         }
745         i++;
746       }
747 
748       String includesMinusExcludesName = overallName + includesMinusExcludesExtensionSuffix();
749       Group includesMinusExcludesGroup = GroupFinder.findByName(grouperSession, includesMinusExcludesName, false);
750 
751       //#############################################
752       //make final structure
753       if (andGroupsLength == 0) {
754             
755         if (isIncludeExclude) {
756           //now the overall group needs to be a composite complement
757           overallGroup.assignCompositeMember(CompositeType.COMPLEMENT, systemOfRecordAndIncludesGroup, excludesGroup);
758         } else {
759           //make sure not a composite group from before
760           if (overallGroup.hasComposite()) {
761             overallGroup.deleteCompositeMember();
762           }
763           overallGroup.addMember(SubjectFinder.findById(systemOfRecordGroup.getUuid(), true), false);
764           LOG.debug("Adding system of record group (" + systemOfRecordName + ") to the overall group: "  
765               + overallGroup.getName());
766 
767         }
768         
769       } else {
770 
771         //######################################
772         //includesMinusExcludes is composite complement: systemOfRecordAndIncludesGroup minus excludesGroup
773         includesMinusExcludesGroup = GroupFinder.findByName(grouperSession, includesMinusExcludesName, false);
774         //if no groups we dont need this one
775         if (isIncludeExclude && includesMinusExcludesGroup == null ) {
776 
777           String includesMinusExcludesGroupDescription = systemOfRecordAndIncludesDescription(overallExtension, overallDisplayExtension);
778           String includesMinusExcludesGroupDisplayExtension = overallDisplayExtension + includesMinusExcludesDisplayExtensionSuffix();
779           includesMinusExcludesGroup = Group.saveGroup(grouperSession, null, null, includesMinusExcludesName, 
780               includesMinusExcludesGroupDisplayExtension, 
781               includesMinusExcludesGroupDescription, null, true);
782         }
783         if (isIncludeExclude) {
784           includesMinusExcludesGroup.assignCompositeMember(CompositeType.COMPLEMENT, systemOfRecordAndIncludesGroup, excludesGroup);
785         }
786 
787         //for one andGroup, structure will be
788         //includesMinusExcludes is composite complement: systemOfRecordAndIncludesGroup minus excludesGroup
789         //overallGroup is composite complement: includesMinusExcludes minus aStem:activeEmployee
790         Group[] andGroupsArray = GrouperUtil.toArray(andGroups, Group.class);
791         if (andGroups.size() == 1) {
792           if (isIncludeExclude) {
793             overallGroup.assignCompositeMember(CompositeType.INTERSECTION, includesMinusExcludesGroup, andGroupsArray[0]);
794           } else {
795             overallGroup.assignCompositeMember(CompositeType.INTERSECTION, systemOfRecordGroup, andGroupsArray[0]);
796           }
797         } else {
798         
799           Group previousGroup = null;
800           
801           //for more than one andGroup, structure will be
802           //includeExcludeMinusAnd2 is composite complement: includesMinusExcludes minus aStem:activeEmployee
803           //includeExcludeMinusAnd1 is composite complement: includeExcludeMinusAnd2 minus aStem:anotherGroup
804           //overallGroup is composite complement: includeExcludeMinusAnd1 minus aStem:yetAnotherGroup
805           int arrayIndex = 0;
806           for (i=andGroupsLength-1;i>=1;i--) {
807             String requireGroupsName = overallName + requireGroupsExtensionSuffix(i);
808             Group requireGroup = GroupFinder.findByName(grouperSession, requireGroupsName, false);
809             //if no groups we dont need this one
810             if (requireGroup == null ) {
811   
812               String requireGroupsDescription = requireGroupsDescription(i, overallExtension, overallDisplayExtension);
813               String requireGroupsDisplayExtension = overallDisplayExtension + requireGroupsDisplayExtensionSuffix(i);
814               requireGroup = Group.saveGroup(grouperSession, null, null, requireGroupsName, requireGroupsDisplayExtension, 
815                   requireGroupsDescription, null, true);
816               LOG.debug("Adding requireGroups group: " + requireGroup.getExtension());
817             }
818             Group leftGroup = i == andGroupsLength-1 
819               ? (isIncludeExclude ? includesMinusExcludesGroup : systemOfRecordGroup) : previousGroup;
820             requireGroup.assignCompositeMember(CompositeType.INTERSECTION, leftGroup, andGroupsArray[arrayIndex]);
821             previousGroup = requireGroup;
822             arrayIndex++;
823           }
824 
825           //cant assign if has one
826           if (overallGroup.hasComposite()) {
827             overallGroup.deleteCompositeMember();
828           }
829           
830           overallGroup.assignCompositeMember(CompositeType.INTERSECTION, previousGroup, andGroupsArray[andGroupsLength-1]);
831             
832         }
833         
834       }
835       
836       for (i=andGroupsLength;i<=highestIndexToDelete;i++) {
837         String requireGroupsSuffixExtension = requireGroupsExtensionSuffix(i);
838         String requireGroupsName = overallName + requireGroupsSuffixExtension;
839         Group includeExcludeMinusAndGroup = GroupFinder.findByName(grouperSession, requireGroupsName, false);
840         if (includeExcludeMinusAndGroup != null) {
841           LOG.debug("Deleting unneeded group: " + includeExcludeMinusAndGroup.getName());
842           includeExcludeMinusAndGroup.delete();
843         }
844       }
845 
846       //if no groups we dont need this one
847       if (andGroupsLength == 0 || !isIncludeExclude) {
848         if (includesMinusExcludesGroup != null ) {
849           LOG.debug("Deleting unneeded group: " + includesMinusExcludesGroup.getName());
850           includesMinusExcludesGroup.delete();
851         }
852       }
853       if (!isIncludeExclude && systemOfRecordAndIncludesGroup != null ) {
854           LOG.debug("Deleting unneeded group: " + systemOfRecordAndIncludesGroup.getName());
855           systemOfRecordAndIncludesGroup.delete();
856       }
857 
858     } catch (Exception e) {
859       throw new RuntimeException("Error doing include/exclude on group: " + groupUuid, e);
860     }
861   }

875    * @return true if include/exclude

877   public static boolean nameIsIncludeExcludeRequireGroup(String groupName) {
878     boolean useGrouperIncludeExclude = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.use", false);
879 
880     //check include exclude
881     if (useGrouperIncludeExclude) {
882       
883       if (groupName.endsWith(excludeExtensionSuffix())
884           || groupName.endsWith(includeExtensionSuffix())
885           || groupName.endsWith(includesMinusExcludesExtensionSuffix())
886           || groupName.endsWith(systemOfRecordAndIncludesExtensionSuffix())
887           || groupName.endsWith(systemOfRecordExtensionSuffix())) {
888         return true;

890

895     if (useRequireGroups) {
896       for (int i=1;i<=10;i++) {
897         if (groupName.endsWith(requireGroupsExtensionSuffix(i))) {
898           return true;
899         }
900       }
901     }
902     return false;
903   }

974   /**
975    * 
976    * @param grouperSession 
977    * @param groupName if the overall or system of record group
978    * @param reason if you want a reason passed back
979    * @param saveIncludesExcludesIfMembers true if not delete only the includes group and excludes group
980    * if they have members.  false, delete anyway
981    * @return the number of groups removed
982    */
983   public static int deleteGroupsIfNotUsed(GrouperSession grouperSession, 
984       String groupName, StringBuilder reason, boolean saveIncludesExcludesIfMembers) {
985     if (reason == null) {
986       reason = new StringBuilder();
987     }
988     int count = 0;
989     try {
990       String overallName = convertToOverall(groupName);
991       Group group = GroupFinder.findByName(grouperSession, overallName, false);
992       if (group != null) {
993         
994         Member member = MemberFinder.internal_findBySubject(SubjectFinder.findByIdAndSource(group.getUuid(),
995             SubjectFinder.internal_getGSA().getId(), true), null, false);
996         
997         if (member != null) {
998           //get any membership
999           Set<Membership> memberships = GrouperDAOFactory.getFactory()
1000             .getMembership().findAllImmediateByMember(member.getUuid(), false);
1001           if (!GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(
1002               "loader.sqlTable.likeString.removeGroupIfMemberOfAnotherGroup", false) && memberships.size() > 0) {
1003             String message = "Not deleting group: " + overallName + " since used in " 
1004               + memberships.size() + " immediate memberships";
1005             LOG.debug(message);
1006             reason.append(message);
1007             return 0;
1008           }
1009           Set<Composite> composites = CompositeFinder.findAsFactor(group);
1010           if (composites.size() > 0) {
1011             String message = "Not deleting group: " + overallName + " since used in " 
1012               + composites.size() + " composites";
1013             LOG.debug(message);
1014             reason.append(message);
1015             return 0;
1016           }
1017         } 
1018         
1019         //we can delete the overall group
1020         group.delete();
1021         count++;
1022         
1023       }
1024       
1025       //now lets pick apart the composites and such
1026       for (int i=1;i<10;i++) {
1027         String requireGroupsName = overallName + requireGroupsExtensionSuffix(i);
1028         group = GroupFinder.findByName(grouperSession, requireGroupsName, false);
1029         if (group != null) {
1030           String logString = "deleting group: " + requireGroupsName;
1031           GrouperUtil.append(reason, ", ", logString);
1032           LOG.debug(logString);
1033           group.delete();
1034           count++;
1035         }
1036       }
1037       
1038       {
1039         //now if the includesMinusExcludes
1040         String includesMinuxExcludesName = overallName + includesMinusExcludesExtensionSuffix();
1041         group = GroupFinder.findByName(grouperSession, 
1042             includesMinuxExcludesName, false);
1043   
1044         if (group != null) {
1045           String logString = "deleting group: " + includesMinuxExcludesName;
1046           GrouperUtil.append(reason, ", ", logString);
1047           LOG.debug(logString);
1048           group.delete();
1049           count++;
1050         }
1051       }
1052       
1053       {
1054         //now if the includesMinusExcludes
1055         String sorAndIncludesName = overallName + systemOfRecordAndIncludesExtensionSuffix();
1056         group = GroupFinder.findByName(grouperSession, 
1057             sorAndIncludesName, false);
1058   
1059         if (group != null) {
1060           String logString = "deleting group: " + sorAndIncludesName;
1061           GrouperUtil.append(reason, ", ", logString);
1062           LOG.debug(logString);
1063           group.delete();
1064           count++;
1065         }
1066       }
1067       
1068       {
1069         //now excludes if no members
1070         String excludesName = overallName + excludeExtensionSuffix();
1071         group = GroupFinder.findByName(grouperSession, 
1072             excludesName, false);
1073   
1074         if (group != null) {
1075           Set<Member> members = group.getMembers();
1076           if (members.size() > 0 && saveIncludesExcludesIfMembers) {
1077             String logString = "not deleting group: " + excludesName + " since has " + members.size() + " members";
1078             GrouperUtil.append(reason, ", ", logString);
1079             LOG.debug(logString);
1080           } else {
1081             String logString = "deleting group: " + excludesName;
1082             GrouperUtil.append(reason, ", ", logString);
1083             LOG.debug(logString);
1084             group.delete();
1085             count++;
1086           }
1087         }
1088       }
1089       
1090       {
1091         //now includes if no members
1092         String includesName = overallName + includeExtensionSuffix();
1093         group = GroupFinder.findByName(grouperSession, 
1094             includesName, false);
1095   
1096         if (group != null) {
1097           Set<Member> members = group.getMembers();
1098           if (members.size() > 0 && saveIncludesExcludesIfMembers) {
1099             String logString = "not deleting group: " + includesName + " since has " + members.size() + " members";
1100             GrouperUtil.append(reason, ", ", logString);
1101             LOG.debug(logString);
1102           } else {
1103             String logString = "deleting group: " + includesName;
1104             GrouperUtil.append(reason, ", ", logString);
1105             LOG.debug(logString);
1106             group.delete();
1107             count++;
1108           }
1109         }
1110       }
1111       
1112       {
1113         //finally system of record
1114         String systemOfRecordName = overallName + systemOfRecordExtensionSuffix();
1115         group = GroupFinder.findByName(grouperSession, 
1116             systemOfRecordName, false);
1117   
1118         if (group != null) {
1119           String logString = "deleting group: " + systemOfRecordName;
1120           GrouperUtil.append(reason, ", ", logString);
1121           LOG.debug(logString);
1122           group.delete();
1123           count++;
1124         }
1125       }      
1126       GrouperUtil.append(reason, ", ", "deleted " + count + " groups");
1127       
1128       return count;
1129     } catch (Exception e) {
1130       throw new RuntimeException("Problem deleting groups for name: " + groupName 
1131           + ", though did delete " + count + " groups", e);
1132     }
1133   }
