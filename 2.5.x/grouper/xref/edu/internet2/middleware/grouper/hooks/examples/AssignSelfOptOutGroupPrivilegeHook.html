31  /**
32   * <pre>
33   * If you want the group or groups in a folder to allow opt out to anyone in group, 
34   * then configure this hook in grouper.properties:
35   * 
36   * hooks.group.class=edu.internet2.middleware.grouper.hooks.examples.AssignSelfOptOutGroupPrivilegeHook
37   * hooks.membership.class=edu.internet2.middleware.grouper.hooks.examples.AssignSelfOptOutMembershipPrivilegeRevocationVetoHook
38   * 
39   * configure an attribute to assign to groups or folders in grouper.properties
40   * 
41   * grouper.optOutRequired.attributeDefName = a:b:c:assignOptOut
42   * 
43   * https://bugs.internet2.edu/jira/browse/GRP-1197
44   * 
45   * grouperSession = GrouperSession.startRootSession(); 
46   * String attributeFolderName = "a:b:c"; 
47   * attributeDef = new AttributeDefSave(grouperSession).assignName(attributeFolderName + ":assignOptOutDef").assignToStem(true).assignAttributeDefType(AttributeDefType.attr).assignCreateParentStemsIfNotExist(true).assignValueType(AttributeDefValueType.marker).save();
48   * attributeDef.getAttributeDefActionDelegate().configureActionList("assign"); 
49   * attributeDefName = new AttributeDefNameSave(grouperSession, attributeDef).assignName(attributeFolderName + ":assignOptOut").assignCreateParentStemsIfNotExist(true).save(); 
50   * 
51   * Test it out
52   * 
53   * stem = new StemSave(grouperSession).assignName("j:k").assignCreateParentStemsIfNotExist(true).save();
54   * stem.getAttributeDelegate().assignAttribute(attributeDefName);
55   * groupSub = new GroupSave(grouperSession).assignName("j:k:l:m").assignCreateParentStemsIfNotExist(true).save();
56   *
57   * </pre>
58   */
59  public class AssignSelfOptOutGroupPrivilegeHook extends GroupHooks {
60  
61    /** logger */
62    private static final Log logger = GrouperUtil.getLog(AssignSelfOptOutGroupPrivilegeHook.class);
63  
64    /**
65     * 
66     * @see edu.internet2.middleware.grouper.hooks.GroupHooks#groupPostCommitInsert(edu.internet2.middleware.grouper.hooks.beans.HooksContext, edu.internet2.middleware.grouper.hooks.beans.HooksGroupBean)
67     */
68    @Override
69    public void groupPostCommitInsert(HooksContext hooksContext,
70        HooksGroupBean postCommitInsertBean) {
71      
72      //only care about this if not grouper loader
73      if (GrouperContextTypeBuiltIn.GROUPER_LOADER.equals(hooksContext
74          .getGrouperContextType())) {
75        return;
76      }
77  
78      Group thisGroup = GroupFinder.findByUuid(GrouperSession.startRootSession(),
79          postCommitInsertBean.getGroup().getId(), false);
80      if (logger.isDebugEnabled()) {
81        logger.debug("The Group: " + thisGroup);
82        logger.debug("Group's subject " + thisGroup.toSubject());
83      }
84      //assign this attribute to a group or folder where opt out is required
85      String optoutRequiredAttributeDefName = GrouperConfig.retrieveConfig().propertyValueStringRequired(
86          AssignSelfOptOutMembershipPrivilegeRevocationVetoHook.GROUPER_OPT_OUT_REQUIRED_ATTRIBUTE_DEF_NAME);
87      
88      if (thisGroup.getAttributeDelegate().hasAttributeOrAncestorHasAttribute(optoutRequiredAttributeDefName, false)) {
89  
90        //assign opt out priv
91        thisGroup.grantPriv(thisGroup.toSubject(), AccessPrivilege.OPTOUT, false);
92      }
93    }
94  }
