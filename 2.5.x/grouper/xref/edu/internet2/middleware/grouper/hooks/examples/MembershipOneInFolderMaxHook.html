146   /**
147    * cache if stem name has the membership one attribute
148    */
149   private static GrouperCache<String, Boolean> stemHasMembershipOneAttribute = new GrouperCache(
150       MembershipOneInFolderMaxHook.class.getName() + ".membershipOneAttribute", 5000, false, 60, 60, false);
151   //TODO remove defaults in 2.3+
152   
153   /**
154    * @see edu.internet2.middleware.grouper.hooks.MembershipHooks#membershipPreAddMember(edu.internet2.middleware.grouper.hooks.beans.HooksContext, edu.internet2.middleware.grouper.hooks.beans.HooksMembershipChangeBean)
155    */
156   @Override
157   public void membershipPreAddMember(HooksContext hooksContext,
158       final HooksMembershipChangeBean preAddMemberBean) {
159 
160     if (!FieldType.LIST.equals(preAddMemberBean.getField().getType())) {
161       return;
162     }
163 
164     if (GrouperCheckConfig.inCheckConfig || !GrouperStartup.isFinishedStartupSuccessfully()) {
165       return;
166     }
167 
168     initObjectsOnce(false);
169 
170     final Set<Group> problemGroups = new HashSet<Group>();
171 
172     GrouperSession.internal_callbackRootGrouperSession(new GrouperSessionHandler() {
173       
174       public Object callback(GrouperSession grouperSession) throws GrouperSessionException {
175         
176         
177         //lets see if we are limiting for this membership
178         Group group = preAddMemberBean.getGroup();
179         
180         String parentStemName = group.getParentStemName();
181         
182         Boolean membershipOne = stemHasMembershipOneAttribute.get(parentStemName);
183 
184         //if checking, see what memberships are in this folder
185         Stem parentStem = null;
186 
187         if (membershipOne == null) {
188 
189           if (parentStem == null) {
190             parentStem = group.getParentStem();
191           }
192           
193           //if not in cache that we check this stem, add it
194           AttributeDefName membershipOneAttributeDefName = membershipOneInFolderAttributeDefName();
195           
196           membershipOne = parentStem.getAttributeDelegate().hasAttribute(membershipOneAttributeDefName);
197 
198           stemHasMembershipOneAttribute.put(parentStemName, membershipOne);
199         }
200           
201         if (membershipOne) {
202 
203           if (parentStem == null) {
204             parentStem = group.getParentStem();
205           }
206 
207           Set<MembershipSubjectContainer> membershipSubjectContainers = new MembershipFinder()
208             .assignStem(parentStem).assignStemScope(Scope.ONE).assignField(Group.getDefaultList())
209             .assignMemberIds(GrouperUtil.toSet(preAddMemberBean.getMember().getId()))
210             .assignEnabled(true).findMembershipResult().getMembershipSubjectContainers();
211           
212           for (MembershipSubjectContainer membershipSubjectContainer : GrouperUtil.nonNull(membershipSubjectContainers)) {
213 
214             Group currentGroup = membershipSubjectContainer.getGroupOwner();
215             if (!StringUtils.equals(currentGroup.getName(), group.getName())) {
216               problemGroups.add(currentGroup);
217             }
218             
219           }
220           
221         }
222         return null;
223       }
224     });
225     
226 
227     //now we are out of the GrouperSystem, and we need to remove memberships
228     //do this with the privileges of the calling user.  if they cant, throw a veto
229     for (Group currentGroup : problemGroups) {
230       try {
231         currentGroup.deleteMember(preAddMemberBean.getMember());
232       } catch (Exception e) {
233         if (LOG.isDebugEnabled()) {
234           LOG.debug("Problem auto removing from " + currentGroup.getName() + ", " + GrouperUtil.subjectToString(GrouperSession.staticGrouperSession().getSubject()), e);
235         }
236         throw new HookVeto(HOOK_VETO_MEMBERSHIP_ONE_IN_FOLDER_CANT_DELETE_MEMBER, "This folder only allows one membership in any group, "
237             + "but there was a problem removing from " + currentGroup.getName() + ", do you have privileges to remove memberships there?");
238       }
239     }
240     
241   }

243   /**
244    * @return attribute def name for this hook
245    */
246   public static AttributeDefName membershipOneInFolderAttributeDefName() {
247     initObjectsOnce(false);
248     return AttributeDefNameFinder.findByName(membershipOneFolderStemName() + ":" + membershipOneFolderExtensionOfAttributeDefName, true);
249   }

251   /**
252    * pass in the stem and assign attribute and clear cache
253    * @param stem
254    */
255   public static void assignMembershipOneInFolderAttributeDefName(Stem stem) {
256     
257     stem.getAttributeDelegate().assignAttribute(MembershipOneInFolderMaxHook.membershipOneInFolderAttributeDefName());
258     stemHasMembershipOneAttribute.clear();
259   }
