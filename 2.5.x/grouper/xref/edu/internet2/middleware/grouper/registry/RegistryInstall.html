59   * Install the Groups Registry.  When there is a newly created (or truncated database),
60   * this will put the base records that grouper needs to operate (e.g. root stem)
61   * <p/>
62   * @author  blair christensen.

69     * @param args
70     */
71    public static void main(String[] args) {
72      install();
73    }

75    /**
76     * install the registry if it is not already installeds
77     */
78    public static void install() {
79      // Install group types, fields and privileges
80      try {
81        GrouperSession s = GrouperSession.start( SubjectFinder.findRootSubject(), false );
82        
83        GrouperSession.callbackGrouperSession(s, new GrouperSessionHandler() {
84  
85          public Object callback(GrouperSession grouperSession)
86              throws GrouperSessionException {
87            //make sure the ALL subject is created before the change log happens
88            MemberFinder.findBySubject(grouperSession, SubjectFinder.findAllSubject(), true);
89            boolean changed = false;
90            try {
91              changed = changed | _installFieldsAndTypes(grouperSession);
92              changed = changed | _installGroupsAndStems(grouperSession);
93  
94              if (changed) {
95                ChangeLogTempToEntity.convertRecords();
96              }
97  
98              if (changed && LOG.isWarnEnabled()) {
99                LOG.warn("Registry was initted (default fields, types, stem, etc inserted)");
100             }
101           } catch (Exception e) {
102             throw new GrouperSessionException(e);
103           }
104           return null;
105         }
106         
107       });
108       
109       s.stop();
110     }
111     catch (Throwable throwable) {
112       //unwrap exception
113       if (throwable instanceof GrouperSessionException && throwable.getCause() != null) {
114         throwable = throwable.getCause();
115       }
116       String msg = "unable to initialize registry: " + throwable.getMessage();
117       if (GrouperStartup.logErrorStatic) {
118         LOG.fatal(msg, throwable);
119       } else {
120         LOG.debug(msg, throwable);
121       }
122       throw new GrouperException(msg, throwable);
123     }
124     GrouperCacheUtils.clearAllCaches();
125 
126   }

132   /**
133    * 
134    * @param s
135    * @return true if any changes were made
136    * @throws InsufficientPrivilegeException
137    * @throws SchemaException
138    */
139   private static boolean _installFieldsAndTypes(GrouperSession s) 
140     throws  InsufficientPrivilegeException,
141             SchemaException {
142     //note, no need for GrouperSession inverse of control
143     boolean changed = false;
144     boolean[] changedArray = {false};
145 
146     // base lists
147     Field.internal_addField( s, "members", FieldType.LIST, AccessPrivilege.READ, 
148         AccessPrivilege.UPDATE, false, false , changedArray, null);
149     changed = changed || changedArray[0];
150     // reserve access privs
151     Field.internal_addField( s, Field.FIELD_NAME_ADMINS,   FieldType.ACCESS, AccessPrivilege.ADMIN,  
152         AccessPrivilege.ADMIN,  false, false , changedArray, null);
153     changed = changed || changedArray[0];
154     Field.internal_addField( s, Field.FIELD_NAME_OPTOUTS,  FieldType.ACCESS, AccessPrivilege.UPDATE,
155         AccessPrivilege.UPDATE, false, false , changedArray, null);
156     changed = changed || changedArray[0];
157     Field.internal_addField( s, Field.FIELD_NAME_OPTINS,   FieldType.ACCESS, AccessPrivilege.UPDATE, 
158         AccessPrivilege.UPDATE, false, false , changedArray, null);
159     changed = changed || changedArray[0];
160     Field.internal_addField( s, Field.FIELD_NAME_READERS,  FieldType.ACCESS, AccessPrivilege.ADMIN,  
161         AccessPrivilege.ADMIN,  false, false , changedArray, null);
162     changed = changed || changedArray[0];
163     Field.internal_addField( s, Field.FIELD_NAME_UPDATERS, FieldType.ACCESS, AccessPrivilege.ADMIN,  
164         AccessPrivilege.ADMIN,  false, false, changedArray , null);
165     changed = changed || changedArray[0];
166     Field.internal_addField( s, Field.FIELD_NAME_VIEWERS,  FieldType.ACCESS, AccessPrivilege.ADMIN,  
167         AccessPrivilege.ADMIN,  false, false, changedArray, null );
168     changed = changed || changedArray[0];
169     Field.internal_addField( s, Field.FIELD_NAME_GROUP_ATTR_READERS,  FieldType.ACCESS, AccessPrivilege.ADMIN,  
170         AccessPrivilege.ADMIN,  false, false, changedArray, null );
171     changed = changed || changedArray[0];
172     Field.internal_addField( s, Field.FIELD_NAME_GROUP_ATTR_UPDATERS,  FieldType.ACCESS, AccessPrivilege.ADMIN,  
173         AccessPrivilege.ADMIN,  false, false, changedArray, null );
174     changed = changed || changedArray[0];
175 
176     // reserve attributeDef privs
177     Field.internal_addField( s, "attrAdmins",   FieldType.ATTRIBUTE_DEF, AttributeDefPrivilege.ATTR_ADMIN,  
178         AttributeDefPrivilege.ATTR_ADMIN,  false, false , changedArray, null);
179     changed = changed || changedArray[0];
180     Field.internal_addField( s, "attrOptouts",  FieldType.ATTRIBUTE_DEF, AttributeDefPrivilege.ATTR_UPDATE,
181         AttributeDefPrivilege.ATTR_UPDATE, false, false , changedArray, null);
182     changed = changed || changedArray[0];
183     Field.internal_addField( s, "attrOptins",   FieldType.ATTRIBUTE_DEF, AttributeDefPrivilege.ATTR_UPDATE, 
184         AttributeDefPrivilege.ATTR_UPDATE, false, false , changedArray, null);
185     changed = changed || changedArray[0];
186     Field.internal_addField( s, "attrReaders",  FieldType.ATTRIBUTE_DEF, AttributeDefPrivilege.ATTR_ADMIN,  
187         AttributeDefPrivilege.ATTR_ADMIN,  false, false , changedArray, null);
188     changed = changed || changedArray[0];
189     Field.internal_addField( s, "attrUpdaters", FieldType.ATTRIBUTE_DEF, AttributeDefPrivilege.ATTR_ADMIN,  
190         AttributeDefPrivilege.ATTR_ADMIN,  false, false, changedArray , null);
191     changed = changed || changedArray[0];
192     Field.internal_addField( s, "attrViewers",  FieldType.ATTRIBUTE_DEF, AttributeDefPrivilege.ATTR_ADMIN,  
193         AttributeDefPrivilege.ATTR_ADMIN,  false, false, changedArray, null );
194     changed = changed || changedArray[0];
195     Field.internal_addField( s, "attrDefAttrReaders",  FieldType.ATTRIBUTE_DEF, AttributeDefPrivilege.ATTR_ADMIN,  
196         AttributeDefPrivilege.ATTR_ADMIN,  false, false, changedArray, null );
197     changed = changed || changedArray[0];
198     Field.internal_addField( s, "attrDefAttrUpdaters",  FieldType.ATTRIBUTE_DEF, AttributeDefPrivilege.ATTR_ADMIN,  
199         AttributeDefPrivilege.ATTR_ADMIN,  false, false, changedArray, null );
200     changed = changed || changedArray[0];
201 
202     // reserve naming privs
203     Field.internal_addField( s, Field.FIELD_NAME_CREATORS, FieldType.NAMING, NamingPrivilege.STEM_ADMIN, 
204         NamingPrivilege.STEM_ADMIN, false, false, changedArray, null);
205     changed = changed || changedArray[0];
206     Field.internal_addField( s, Field.FIELD_NAME_STEM_ADMINS, FieldType.NAMING, 
207         NamingPrivilege.STEM_ADMIN, NamingPrivilege.STEM_ADMIN, false, false, changedArray, null);
208     changed = changed || changedArray[0];
209     Field.internal_addField( s, Field.FIELD_NAME_STEM_ATTR_READERS, FieldType.NAMING, 
210         NamingPrivilege.STEM_ADMIN, NamingPrivilege.STEM_ADMIN, false, false, changedArray, null);
211     changed = changed || changedArray[0];
212     Field.internal_addField( s, Field.FIELD_NAME_STEM_ATTR_UPDATERS, FieldType.NAMING, 
213         NamingPrivilege.STEM_ADMIN, NamingPrivilege.STEM_ADMIN, false, false, changedArray, null);
214     changed = changed || changedArray[0];
215     FieldFinder.clearCache();
216     GroupTypeFinder.clearCache();
217     return changed;
218   } // private static void _installFieldsAndTypes(s)

220   /**
221    * 
222    * @param s
223    * @return true if there were changes, false if not
224    * @throws GrouperException
225    */
226   private static boolean _installGroupsAndStems(GrouperSession s) 
227     throws  GrouperException
228   {
229     //note, no need for GrouperSession inverse of control
230     boolean[] changed = {false};
231     Stem.internal_addRootStem(s, changed);
232     return changed[0];
233   } // private static void _installGroupsAndStems(s)
