216   /**
217    * <pre>
218    * create or update a external subject.
219    * 
220    * Steps:
221    * 
222    * 1. Find the external subject by identifierToEdit
223    * 2. Internally set all the fields of the stem (no need to reset if already the same)
224    * 3. Store the external subject (insert or update) if needed
225    * 4. Return the external subject object
226    * 
227    * This runs in a tx so that if part of it fails the whole thing fails, and potentially the outer
228    * transaction too
229    * </pre>
230    * @return the external subject
231    * @throws InsufficientPrivilegeException
232    */
233   public ExternalSubject save() {
234 
235     //help with incomplete entries
236     if (StringUtils.isBlank(this.identifier)) {
237       this.identifier = this.identifierToEdit;
238     }
239 
240     //get from uuid since could be a rename
241     if (StringUtils.isBlank(this.identifierToEdit) && !StringUtils.isBlank(this.uuid)) {
242       ExternalSubject externalSubject = HibernateSession.byObjectStatic().load(ExternalSubject.class, this.uuid);
243       if (externalSubject != null) {
244         this.identifierToEdit = externalSubject.getUuid();
245       }
246     }
247     
248     if (StringUtils.isBlank(this.identifierToEdit)) {
249       this.identifierToEdit = this.identifier;
250     }
251     
252     //default to insert or update
253     ExternalSubjectSave.this.saveMode = (SaveMode)ObjectUtils.defaultIfNull(ExternalSubjectSave.this.saveMode, SaveMode.INSERT_OR_UPDATE);
254     final SaveMode SAVE_MODE = ExternalSubjectSave.this.saveMode;
255 
256     try {
257       //do this in a transaction
258       ExternalSubject externalSubject = (ExternalSubject)GrouperTransaction.callbackGrouperTransaction(new GrouperTransactionHandler() {
259   
260         @SuppressWarnings("cast")
261         @Override
262         public Object callback(GrouperTransaction grouperTransaction)
263             throws GrouperDAOException {
264           
265           grouperTransaction.setCachingEnabled(false);
266           
267           return (ExternalSubject)GrouperSession.callbackGrouperSession(ExternalSubjectSave.this.grouperSession, new GrouperSessionHandler() {
268 
269               @Override
270               public Object callback(GrouperSession theGrouperSession)
271                   throws GrouperSessionException {
272                 
273                 String identifierForError = GrouperUtil.defaultIfBlank(ExternalSubjectSave.this.identifierToEdit, ExternalSubjectSave.this.identifier);
274                 
275                 ExternalSubject theExternalSubject = null;
276 
277                 //see if update
278                 boolean isUpdate = SAVE_MODE.isUpdate(ExternalSubjectSave.this.identifierToEdit, ExternalSubjectSave.this.identifier);
279 
280                 theExternalSubject = GrouperDAOFactory.getFactory().getExternalSubject().findByIdentifier(ExternalSubjectSave.this.identifierToEdit, false, null);
281 
282                 if (theExternalSubject != null) {
283                   //while we are here, make sure uuid's match if passed in
284                   if (!StringUtils.isBlank(ExternalSubjectSave.this.uuid) && !StringUtils.equals(ExternalSubjectSave.this.uuid, theExternalSubject.getUuid())) {
285                     throw new RuntimeException("UUID external subject changes are not supported: new: " + ExternalSubjectSave.this.uuid + ", old: " 
286                         + theExternalSubject.getUuid() + ", " + identifierForError);
287                   }
288                   
289                 } else {
290                   
291                   if (SAVE_MODE.equals(SaveMode.INSERT_OR_UPDATE) || SAVE_MODE.equals(SaveMode.INSERT)) {
292                     isUpdate = false;
293                   } else {
294                     throw new RuntimeException("Cant update an external subject which doesnt already exist! " + ExternalSubjectSave.this.identifierToEdit);
295                   }
296                 }
297 
298                 //default
299                 ExternalSubjectSave.this.saveResultType = SaveResultType.NO_CHANGE;
300                 boolean needsSave = false;
301                 //if inserting
302                 if (!isUpdate) {
303 
304                   ExternalSubjectSave.this.saveResultType = SaveResultType.INSERT;
305                   needsSave = true;
306                   theExternalSubject = new ExternalSubject();
307                   theExternalSubject.setIdentifier(ExternalSubjectSave.this.identifierToEdit);
308 
309                 }
310                 
311                 //now compare and put all attributes (then store if needed)
312                 if (GrouperUtil.booleanValue(ExternalSubjectSave.this.enabled) != theExternalSubject.isEnabled()) {
313                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
314                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
315                   }
316                   theExternalSubject.setEnabled(ExternalSubjectSave.this.enabled);
317                   needsSave = true;
318                 }
319 
320                 if (!StringUtils.equals(StringUtils.trimToEmpty(ExternalSubjectSave.this.email), StringUtils.trimToEmpty(theExternalSubject.getEmail()))) {
321                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
322                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
323                   }
324                   theExternalSubject.setEmail(ExternalSubjectSave.this.email);
325                   needsSave = true;
326                 }
327 
328                 if (!StringUtils.equals(StringUtils.trimToEmpty(ExternalSubjectSave.this.institution), StringUtils.trimToEmpty(theExternalSubject.getInstitution()))) {
329                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
330                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
331                   }
332                   theExternalSubject.setInstitution(ExternalSubjectSave.this.institution);
333                   needsSave = true;
334                 }
335 
336                 if (!StringUtils.equals(StringUtils.trimToEmpty(ExternalSubjectSave.this.name), StringUtils.trimToEmpty(theExternalSubject.getName()))) {
337                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
338                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
339                   }
340                   theExternalSubject.setName(ExternalSubjectSave.this.name);
341                   needsSave = true;
342                 }
343                 if (!StringUtils.equals(StringUtils.trimToEmpty(ExternalSubjectSave.this.vettedEmailAddresses), StringUtils.trimToEmpty(theExternalSubject.getVettedEmailAddresses()))) {
344                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
345                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
346                   }
347                   theExternalSubject.setVettedEmailAddresses(ExternalSubjectSave.this.vettedEmailAddresses);
348                   needsSave = true;
349                 }
350 
351                 Map<String, ExternalSubjectAttribute> mapNewAttributeNameToObject = new HashMap<String, ExternalSubjectAttribute>();
352                 Map<String, ExternalSubjectAttribute> mapOldAttributeNameToObject = new HashMap<String, ExternalSubjectAttribute>();
353 
354                 if (!StringUtils.isBlank(theExternalSubject.getUuid())) {
355                   Set<ExternalSubjectAttribute> setOldAttributes = new HashSet<ExternalSubjectAttribute>();
356                   setOldAttributes.addAll(GrouperUtil.nonNull(theExternalSubject.retrieveAttributes()));
357                   for (ExternalSubjectAttribute externalSubjectAttribute : setOldAttributes) {
358                     mapOldAttributeNameToObject.put(externalSubjectAttribute.getAttributeSystemName(), externalSubjectAttribute);
359                   }
360                 }
361                 for (String attributeName : GrouperUtil.nonNull(ExternalSubjectSave.this.attributes).keySet()) {
362                   String newValue = ExternalSubjectSave.this.attributes.get(attributeName);
363                   {
364                     ExternalSubjectAttribute oldExternalSubjectAttribute = mapOldAttributeNameToObject.get(attributeName);
365                     if (oldExternalSubjectAttribute != null) {
366                       mapOldAttributeNameToObject.remove(attributeName);
367                       mapNewAttributeNameToObject.put(attributeName, oldExternalSubjectAttribute);
368                       if (StringUtils.equals(newValue, oldExternalSubjectAttribute.getAttributeValue())) {
369                         continue;
370                       }
371                       oldExternalSubjectAttribute.setAttributeValue(newValue);
372                       if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
373                         ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
374                       }
375                       needsSave = true;
376                       continue;
377                     }
378                   }
379                   ExternalSubjectAttributejectAttribute.html#ExternalSubjectAttribute">ExternalSubjectAttribute externalSubjectAttribute = new ExternalSubjectAttribute();
380                   externalSubjectAttribute.setAttributeSystemName(attributeName);
381                   externalSubjectAttribute.setAttributeValue(newValue);
382                   mapNewAttributeNameToObject.put(attributeName, externalSubjectAttribute);
383                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
384                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
385                   }
386                   needsSave = true;
387                 }
388                 
389                 if (mapOldAttributeNameToObject.size() > 0) {
390                   if (ExternalSubjectSave.this.saveResultType == SaveResultType.NO_CHANGE) {
391                     ExternalSubjectSave.this.saveResultType = SaveResultType.UPDATE;
392                   }
393                   needsSave = true;
394                   for (ExternalSubjectAttribute externalSubjectAttribute : mapOldAttributeNameToObject.values()) {
395                     externalSubjectAttribute.setAttributeValue(null);
396                     mapNewAttributeNameToObject.put(externalSubjectAttribute.getAttributeSystemName(), externalSubjectAttribute);
397                   }
398                 }
399                 
400                 //only store once, include attributes
401                 if (needsSave) {
402                   theExternalSubject.store(new HashSet<ExternalSubjectAttribute>(mapNewAttributeNameToObject.values()), null, true, true, false);
403                 }
404                 
405 			          return theExternalSubject;
406               }
407           });
408         }
409       });
410       return externalSubject;
411     } catch (RuntimeException re) {
412       
413       GrouperUtil.injectInException(re, "Problem saving externalSubject: " + this.identifier + ", thread: " + Integer.toHexString(Thread.currentThread().hashCode()));
414       
415       Throwable throwable = re.getCause();
416 //      if (throwable instanceof StemNotFoundException) {
417 //        throw (StemNotFoundException)throwable;
418 //      }
419       if (throwable instanceof InsufficientPrivilegeException) {
420         throw (InsufficientPrivilegeException)throwable;
421       }
422       //must just be runtime
423       throw re;
424     }
