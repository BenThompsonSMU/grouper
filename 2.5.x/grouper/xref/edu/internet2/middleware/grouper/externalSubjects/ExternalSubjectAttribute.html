323   /**
324    * delete this object from the DB.
325    * @param externalSubject 
326    */
327   void delete(final ExternalSubject externalSubject) {    
328     
329     assertValidAttribute(this.getAttributeSystemName());
330 
331     HibernateSession.callbackHibernateSession(
332       GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, AuditControl.WILL_AUDIT,
333       new HibernateHandler() {
334   
335         public Object callback(HibernateHandlerBean hibernateHandlerBean)
336             throws GrouperDAOException {
337   
338           hibernateHandlerBean.getHibernateSession().setCachingEnabled(false);
339   
340           ExternalSubjectAttributeStorageController.delete( ExternalSubjectAttribute.this );
341             
342           if (!hibernateHandlerBean.isCallerWillCreateAudit()) {
343             AuditEntry auditEntry = null;
344             
345             auditEntry = new AuditEntry(AuditTypeBuiltin.EXTERNAL_SUBJ_ATTR_DELETE, "id", 
346                 ExternalSubjectAttribute.this.getUuid(), "name", ExternalSubjectAttribute.this.getAttributeSystemName(), 
347                 "identifier", externalSubject.getIdentifier(), "value", ExternalSubjectAttribute.this.getAttributeValue());
348             auditEntry.setDescription("Deleted external subject attribute: " + ExternalSubjectAttribute.this.getAttributeSystemName());
349             auditEntry.saveOrUpdate(true);
350           }
351   
352           return null;
353         }
354       });
355     
356   }

358   /**
359    * @see GrouperAPI#onPreSave(HibernateSession)
360    */
361   @Override
362   public void onPreSave(HibernateSession hibernateSession) {
363     super.onPreSave(hibernateSession);
364     
365     if (StringUtils.isBlank(this.getUuid())) {
366       this.setUuid(GrouperUuid.getUuid());
367     }
368     
369     this.setModifierMemberId( GrouperSession.staticGrouperSession().getMember().getUuid() );
370     this.setModifyTimeDb( System.currentTimeMillis() );
371   
372     this.setCreatorMemberId( GrouperSession.staticGrouperSession().getMember().getUuid() );
373     this.setCreateTimeDb( System.currentTimeMillis() );
374   }

395   /**
396    * assert that the attribute name is valid
397    * @param attributeName
398    * @param errorOnNotfound
399    * @return if found
400    */
401   public static boolean validAttribute(String attributeName, boolean errorOnNotfound) {
402     //make sure attribute name is ok
403     boolean foundAttribute = false;
404     for (ExternalSubjectAttributeConfigBean externalSubjectAttributeConfigBean : 
405         GrouperUtil.nonNull(ExternalSubjectConfig.externalSubjectConfigBean().getExternalSubjectAttributeConfigBeans())) {
406       if (StringUtils.equals(attributeName, externalSubjectAttributeConfigBean.getSystemName())) {
407         foundAttribute = true;
408         break;
409       }
410     }
411     if (!foundAttribute) {
412       if (errorOnNotfound) {
413         throw new RuntimeException("Invalid attribute name: " + attributeName 
414             + ", not found in grouper.properties");
415       }
416       return false;
417     }
418     return true;
419   }

421   /**
422    * store this object to the DB.
423    * @param externalSubject reference back to owner
424    */
425   public void store(final ExternalSubject externalSubject) {    
426 
427     if (StringUtils.isBlank(this.getAttributeSystemName())) {
428       throw new RuntimeException("Attribute system name cannot be blank: " + this);
429     }
430     
431     assertValidAttribute(this.getAttributeSystemName());
432 
433     HibernateSession.callbackHibernateSession(
434       GrouperTransactionType.READ_WRITE_OR_USE_EXISTING, AuditControl.WILL_AUDIT,
435       new HibernateHandler() {
436   
437         public Object callback(HibernateHandlerBean hibernateHandlerBean)
438             throws GrouperDAOException {
439   
440           hibernateHandlerBean.getHibernateSession().setCachingEnabled(false);
441   
442           ExternalSubjectAttribute.this.setSubjectUuid(externalSubject.getUuid());
443           
444           boolean isInsert = HibUtilsMapping.isInsert(ExternalSubjectAttribute.this);
445           
446           ExternalSubjectAttributeStorageController.saveOrUpdate( ExternalSubjectAttribute.this );
447             
448           if (!hibernateHandlerBean.isCallerWillCreateAudit()) {
449             AuditEntry auditEntry = null;
450             
451             if (isInsert) {
452               auditEntry = new AuditEntry(AuditTypeBuiltin.EXTERNAL_SUBJ_ATTR_ADD, "id", 
453                   ExternalSubjectAttribute.this.getUuid(), "name", ExternalSubjectAttribute.this.getAttributeSystemName(), "identifier", externalSubject.getIdentifier(),
454                   "value", ExternalSubjectAttribute.this.getAttributeValue());
455               auditEntry.setDescription("Added external subject attr: " + externalSubject.getIdentifier() + ", attr: " + ExternalSubjectAttribute.this.getAttributeSystemName());
456             } else {
457               auditEntry = new AuditEntry(AuditTypeBuiltin.EXTERNAL_SUBJ_ATTR_UPDATE, "id", 
458                   ExternalSubjectAttribute.this.getUuid(), "name", ExternalSubjectAttribute.this.getAttributeSystemName(), "identifier", externalSubject.getIdentifier(),
459                   "value", ExternalSubjectAttribute.this.getAttributeValue());
460               auditEntry.setDescription("Updated external subject attr: " + externalSubject.getIdentifier() + ", attr: " + ExternalSubjectAttribute.this.getAttributeSystemName());
461               
462             }
463             auditEntry.saveOrUpdate(true);
464           }
465   
466           return null;
467         }
468       });
469     
470   }

487       }

489         result.append("subjectUuid: ").append(this.subjectUuid).append(", ");
490       }
491     } catch (Exception e) {
492       //ignore, we did the best we could
493     }
494     return result.toString();
495   }
