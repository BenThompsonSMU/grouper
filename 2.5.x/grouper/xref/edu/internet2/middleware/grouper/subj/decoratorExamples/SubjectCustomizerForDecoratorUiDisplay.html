40  /**
41   * decorate an attribute for some subjects
42   */
43  public class SubjectCustomizerForDecoratorUiDisplay extends SubjectCustomizerBase {
44    
45    /**
46     * cache who has access
47     */
48    private static GrouperCache<MultiKey, Boolean> subjectIsPrivileged = new GrouperCache(
49        "subjectIsPrivileged", 10000, false, 120, 120, false);
50    
51    /**
52     * @see SubjectCustomizer#filterSubjects(GrouperSession, Set, String)
53     */
54    @Override
55    public Set<Subject> filterSubjects(GrouperSession grouperSession,
56        Set<Subject> subjects, String findSubjectsInStemName) {
57  
58      //if not the UI then dont worry about it
59      GrouperContextType grouperContextType = GrouperContextTypeBuiltIn.currentGrouperContext();
60      if (grouperContextType == null || grouperContextType != GrouperContextTypeBuiltIn.GROUPER_UI) {
61        return subjects;
62      }
63      
64      final Subject subjectCallingGrouperSession = grouperSession.getSubject();
65  
66      MultiKey subjectMultiKey = new MultiKey(subjectCallingGrouperSession.getSourceId(), subjectCallingGrouperSession.getId());
67  
68      //see if decision in cache
69      Boolean allowExtendedDisplay = subjectIsPrivileged.get(subjectMultiKey);
70      
71      if (allowExtendedDisplay == null) {
72        
73        //dont allow grouperSysAdmin or wheel group
74        boolean grouperSystem = PrivilegeHelper.isSystemSubject(subjectCallingGrouperSession);
75        boolean wheel = PrivilegeHelper.isWheel(grouperSession);
76        
77        //if grouper system then dont see the data, some thing happen as grouper system, dont allow
78        if (grouperSystem) {
79          allowExtendedDisplay = false;
80        } else if (wheel) {
81          
82          //if someone is a wheel group, they can see everything
83          allowExtendedDisplay = true;
84          
85        } else {
86  
87          //do this check as root so anyone can do it
88          allowExtendedDisplay = (Boolean)GrouperSession.callbackGrouperSession(grouperSession.internal_getRootSession(), new GrouperSessionHandler() {
89            
90            @Override
91            public Object callback(GrouperSession rootSession) throws GrouperSessionException {
92              Group privilegedGroup = GroupFinder.findByName(rootSession, 
93                  GrouperConfig.retrieveConfig().propertyValueString("grouper.rootStemForBuiltinObjects", "etc") + ":privilegedGroup", true);
94              return privilegedGroup.hasMember(subjectCallingGrouperSession);
95            }
96          });
97          
98        }
99        //put back in cache
100       subjectIsPrivileged.put(subjectMultiKey, allowExtendedDisplay);
101     }
102     
103     Set<Subject> newSubjectList = new LinkedHashSet<Subject>();
104     
105     for (Subject originalSubject : subjects) {
106 
107       if (StringUtils.equals(originalSubject.getSourceId(), "jdbc" )) {
108         
109         String uiLabel = originalSubject.getName();
110         if (allowExtendedDisplay) {
111           uiLabel += " - " + originalSubject.getId();
112         }
113         Subject newSubject = originalSubject;
114         if (!(newSubject instanceof SubjectImpl)) {
115           newSubject = new SubjectImpl(originalSubject.getId(), originalSubject.getName(), 
116               originalSubject.getDescription(), originalSubject.getTypeName(), originalSubject.getSourceId(), 
117               originalSubject.getAttributes(false));
118         }
119 
120         //this should return a modifiable map of attributes for us to work with
121         newSubject.getAttributes(false).put("uiLabel", GrouperUtil.toSet(uiLabel));
122         newSubjectList.add(newSubject);
123       } else {
124         newSubjectList.add(originalSubject);
125       }
126     }
127     
128     return newSubjectList;
129     
130   }
