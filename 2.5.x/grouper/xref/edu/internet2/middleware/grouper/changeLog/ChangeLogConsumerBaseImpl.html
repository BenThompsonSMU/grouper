73    /** Maps supported changeLogEntry category and action to methods */
74    enum ChangeLogEventType {
75      group_addGroup {
76  
77        public void process(ChangeLogEntry changeLogEntry,
78            ChangeLogConsumerBaseImpl consumer) {
79          // does this event pertain to us? was the group or one of its parent folders marked for sync
80          final String groupName = changeLogEntry
81              .retrieveValueForLabel(ChangeLogLabels.GROUP_ADD.name);
82          final Group group = GroupFinder.findByName(
83              GrouperSession.staticGrouperSession(false), groupName, false);
84          if (group != null) {
85            if (consumer.isGroupMarkedForSync(groupName)) {
86              if (consumer.debugLog != null) {
87                consumer.debugLog.put("callingAddGroup", true);
88              }
89              consumer.addGroup(group, changeLogEntry);
90            } else {
91              if (consumer.debugLog != null) {
92                consumer.debugLog.put("skippingAddGroupNotMarked", true);
93                consumer.debugLog.put("notMarked", true);
94              }
95            }
96          } else {
97            if (consumer.debugLog != null) {
98              consumer.debugLog.put("skippingAddGroup", true);
99              consumer.debugLog.put("groupDeleted", true);
100           }
101         }
102       }

104     group_updateGroup {
105 
106       public void process(ChangeLogEntry changeLogEntry,
107           ChangeLogConsumerBaseImpl consumer) {
108         final String propertyChanged = changeLogEntry
109             .retrieveValueForLabel(ChangeLogLabels.GROUP_UPDATE.propertyChanged);
110         if (consumer.debugLog != null) {
111           consumer.debugLog.put("propertyChanged", propertyChanged);
112         }
113         // a group "move" is a change in group name and shows up in change log as a group update
114         if ("name".equals(changeLogEntry
115             .retrieveValueForLabel(ChangeLogLabels.GROUP_UPDATE.propertyChanged))) {
116           // group was moved to a new folder, get the new name and the old name
117           final String newGroupName = changeLogEntry
118               .retrieveValueForLabel(ChangeLogLabels.GROUP_UPDATE.propertyNewValue);
119           final String oldGroupName = changeLogEntry
120               .retrieveValueForLabel(ChangeLogLabels.GROUP_UPDATE.propertyOldValue);
121           if (consumer.debugLog != null) {
122             consumer.debugLog.put("oldGroupName", oldGroupName);
123             consumer.debugLog.put("newGroupName", newGroupName);
124           }
125 
126           // if newGroupName is marked for sync then rename or add group and membership at target
127           // this covers three cases:
128           //  1) directly marked group moved
129           //  2) indirectly marked group moved and is still indirectly marked
130           //  3) unmarked group move and is now marked (indirectly)
131           if (consumer.isGroupMarkedForSync(newGroupName)) {
132             if (consumer.debugLog != null) {
133               consumer.debugLog.put("callingRenameGroup", true);
134               consumer.debugLog.put("oldGroupName", oldGroupName);
135               consumer.debugLog.put("newGroupName", newGroupName);
136             }
137             consumer.renameGroup(oldGroupName, newGroupName, changeLogEntry);
138           } else {
139             // newGroupName not marked, check oldGroupName folders for indirect mark, if so delete at target, otherwise no-op
140 
141             // get the parent folder path name from some:parent:path:oldGroupName
142             final int lastPathSeparator = oldGroupName.lastIndexOf(":");
143             final String oldGroupNameFolderPath = oldGroupName.substring(0,
144                 lastPathSeparator);
145             final Stem oldGroupParentFolder = StemFinder
146                 .findByName(GrouperSession.staticGrouperSession(false),
147                     oldGroupNameFolderPath, false);
148             if (oldGroupParentFolder != null
149                 && oldGroupParentFolder.getAttributeDelegate()
150                     .hasAttributeOrAncestorHasAttribute(consumer.syncAttribute.getName(),
151                         false)) {
152               // oldGroupName previously marked, so remove from target
153               if (consumer.debugLog != null) {
154                 consumer.debugLog.put("callingRemoveMovedGroup", true);
155               }
156               consumer.removeMovedGroup(oldGroupName, changeLogEntry);
157             } else {
158               // could find moved parent folder (already deleted?) or not previously marked, no-op in either case
159               if (consumer.debugLog != null) {
160                 consumer.debugLog.put("skippingUpdateGroup", true);
161                 consumer.debugLog.put("noParentOrNotMarked", true);
162               }
163             }
164           }
165         } else {
166           // some other group property update; description, parentStemId, displayName, etc
167           final String groupName = changeLogEntry
168               .retrieveValueForLabel(ChangeLogLabels.GROUP_UPDATE.name);
169           if (consumer.debugLog != null) {
170             consumer.debugLog.put("skipingUpdateGroup", true);
171             consumer.debugLog.put("invalidPropertyChanged", true);
172             consumer.debugLog.put("groupName", groupName);
173           }
174         }
175       }

180       public void process(ChangeLogEntry changeLogEntry,
181           ChangeLogConsumerBaseImpl consumer) {
182         final String groupName = changeLogEntry
183             .retrieveValueForLabel(ChangeLogLabels.GROUP_DELETE.name);
184         PITGroup pitGroup = PITGroupFinder.findMostRecentByName(groupName, false);
185         if (pitGroup != null) {
186           if (consumer.isGroupMarkedForSync(pitGroup.getName())) {
187             // case when marker is on parent folder of deleted group
188             // TODO handle deleted group when marker is on parent folder of deleted group...
189             if (consumer.debugLog != null) {
190               consumer.debugLog.put("callingRemoveDeletedGroup", true);
191             }
192             consumer.removeDeletedGroup(pitGroup, changeLogEntry);
193           } else {
194             // skipping changeLogEntry that doesn't pertain to us
195             if (consumer.debugLog != null) {
196               consumer.debugLog.put("skippingDeleteGroup", true);
197               consumer.debugLog.put("notMarkedForSync", true);
198             }
199           }
200         } else {
201           //Couldn't find deleted group {} in PIT, already purged?. Let fullSync sort it out
202           if (consumer.debugLog != null) {
203             consumer.debugLog.put("skippingDeleteGroup", true);
204             consumer.debugLog.put("couldntFindDeletedGroupInPit", true);
205           }
206         }

211       public void process(ChangeLogEntry changeLogEntry,
212           ChangeLogConsumerBaseImpl consumer) {
213         final String groupName = changeLogEntry
214             .retrieveValueForLabel(ChangeLogLabels.MEMBERSHIP_ADD.groupName);
215         final String subjectId = changeLogEntry
216             .retrieveValueForLabel(ChangeLogLabels.MEMBERSHIP_ADD.subjectId);
217         final String sourceId = changeLogEntry
218             .retrieveValueForLabel(ChangeLogLabels.MEMBERSHIP_ADD.sourceId);
219         if (consumer.debugLog != null) {
220           consumer.debugLog.put("groupName", groupName);
221           consumer.debugLog.put("subjectSourceId", sourceId);
222           consumer.debugLog.put("subjectId", subjectId);
223         }
224         final Group group = GroupFinder.findByName(
225             GrouperSession.staticGrouperSession(false), groupName, false);
226         final Subject subject = SubjectFinder.findByIdAndSource(subjectId, sourceId, false);
227         
228         if (consumer.debugLog != null) {
229           consumer.debugLog.put("foundGroup", group != null);
230           consumer.debugLog.put("foundSubject", subject != null);
231         }
232         
233         // we only care about effective membership for people in marked groups, skip addMembership for everything else
234         if (!subject.getType().equals(SubjectTypeEnum.PERSON)) {
235           if (consumer.debugLog != null) {
236             consumer.debugLog.put("skippingAddMembership", true);
237             consumer.debugLog.put("notPerson", true);
238             consumer.debugLog.put("subjectType", subject.getType());
239           }
240         } else {
241           if (consumer.isGroupMarkedForSync(groupName)) {
242             if (consumer.debugLog != null) {
243               consumer.debugLog.put("callingAddMembership", true);
244             }
245             consumer.addMembership(subject, group, changeLogEntry);
246           } else {
247             // skipping changeLogEntry that doesn't pertain to us
248             if (consumer.debugLog != null) {
249               consumer.debugLog.put("skippingAddMembership", true);
250               consumer.debugLog.put("notMarkedForSync", true);
251             }
252           }
253         }

314       /**
315        * On assignment of the syncAttribute marker, create all the groups or group (if directly assigned)
316        * and any current memberships at the target.
317        */
318       public void process(ChangeLogEntry changeLogEntry,
319           ChangeLogConsumerBaseImpl consumer) {
320         // check if this is our syncAttribute, otherwise nothing to do.
321         final String attributeDefNameName = changeLogEntry
322             .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_ADD.attributeDefNameName);
323         if (consumer.syncAttribute.getName().equals(attributeDefNameName)) {
324           // syncAttribute applied to a group, then create the group at the target
325           String assignType = changeLogEntry
326               .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_ADD.assignType);
327           String ownerId1 = changeLogEntry
328               .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_ADD.ownerId1);
329           if (consumer.debugLog != null) {
330             consumer.debugLog.put("assignType", assignType);
331             consumer.debugLog.put("ownerId", ownerId1);
332           }
333           if (GROUP.equals(assignType)) {
334             Group markedGroup = GroupFinder.findByUuid(
335                 GrouperSession.staticGrouperSession(false), ownerId1, false);
336             if (markedGroup != null) {
337               if (consumer.debugLog != null) {
338                 consumer.debugLog.put("callingAddGroupAndMemberships", true);
339                 consumer.debugLog.put("groupName", markedGroup.getName());
340               }
341               consumer.addGroupAndMemberships(markedGroup, changeLogEntry);
342             } else { // couldn't find group, already deleted?
343               if (consumer.debugLog != null) {
344                 consumer.debugLog.put("skippingAddGroup", true);
345                 consumer.debugLog.put("cantFindGroup", true);
346               }
347             }
348           } else if (FOLDER.equals(assignType)) {
349             // syncAttribute applied to a folder, get all the groups below this folder and sub folders and create them at the target
350             Stem markedFolder = StemFinder.findByUuid(
351                 GrouperSession.staticGrouperSession(false), ownerId1, false);
352             if (markedFolder != null) {
353               Set<Group> markedGroups = markedFolder.getChildGroups(Stem.Scope.SUB);
354               for (Group group : markedGroups) {
355                 if (consumer.debugLog != null) {
356                   consumer.debugLog.put("folderName", markedFolder.getName());
357                   consumer.debugLog.put("callingAddGroupAndMemberships", true);
358                   consumer.debugLog.put("groupName", group.getName());
359                 }
360                 consumer.addGroupAndMemberships(group, changeLogEntry);
361               }
362             } else { // couldn't find folder, already deleted?
363               if (consumer.debugLog != null) {
364                 consumer.debugLog.put("folderId", ownerId1);
365                 consumer.debugLog.put("skippingAddGroupAndMemberships", true);
366                 consumer.debugLog.put("cantFindFolder", true);
367               }
368             }
369           }
370         } else {
371           // not our syncAttribute, so nothing to do
372           if (consumer.debugLog != null) {
373             consumer.debugLog.put("skippingAddAttribute", true);
374           }
375         }
376       }

378     attributeAssign_deleteAttributeAssign {
379 
380       /**
381        * On removal of the syncAttribute marker, delete all the groups or group (if directly assigned) at the target, unless
382        * otherwise still marked by direct assignment or a parent folder.
383        */
384       public void process(ChangeLogEntry changeLogEntry,
385           ChangeLogConsumerBaseImpl consumer) {
386         // is this our syncAttribute? otherwise nothing to do.
387         final String attributeDefNameName = changeLogEntry
388             .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_DELETE.attributeDefNameName);
389         if (consumer.syncAttribute.getName().equals(attributeDefNameName)) {
390           String assignType = changeLogEntry
391               .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_DELETE.assignType);
392           String ownerId1 = changeLogEntry
393               .retrieveValueForLabel(ChangeLogLabels.ATTRIBUTE_ASSIGN_DELETE.ownerId1);
394           // is it for a group? then delete the group at the target, unless a parent folder is also marked.
395           if (consumer.debugLog != null) {
396             consumer.debugLog.put("assignType", assignType);
397             consumer.debugLog.put("ownerId", ownerId1);
398           }
399           if (GROUP.equals(assignType)) {
400             // get the group and then check for a marked parent folder before deleting at target.
401             Group group = GroupFinder.findByUuid(
402                 GrouperSession.staticGrouperSession(false), ownerId1, false);
403             if (group != null) {
404               if (consumer.debugLog != null) {
405                 consumer.debugLog.put("groupName", group.getName());
406               }
407               // case when group had a direct syncAttribute marker assignment removed, does it still have a parent marker?
408               // need to flush edu.internet2.middleware.grouper.attr.assign.AttributeAssignBaseDelegate.objectHasAttributeCache to make sure syncAttribute is cleared
409               EhcacheController
410                   .ehcacheController()
411                   .getGrouperCache(
412                       "edu.internet2.middleware.grouper.attr.assign.AttributeAssignBaseDelegate.objectHasAttributeCache")
413                   .clear();
414               if (group.getAttributeDelegate().hasAttributeOrAncestorHasAttribute(
415                   consumer.syncAttribute.getName(), false)) {
416                 if (consumer.debugLog != null) {
417                   consumer.debugLog.put("skippingDeleteAttribute", true);
418                   consumer.debugLog.put("markedByParentFolder", true);
419                 }
420               } else {
421                 // marker syncAttribute removed from group and no other parent folder marked so delete at target
422                 if (consumer.debugLog != null) {
423                   consumer.debugLog.put("callingRemoveGroup", true);
424                   consumer.debugLog.put("markedByGroup", true);
425                 }
426                 consumer.removeGroup(group, changeLogEntry);
427               }
428             } else {
429               // case when a group which had a direct syncAttribute marker was deleted, always delete at target
430               PITGroup pitGroup = PITGroupFinder.findBySourceId(ownerId1, false)
431                   .iterator().next();
432               if (pitGroup != null) {
433                 String pitGroupName = pitGroup.getName();
434                 // marker syncAttribute removed when deleting a group, always delete at target
435                 if (consumer.debugLog != null) {
436                   consumer.debugLog.put("callingRemoveGroup", true);
437                   consumer.debugLog.put("deletedGroup", true);
438                   consumer.debugLog.put("pitGroupName", pitGroupName);
439                 }
440                 consumer.removeDeletedGroup(pitGroup, changeLogEntry);
441               } else {
442                 // couldn't find group anywhere? so can't determine its name.
443                 if (consumer.debugLog != null) {
444                   consumer.debugLog.put("skippingRemoveGroup", true);
445                   consumer.debugLog.put("cantFindGroup", true);
446                 }
447               }
448             }
449           } else if (FOLDER.equals(assignType)) {
450             // is it a folder, then delete all the containing groups at the target, unless they are still marked otherwise (direct or indirect)
451             Stem unMarkedFolder = StemFinder.findByUuid(
452                 GrouperSession.staticGrouperSession(false), ownerId1, false);
453             if (unMarkedFolder != null) {
454               if (consumer.debugLog != null) {
455                 consumer.debugLog.put("folderName", unMarkedFolder.getName());
456               }
457               // get all the groups below this folder and sub folders and to see if they are still marked, otherwise delete them at the target
458               Set<Group> unMarkedGroups = unMarkedFolder.getChildGroups(Stem.Scope.SUB);
459               // need to flush edu.internet2.middleware.grouper.attr.assign.AttributeAssignBaseDelegate.objectHasAttributeCache to make sure syncAttribute is cleared
460               EhcacheController
461                   .ehcacheController()
462                   .getGrouperCache(
463                       "edu.internet2.middleware.grouper.attr.assign.AttributeAssignBaseDelegate.objectHasAttributeCache")
464                   .clear();
465               for (Group group : unMarkedGroups) {
466                 // check that the group isn't marked directly or from some other parent folder
467                 if (consumer.isGroupMarkedForSync(group.getName())) {
468                   if (consumer.debugLog != null) {
469                     consumer.debugLog.put("skippingRemoveGroup", true);
470                     consumer.debugLog.put("foundMarkForGroup", true);
471                   }
472                 } else {
473                   if (consumer.debugLog != null) {
474                     consumer.debugLog.put("callingRemoveGroup", true);
475                     consumer.debugLog.put("groupName", group.getName());
476                   }
477                   consumer.removeGroup(group, changeLogEntry);
478                 }
479               }
480             } else {
481               // can't delete a folder without first deleting child objects, so there should be nothing to delete at the target
482               // deleting a folder with an attribute assign will cause a deleteAttributeAssign to come after folder already deleted.
483               if (consumer.debugLog != null) {
484                 consumer.debugLog.put("skippingRemoveGroup", true);
485                 consumer.debugLog.put("deletedFolder", true);
486                 consumer.debugLog.put("folderId", true);
487               }
488             }
489           }
490         }
491       }

573   /**
574    * If syncAttribute was applied to the group or one of the parent folders return true
575    * Method keeps an internal cache of results per run in markedFoldersAndGroups
576    * Will also check the PIT for recently deleted groups
577    */
578   private boolean isGroupMarkedForSync(String groupName) {
579 
580     boolean markedForSync = false;
581 
582     String markedForSyncString = markedFoldersAndGroups.get(groupName);
583     // have we seen this group already in this run
584     if (!StringUtils.isBlank(markedForSyncString)) {
585       markedForSync = markedForSyncString.equals(MARKED);
586       if (this.debugLog != null) {
587         this.debugLog.put("foundGroupInMarkedCache", true);
588         this.debugLog.put("markedForSync", markedForSync);
589       }
590       return markedForSync;
591     }
592 
593     // looking for the group
594     final Group group = GroupFinder.findByCurrentName(
595         GrouperSession.staticGrouperSession(false), groupName, false);
596 
597     if (group != null) {
598       // is it marked with the syncAttribute?
599       markedForSync = group.getAttributeDelegate().hasAttributeOrAncestorHasAttribute(
600           syncAttribute.getName(), false);
601       if (this.debugLog != null) {
602         this.debugLog.put("foundGroup", true);
603       }
604     } else {
605       // looking for the deleted group in the PIT
606       PITGroup pitGroup = PITGroupFinder.findMostRecentByName(groupName, false);
607       if (pitGroup != null) {
608         if (this.debugLog != null) {
609           this.debugLog.put("foundPitGroup", true);
610         }
611         // looking for syncAttribute assignment in the PIT
612         Set<PITAttributeDefName> pitSyncAttributes = PITAttributeDefNameFinder
613             .findByName(syncAttribute.getName(), false, true);
614         PITAttributeDefName pitSyncAttribute = pitSyncAttributes.iterator().next();
615         //System.out.println(pitGroup.getStartTime() + " -> " + pitGroup.getEndTime());
616         Set<PITAttributeAssign> pitAttributeAssigns = PITAttributeAssignFinder
617             .findByOwnerPITGroupAndPITAttributeDefName(pitGroup, pitSyncAttribute,
618                 pitGroup.getEndTime(), pitGroup.getEndTime());
619         if (!pitAttributeAssigns.isEmpty()) {
620           markedForSync = true;
621         } else {
622           // only gets direct group assignments, check folder next if no direct mark
623           // check for folders for mark
624           String parentStemName = GrouperUtil.parentStemNameFromName(pitGroup.getName());
625           Stem folder = StemFinder.findByName(GrouperSession.staticGrouperSession(false),
626               parentStemName, false);
627           if (folder != null) {
628             markedForSync = folder.getAttributeDelegate()
629                 .hasAttributeOrAncestorHasAttribute(syncAttribute.getName(), false);
630             if (this.debugLog != null) {
631               this.debugLog.put("foundPitGroupFolder", true);
632             }
633           }
634         }
635       } else {
636         // couldn't find group anywhere including the PIT
637         if (this.debugLog != null) {
638           this.debugLog.put("couldntFindGroup", true);
639         }
640         markedForSync = false;
641       }
642     }
643 
644     if (this.debugLog != null) {
645       debugLog.put("markedForSync", markedForSync);
646     }
647 
648     // remember this for next time
649     if (markedForSync) {
650       markedFoldersAndGroups.put(groupName, MARKED);
651       return true;
652     } 
653     markedFoldersAndGroups.put(groupName, NOT_MARKED);
654     return false;
655   }

836   /**
837    * Call the method of the {@link ChangeLogEventType} enum which matches the {@link ChangeLogEntry} category and action (i.e. the change log event type).
838    *
839    * @param changeLogEntry the change log entry
840    */
841   private void processChangeLogEntry(ChangeLogEntry changeLogEntry) {
842 
843     if (LOG.isDebugEnabled()) {
844        this.debugLog = new LinkedHashMap<String, Object>();
845        this.debugLog.put("consumerName", consumerName);
846        this.debugLog.put("startingSequenceNumber", changeLogEntry.getSequenceNumber());
847        this.debugLog.put("method", "processChangeLogEntry");
848     }
849 
850     // construct key from changeLogEntry in the form of <category>_<action>
851     final String changeLogEventTypeKey = changeLogEntry.getChangeLogType()
852         .getChangeLogCategory() + "_"
853         + changeLogEntry.getChangeLogType().getActionName();
854 
855     if (this.debugLog != null) {
856       this.debugLog.put("changeLogEventTypeKey", changeLogEventTypeKey);
857     }
858     
859     ChangeLogEventType changeLogEventType = null;
860 
861     // look up method to map to call it
862     try {
863       try {
864         changeLogEventType = ChangeLogEventType.valueOf(changeLogEventTypeKey);
865       } catch (IllegalArgumentException e) {
866         if (this.debugLog != null) {
867           this.debugLog.put("invalidEvent", true);
868         }      
869       }
870       if (changeLogEventType != null) {
871         changeLogEventType.process(changeLogEntry, this);
872       }
873     } catch (RuntimeException e) {
874       if (this.debugLog != null) {
875         this.debugLog.put("error", GrouperUtil.getFullStackTrace(e));
876       }
877       throw e;
878     } finally {
879       if (this.debugLog != null) {
880         LOG.debug(GrouperUtil.mapToString(this.debugLog));
881       }
882     }
