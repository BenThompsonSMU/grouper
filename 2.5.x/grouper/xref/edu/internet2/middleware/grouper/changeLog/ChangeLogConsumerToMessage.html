17  package edu.internet2.middleware.grouper.changeLog;
18  
19  import java.util.List;
20  
21  import org.apache.commons.logging.Log;
22  
23  import edu.internet2.middleware.grouper.app.loader.GrouperLoaderConfig;
24  import edu.internet2.middleware.grouper.util.GrouperUtil;
25  import edu.internet2.middleware.grouperClient.messaging.GrouperMessageQueueType;
26  import edu.internet2.middleware.grouperClient.messaging.GrouperMessageSendParam;
27  import edu.internet2.middleware.grouperClient.messaging.GrouperMessagingEngine;
28  
29  
30  /**
31   *
32   */
33  public class ChangeLogConsumerToMessage extends ChangeLogConsumerBase {
34  
35    /**
36     * 
37     */
38    public ChangeLogConsumerToMessage() {
39    }
40  
41    /**
42     * @see edu.internet2.middleware.grouper.changeLog.ChangeLogConsumerBase#processChangeLogEntries(java.util.List, edu.internet2.middleware.grouper.changeLog.ChangeLogProcessorMetadata)
43     */
44    @Override
45    public long processChangeLogEntries(List<ChangeLogEntry> changeLogEntryList,
46        ChangeLogProcessorMetadata changeLogProcessorMetadata) {
47  
48      long lastProcessed = -1;
49  
50      String messagingSystemName = GrouperLoaderConfig.retrieveConfig().propertyValueString("changeLog.consumer." 
51          + changeLogProcessorMetadata.getConsumerName() + ".messagingSystemName");
52      String queueOrTopicName = GrouperLoaderConfig.retrieveConfig().propertyValueStringRequired("changeLog.consumer." 
53          + changeLogProcessorMetadata.getConsumerName() + ".queueOrTopicName");
54      String messageQueueType = GrouperLoaderConfig.retrieveConfig().propertyValueStringRequired("changeLog.consumer." 
55          + changeLogProcessorMetadata.getConsumerName() + ".messageQueueType");
56      GrouperMessageQueueType grouperMessageQueueType = GrouperMessageQueueType.valueOfIgnoreCase(messageQueueType, true);
57      
58      String routingKey = GrouperLoaderConfig.retrieveConfig().propertyValueString("changeLog.consumer." 
59          + changeLogProcessorMetadata.getConsumerName() + ".routingKey", "");
60     
61      String exchangeType = GrouperLoaderConfig.retrieveConfig().propertyValueString("changeLog.consumer." 
62          + changeLogProcessorMetadata.getConsumerName() + ".exchangeType", "");
63      
64      boolean autocreateObjects = GrouperLoaderConfig.retrieveConfig().propertyValueBoolean("loader.messaging.settings.autocreate.objects", true);
65      
66      for (ChangeLogEntry changeLogEntry : GrouperUtil.nonNull(changeLogEntryList)) {
67  
68        try {
69          String json = changeLogEntry.toJson(true);
70          GrouperMessagingEngine.send(new GrouperMessageSendParam().assignGrouperMessageSystemName(messagingSystemName)
71              .assignAutocreateObjects(autocreateObjects)
72              .assignQueueType(grouperMessageQueueType)
73              .assignQueueOrTopicName(queueOrTopicName)
74              .addMessageBody(json)
75              .assignRoutingKey(routingKey)
76              .assignExchangeType(exchangeType));
77          lastProcessed = changeLogEntry.getSequenceNumber();
78        } catch (Exception e) {
79          LOG.error("Error processing event: " + changeLogEntry.getId(), e);
80          changeLogProcessorMetadata.registerProblem(e, "Error processing event: " + changeLogEntry.getId() + ", " + e.getMessage(), changeLogEntry.getSequenceNumber());
81          break;
82        }
83      }
84      return lastProcessed;
85    
86    }
87  
88    /** logger */
89    private static final Log LOG = GrouperUtil.getLog(ChangeLogConsumerToMessage.class);
90    
91  
92  }
