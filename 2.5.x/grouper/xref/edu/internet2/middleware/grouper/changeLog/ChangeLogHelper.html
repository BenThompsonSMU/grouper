58    static class TestChangeLogHelper extends ChangeLogConsumerBase {
59  
60      /**
61       * 
62       * @see edu.internet2.middleware.grouper.changeLog.ChangeLogConsumerBase#processChangeLogEntries(java.util.List, edu.internet2.middleware.grouper.changeLog.ChangeLogProcessorMetadata)
63       */
64      @Override
65      public long processChangeLogEntries(List<ChangeLogEntry> changeLogEntryList,
66          ChangeLogProcessorMetadata changeLogProcessorMetadata) {
67        
68        long currentId = -1;
69  
70        for (ChangeLogEntry changeLogEntry : changeLogEntryList) {
71          //try catch so we can track that we made some progress
72          try {
73  
74            currentId = changeLogEntry.getSequenceNumber();
75            
76            if (changeLogEntry.equalsCategoryAndAction(ChangeLogTypeBuiltin.MEMBERSHIP_ADD)) {
77  
78              System.out.println("Member add, name: " 
79                  + changeLogEntry.retrieveValueForLabel(ChangeLogLabels.MEMBERSHIP_ADD.groupName)
80                  + ", " + changeLogEntry.retrieveValueForLabel(ChangeLogLabels.MEMBERSHIP_ADD.subjectId));
81  
82            } else if (changeLogEntry.equalsCategoryAndAction(ChangeLogTypeBuiltin.MEMBERSHIP_DELETE)) {
83  
84              System.out.println("Member delete, name: " 
85                  + changeLogEntry.retrieveValueForLabel(ChangeLogLabels.MEMBERSHIP_DELETE.groupName)
86                  + ", " + changeLogEntry.retrieveValueForLabel(ChangeLogLabels.MEMBERSHIP_DELETE.subjectId));
87  
88            }
89            
90          } catch (Exception e) {
91            //we unsuccessfully processed this record... decide whether to wait, throw, ignore, log, etc...
92            LOG.error("problem with id: " + currentId, e);
93            //continue
94          }
95        }
96  
97        return currentId;
98      }
99    };

169   /**
170    * <pre>
171    * call this method to process a batch of 100k (max) records of the change log... 
172    * pass in a consumer name (nothing that people would use for a real change log consumer), that is used
173    * to keep track of the last processed record, the loader log which will log process in the grouper loader
174    * log table, and the processor which is the change log consumer base...
175    * 
176    * to test this, do your changes, e.g. add a member, delete a member, then call this:
177    * 
178    * GrouperLoader.runOnceByJobName(grouperSession, GrouperLoaderType.GROUPER_CHANGE_LOG_TEMP_TO_CHANGE_LOG);
179    * 
180    * then call this method...  e.g. the static example() method in this class
181    * 
182    * 
183    * </pre>
184    * @param consumerName name of configured consumer, or another name that is not configured (e.g. ldappcng)
185    * @param hib3GrouploaderLog send an instance of this in so it can be logged to the DB...
186    * @param changeLogConsumerBase is the instance that should handle the requests
187    */
188   public static void processRecords(String consumerName, Hib3GrouperLoaderLog hib3GrouploaderLog, ChangeLogConsumerBase changeLogConsumerBase) {
189     
190     Map<String, Object> debugMap = LOG.isDebugEnabled() ? new LinkedHashMap<String, Object>() : null;
191 
192     if (LOG.isDebugEnabled()) {
193       debugMap.put("consumerName", consumerName);
194     }
195     
196     try {
197 
198       ChangeLogConsumer changeLogConsumer = GrouperDAOFactory.getFactory().getChangeLogConsumer().findByName(consumerName, false);
199       boolean error = false;
200       
201       //if this is a new job
202       if (changeLogConsumer == null) {
203 
204         if (LOG.isDebugEnabled()) {
205           debugMap.put("creating consumer row", true);
206         }
207         
208         changeLogConsumer = new ChangeLogConsumer();
209         changeLogConsumer.setName(consumerName);
210         GrouperDAOFactory.getFactory().getChangeLogConsumer().saveOrUpdate(changeLogConsumer);
211       }
212       
213       //if the sequence number is not set
214       if (changeLogConsumer.getLastSequenceProcessed() == null) {
215         changeLogConsumer.setLastSequenceProcessed(GrouperUtil.defaultIfNull(ChangeLogEntry.maxSequenceNumber(true), 0l));
216         GrouperDAOFactory.getFactory().getChangeLogConsumer().saveOrUpdate(changeLogConsumer);
217       }
218       
219       if (LOG.isDebugEnabled()) {
220         debugMap.put("last sequence processed", changeLogConsumer.getLastSequenceProcessed());
221       }
222       
223       int batchSize = GrouperLoaderConfig.retrieveConfig().propertyValueInt("changeLog.changeLogConsumerBatchSize", 1000);
224 
225       for (int i=0;i<1000;i++) {
226         
227         ChangeLogProcessorMetadatadata.html#ChangeLogProcessorMetadata">ChangeLogProcessorMetadata changeLogProcessorMetadata = new ChangeLogProcessorMetadata();
228         changeLogProcessorMetadata.setHib3GrouperLoaderLog(hib3GrouploaderLog);
229         changeLogProcessorMetadata.setConsumerName(consumerName);
230                 
231         List<ChangeLogEntry> changeLogEntryList = GrouperDAOFactory.getFactory().getChangeLogEntry()
232           .retrieveBatch(changeLogConsumer.getLastSequenceProcessed(), batchSize);
233 
234         if (LOG.isDebugEnabled()) {
235           debugMap.put(i + ": number of records found to process", changeLogEntryList.size());
236         }
237         
238         if (changeLogEntryList.size() == 0) {
239           break;
240         }
241         
242         //pass this to the consumer
243         long lastProcessed = -1;
244         try {
245           lastProcessed = changeLogConsumerBase.processChangeLogEntries(changeLogEntryList, changeLogProcessorMetadata);
246           
247           if (LOG.isDebugEnabled()) {
248             debugMap.put(i + ": processed to record number", lastProcessed);
249           }
250 
251         } catch (Exception e) {
252           if (LOG.isDebugEnabled()) {
253             debugMap.put(i + ": error processing records", true);
254           }
255           LOG.error("Error", e);
256           hib3GrouploaderLog.appendJobMessage("Error: " 
257               + ExceptionUtils.getFullStackTrace(e));
258           error = true;
259         }
260         if (lastProcessed != -1 && (changeLogConsumer.getLastSequenceProcessed() == null || changeLogConsumer.getLastSequenceProcessed() != lastProcessed)) {
261           changeLogConsumer.setLastSequenceProcessed(lastProcessed);
262           GrouperDAOFactory.getFactory().getChangeLogConsumer().saveOrUpdate(changeLogConsumer);
263         }
264         
265         long lastSequenceInBatch = changeLogEntryList.get(changeLogEntryList.size()-1).getSequenceNumber();
266   
267         
268         if (changeLogProcessorMetadata.isHadProblem()) {
269           if (LOG.isDebugEnabled()) {
270             debugMap.put(i + ": hadProblem", true + ", " + changeLogProcessorMetadata.getRecordProblemText());
271           }
272           String errorString = "Error: " 
273               + changeLogProcessorMetadata.getRecordProblemText()
274               + ", sequenceNumber: " + changeLogProcessorMetadata.getRecordExceptionSequence()
275               + ", " + ExceptionUtils.getFullStackTrace(changeLogProcessorMetadata.getRecordException());
276           LOG.error(errorString);
277           hib3GrouploaderLog.appendJobMessage(errorString);
278           error = true;
279         }
280         if (lastProcessed != lastSequenceInBatch) {
281           String errorString = "Did not get all the way through the batch! " + lastProcessed
282               + " != " + lastSequenceInBatch;
283           if (LOG.isDebugEnabled()) {
284             debugMap.put(i + ": allThroughBatch", errorString);
285           }
286           LOG.error(errorString);
287           hib3GrouploaderLog.appendJobMessage(errorString);
288           //didnt get all the way through
289           error = true;
290         }
291         
292         if (error) {
293           hib3GrouploaderLog.setStatus(GrouperLoaderStatus.ERROR.name());
294           break;
295         }
296         
297         hib3GrouploaderLog.addTotalCount(changeLogEntryList.size());
298         
299         if (changeLogEntryList.size() < batchSize) {
300           break;
301         }
302         hib3GrouploaderLog.store();
303       }
304       if (LOG.isDebugEnabled()) {
305         debugMap.put("totalRecordsProcessed", hib3GrouploaderLog.getTotalCount());
306       }
307   
308       if (!error) {
309   
310         hib3GrouploaderLog.setStatus(GrouperLoaderStatus.SUCCESS.name());
311       }
312     } finally {
313       if (LOG.isDebugEnabled()) {
314         LOG.debug(GrouperUtil.mapToString(debugMap));
315       }
316 
317     }
318   }
