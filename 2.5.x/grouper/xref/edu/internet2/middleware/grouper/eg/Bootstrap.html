56  /**
57   * EXAMPLE Bootstrap your Groups Registry by creating a sysadmin (wheel) group.
58   * <p>Running <code>ant eg.bootstrap</code> will run this example code.</p>
59   * <h4>Step 0: Import Required Packages</h4>
60   * <p>To begin using the Grouper API you will first need to import the Grouper API,
61   * located in the <i>edu.internet2.middleware.grouper</i> package.  For many operations,
62   * including those in this example, you will also need to import the I2MI Subject API
63   * which is located in the <i>edu.internet2.middleware.subject</i> package.<p>
64   * <pre class="eg">
65   * import edu.internet2.middleware.grouper.*; // Import Grouper API
66   * import edu.internet2.middleware.subject.*; // Import Subject API
67   * </pre>
68   * <h4>Step 1: Find <i>GrouperSystem</i> Subject</h4>
69   * <p>The next step is to find the <i>GrouperSystem</i> subject.  This subject is internal
70   * to the Grouper API and is the "root" user for your Groups Registry.  As normal access
71   * and naming privileges do not apply to <i>GrouperSystem</i> you must as as this subject
72   * to bootstrap your registry for use by others.</p>
73   * <pre class="eg">
74   * Subject grouperSystem = SubjectFinder.findRootSubject();
75   * </pre>
76   * <h4>Step 2: Start Session</h4>
77   * <p>Almost all Grouper API operations take place within the context of a
78   * <code>GrouperSession</code> and each session has an associated subject.  The privilegs
79   * of the session's subject determine what API actions can be performed.  As this session
80   * will be acting as <i>GrouperSystem</i> there will be no restrictions.</code>
81   * <pre class="eg">
82   * try {
83   *  GrouperSession s = GrouperSession.start(grouperSystem);  
84   * }
85   * catch (SessionException eS) {
86   *   // Error starting session
87   * }
88   * </pre>
89   * <h4>Step 3: Find Root Stem</h4>
90   * <p>At the very base of the Groups Registry is the root stem.  All other stems and
91   * groups in the registry descend from this stem.</p>
92   * <pre class="eg">
93   * Stem root = StemFinder.findRootStem(s);
94   * </pre>
95   * <h4>Step 4: Find Or Add Top-Level Stem "etc" (or whatever is configured in grouper.properties)</h4>
96   * <p>After retrieving the root stem you can create a new top-level stem, in this case
97   * named "etc", beneath the root stem.  
98   * <pre class="eg">
99   * Stem   etc;
100  * String extn         = "etc";
101  * String displayExtn  = "Grouper Administration";
102  * // check to see if stem already exists
103  * try {
104  *   etc = StemFinder.findByName(s, extn);
105  * }
106  * catch (StemNotFoundException eNSNF) {
107  *   // create stem if it doesn't exist
108  *   try {
109  *     etc = root.addChildStem(extn, displayExtn);
110  *   }
111  *   catch (InsufficientPrivilegeException eIP) {
112  *     // not privileged to create top-level stem
113  *   }
114  *   catch (StemAddException eNSA) {
115  *     // error adding top-level stem
116  *   }
117  * }
118  * </pre>
119  * However, we are creating a dynamically named stem based on the wheel group named in grouper.properties:
120  * <pre class="eg">
121  *    try {
122  *      this.etc = Stem.saveStem(this.s, etcStem, null, etcStem, displayExtn, null, SaveMode.INSERT_OR_UPDATE, true);
123  *      System.err.println("created top-level stem: " + this.etc);
124  *    }
125  *    catch (Exception eIP) {
126  *      throw new GrouperRuntimeException( "error adding top-level stem: " + eIP.getMessage() + ", " + etcStem, eIP );
127  *    }
128  * </pre> 
129  * <h4>Step 5: Find Or Add Sysadmin Group "etc:sysadmin" (or whatever is configured in grouper.properties)</h4>
130  * <p>After adding the top-level "etc" stem you can then create the wheel group
131  * ("etc:sysadmin") beneath it.</p>
132  * <pre class="eg">
133  * Group   wheel;
134  * String  extn        = "wheel";
135  * String  displayExtn = "Wheel Group";
136  * // check to see if group exists
137  * try {
138  *   wheel = GroupFinder.findByName( s, etc.getName() + ":" + extn );
139  * }
140  * catch (GroupNotFoundException eGNF) {
141  *   try {
142  *     // create group if it doesn't exist
143  *     wheel = etc.addChildGroup(extn, displayExtn);
144  *   }
145  *   catch (GroupAddException eGA) {
146  *     // error adding wheel group
147  *   } 
148  *   catch (InsufficientPrivilegeException eIP) {
149  *     // not privileged to create wheel group
150  *   }
151  * }
152  * </pre>
153  * However, we are doing things more dynamically:
154  * <pre class="eg">
155  *    try {
156  *      // create group if it doesn't exist
157  *      this.wheel = Group.saveGroup(this.s, wheelGroupName, null, wheelGroupName, displayExtn, null, SaveMode.INSERT_OR_UPDATE, true);
158  *      System.err.println("created sysadmin (wheel) group: " + this.wheel);
159  *    }
160  *    catch (Exception eGA) {
161  *      throw new GrouperRuntimeException( "error adding sysadmin group: " + eGA.getMessage() + ", " + wheelGroupName, eGA );
162  *    }
163  * </pre>
164  * <h4>Step 6: Find <i>GrouperAll</i> Subject</h4>
165  * <p><i>GrouperAll</i> is another subject internal to Grouper.  When you assign a
166  * membership or grant a privilege to <i>GrouperAll</i> it is the equivalent of performing
167  * that operation for <b>*all subjects*.</b>.</p>
168  * <pre class="eg">
169  * Subject grouperAll = SubjectFinder.findAllSubject();
170  * </pre>
171  * <h4>Step 7: Add <i>GrouperAll</i> As Member Of The Wheel Group</h4>
172  * <p>Now that the wheel group exists you may add members to it.  Provided you have
173  * enabled use of a wheel group in <code>conf/grouper.properties</code> all members added
174  * to this group will now have root-like privileges over the entire Groups Registry.</p>
175  * <pre class="eg">
176  * // verify GrouperAll is not already a member before attempting to add it
177  * if ( !wheel.hasMember(grouperAll) ) {
178  *   try {
179  *     // this is *not* recommend for most grouper deployments as it will give every
180  *     // subject root-like privileges over the entire groups registry. 
181  *     wheel.addMember(grouperAll);
182  *   }
183  *   catch (InsufficientPrivilegeException eIP) {
184  *     // not privileged to add GrouperAll as member to wheel group
185  *   }
186  *   catch (MemberAddException eMA) {
187  *     // error adding GrouperAll as member to wheel group
188  *   }
189  * }
190  * </pre>
191  * <h4>Step 8: Stop Session</h4>
192  * <p>The final action after bootstrapping your Groups Registry is to stop the session you
193  * started at the beginning of the bootstrapping process.</p>
194  * <pre class="eg">
195  * try {
196  *  s.stop();
197  * }
198  * catch (SessionException eS) {
199  *   // Error stopping session
200  * }
201  * </pre>
202  * @author  blair christensen.
203  * @version $Id: Bootstrap.java,v 1.6 2009-03-15 06:37:24 mchyzer Exp $
204  * @see     <a href="http://viewvc.internet2.edu/viewvc.py/grouper/src/grouper/edu/internet2/middleware/grouper/eg/Bootstrap.java?root=I2MI&view=markup">Source</a>
205  * @since   1.2.0
206  */

228   public static void main(String args[]) {
229     int exitValue = 1; // indicate failure by default
230 
231     try {
232       final Bootstrap/grouper/eg/Bootstrap.html#Bootstrap">Bootstrap bs = new Bootstrap();
233       bs.findGrouperSystem();
234       bs.startSession();
235       GrouperSession.callbackGrouperSession(bs.s, new GrouperSessionHandler() {
236 
237         public Object callback(GrouperSession grouperSession)
238             throws GrouperSessionException {
239           bs.findRootStem();
240           bs.findOrAddTopLevelStem();
241           bs.findOrAddWheelGroup();
242           bs.findGrouperAll();
243           bs.addGrouperAllToWheelGroupIfNotAlreadyAMember();
244           return null;
245         }
246         
247       });
248       bs.stopSession();
249       exitValue = 0;
250     }
251     catch (GrouperException eGRE) {
252       System.err.println( eGRE.getMessage() );
253     }
254 
255     System.exit(exitValue);
256   }

268   private void addGrouperAllToWheelGroupIfNotAlreadyAMember() {
269     // verify GrouperAll is not already a member before attempting to add it
270     if ( !this.wheel.hasMember(grouperAll) ) {
271       try {
272         // this is *not* recommend for most grouper deployments as it will give every
273         // subject root-like privileges over the entire groups registry. 
274         this.wheel.addMember(grouperAll);
275         System.err.println("added GrouperAll to sysadmin (wheel) group");
276       }
277       catch (InsufficientPrivilegeException eIP) {
278         throw new GrouperException( "error adding GrouperAll to sysadmin (wheel) group: " + eIP.getMessage(), eIP );
279       }
280       catch (MemberAddException eMA) {
281         throw new GrouperException( "error adding GrouperAll to sysadmin (wheel) group: " + eMA.getMessage(), eMA );
282       }
283     }
284     else {
285       System.err.println("GrouperAll is already a member of the wheel group");
286     }
287   }
288 
289   private void findGrouperAll() {
290     this.grouperAll = SubjectFinder.findAllSubject();
291     System.err.println("found GrouperAll subject: " + this.grouperAll);
292   }
293 
294   private void findGrouperSystem() {
295     this.grouperSystem = SubjectFinder.findRootSubject();
296     System.err.println("found GrouperSystem subject: " + this.grouperSystem);
297   }
298 
299   private void findOrAddTopLevelStem() {
300     String etcStem = GrouperUtil.parentStemNameFromName(this.wheelGroupName());
301     String displayExtn  = "Grouper Administration";
302     // check to see if stem exists
303     try {
304       this.etc = StemFinder.findByName(this.s, etcStem, true);
305       System.err.println("found top-level stem: " + this.etc);
306     }
307     catch (StemNotFoundException eNSNF) {
308       // create stem if it doesn't exist
309       try {
310         this.etc = Stem.saveStem(this.s, etcStem, null, etcStem, displayExtn, null, SaveMode.INSERT_OR_UPDATE, true);
311         System.err.println("created top-level stem: " + this.etc);
312       }
313       catch (Exception eIP) {
314         throw new GrouperException( "error adding top-level stem: " + eIP.getMessage() + ", " + etcStem, eIP );
315       }
316     }
317   }
318 
319   private void findRootStem() {
320     this.root = StemFinder.findRootStem(this.s);
321     System.err.println("found root stem: " + this.root);
322   }
323 
324   private void findOrAddWheelGroup() {
325     String wheelGroupName = wheelGroupName();
326     String  extn        = GrouperUtil.extensionFromName(wheelGroupName);
327     
328     //if etc:wheel (the old way), use that as the friendly name
329     String  displayExtn = StringUtils.equals(wheelGroupName, GrouperConfig.retrieveConfig().propertyValueString("grouper.rootStemForBuiltinObjects", "etc") + ":wheel") 
330         ? "Wheel Group" 
331         : "Sysadmin group";
332     // check to see if group exists
333     try {
334       this.wheel = GroupFinder.findByName( s, this.etc.getName() + ":" + extn , true);
335       System.err.println("found sysadmin (wheel) group: " + this.wheel);
336     }
337     catch (GroupNotFoundException eGNF) {
338       try {
339         // create group if it doesn't exist
340         this.wheel = Group.saveGroup(this.s, wheelGroupName, null, wheelGroupName, displayExtn, displayExtn, SaveMode.INSERT_OR_UPDATE, true);
341         System.err.println("created sysadmin (wheel) group: " + this.wheel);
342       }
343       catch (Exception eGA) {
344         throw new GrouperException( "error adding sysadmin group: " + eGA.getMessage() + ", " + wheelGroupName, eGA );
345       }
346     }
347   }
348 
349 
350   /**
351    * @return
352    */
353   private String wheelGroupName() {
354     String wheelGroupName = GrouperConfig.retrieveConfig().propertyValueString("groups.wheel.group");
355     
356     if (StringUtils.isBlank(wheelGroupName)) {
357       throw new RuntimeException("grouper.properties must have an extry for " +
358           "groups.wheel.group, e.g. etc:sysadmingroup");
359     }
360     return wheelGroupName;
361   }
362 
363   private void startSession() {
364     try {
365       this.s = GrouperSession.start(this.grouperSystem, false);  
366       System.err.println("started session: " + this.s);
367     }
368     catch (SessionException eS) {
369       throw new GrouperException( "error starting session: " + eS.getMessage(), eS );
370     }
371   }
372 
373   private void stopSession() {
374     try {
375       this.s.stop();
376       System.err.println("stopped session");
377     }
378     catch (SessionException eS) {
379       throw new GrouperException( "error stopping session: " + eS.getMessage(), eS );
380     }
381   }
