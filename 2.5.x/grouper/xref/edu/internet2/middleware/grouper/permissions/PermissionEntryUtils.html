33    /**
34     * if internal heuristic is not set, set it, order by so most important as at top...
35     * @param permissionEntries
36     */
37    public static void orderByAndSetFriendlyHeuristic(List<PermissionEntry> permissionEntries) {
38      if (GrouperUtil.length(permissionEntries) < 1) {
39        return;
40      }
41      
42      for (PermissionEntry permissionEntry : permissionEntries) {
43        PermissionHeuristics permissionHeuristics = permissionEntry.getPermissionHeuristics();
44        if (permissionHeuristics == null) {
45          permissionHeuristics = new PermissionHeuristics();
46          permissionEntry.setPermissionHeuristics(permissionHeuristics);
47        }
48        if (permissionHeuristics.getInternalScore() == -1) {
49          long internalScore = PermissionHeuristic.computePermissionHeuristic(permissionEntry);
50          permissionHeuristics.setInternalScore(internalScore);
51        }
52      }
53      
54      Collections.sort(permissionEntries, new Comparator<PermissionEntry>() {
55  
56        public int compare(PermissionEntry../../../../edu/internet2/middleware/grouper/permissions/PermissionEntry.html#PermissionEntry">PermissionEntry o1, PermissionEntry o2) {
57          PermissionHeuristics permissionHeuristics1 = o1.getPermissionHeuristics();
58          PermissionHeuristics permissionHeuristics2 = o2.getPermissionHeuristics();
59          
60          Long score1 = permissionHeuristics1.getInternalScore();
61          Long score2 = permissionHeuristics2.getInternalScore();
62          
63          return score2.compareTo(score1);
64        }
65  
66      });
67      
68      int previousFriendlyScore = 0;
69      long previousHeuristic = -1;
70      for (PermissionEntry permissionEntry : permissionEntries) {
71        PermissionHeuristics permissionHeuristics = permissionEntry.getPermissionHeuristics();
72        if (previousHeuristic == -1) {
73          permissionHeuristics.setFriendlyScore(1);
74        } else {
75          //if equal then same score
76          if (permissionHeuristics.getInternalScore() == previousHeuristic) {
77            permissionHeuristics.setFriendlyScore(previousFriendlyScore);
78          } else {
79            permissionHeuristics.setFriendlyScore(previousFriendlyScore + 1);
80          }
81        }
82        
83        
84        previousFriendlyScore = permissionHeuristics.getFriendlyScore();
85        previousHeuristic = permissionHeuristics.getInternalScore();
86        
87        
88      }  
89    }

91    /**
92     * see if a permission is in the list of entries
93     * @param permissionEntries
94     * @param roleName
95     * @param attributeDefNameName
96     * @param action
97     * @param subjectSourceId
98     * @param subjectId
99     * @return true if the item is in the list
100    */
101   public static boolean collectionContains(Collection<PermissionEntry> permissionEntries, String roleName, 
102       String attributeDefNameName, String action, String subjectSourceId, String subjectId) {
103     return collectionFindFirst(permissionEntries, roleName, attributeDefNameName, action, subjectSourceId, subjectId, null, false) != null;
104   }

123   /**
124    * find the first permission entry in the list of entries
125    * @param permissionEntries
126    * @param roleName
127    * @param attributeDefNameName
128    * @param action
129    * @param subjectSourceId
130    * @param subjectId
131    * @param permissionType e.g. role or role_subject
132    * @param considerPermissionType 
133    * @return true if the item is in the list
134    */
135   public static PermissionEntry collectionFindFirst(Collection<PermissionEntry> permissionEntries, String roleName, 
136       String attributeDefNameName, String action, String subjectSourceId, String subjectId, String permissionType, boolean considerPermissionType) {
137     for (PermissionEntry permissionEntry : GrouperUtil.nonNull(permissionEntries)) {
138       if (StringUtils.equals(roleName, permissionEntry.getRoleName())
139           && StringUtils.equals(attributeDefNameName, permissionEntry.getAttributeDefNameName())
140           && StringUtils.equals(action, permissionEntry.getAction())
141           && StringUtils.equals(subjectSourceId, permissionEntry.getSubjectSourceId())
142           && StringUtils.equals(subjectId, permissionEntry.getSubjectId())
143           && (considerPermissionType ? StringUtils.equals(permissionType, permissionEntry.getPermissionTypeDb()) : true)
144            ) {
145         return permissionEntry;
146       }
147     }
148     return null;
149   }
