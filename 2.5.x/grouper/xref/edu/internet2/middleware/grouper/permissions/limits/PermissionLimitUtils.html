342   /**
343    * add standard limit variables
344    * @param limitEnvVarsObject
345    */
346   public static void addStandardLimitVariablesIfNotExist(Map<String, Object> limitEnvVarsObject) {
347     Calendar calendar = new GregorianCalendar();
348     calendar.setTimeInMillis(System.currentTimeMillis());
349     
350     //sunday is one
351     int dayOfWeek = calendar.get(Calendar.DAY_OF_WEEK);
352     
353     int hourOfDay = calendar.get(Calendar.HOUR_OF_DAY);
354     int minuteOfHour = calendar.get(Calendar.MINUTE);
355     int minuteOfDay = (hourOfDay * 60) + minuteOfHour;
356     
357     //january is 0
358     int monthOfYear = calendar.get(Calendar.MONTH);
359     
360     if (!limitEnvVarsObject.containsKey(CALENDAR)) {
361       limitEnvVarsObject.put(CALENDAR, calendar);
362     }
363     if (!limitEnvVarsObject.containsKey(DAY_OF_WEEK)) {
364       limitEnvVarsObject.put(DAY_OF_WEEK, dayOfWeek);
365     }
366     if (!limitEnvVarsObject.containsKey(HOUR_OF_DAY)) {
367       limitEnvVarsObject.put(HOUR_OF_DAY, hourOfDay);
368     }
369     if (!limitEnvVarsObject.containsKey(MINUTE_OF_HOUR)) {
370       limitEnvVarsObject.put(MINUTE_OF_HOUR, minuteOfHour);
371     }
372     if (!limitEnvVarsObject.containsKey(MINUTE_OF_DAY)) {
373       limitEnvVarsObject.put(MINUTE_OF_DAY, minuteOfDay);
374     }
375     if (!limitEnvVarsObject.containsKey(MONTH_OF_YEAR)) {
376       limitEnvVarsObject.put(MONTH_OF_YEAR, monthOfYear);
377     }
378     
379   }

381   /** limit logic map */
382   static ExpirableCache<Boolean, Set<String>> limitRealms = new ExpirableCache<Boolean, Set<String>>(5);
383 
384   /** 
385    * pattern to parse the limit realms
386    * "grouper.permissions.limits.realm." + realmName
387    */
388   private static Pattern limitRealmPattern = Pattern.compile("^grouper\\.permissions\\.limits\\.realm\\.([^.]+)$");
389 
390   /**
391    * get a logic instance based on attributeDefName of the limit
392    * @return an instance of the interface
393    */
394   @SuppressWarnings("unchecked")
395   public static Set<String> limitRealms() {
396     
397     Set<String> limitRealmsSet = limitRealms.get(Boolean.TRUE);
398 
399     if (limitRealmsSet == null) {
400       
401       synchronized (PermissionLimitUtils.class) {
402         
403         limitRealmsSet = limitRealms.get(Boolean.TRUE);
404         
405         if (limitRealmsSet == null) {
406           limitRealmsSet = new TreeSet<String>();
407           
408           Matcher matcher = null;
409           
410           //lets get the sources
411           for (String sourcePropertyName : GrouperConfig.retrieveConfig().propertyNames()) {
412             
413             matcher = limitRealmPattern.matcher(sourcePropertyName);
414             if (matcher.matches()) {
415               String limitRealm = matcher.group(1);
416               limitRealmsSet.add(limitRealm);
417             }
418           }
419           limitRealms.put(Boolean.TRUE, limitRealmsSet);
420           
421         }
422       }
423     }
424     
425     return limitRealmsSet;
426     
427   }

441   /** 
442    * pattern to parse the limit configs
443    * grouper.permissions.limits.logic.someName.limitName
444    */
445   private static Pattern limitNamePattern = Pattern.compile("^grouper\\.permissions\\.limits\\.logic\\.([^.]+)\\.limitName$");
446   
447   
448   /**
449    * get a logic instance based on attributeDefName of the limit
450    * @param limitName name of the attribute def name
451    * @return an instance of the interface
452    */
453   @SuppressWarnings("unchecked")
454   public static PermissionLimitInterface logicInstance(String limitName) {
455     
456     Map<String, Class<PermissionLimitInterface>> limitMap = limitLogicMap.get(Boolean.TRUE);
457 
458     if (limitMap == null) {
459       
460       synchronized (PermissionLimitUtils.class) {
461         
462         limitMap = limitLogicMap.get(Boolean.TRUE);
463         
464         if (limitMap == null) {
465           limitMap = new HashMap<String, Class<PermissionLimitInterface>>();
466           //lets add standard entries, do this first so they can be overridden
467           limitMap.put(PermissionLimitUtils.limitElName(), (Class<PermissionLimitInterface>)(Object)PermissionLimitElLogic.class);
468           limitMap.put(PermissionLimitUtils.limitWeekday9to5Name(), (Class<PermissionLimitInterface>)(Object)PermissionLimitWeekday9to5Logic.class);
469           limitMap.put(PermissionLimitUtils.limitAmountLessThanName(), (Class<PermissionLimitInterface>)(Object)PermissionLimitAmountLessThan.class);
470           limitMap.put(PermissionLimitUtils.limitAmountLessThanOrEqualName(), (Class<PermissionLimitInterface>)(Object)PermissionLimitAmountLessThanEquals.class);
471           limitMap.put(PermissionLimitUtils.limitIpOnNetworksName(), (Class<PermissionLimitInterface>)(Object)PermissionLimitIpOnNetworks.class);
472           limitMap.put(PermissionLimitUtils.limitIpOnNetworkRealmName(), (Class<PermissionLimitInterface>)(Object)PermissionLimitIpOnNetworkRealm.class);
473           limitMap.put(PermissionLimitUtils.limitLabelsContainName(), (Class<PermissionLimitInterface>)(Object)PermissionLimitLabelsContain.class);
474           
475           Matcher matcher = null;
476           
477           //lets get the sources
478           for (String sourcePropertyName : GrouperConfig.retrieveConfig().propertyNames()) {
479             
480             matcher = limitNamePattern.matcher(sourcePropertyName);
481             if (matcher.matches()) {
482               String limitKey = matcher.group(1);
483               String className = GrouperConfig.retrieveConfig().propertyValueString("grouper.permissions.limits.logic." + limitKey + ".logicClass");
484               String limitAttributeName = GrouperConfig.retrieveConfig().propertyValueString(sourcePropertyName);
485               
486               try {
487                 Class<PermissionLimitInterface> theClass = GrouperUtil.forName(className);
488                 
489                 limitMap.put(limitAttributeName, theClass);
490                 
491               } catch (RuntimeException runtimeException) {
492                 LOG.error("Error with classname for limit, maybe class not on classpath or it is misconfigured in grouper.properties: " + sourcePropertyName + ", classname: " + className, runtimeException);
493                 //we will need to continue so one problem doesnt ruin it for all
494                 continue;
495               }
496             }
497           }
498           limitLogicMap.put(Boolean.TRUE, limitMap);
499           
500         }
501       }
502     }
503     Class<PermissionLimitInterface> theClass = limitMap.get(limitName);
504     
505     if (theClass == null) {
506       throw new RuntimeException("Cant find configuration for limit: " + limitName + " in grouper.properties or built-ins");
507     }
508     
509     return GrouperUtil.newInstance(theClass);
510     
511   }

513   /** limit logic map */
514   static ExpirableCache<Boolean, Set<Class<?>>> limitElClassesMap = new ExpirableCache<Boolean, Set<Class<?>>>(5);
515 
516   /**
517    * custom el instances to add to the variable map for limit EL
518    * @return the map
519    */
520   public static Map<String, Object> limitElClasses() {
521     
522     //grouper.permissions.limits.el.classes
523 
524     Set<Class<?>> limitElClasses = limitElClassesMap.get(Boolean.TRUE);
525 
526     if (limitElClasses == null) {
527 
528       synchronized (PermissionLimitUtils.class) {
529 
530         limitElClasses = limitElClassesMap.get(Boolean.TRUE);
531 
532         if (limitElClasses == null) {
533 
534           limitElClasses = new HashSet<Class<?>>();
535 
536           //middleware.grouper.rules.MyRuleUtils
537           String customElClasses = GrouperConfig.retrieveConfig().propertyValueString("grouper.permissions.limits.el.classes");
538 
539           if (!StringUtils.isBlank(customElClasses)) {
540             String[] customElClassesArray = GrouperUtil.splitTrim(customElClasses, ",");
541             for (String customElClass : customElClassesArray) {
542               Class<?> customClassClass = GrouperUtil.forName(customElClass);
543               limitElClasses.add(customClassClass);
544             }
545           }
546 
547 
548           //lets add standard entries, do this first so they can be overridden
549           limitElClassesMap.put(Boolean.TRUE, limitElClasses);
550 
551         }
552       }
553     }
554     
555     //get new instances each time
556     Map<String,Object> result = new HashMap<String, Object>();
557     for (Class<?> customClassClass : limitElClasses) {
558       String simpleName = StringUtils.uncapitalize(customClassClass.getSimpleName());
559       result.put(simpleName, GrouperUtil.newInstance(customClassClass));
560       
561     }
562 
563     return result;
564     
565   }
