31  package edu.internet2.middleware.grouper.privs;
32  
33  import edu.internet2.middleware.grouper.GrouperSession;
34  import edu.internet2.middleware.grouper.cfg.GrouperConfig;
35  import edu.internet2.middleware.grouper.exception.GrouperException;
36  import edu.internet2.middleware.grouper.internal.util.ParameterHelper;
37  import edu.internet2.middleware.grouper.internal.util.Realize;
38  
39  /** 
40   * Factory for returning a <code>AttributeDefResolver</code>.
41   * @author  blair christensen.
42   * @version $Id: AttributeDefResolverFactory.java,v 1.1 2009-09-21 06:14:26 mchyzer Exp $
43   * @since   1.2.1
44   */
45  public class AttributeDefResolverFactory {
46  
47    /** */
48    private static AttributeDefResolver resolver;
49  
50    /** */
51    private static ParameterHelperinternal/util/ParameterHelper.html#ParameterHelper">ParameterHelper param = new ParameterHelper();;
52  
53    /**
54     * @param session 
55     * @return  <code>AttributeDefResolver</code> instance.
56     * @throws  IllegalArgumentException if any parameter is null.
57     */
58    public static AttributeDefResolver getInstance(GrouperSession session)
59        throws IllegalArgumentException {
60      return getInstance(
61          session, (AttributeDefAdapter) Realize.instantiate(GrouperAttributeDefAdapter.class.getName()));
62    }
63  
64    /**
65     * Returns chain of attrDef resolvers.
66     * <p>Order of execution:</p>
67     * <ol>
68     *  <li>{@link ValidatingAccessResolver}</li>
69     *  <li>{@link WheelAccessResolver}</li>
70     *  <li>{@link CachingAccessResolver}</li>
71     *  <li>{@link GrouperSystemAccessResolver}</li>
72     *  <li>{@link GrouperAllAccessResolver}</li>
73     *  <li>{@link AccessWrapper}</li>
74     * </ol>
75     * @param session 
76     * @param attributeDefAdapter 
77     * @return  <code>AccessResolver</code> instance.
78     * @throws  GrouperException if unable to get instance.
79     * @throws  IllegalArgumentException if any parameter is null.
80     */
81    public static AttributeDefResolver getInstance(GrouperSession session, AttributeDefAdapter attributeDefAdapter)
82        throws GrouperException,
83        IllegalArgumentException {
84      param.notNullGrouperSession(session).notNullAttrDefAdapter(attributeDefAdapter);
85      return new ValidatingAttrDefResolver (new WheelAttrDefResolver(
86          new CachingAttrDefResolver(new GrouperSystemAttrDefResolver(
87          new GrouperAllAttrDefResolver(new AttributeDefWrapper(session, attributeDefAdapter))))));
88    }
89  
90    /**
91     * @param session 
92     * @return  Singleton <code>AttributeDefResolver</code>.
93     * @throws  GrouperException if unable to get resolver.
94     * @throws  IllegalArgumentException if any parameter is null.
95     */
96    public static AttributeDefResolver getResolver(GrouperSession session)
97        throws GrouperException,
98        IllegalArgumentException {
99      if (resolver == null) {
100       resolver = getInstance(session);
101     }
102     return resolver;
103   }
