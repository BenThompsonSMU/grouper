74    /**
75     * 
76     * @param attributeDef
77     * @param subj
78     * @param priv
79     * @return if has priv, or null if not known
80     */
81    private Boolean getFromHasPrivilegeCache(AttributeDef attributeDef, Subject subj,
82        Privilege priv) {
83      // TODO 20070823 are these the right element keys to use?
84      Element el = EhcacheController.ehcacheController().getCache(CACHE_HASPRIV).get(
85          new MultiKey(attributeDef.getId(), subj.getSourceId(), subj.getId(), priv));
86      if (el != null) {
87        return (Boolean) el.getObjectValue();
88      }
89      return null;
90    }

122   /**
123    * 
124    * @see edu.internet2.middleware.grouper.privs.AttributeDefResolver#getPrivileges(edu.internet2.middleware.grouper.attr.AttributeDef, edu.internet2.middleware.subject.Subject)
125    */
126   public Set<AttributeDefPrivilege> getPrivileges(AttributeDef attributeDef,
127       Subject subject)
128       throws IllegalArgumentException {
129     //2007-11-02 Gary Brown
130     //https://bugs.internet2.edu/jira/browse/GRP-30
131     //Needs to return actual privileges but also
132     //cache true/false for each possible Privilege
133     Set<AttributeDefPrivilege> privs = super.getDecoratedResolver().getPrivileges(
134         attributeDef, subject);
135     Set<String> privsSet = new HashSet<String>();
136     AttributeDefPrivilege ap = null;
137     Iterator it = privs.iterator();
138     while (it.hasNext()) {
139       ap = (AttributeDefPrivilege) it.next();
140       privsSet.add(ap.getName());
141     }
142     Set<Privilege> attrDefPrivs = Privilege.getAttributeDefPrivs();
143     Iterator<Privilege> attributeDefPrivsIterator = attrDefPrivs.iterator();
144     Privilege p = null;
145     while (attributeDefPrivsIterator.hasNext()) {
146       p = attributeDefPrivsIterator.next();
147       putInHasPrivilegeCache(attributeDef, subject, p, 
148           new Boolean(privsSet.contains(p.getName())));
149     }
150     return privs;
151   }

176   public void grantPrivilege(AttributeDef attributeDef, Subject subject, Privilege privilege, String uuid)
177       throws IllegalArgumentException,
178       UnableToPerformException {
179     // TODO 20070816 add caching
180     super.getDecoratedResolver().grantPrivilege(attributeDef, subject, privilege, uuid);
181     this.flushCache();
182     //there is a problem where if this action happens in root session, the
183     //normal session doesnt get flushed
184     GrouperSession grouperSession = GrouperSession.staticGrouperSession();
185     grouperSession.getAttributeDefResolver().flushCache();
186     this.putInHasPrivilegeCache(attributeDef, subject, privilege, Boolean.TRUE);

189   /**
190    * 
191    * @see edu.internet2.middleware.grouper.privs.AttributeDefResolver#hasPrivilege(edu.internet2.middleware.grouper.attr.AttributeDef, edu.internet2.middleware.subject.Subject, edu.internet2.middleware.grouper.privs.Privilege)
192    */
193   public boolean hasPrivilege(AttributeDef attributeDef, Subject subject, Privilege privilege)
194       throws IllegalArgumentException {
195     Boolean rv = this.getFromHasPrivilegeCache(attributeDef, subject, privilege);
196     if (rv == null) {
197       //2007-11-02 Gary Brown
198       //https://bugs.internet2.edu/jira/browse/GRP-30
199       //Get all the privileges - which will then be cached
200       //on the assumption we will be checking other privileges
201       getPrivileges(attributeDef, subject);
202       //must be in the cache now
203       rv = this.getFromHasPrivilegeCache(attributeDef, subject, privilege);
204     }
205     //Hopefully redundant
206     if (rv == null) {
207       rv = super.getDecoratedResolver().hasPrivilege(attributeDef, subject, privilege);
208       this.putInHasPrivilegeCache(attributeDef, subject, privilege, rv);
209     }
210     return rv;
211   }

304    */
305   public Set<AttributeDef> postHqlFilterAttrDefs(Set<AttributeDef> attributeDefs, Subject subject,
306       Set<Privilege> privInSet) {
307 
308     Set<AttributeDef> filteredAttrDefs = super.getDecoratedResolver().postHqlFilterAttrDefs(attributeDefs,
309         subject, privInSet);
310 
311     //add to cache
312     for (AttributeDef attributeDef : attributeDefs) {
313       putInHasPrivilegeCache(attributeDef, subject, AttributeDefPrivilege.ATTR_VIEW, filteredAttrDefs
314           .contains(attributeDef));
315     }
316 
317     //return filtered groups
318     return filteredAttrDefs;
319   }

395    * @see edu.internet2.middleware.grouper.privs.AttributeDefResolver#postHqlFilterPermissions(edu.internet2.middleware.subject.Subject, java.util.Set)
396    */
397   public Set<PermissionEntry> postHqlFilterPermissions(Subject subject,
398       Set<PermissionEntry> permissionsEntries) {
399     AttributeDefResolver decoratedResolver = super.getDecoratedResolver();
400 
401     Set<PermissionEntry> filteredPermissions = decoratedResolver.postHqlFilterPermissions(
402         subject, permissionsEntries);
403 
404     for (PermissionEntry permissionEntry : permissionsEntries) {
405       
406       // make sure we're not putting an inactive point in time entry in the cache...
407       if (permissionEntry.isActive()) {
408         putInHasPrivilegeCache(permissionEntry, subject, AttributeDefPrivilege.ATTR_VIEW,
409             filteredPermissions.contains(permissionEntry));
410       }
411     }
412 
413     return filteredPermissions;
414   }
