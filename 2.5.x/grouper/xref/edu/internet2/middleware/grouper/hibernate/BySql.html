21  import java.sql.Connection;
22  import java.sql.PreparedStatement;
23  import java.util.List;
24  
25  import org.apache.commons.logging.Log;
26  import org.hibernate.internal.SessionImpl;
27  import org.hibernate.type.Type;
28  
29  import edu.internet2.middleware.grouper.util.GrouperUtil;
30  
31  /**
32   * 
33   * for simple HQL, use this instead of inverse of control.
34   * this will do proper error handling and descriptive exception
35   * handling.  This will by default use the transaction modes
36   * GrouperTransactionType.READONLY_OR_USE_EXISTING, and 
37   * GrouperTransactionType.READ_WRITE_OR_USE_EXISTING depending on
38   * if a transaction is needed.
39   * 
40   * @author mchyzer
41   *
42   */
43  public class BySql extends HibernateDelegate {
44    
45    /** logger */
46    @SuppressWarnings("unused")
47    private static final Log LOG = GrouperUtil.getLog(BySql.class);
48  
49    /** query count exec queries, used for testing */
50    public static int queryCountQueries = 0;
51    
52  
53    /**
54     * assign a different grouperTransactionType (e.g. for autonomous transactions)
55     * @param theGrouperTransactionType
56     * @return the same object for chaining
57     * @TODO remove in future grouper version 2.4
58     */
59    public BySql setGrouperTransactionType(GrouperTransactionType 
60        theGrouperTransactionType) {
61      return this;
62    }
63    
64    /**
65     * string value for error handling
66     * @return the string value
67     * @TODO remove in future grouper release 2.4
68     */
69    @Override
70    public String toString() {
71      return super.toString();
72    }
73  
74    /**
75     * set the query to run
76     * @param theHqlQuery
77     * @return this object for chaining
78     * @TODO remove in future grouper release 2.4
79     */
80    public BySql createQuery(String theHqlQuery) {
81      return this;
82    }
83  
84    /**
85     * execute some sql
86     * @param sql can be insert, update, delete, or ddl
87     * @param params prepared statement params
88     * @return the number of rows affected or 0 for ddl
89     * @deprecated doesnt work with postgres, need to pass in param types explicitly since cant determine them if null
90     */
91    @Deprecated
92    public int executeSql(final String sql, final List<Object> params) {
93    
94  
95      return executeSql(sql, params, BySqlStatic.convertParamsToTypes(params));
96    }
97  
98    /**
99     * execute some sql
100    * @param sql can be insert, update, delete, or ddl
101    * @param params prepared statement params
102    * @param types
103    * @return the number of rows affected or 0 for ddl
104    */
105   public int executeSql(final String sql, final List<Object> params, final List<Type> types) {
106   
107     HibernateSession hibernateSession = this.getHibernateSession();
108     hibernateSession.misc().flush();
109     
110     PreparedStatement preparedStatement = null;
111     try {
112       
113       //we dont close this connection or anything since could be pooled
114       Connection connection = ((SessionImpl)hibernateSession.getSession()).connection();
115       preparedStatement = connection.prepareStatement(sql);
116   
117       BySqlStatic.attachParams(preparedStatement, params, types);
118       
119       GrouperContext.incrementQueryCount();
120       int result = preparedStatement.executeUpdate();
121       
122       return result;
123 
124     } catch (Exception e) {
125       throw new RuntimeException("Problem with query in bysqlstatic: " + sql, e);
126     } finally {
127       GrouperUtil.closeQuietly(preparedStatement);
128     }
129   
130   }
131 
132 
133   /**
134    * @param theHibernateSession
135    */
136   public BySql(HibernateSession theHibernateSession) {
137     super(theHibernateSession);
138   }
