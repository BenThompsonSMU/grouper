153     dataForTier.put("tbProductVersion", GrouperVersion.GROUPER_VERSION);
154         
155     if (!StringUtils.isEmpty(System.getenv("GROUPER_CONTAINER_VERSION"))) {

182           } else {
183             if (othersAfter) {
184               patchStrings.add(engineLabel + maxConsecutive + "AndUnknownPatches");
185             } else {
186               patchStrings.add(engineLabel + maxConsecutive);
187             }
188           }
189         }
190       }
191           
192       if (patchStrings.size() > 0) {
193         release = release + "-" + String.join("-", patchStrings);
194       }
195       
196       dataForTier.put("tbTIERRelease", release); 
197     }
198     
199     // sleep for random time between 0 and 10 minutes so collector doesn't get hit all at once..
200     int sleepTime = ThreadLocalRandom.current().nextInt(0, 600000);
201     LOG.info("Sleeping for " + sleepTime + " milliseconds");
202 
203     try {
204       Thread.sleep(sleepTime);
205     } catch (InterruptedException e) {
206       // ignore
207     }
208     
209     try {
210       sendToTier(dataForTier);
211       otherJobInput.getHib3GrouperLoaderLog().setJobMessage("Finished successfully running TIER instrumentation daemon.");

215     } finally {
216       otherJobInput.getHib3GrouperLoaderLog().store();
217     }
218 
219     return null;
220   }

222   private static void sendToTier(Map<String, String> data) throws IOException {
223     String dataJson = GrouperUtil.jsonConvertTo(data, false);
224     //System.out.println(dataJson);
225     
226     HttpClient httpClient = new HttpClient();
227     httpClient.getParams().setSoTimeout(60000);
228     httpClient.getParams().setParameter("http.connection.timeout", 60000);
229     String collectorUrl = GrouperLoaderConfig.retrieveConfig().propertyValueStringRequired("otherJob.tierInstrumentationDaemon.collectorUrl");
230     
231     PostMethod method = new PostMethod(collectorUrl);
232     method.setRequestEntity(new StringRequestEntity(dataJson, "application/json", "UTF-8"));
233     int collectorReturnCode = httpClient.executeMethod(method);
234     String collectorBody = IOUtils.toString(method.getResponseBodyAsStream());
235 
236     if (collectorReturnCode == 200 || collectorReturnCode == 201) {
237       // we're good
238       LOG.info("Successfully sent data to endpoint " + collectorUrl);
239     } else {    
240       throw new RuntimeException("Failed to send data to endpoint " + collectorUrl + ".  Code=" + collectorReturnCode + ", body=" + collectorBody);    
241     }
242   }

358     
359     List<JSONObject> list = new LinkedList<JSONObject>();
360     
361     List<InstrumentationDataInstance> instances = InstrumentationDataInstanceFinder.findAll(true);
362     for (InstrumentationDataInstance instance : instances) {
363       JSONObject jsonObject = new JSONObject();
364       jsonObject.put("uuid", instance.getUuid());
365       jsonObject.put("engineName", instance.getEngineName());
366       jsonObject.put("serverLabel", instance.getServerLabel());
367       
368       if (instance.getLastUpdate() != null) {
369         jsonObject.put("lastUpdate", instance.getLastUpdate().getTime());
370       }
371       
372       Set<String> newCounts = new LinkedHashSet<String>();
373 
374       List<InstrumentationDataInstanceCounts> instanceCountsList = instance.getCounts();
375       for (InstrumentationDataInstanceCounts instanceCounts : instanceCountsList) {
376         if (instanceCounts.getCreatedOn().getTime() >= lastCollectorUpdate && instanceCounts.getCreatedOn().getTime() < currentCollectorStart) {
377           Map<String, Long> currentNewCounts = new LinkedHashMap<String, Long>();
378           currentNewCounts.put("startTime", instanceCounts.getStartTime().getTime());
379           currentNewCounts.put("duration", instanceCounts.getDuration());
380           for (String key : instanceCounts.getCounts().keySet()) {
381             currentNewCounts.put(key, instanceCounts.getCounts().get(key));
382           }
383           
384           newCounts.add(GrouperUtil.jsonConvertTo(currentNewCounts, false));
385         }
386       }
387       
388       jsonObject.put("newCounts", newCounts);
389       
390       list.add(jsonObject);
391     }
392     
393     return list;

396   private static String getInstitution() {
397     String institution = GrouperConfig.retrieveConfig().getProperty("grouper.institution.name", null);
398     if (!StringUtils.isBlank(institution)) {
399       return institution;
400     }
401     
402     String hostname = GrouperUtil.hostname();
403     if (StringUtils.isBlank(hostname)) {
404       return "";
405     }
406     
407     int index = StringUtils.indexOf(hostname, ".");
408     
409     if (index == -1) {
410       return "";
411     }
412     
413     return hostname.substring(index + 1);
414   }

443   private static void sendToTier(Map<String, Object> data) throws HttpException, IOException {
444     String dataJson = GrouperUtil.jsonConvertTo(data, false);
445     
446     HttpClient httpClient = new HttpClient();
447     String discoveryUrl = GrouperLoaderConfig.retrieveConfig().propertyValueString("otherJob.tierInstrumentationDaemon.discoveryUrl", "https://id.internet2.edu/ti/jrd/collector");
448     GetMethod discoveryMethod = new GetMethod(discoveryUrl);
449     int discoveryReturnCode = httpClient.executeMethod(discoveryMethod);
450     String discoveryBody = IOUtils.toString(discoveryMethod.getResponseBodyAsStream());
451 
452     if (discoveryReturnCode != 200) {
453       throw new RuntimeException("Bad response code from discovery url " + discoveryUrl + ".  Code=" + discoveryReturnCode + ", body=" + discoveryBody);
454     }
455     
456     TierDiscovery tierDiscovery = GrouperUtil.jsonConvertFrom(discoveryBody, TierDiscovery.class);
457     if (!tierDiscovery.isServiceEnabled()) {
458       LOG.warn("TIER discovery service indicates that the TIER collector is disabled currently.");
459       return;
460     }
461     
462     List<String> uris = new LinkedList<String>();
463     
464     for (Map<String, String> endpoint : tierDiscovery.getEndpoints()) {
465       String uri = endpoint.get("uri");
466       if (!StringUtils.isBlank(uri)) {
467         uris.add(uri);
468       }
469     }
470     
471     if (uris.size() == 0) {
472       LOG.warn("TIER discovery service doesn't have any URIs for the TIER collector.");
473       return;
474     }
475     
476     Iterator<String> uriIterator = uris.iterator();
477     while (uriIterator.hasNext()) {
478       String uri = uriIterator.next();
479       
480       // try to send results here
481       httpClient = new HttpClient();
482       PostMethod collectorMethod = new PostMethod(uri);
483       try {
484         collectorMethod.setRequestEntity(new StringRequestEntity(dataJson, "application/json", "UTF-8"));
485         int collectorReturnCode = httpClient.executeMethod(collectorMethod);
486         String collectorBody = IOUtils.toString(collectorMethod.getResponseBodyAsStream());
487 
488         if (collectorReturnCode == 200 || collectorReturnCode == 201) {
489           // we're good
490           LOG.info("Successfully sent data to endpoint " + uri);
491           break;
492         }
493         
494         throw new RuntimeException("Failed to send data to endpoint " + uri + ".  Code=" + collectorReturnCode + ", body=" + collectorBody);
495       } catch (Exception e) {
496         if (uriIterator.hasNext()) {
497           LOG.warn("Failed to send data to endpoint " + uri + ".  Will try another endpoint", e);
498         } else {
499           throw new RuntimeException(e);
500         }
501       }
502     }
