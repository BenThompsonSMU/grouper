105   /**
106    * @param args 
107    * @since   1.3.1
108    */
109   public static void main(String[] args) {
110     Options options = new Options();
111     OptionGroup optionGroup = new OptionGroup();
112     optionGroup.addOption(new Option("all", false, 
113       "Find bad memberships."));
114     optionGroup.setRequired(true);
115     options.addOptionGroup(optionGroup);
116 
117     if (args.length == 0) {
118       printUsage(options);
119       System.exit(0);
120     }
121 
122     CommandLineParser parser = new GnuParser();
123     CommandLine line = null;
124     try {
125       line = parser.parse(options, args);
126     } catch (ParseException e) {
127       System.err.println(e.getMessage());
128       printUsage(options);
129       System.exit(1);
130     }
131     clearResults();
132     // maybe this should go to a log file instead?
133     printErrorsToSTOUT = true;
134     
135     GrouperSession grouperSession = null;
136     
137     try {
138       grouperSession = GrouperSession.startRootSession();
139 
140       if (line.hasOption("all")) {
141         checkAll(out);
142       } else {
143         printUsage(options);
144         System.exit(0);
145       }
146 
147     } catch (Exception e) {
148       System.err.println(e.getMessage());
149       System.exit(1);
150     } finally {
151       GrouperSession.stopQuietly(grouperSession);
152     }
153 
154     out.println();
155     out.println();
156     if (gshScript == null) {
157       out.println("No membership errors found.");
158     } else {
159       out.println("Membership errors have been found.  Do the following to resolve the errors:");
160       writeFile(gshScript, gshScriptFilename);
161       out.println(" - Review the GSH script before applying any changes to your database.");
162       out.println(" - Execute the GSH Script " + gshScriptFilename);
163       out.println(" - Re-run the bad membership finder utility to verify that bad memberships have been fixed.");
164     }
165     System.exit(0);
166   }

168   /**
169    * @param printStream 
170    * @throws SessionException
171    */
172   public static void checkAll(PrintStream printStream) throws SessionException {
173     PrintStream oldPrintStream = out;
174     out = printStream;
175     try {  
176       checkAll();
177     } finally {
178       out = oldPrintStream;
179     }
180   }

182   /**
183    * @return count of bad and missing memberships
184    */
185   public static long checkAll() {
186     long errors = 0;
187     
188     if (printErrorsToSTOUT) {
189       out.println();
190       out.println("Checking Composite Memberships");
191     }
192     
193     errors += checkComposites();
194     
195     if (errors == 0) {
196       // only check group sets if composites are okay
197       if (printErrorsToSTOUT) {
198         out.println("Checking Group Sets");
199       }
200       errors += checkGroupSets();
201     } else {
202       if (printErrorsToSTOUT) {
203         out.println("*** Skipping group set check since there were composite issues. Re-run after fixing composites. ***\n");
204       }
205     }
206     
207     if (printErrorsToSTOUT) {
208       out.println("Checking memberships where member is a deleted group");
209     }
210     errors += checkDeletedGroupAsMember();
211     
212     
213     if (printErrorsToSTOUT) {
214       out.println("Checking GrouperAll");
215     }
216     errors += checkGrouperAll();
217     
218     return errors;
219   }

253   /**
254    * Note that this isn't a complete check of the group set table..
255    * @return count of bad and missing group sets
256    */
257   public static long checkGroupSets() {
258     Set<GroupSet> badType = GrouperDAOFactory.getFactory().getGroupSet().findTypeMismatch();
259 
260     for (GroupSet gs : badType) {
261       if (gs.getDepth() != 0) {
262         throw new RuntimeException("Unexpected depth of " + gs.getDepth());
263       }
264       
265       Group g = GrouperDAOFactory.getFactory().getGroup().findByUuid(gs.getOwnerGroupId(), true);
266       Composite c = GrouperDAOFactory.getFactory().getComposite().findAsOwner(g, false);
267       String currentType = gs.getType();
268       String newType = c == null ? "immediate" : "composite";
269       if (printErrorsToSTOUT) {
270         out.println("Bad group set type: owner groupId=" + g.getUuid() + ", owner group name=" + g.getName() + ", currentType=" + currentType + ", newType=" + newType + ".");
271       }
272       
273       logGshScript("sqlRun(\"update grouper_group_set set mship_type='" + newType + "' where id='" + gs.getId() + "'\");\n");
274     }
275     
276     Set<GroupSet> badCompositeGroupSets = GrouperDAOFactory.getFactory().getGroupSet().findBadGroupSetsForCompositeGroups();
277     for (GroupSet gs : badCompositeGroupSets) {
278       Group ownerGroup = GrouperDAOFactory.getFactory().getGroup().findByUuid(gs.getOwnerGroupId(), true);
279       Group memberGroup = GrouperDAOFactory.getFactory().getGroup().findByUuid(gs.getMemberGroupId(), true);
280 
281       if (printErrorsToSTOUT) {
282         out.println("Bad group set for composite: owner groupId=" + ownerGroup.getUuid() + ", owner group name=" + ownerGroup.getName() + ", member groupId=" + memberGroup.getUuid() + ", member group name=" + memberGroup.getName() + ".");
283       }
284       
285       logGshScript("GrouperDAOFactory.getFactory().getGroupSet().findById(\"" + gs.getId() + "\").delete(true);\n");
286     }
287     
288     Set<GroupSet> immediateGroupSetsWithMissingEffective = getImmediateGroupSetsWithMissingEffective(GrouperDAOFactory.getFactory().getGroupSet().findMissingEffectiveGroupSets());
289     for (GroupSet gs : immediateGroupSetsWithMissingEffective) {
290       if (gs.getOwnerGroupId() == null || gs.getMemberGroupId() == null || gs.getDepth() != 1 || !gs.getFieldId().equals(Group.getDefaultList().getUuid())) {
291         throw new RuntimeException("Excepted an immediate group set with an ownerGroup, memberGroup, and member field.  id=" + gs.getId());
292       }
293       
294       Group ownerGroup = GrouperDAOFactory.getFactory().getGroup().findByUuid(gs.getOwnerGroupId(), true);
295       Group memberGroup = GrouperDAOFactory.getFactory().getGroup().findByUuid(gs.getMemberGroupId(), true);
296       
297       if (printErrorsToSTOUT) {
298         out.println("Incomplete group set hierarchy (GSH script will attempt to delete and recreate it): owner groupId=" + ownerGroup.getUuid() + ", owner group name=" + ownerGroup.getName() + ", member groupId=" + memberGroup.getUuid() + ", member group name=" + memberGroup.getName() + ".");
299       }
300       
301       // TODO: this should be improved to fix group sets directly without having to delete and recreate the membership
302       // simplifying this right now to avoid timing issues with point in time.
303       logGshScript("delMember(\"" + ownerGroup.getName() + "\", \"" + memberGroup.getId() + "\");\n");      
304       logGshScript("addMember(\"" + ownerGroup.getName() + "\", \"" + memberGroup.getId() + "\");\n");      
305     }
306     
307     List<GroupSet> badEffectiveGroupSets = getBadEffectiveGroupSets(new ArrayList<GroupSet>(GrouperDAOFactory.getFactory().getGroupSet().findBadEffectiveGroupSets()));
308     for (GroupSet gs : badEffectiveGroupSets) {
309 
310       if (printErrorsToSTOUT) {
311         out.println("Bad effective group set: group set id=" + gs.getId() + ", depth=" + gs.getDepth() + ", owner id=" + gs.getOwnerId() + ", member id = " + gs.getMemberId() + ".");
312       }
313 
314       logGshScript("sqlRun(\"delete from grouper_group_set where id='" + gs.getId() + "'\");\n");
315     }
316     
317     // note this doesn't handle if in a set of duplicates, multiple group sets have foreign keys - that seems unlikely.
318     // this also doesn't fix PIT if needed.  should run pit sync after this.
319     Set<GroupSet> duplicates = GrouperDAOFactory.getFactory().getGroupSet().findDuplicateSelfGroupSets();
320 
321     for (GroupSet gs : duplicates) {
322       if (gs.getDepth() != 0) {
323         throw new RuntimeException("Unexpected depth of " + gs.getDepth());
324       }
325 
326       if (printErrorsToSTOUT) {
327         out.println("Duplicate group set: owner=" + gs.getOwnerId() + ", member=" + gs.getMemberId() + ", field=" + gs.getFieldId() + ".");
328       }
329       
330       logGshScript("sqlRun(\"update grouper_group_set set parent_id=null where id='" + gs.getId() + "'\");\n");
331       logGshScript("sqlRun(\"delete from grouper_group_set where id='" + gs.getId() + "'\");\n");
332     }
333     
334     return duplicates.size() + badType.size() + badCompositeGroupSets.size() + immediateGroupSetsWithMissingEffective.size() + badEffectiveGroupSets.size();
335   }

337   /**
338    * GrouperAll shouldn't have manage privileges or be a member of a group
339    * @return count of bad memberships
340    */
341   public static long checkGrouperAll() {
342     
343     Member grouperAll = GrouperDAOFactory.getFactory().getMember().findBySubject(SubjectFinder.findAllSubject().getId(), SubjectFinder.findAllSubject().getSourceId(), SubjectFinder.findAllSubject().getTypeName(), false);
344     if (grouperAll == null) {
345       return 0;
346     }
347     
348     Set<Membership> badMemberships = new LinkedHashSet<Membership>();
349     
350     for (Privilege privilege : AccessPrivilege.MANAGE_PRIVILEGES) {
351       Field field = privilege.getField();
352       badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findAllImmediateByMemberAndField(grouperAll.getId(), field, false));
353     }
354     
355     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findAllImmediateByMemberAndField(grouperAll.getId(), Group.getDefaultList(), false));
356     
357     for (Membership ms : badMemberships) {
358       if (printErrorsToSTOUT) {
359         out.println("Bad GrouperAll membership: groupId=" + ms.getOwnerGroupId() + ", group name=" + ms.getOwnerGroup().getName() + ", field=" + ms.getField().getName() + ".");
360       }
361       
362       logGshScript("GrouperDAOFactory.getFactory().getMembership().findByImmediateUuid(\"" + ms.getImmediateMembershipId() + "\", true).delete();\n");
363     }
364     
365     return badMemberships.size();
366   }

368   /**
369    * GrouperAll shouldn't have manage privileges or be a member of a group.  Use this to force a cleanup without outputting a GSH script.
370    */
371   public static void checkAndFixGrouperAll() {
372     
373     Member grouperAll = GrouperDAOFactory.getFactory().getMember().findBySubject(SubjectFinder.findAllSubject().getId(), SubjectFinder.findAllSubject().getSourceId(), SubjectFinder.findAllSubject().getTypeName(), false);
374     if (grouperAll == null) {
375       System.out.println("Finished running successfully.  0 changes made.");
376       return;
377     }
378     
379     Set<Membership> badMemberships = new LinkedHashSet<Membership>();
380     
381     for (Privilege privilege : AccessPrivilege.MANAGE_PRIVILEGES) {
382       Field field = privilege.getField();
383       badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findAllImmediateByMemberAndField(grouperAll.getId(), field, false));
384     }
385     
386     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findAllImmediateByMemberAndField(grouperAll.getId(), Group.getDefaultList(), false));
387     
388     for (Membership ms : badMemberships) {
389       System.out.println("Removing GrouperAll membership: groupId=" + ms.getOwnerGroupId() + ", group name=" + ms.getOwnerGroup().getName() + ", field=" + ms.getField().getName() + ".");
390       
391       ms.delete();
392     }
393     
394     System.out.println("Finished running successfully.  " + badMemberships.size() + " changes made.");
395   }

397   /**
398    * @return count of bad and missing memberships
399    */
400   public static long checkComposites() {
401     
402     Set<Membership> badMemberships = new LinkedHashSet<Membership>();
403     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findBadMembershipsOnCompositeGroup());
404     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findBadCompositeMembershipsOnNonCompositeGroup());
405     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findBadComplementMemberships());
406     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findBadUnionMemberships());
407     badMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findBadIntersectionMemberships());
408     
409     Set<Object[]> missingMemberships = new LinkedHashSet<Object[]>();
410     missingMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findMissingComplementMemberships());
411     missingMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findMissingUnionMemberships());
412     missingMemberships.addAll(GrouperDAOFactory.getFactory().getMembership().findMissingIntersectionMemberships());
413 
414     for (Membership ms : badMemberships) {
415       if (printErrorsToSTOUT) {
416         out.println("Bad composite membership: groupId=" + ms.getOwnerGroupId() + ", group name=" + ms.getOwnerGroup().getName() + ", subjectId=" + ms.getMember().getSubjectId() + ".");
417       }
418       
419       logGshScript("GrouperDAOFactory.getFactory().getMembership().findByImmediateUuid(\"" + ms.getImmediateMembershipId() + "\", true).delete();\n");
420     }
421     
422     for (Object[] ownerAndCompositeAndMember : missingMemberships) {
423       String ownerGroupId = (String)ownerAndCompositeAndMember[0];
424       String compositeId = (String)ownerAndCompositeAndMember[1];
425       String memberId = (String)ownerAndCompositeAndMember[2];
426       
427       if (printErrorsToSTOUT) {
428         Group group = GrouperDAOFactory.getFactory().getGroup().findByUuid(ownerGroupId, true);
429         Member member = GrouperDAOFactory.getFactory().getMember().findByUuid(memberId, true);
430         out.println("Missing composite membership: groupId=" + group.getId() + ", group name=" + group.getName() + ", subjectId=" + member.getSubjectId() + ".");
431       }
432       
433       logGshScript("GrouperDAOFactory.getFactory().getMembership().save(Composite.internal_createNewCompositeMembershipObject(\"" + ownerGroupId + "\", \"" + memberId + "\", \"" + compositeId + "\"));\n");
434     }
435     
436     return badMemberships.size() + missingMemberships.size();
437   }

463

465       GroupSet/../../../../edu/internet2/middleware/grouper/group/GroupSet.html#GroupSet">GroupSet parent = (GroupSet)parentAndImmediate[0];
466       GroupSet/../../../edu/internet2/middleware/grouper/group/GroupSet.html#GroupSet">GroupSet immediate = (GroupSet)parentAndImmediate[1];
467

469       GroupSetper/group/GroupSet.html#GroupSet">GroupSet groupSet = new GroupSet();
470       groupSet.setId(GrouperUuid.getUuid());
471       groupSet.setDepth(parent.getDepth() + 1);

473       groupSet.setFieldId(parent.getFieldId());

475       groupSet.setOwnerGroupId(parent.getOwnerGroupId());

511   /**
512    * Write data to a file.
513    *
514    * @param data to write
515    * @param filename to write data to.
516    */
517   private static void writeFile(StringWriter data, String filename) {
518     FileWriter fw = null;
519     try {
520       out.println("Writing file: " + GrouperUtil.fileCanonicalPath(new File(filename)));
521       fw = new FileWriter(filename, false);
522       fw.write(data.toString());
523     } catch (IOException e) {
524       out.println("Exception while writing out to file " + filename + ": " + e.toString());
525       out.println("Writing data out here instead: ");
526       out.println(data.toString());
527     } finally {
528       try {
529         if (fw != null) {
530           fw.close();
531         }
532       } catch (IOException e) {
533         // do nothing
534       }
535     }
536   }

538   private static List<GroupSet> getBadEffectiveGroupSets(List<GroupSet> groupSets) {
539     List<GroupSet> groupSetsToReturn = new ArrayList<GroupSet>();
540     for (GroupSet groupSet : groupSets) {
541       // get children
542       List<GroupSet> childGroupSets = new ArrayList<GroupSet>(GrouperDAOFactory.getFactory().getGroupSet().findAllByParentId(groupSet.getId()));
543       
544       groupSetsToReturn.addAll(getBadEffectiveGroupSets(childGroupSets));
545       groupSetsToReturn.add(groupSet);
546     }
547     
548     return groupSetsToReturn;
549   }
