115   /**
116    * print where config is read from, to sys out and log warn
117    */
118   private static void printConfigOnce() {
119     
120     if (printedConfigLocation) {
121       return;
122     }
123 
124     printedConfigLocation = true;
125 
126     boolean displayMessageString = GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.display.startup.message", true);
127     
128     String grouperStartup = "Grouper starting up: " + versionTimestamp();
129     if (!displayMessageString) {
130       //just log this to make sure we can
131       try {
132         LOG.warn(grouperStartup);
133       } catch (RuntimeException re) {
134         //this is bad, print to stderr rightaway (though might dupe)
135         System.err.println(GrouperUtil.LOG_ERROR);
136         re.printStackTrace();
137         throw new RuntimeException(GrouperUtil.LOG_ERROR, re);
138       }
139       
140       return;
141     }
142 
143     StringBuilder resultString = new StringBuilder();
144     resultString.append(grouperStartup + "\n");
145     
146     appendPatchLevelsToStartupString(resultString);
147     String propertiesFileLocation = GrouperUtil.getLocationFromResourceName("grouper.properties");
148     if (propertiesFileLocation == null) {
149       propertiesFileLocation = "not found";
150     }
151     resultString.append("grouper.properties read from: " + propertiesFileLocation + "\n");
152 
153     if (GrouperConfig.retrieveConfig().propertyValueBoolean("grouper.api.readonly", false)) {
154       resultString.append("grouper.api.readonly:         true\n");
155     }
156     
157     resultString.append("Grouper current directory is: " + new File("").getAbsolutePath() + "\n");
158     //get log4j file
159     String log4jFileLocation = GrouperUtil.getLocationFromResourceName("log4j.properties");
160     if (log4jFileLocation == null) {
161       log4jFileLocation = " [cant find log4j.properties]";
162     }
163     resultString.append("log4j.properties read from:   " + log4jFileLocation + "\n");
164     
165     resultString.append(GrouperUtil.logDirPrint());
166     String hibPropertiesFileLocation = GrouperUtil.getLocationFromResourceName("grouper.hibernate.properties");
167     if (hibPropertiesFileLocation == null) {
168       hibPropertiesFileLocation = " [cant find grouper.hibernate.properties]";
169     }
170     resultString.append("grouper.hibernate.properties: " + hibPropertiesFileLocation + "\n");
171     
172     String url = StringUtils.trim(GrouperHibernateConfig.retrieveConfig().propertyValueString("hibernate.connection.url"));
173     String user = StringUtils.trim(GrouperHibernateConfig.retrieveConfig().propertyValueString("hibernate.connection.username"));
174     resultString.append("grouper.hibernate.properties: " + user + "@" + url + "\n");
175     String sourcesString = "problem with sources";
176     try {
177       sourcesString = SourceManager.getInstance().printConfig();
178     } catch (Exception e) {
179       LOG.error("problem with sources", e);
180     }
181     resultString.append(sourcesString);
182     System.out.println(resultString);
183     try {
184       if (!GrouperUtil.isPrintGrouperLogsToConsole()) {
185         LOG.warn(resultString);
186       } else {
187         //print something to log to make sure we can
188         LOG.warn(grouperStartup);
189       }
190     } catch (RuntimeException re) {
191       //this is bad, print to stderr rightaway (though might dupe)
192       System.err.println(GrouperUtil.LOG_ERROR);
193       re.printStackTrace();
194       throw new RuntimeException(GrouperUtil.LOG_ERROR, re);
195     }
196   }

199   /**
200    * @param resultString
201    */
202   private static void appendPatchLevelsToStartupString(StringBuilder resultString) {
203     try {
204       if (GrouperConfig.retrieveConfig().propertyValueBoolean("grouper.print.patches.on.startup", true)) {
205   
206         File grouperPatchStatusFile = GrouperVersion.grouperPatchStatusFile(false);
207         String grouperPatchStatusFileLocation = grouperPatchStatusFile == null ? "not found" 
208             : GrouperUtil.fileCanonicalPath(grouperPatchStatusFile); 
209         resultString.append("grouperPatchStatus read from: " + grouperPatchStatusFileLocation + "\n");
210   
211         Map<String, Set<Integer>> patchesInstalled = GrouperVersion.patchesInstalled();
212         if (GrouperUtil.length(patchesInstalled) == 0) {
213           resultString.append("No patches detected to be installed\n");
214         } else {
215           for (String engine : GrouperUtil.nonNull(patchesInstalled).keySet()) {
216             String engineLabel = engine + " patches installed: ";
217             resultString.append(engineLabel);
218             //get things to line up
219             if (engineLabel.length() < 30) {
220               resultString.append(GrouperUtil.repeat(" ", 30-engineLabel.length()));
221             }
222             boolean first = true;
223             for (Integer patchNumber : patchesInstalled.get(engine)) {
224               if (!first) {
225                 resultString.append(", ");
226               }
227                 
228               resultString.append(patchNumber);
229               
230               first = false;
231             }
232             resultString.append("\n");
233           }
234         }
235       }
236     } catch (Exception e) {
237       try {
238         LOG.error("Cant print patch info", e);
239       } catch (Exception e2) {
240         System.out.println("Cant print patch info");
241         e.printStackTrace();
242       }
243     }
244   }

267   /**
268    * call this when grouper starts up
269    * @return false if already started, true if this started it
270    */
271   public synchronized static boolean startup() {
272     try {
273       if (started) {
274         return false;
275       }
276       started = true;
277       finishedStartupSuccessfully = false;
278       
279       printConfigOnce();
280   
281       //check java version
282 //      if (GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.checkJavaVersion", true)) {
283 //        String javaVersion = System.getProperty("java.version");
284 //        if (javaVersion != null && !javaVersion.startsWith("1.6") && !javaVersion.startsWith("1.7")) {
285 //          String error = "Error: Java should be version 6 or 7 (1.6 or 1.7), but is detected as: " + javaVersion;
286 //          LOG.error(error);
287 //          System.out.println(error);
288 //        }
289 //      }
290 
291       //add in custom sources
292       SourceManager.getInstance().loadSource(SubjectFinder.internal_getGSA());
293       SourceManager.getInstance().loadSource(InternalSourceAdapter.instance());
294       
295       if (GrouperConfig.retrieveConfig().propertyValueBoolean("externalSubjects.autoCreateSource", true)) {
296         
297         SourceManager.getInstance().loadSource(ExternalSubjectAutoSourceAdapter.instance());
298         
299       }
300       
301 //      if (GrouperConfig.retrieveConfig().propertyValueBoolean("entities.autoCreateSource", true)) {
302 //        
303 //        SourceManager.getInstance().loadSource(EntitySourceAdapter.instance());
304 //        
305 //      }
306       
307       //dont print big classname, dont print nulls
308       ToStringBuilder.setDefaultStyle(new GrouperToStringStyle());
309 
310       //first check databases
311       
312       if (!ignoreCheckConfig) {
313         GrouperCheckConfig.checkGrouperDb();
314       }
315       
316       if (runDdlBootstrap) {
317         //first make sure the DB ddl is up to date
318         GrouperDdlUtils.bootstrap(false, false, false);
319       }
320 
321       // we are ready to use the database
322       ConfigPropertiesCascadeBase.assignInitted();
323       
324       if (!ignoreCheckConfig) {
325         //make sure configuration is ok
326         GrouperCheckConfig.checkConfig();
327       }
328       
329       //startup hooks
330       GrouperHooksUtils.fireGrouperStartupHooksIfNotFiredAlready();
331   
332       //register hib objects
333       Hib3DAO.initHibernateIfNotInitted();
334       
335       initData(true);
336       
337       boolean legacyAttributeMigrationIncomplete = GrouperDdlUtils.getTableCount("grouper_types_legacy", false) > 0;
338       
339       // skip for now if legacy attribute migration is not done
340       if (!legacyAttributeMigrationIncomplete) {
341         //init include exclude type
342         initIncludeExcludeType();
343         
344         //init loader types and attributes if configured
345         initLoaderType();
346         
347         //init membership lite config type
348         initMembershipLiteConfigType();
349       }
350       
351       // verify member search and sort config
352       verifyMemberSortAndSearchConfig();
353       
354       // verify id indexes
355       verifyTableIdIndexes();
356 
357       verifyUtf8andTransactions();
358       
359       finishedStartupSuccessfully = true;
360       
361       //uncache config settings
362       GrouperConfig.retrieveConfig().clearCachedCalculatedValues();
363       
364       return true;
365     } catch (RuntimeException re) {
366       if (logErrorStatic) {
367         //NOTE, the caller might not handle this exception, so print now. 
368         //ALSO, the logger might not work, so print to stderr first
369         String error = "Couldnt startup grouper: " + re.getMessage();
370         System.err.println(error);
371         re.printStackTrace();
372         LOG.error(error, re);
373       }
374       throw re;
375     }
376   }

406   /**
407    * make sure grouper can handle utf8
408    */
409   public static void verifyUtf8andTransactionsHelper() {
410 
411     //hibernate not ok yet, try next time starting up
412     if (!GrouperDdlUtils.okToUseHibernate()) {
413       return;
414     }
415     
416     boolean detectTransactionProblems = GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.detect.db.transaction.problems", true);
417 
418     boolean detectUtf8Problems = GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.detect.utf8.problems", true);
419 
420     boolean detectUtf8FileProblems = GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.detect.utf8.file.problems", true);
421     
422     boolean detectCaseSensitiveProblems = GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.detect.db.caseSensitive.problems", true);
423 
424     boolean utfProblems = false;
425     
426     if (!detectUtf8Problems && !detectTransactionProblems && !detectCaseSensitiveProblems) {
427       return;
428     }
429 
430     final String someUtfString = "ٹٺٻټكلل";
431 
432     final String name = "grouperUtf_" + GrouperUuid.getUuid();
433     final String id = GrouperUuid.getUuid();
434     
435     if (detectUtf8FileProblems && !utfProblems) {
436       String theStringFromFile = null;
437       try {
438         theStringFromFile = GrouperUtil.readResourceIntoString("grouperUtf8.txt", false);
439       } catch (Exception e) {
440         String error = "Error: Cannot read string from resource grouperUtf8.txt";
441         LOG.error(error, e);
442         System.out.println(error);
443         e.printStackTrace();
444         utfProblems = true;
445       }
446       if (!utfProblems && !StringUtils.equals(theStringFromFile, someUtfString)) {
447         String error = "Error: Cannot properly read UTF string from resource: grouperUtf8.txt: '" + theStringFromFile
448             + "'";
449 
450         String fileEncoding = System.getProperty("file.encoding");
451         if (fileEncoding == null || !fileEncoding.toLowerCase().startsWith("utf")) {
452           error += ", make sure you pass in the JVM switch -Dfile.encoding=utf-8 (currently is '" 
453               + fileEncoding + "')";
454         }
455 
456         fileEncoding = GrouperConfig.retrieveConfig().propertyValueString("grouper.default.fileEncoding");
457         if (fileEncoding == null || !fileEncoding.toLowerCase().startsWith("utf")) {
458           error += ", make sure you have grouper.default.fileEncoding set to UTF-8 in the grouper.properties (or leave it out since the default should be UTF-8)";
459         }
460 
461         LOG.error(error);
462         System.out.println(error);
463         utfProblems = true;
464       }
465       if (!utfProblems & GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.display.utf8.success.message", false)) {
466         System.out.println("Grouper can read UTF characters correctly from files");
467       }
468     }
469 
470     //lets check case sensitive
471     if (detectCaseSensitiveProblems) {
472       //load the grouper object in app upper
473 
474       Hib3GrouperDdl grouperDdl = GrouperDdlUtils.retrieveDdlByNameFromDatabase("GROUPER");
475 
476       if (grouperDdl != null) {
477 
478         String error = "Error: Queries in your database seem to be case insensitive, "
479             + "this can be a problem for Grouper, if you are using MySQL you should use a bin collation";
480         LOG.error(error);
481         System.out.println(error);
482 
483       } else if (GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.display.db.caseSensitive.success.message", false)) {
484 
485         System.out.println("Your database can handle case sensitive queries correctly");
486 
487       }
488     }
489 
490     if (!HibernateSession.isReadonlyMode()) {
491       //this shouldnt exist, just make sure
492       GrouperDdlUtils.deleteUtfDdls();
493   
494       Hib3GrouperDdl grouperDdl = GrouperDdlUtils.storeAndReadUtfString(someUtfString, id, name);
495   
496       //lets check transactions
497       if (detectTransactionProblems) {
498         Hib3GrouperDdl grouperDdlNew = GrouperDdlUtils.retrieveDdlByIdFromDatabase(id);
499         if (grouperDdlNew != null) {
500   
501           String error = "Error: Your database does not seem to support transactions, Grouper requires a transactional database";
502           LOG.error(error);
503           System.out.println(error);
504   
505           //delete it again
506           GrouperDdlUtils.deleteDdlById(id);
507   
508         } else if (GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.display.transaction.success.message", false)) {
509           System.out.println("Your database can handle transactions correctly");
510         }
511       }
512   
513       if (grouperDdl == null) {
514         String error = "Error: Why is grouperDdl utf null???";
515         LOG.error(error);
516         System.out.println(error);
517         utfProblems = true;
518       } else if (!utfProblems) {
519   
520         if (!StringUtils.equals(grouperDdl.getHistory(), someUtfString)) {
521           String error = "Error: Cannot properly read UTF string from database: '" + grouperDdl.getHistory()
522               + "', make sure your database has UTF tables and perhaps a hibernate.connection.url in grouper.hibernate.properties";
523           LOG.error(error);
524           System.out.println(error);
525           utfProblems = true;
526         }
527       }
528       if (!utfProblems & GrouperConfig.retrieveConfig().propertyValueBoolean("configuration.display.utf8.success.message", false)) {
529         System.out.println("Grouper and the database can handle UTF characters correctly");
530       }
531     }
532   }

549   /**
550    * verify that table id indexes
551    */
552   public static void verifyTableIdIndexes() {
553     if (!GrouperConfig.retrieveConfig().propertyValueBoolean("grouper.tableIndex.verifyOnStartup", true)) {
554       return;
555     }
556         
557     //lets see if there are any nulls
558     for (TableIndexType tableIndexType : TableIndexType.values()) {
559       
560       List<String> ids = HibernateSession.bySqlStatic().listSelect(
561           String.class, "select id from " + tableIndexType.tableName() + " where id_index is null order by id" , null, null);
562 
563       if (GrouperUtil.length(ids) > 0) {
564         LOG.error("Found " + GrouperUtil.length(ids) + " " + tableIndexType.name() + " records with null id_index... correcting... please wait...");
565         for (int i=0;i<GrouperUtil.length(ids); i++) {
566           HibernateSession.bySqlStatic().executeSql("update " + tableIndexType.tableName() 
567               + " set id_index = ? where id = ?", 
568               GrouperUtil.toListObject(TableIndex.reserveId(tableIndexType), ids.get(i)),
569               HibUtils.listType(LongType.INSTANCE, StringType.INSTANCE));
570           if (i+1 % 1000 == 0) {
571             LOG.warn("Updated " + (i-1) + "/" + GrouperUtil.length(ids) + " " + tableIndexType + " id_index records...");
572           }
573         }
574         LOG.warn("Finished " + GrouperUtil.length(ids) + " " + tableIndexType + " id_index records...");
575       }
576     }
577   }

678   /**
679    * init the include/exclude type if configured in the grouper.properties
680    */
681   public static void initIncludeExcludeType() {
682     
683     final boolean useGrouperIncludeExclude = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.use", false);
684     final boolean useGrouperRequireGroups = GrouperConfig.retrieveConfig().propertyValueBoolean("grouperIncludeExclude.requireGroups.use", false);
685     
686     final String includeExcludeGroupTypeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.type.name");
687     final String requireGroupsTypeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroups.type.name");
688 
689     GrouperSession grouperSession = null;
690 
691     try {
692 
693       grouperSession = GrouperSession.startRootSession(false);
694 
695       GrouperSession.callbackGrouperSession(grouperSession, new GrouperSessionHandler() {
696 
697         public Object callback(GrouperSession grouperSession)
698             throws GrouperSessionException {
699           try {
700             
701             @SuppressWarnings("unused")
702             GroupType includeExcludeGroupType = useGrouperIncludeExclude ? 
703                 GroupType.createType(grouperSession, includeExcludeGroupTypeName, false) : null;
704 
705             GroupType requireGroupsType = useGrouperRequireGroups ? 
706                 GroupType.createType(grouperSession, requireGroupsTypeName, false) : null;
707 
708             //first the requireGroups
709             String attributeName = GrouperConfig.retrieveConfig().propertyValueString("grouperIncludeExclude.requireGroups.attributeName");
710 
711             if (useGrouperRequireGroups && !StringUtils.isBlank(attributeName)) {
712               requireGroupsType.addAttribute(grouperSession,attributeName, false);
713             }
714 
715             if (useGrouperRequireGroups) {
716               //add types/attributes from grouper.properties
717               int i=0;
718               while (true) {
719                 String propertyName = "grouperIncludeExclude.requireGroup.name." + i;
720                 String attributeOrTypePropertyName = "grouperIncludeExclude.requireGroup.attributeOrType." + i;
721 
722                 String propertyValue = GrouperConfig.retrieveConfig().propertyValueString(propertyName);
723                 if (StringUtils.isBlank(propertyValue)) {
724                   break;
725                 }
726                 String attributeOrTypeValue = GrouperConfig.retrieveConfig().propertyValueString(attributeOrTypePropertyName);
727                 boolean attributeOrType = StringUtils.equals("attribute", attributeOrTypeValue);
728                 if (attributeOrType) {
729                   requireGroupsType.addAttribute(grouperSession, propertyValue, false);
730                 } else {
731                   GroupType.createType(grouperSession, propertyValue, false);
732                 }
733                 i++;
734               }
735             }
736             
737           } catch (Exception e) {
738             throw new RuntimeException(e.getMessage(), e);
739           } finally {
740             GrouperSession.stopQuietly(grouperSession);
741           }
742           return null;
743         }
744         
745       });
746       
747       //register the hook if not already
748       GroupTypeTupleIncludeExcludeHook.registerHookIfNecessary(true);
749       
750     } catch (Exception e) {
751       throw new RuntimeException("Problem adding include/exclude type: " + includeExcludeGroupTypeName, e);
752     }
753 
754   }

756   /**
757    * init data
758    * @param logError
759    */
760   public static void initData(boolean logError) {
761     try {
762       //lets see if we need to
763       boolean needsInit;
764       GrouperSession grouperSession = null;
765       try {
766         grouperSession = GrouperSession.start(SubjectFinder.findRootSubject());
767       } catch (SessionException se) {
768         throw new RuntimeException(se);
769       }
770       try {
771         needsInit = StemFinder.findRootStem(grouperSession) == null;
772         needsInit = needsInit || FieldFinder.find(Field.FIELD_NAME_ADMINS, true) == null ;
773       } catch (Exception e) {
774         if (logError && logErrorStatic) {
775           LOG.error("Error initializing data, might just need to auto-create some data to fix...", e);
776         }
777         needsInit = true;
778       } finally {
779         GrouperSession.stopQuietly(grouperSession);
780       }
781       if (needsInit) {
782         if (GrouperConfig.retrieveConfig().propertyValueBoolean("registry.autoinit", true)) {
783           try {
784             
785             RegistryInstall.install();
786             
787           } catch (Exception e) {
788             if (logError && logErrorStatic) {
789               String error = "Couldnt auto-create data: " + e.getMessage();
790               LOG.fatal(error, e);
791             }
792           }
793         } else {
794           
795           if (logError && logErrorStatic) {
796             LOG.fatal("grouper.properties registry.autoinit is false, so not auto initting.  " +
797               "But the registry needs to be auto-initted.  Please init the registry with GSH: registryInstall()  " +
798               "Initting means adding some default data like the root stem, built in fields, etc.");
799           }
800         }
801       }
802     } catch (Exception e) {
803       if (logError && logErrorStatic) {
804         LOG.error("Error initting data", e);
805       }
806     }
807   }
