159   /**
160    * convert string to version like an enum would
161    * 
162    * @param string
163    * @param exceptionOnNull will not allow null or blank entries
164    * @return the enum or null or exception if not found
165    */
166   public static GrouperVersion valueOfIgnoreCase(String string, boolean exceptionOnNull) {
167 
168     if (StringUtils.isBlank(string)) {
169       if (exceptionOnNull) {
170         throw new RuntimeException("Not expecting a blank string for version");
171       }
172       return null;
173     }
174     
175     //see if in cache
176     GrouperVersion grouperVersion = versionCache.get(string);
177     if (grouperVersion != null) {
178       return grouperVersion;
179     }
180     
181     grouperVersion = new GrouperVersion(string);
182     
183     //dont cause a memory leak
184     if (versionCache.size() < 1000) {
185       versionCache.put(string, grouperVersion);
186     }
187     return grouperVersion;
188   }

190   /**
191    * private constructor
192    * @param versionString
193    */
194   public GrouperVersion(String versionString) {
195     Matcher grouperMatcher = pattern.matcher(versionString);
196     if (!grouperMatcher.matches()) {
197       grouperMatcher = rcPattern.matcher(versionString);
198       if (!grouperMatcher.matches()) {
199         
200         throw new RuntimeException("Invalid grouper version: " + versionString
201             + ", expecting something like: 1.2.3 or 1.2.3rc4");
202       }
203       this.rc = GrouperUtil.intValue(grouperMatcher.group(4));
204     }
205     //get the grouper versions
206     this.major = GrouperUtil.intValue(grouperMatcher.group(1));
207     this.minor = GrouperUtil.intValue(grouperMatcher.group(2));
208     this.build = GrouperUtil.intValue(grouperMatcher.group(3));
209 
210   }

306   /**
307    * see if this version is greater than or equal the argument
308    * @param anotherVersion
309    * @param orEqual if true then count equals as a true
310    * @return true if this is less
311    */
312   private boolean thisGreaterThanArgument(GrouperVersion anotherVersion, boolean orEqual) {
313     //compare, first with major, minor, then build
314     if (this.major > anotherVersion.major) {
315       return true;
316     } else if (this.major < anotherVersion.major) {
317       return false;
318     }
319     if (this.minor > anotherVersion.minor) {
320       return true;
321     } else if (this.minor < anotherVersion.minor) {
322       return false;
323     }
324     //note its greater or equal
325     if (this.build > anotherVersion.build) {
326       return true;
327     } else if (this.build < anotherVersion.build) {
328       return false;
329     }
330     if (GrouperUtil.equals(this.rc, anotherVersion.rc)) {
331       return orEqual;
332     }
333     //not having a release candidate version is higher than having one
334     if (this.rc == null) {
335       return true;
336     }
337     if (anotherVersion.rc == null) {
338       return false;
339     }
340     return this.rc > anotherVersion.rc;
341   }

353   /**
354    * get the patch index file if can find or null if not
355    * @param errorIfProblem if should throw error if problem
356    * @return the file
357    */
358   public static File grouperPatchStatusFile(boolean errorIfProblem) {
359     File log4jFile = GrouperUtil.fileFromResourceName("log4j.properties");
360     if (log4jFile.exists()) {
361       File confDir = log4jFile.getParentFile();
362       if (confDir.exists()) {
363         File grouperDir = confDir.getParentFile();
364         if (grouperDir.exists()) {
365           File patchIndexFile = new File(GrouperUtil.fileCanonicalPath(grouperDir) + File.separator + "grouperPatchStatus.properties");
366           return patchIndexFile;
367         } else {
368           if (errorIfProblem) {
369             throw new RuntimeException("grouper dir doesnt exist");
370           }
371         }
372       } else {
373         if (errorIfProblem) {
374           throw new RuntimeException("conf dir doesnt exist");
375         }
376       }
377     } else {
378       if (errorIfProblem) {
379         throw new RuntimeException("log4j.properties doesnt exist");
380       }
381     }
382     if (errorIfProblem) {
383       throw new RuntimeException("Problem finding grouperPatchStatus.properties");
384     }
385     return null;
386   }

388   /**
389    * map of patch type (e.g. api, ui), and numbers of patches installed in order
390    * @return the map of patches installed, never returns null
391    */
392   public static Map<String, Set<Integer>> patchesInstalled() {
393     Map<String, Set<Integer>> patchesInstalled = patchesInstalledCache.get(Boolean.TRUE);
394     
395     if (patchesInstalled == null) {
396       synchronized (patchesInstalledCache) {
397         
398         patchesInstalled = patchesInstalledCache.get(Boolean.TRUE);
399         if (patchesInstalled == null) {
400           
401           patchesInstalled = new TreeMap<String, Set<Integer>>();
402           
403           try {
404 
405             File patchIndexFile = grouperPatchStatusFile(true);
406             if (patchIndexFile.exists()) {
407               Properties props = GrouperUtil.propertiesFromFile(patchIndexFile, false);
408               
409               Pattern patchStatePattern = Pattern.compile("^grouper_v" + GROUPER_VERSION.replace(".", "_") + "_(.*)_patch_(.*)\\.state$");
410               for (String key : (Set<String>)(Object)props.keySet()) {
411                 Matcher matcher = patchStatePattern.matcher(key);
412 
413                 if (matcher.matches() && "applied".equals(props.getProperty(key))) {
414                   String engine = matcher.group(1);
415                   String patchNumberString = matcher.group(2);
416                   Set<Integer> patchNumbers = patchesInstalled.get(engine);
417                   if (patchNumbers == null) {
418                     patchNumbers = new TreeSet<Integer>();
419                     patchesInstalled.put(engine, patchNumbers);
420                   }
421                   
422                   try {
423                     int patchNumberInt = GrouperUtil.intValue(patchNumberString); 
424                     patchNumbers.add(patchNumberInt);
425                   } catch (Exception e) {
426                     String logMessage = "Error: cant parse version number from grouperPatchStatus.properties entry: " + key + " in " + patchIndexFile.getAbsolutePath();
427                     try {
428                       LOG.error(logMessage);
429                     } catch (Exception e2) {
430                       System.out.println(logMessage);
431                     }
432                     //swallow this since could be in startup
433                   }
434                 }
435               }
436             }
437           } catch (Exception e) {
438             String logMessage = "Error: cant process grouperPatchStatus.properties";
439             try {
440               LOG.error(logMessage, e);
441             } catch (Exception e2) {
442               System.out.println(logMessage);
443               e.printStackTrace();
444             }
445             //swallow this since could be in startup
446           }
447           patchesInstalledCache.put(Boolean.TRUE, patchesInstalled);
448         }
449       }
450     }
451     return patchesInstalled;
452   }
