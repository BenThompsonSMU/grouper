105    * 
106    */
107   private static GrouperDAOFactory gdf;

110   /**
111    * Return singleton {@link GrouperDAOFactory} implementation.
112    * <p/>
113    * @return factory
114    * @since   1.2.0
115    */
116   public static GrouperDAOFactory getFactory() {
117     if (gdf == null) {
118       gdf = getFactoryHelper( );
119     }
120     return gdf;
121   } 
122 
123   /**
124    * Return singleton {@link GrouperDAOFactory} implementation using the specified
125    * configuration.
126    * <p/>
127    * @return factory
128    * @throws  IllegalArgumentException if <i>cfg</i> is null.
129    * @since   1.2.1
130    */
131   private static GrouperDAOFactory getFactoryHelper() 
132     throws  IllegalArgumentException
133   {
134     String            klass = GrouperConfig.retrieveConfig().propertyValueString(GrouperConfig.PROP_DAO_FACTORY);
135     GrouperValidator  v     = NotNullOrEmptyValidator.validate(klass);
136     if ( v.isInvalid() ) {
137       klass = GrouperConfig.DEFAULT_DAO_FACTORY;
138     }
139     return (GrouperDAOFactory) Realize.instantiate(klass);
140   }
