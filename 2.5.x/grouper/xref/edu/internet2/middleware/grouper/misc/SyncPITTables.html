277    * Add missing point in time memberships.

280   public long processMissingActivePITMemberships() {
281     showStatus("\n\nSearching for missing active point in time memberships");
282     
283     long totalProcessed = 0;
284 
285     Set<Membership> mships = GrouperDAOFactory.getFactory().getPITMembership().findMissingActivePITMemberships();
286     showStatus("Found " + mships.size() + " missing active point in time memberships");
287 
288     for (Membership mship : mships) {
289       
290       logDetail("Found missing point in time membership with ownerId: " + mship.getOwnerId() + 
291           ", memberId: " + mship.getMemberUuid() + ", fieldId: " + mship.getFieldId());
292             
293       if (saveUpdates) {
294         PITField pitField = GrouperDAOFactory.getFactory().getPITField().findBySourceIdActive(mship.getFieldId(), true);
295         PITMember pitMember = GrouperDAOFactory.getFactory().getPITMember().findBySourceIdActive(mship.getMemberUuid(), true);
296         
297         PITMembershipTMembership.html#PITMembership">PITMembership pitMembership = new PITMembership();
298         pitMembership.setId(GrouperUuid.getUuid());
299         pitMembership.setSourceId(mship.getImmediateMembershipId());
300         pitMembership.setMemberId(pitMember.getId());
301         pitMembership.setFieldId(pitField.getId());
302         pitMembership.setActiveDb("T");
303         pitMembership.setStartTimeDb(System.currentTimeMillis() * 1000);
304 
305         if (mship.getOwnerGroupId() != null) {
306           pitMembership.setOwnerGroupId(GrouperDAOFactory.getFactory().getPITGroup().findBySourceIdActive(mship.getOwnerGroupId(), true).getId());
307         } else if (mship.getOwnerStemId() != null) {
308           pitMembership.setOwnerStemId(GrouperDAOFactory.getFactory().getPITStem().findBySourceIdActive(mship.getOwnerStemId(), true).getId());
309         } else if (mship.getOwnerAttrDefId() != null) {
310           pitMembership.setOwnerAttrDefId(GrouperDAOFactory.getFactory().getPITAttributeDef().findBySourceIdActive(mship.getOwnerAttrDefId(), true).getId());
311         } else {
312           throw new RuntimeException("Unexpected -- Membership with id " + mship.getUuid() + " does not have an ownerGroupId, ownerStemId, or ownerAttrDefId.");
313         }
314         
315         if (!GrouperUtil.isEmpty(mship.getContextId())) {
316           pitMembership.setContextId(mship.getContextId());
317         }
318         
319         if (sendFlattenedNotifications) {
320           pitMembership.setFlatMembershipNotificationsOnSaveOrUpdate(includeFlattenedMemberships);
321           pitMembership.setFlatPrivilegeNotificationsOnSaveOrUpdate(includeFlattenedPrivileges);
322         }
323         
324         if (sendPermissionNotifications) {
325           pitMembership.setNotificationsForRolesWithPermissionChangesOnSaveOrUpdate(includeRolesWithPermissionChanges);
326         }
327 
328         pitMembership.save();
329       }
330       
331       totalProcessed++;
332     }
333     
334     if (mships.size() > 0 && saveUpdates) {
335       showStatus("Done making " + totalProcessed + " updates");
336     }
337     
338     return totalProcessed;
339   }

345   public long processMissingActivePITAttributeAssigns() {
346     showStatus("\n\nSearching for missing active point in time attribute assigns");
347     
348     long totalProcessed = 0;
349 
350     Set<AttributeAssign> assigns = GrouperDAOFactory.getFactory().getPITAttributeAssign().findMissingActivePITAttributeAssigns();
351     showStatus("Found " + assigns.size() + " missing active point in time attribute assigns");
352 
353     // sort to avoid foreign key issues
354     LinkedHashSet<AttributeAssign> assignsSorted = new LinkedHashSet<AttributeAssign>();
355     for (AttributeAssign assign : assigns) {
356       if (assign.getOwnerAttributeAssignId() == null) {
357         assignsSorted.add(assign);
358       }
359     }
360     
361     for (AttributeAssign assign : assigns) {
362       if (assign.getOwnerAttributeAssignId() != null) {
363         assignsSorted.add(assign);
364       }
365     }
366     
367     for (AttributeAssign assign : assignsSorted) {
368       
369       logDetail("Found missing point in time attribute assign with id: " + assign.getId());
370             
371       if (saveUpdates) {
372         PITAttributeDefName pitAttributeDefName = GrouperDAOFactory.getFactory().getPITAttributeDefName().findBySourceIdActive(assign.getAttributeDefNameId(), true);
373         PITAttributeAssignAction pitAttributeAssignAction = GrouperDAOFactory.getFactory().getPITAttributeAssignAction().findBySourceIdActive(assign.getAttributeAssignActionId(), true);
374         
375         PITAttributeAssignAssign.html#PITAttributeAssign">PITAttributeAssign pitAttributeAssign = new PITAttributeAssign();
376         pitAttributeAssign.setId(GrouperUuid.getUuid());
377         pitAttributeAssign.setSourceId(assign.getId());
378         pitAttributeAssign.setAttributeDefNameId(pitAttributeDefName.getId());
379         pitAttributeAssign.setAttributeAssignActionId(pitAttributeAssignAction.getId());
380         pitAttributeAssign.setAttributeAssignTypeDb(assign.getAttributeAssignTypeDb());
381         pitAttributeAssign.setDisallowedDb(assign.getDisallowedDb());
382         pitAttributeAssign.setActiveDb("T");
383         pitAttributeAssign.setStartTimeDb(System.currentTimeMillis() * 1000);
384         
385         if (AttributeAssignType.group.name().equals(pitAttributeAssign.getAttributeAssignTypeDb())) {
386           PITGroup pitOwner1 = GrouperDAOFactory.getFactory().getPITGroup().findBySourceIdActive(assign.getOwnerGroupId(), true);
387           pitAttributeAssign.setOwnerGroupId(pitOwner1.getId());
388         } else if (AttributeAssignType.stem.name().equals(pitAttributeAssign.getAttributeAssignTypeDb())) {
389           PITStem pitOwner1 = GrouperDAOFactory.getFactory().getPITStem().findBySourceIdActive(assign.getOwnerStemId(), true);
390           pitAttributeAssign.setOwnerStemId(pitOwner1.getId());
391         } else if (AttributeAssignType.member.name().equals(pitAttributeAssign.getAttributeAssignTypeDb())) {
392           PITMember pitOwner1 = GrouperDAOFactory.getFactory().getPITMember().findBySourceIdActive(assign.getOwnerMemberId(), true);
393           pitAttributeAssign.setOwnerMemberId(pitOwner1.getId());
394         } else if (AttributeAssignType.attr_def.name().equals(pitAttributeAssign.getAttributeAssignTypeDb())) {
395           PITAttributeDef pitOwner1 = GrouperDAOFactory.getFactory().getPITAttributeDef().findBySourceIdActive(assign.getOwnerAttributeDefId(), true);
396           pitAttributeAssign.setOwnerAttributeDefId(pitOwner1.getId());
397         } else if (AttributeAssignType.any_mem.name().equals(pitAttributeAssign.getAttributeAssignTypeDb())) {
398           PITGroup pitOwner1 = GrouperDAOFactory.getFactory().getPITGroup().findBySourceIdActive(assign.getOwnerGroupId(), true);
399           PITMember pitOwner2 = GrouperDAOFactory.getFactory().getPITMember().findBySourceIdActive(assign.getOwnerMemberId(), true);
400           pitAttributeAssign.setOwnerGroupId(pitOwner1.getId());
401           pitAttributeAssign.setOwnerMemberId(pitOwner2.getId());
402         } else if (AttributeAssignType.imm_mem.name().equals(pitAttributeAssign.getAttributeAssignTypeDb())) {
403           PITMembership pitOwner1 = GrouperDAOFactory.getFactory().getPITMembership().findBySourceIdActive(assign.getOwnerMembershipId(), false);
404           if (pitOwner1 == null) {
405             // assignment must be disabled..
406             logDetail("Skipping " + assign.getId() + " since active owner was not found in point in time.");
407             continue;
408           }
409           pitAttributeAssign.setOwnerMembershipId(pitOwner1.getId());
410         } else {
411           // this must be an attribute assign of an attribute assign.  foreign keys will make sure we're right.
412           PITAttributeAssign pitOwner1 = GrouperDAOFactory.getFactory().getPITAttributeAssign().findBySourceIdActive(assign.getOwnerAttributeAssignId(), false);
413           if (pitOwner1 == null) {
414             // assignment must be disabled..
415             logDetail("Skipping " + assign.getId() + " since active owner was not found in point in time.");
416             continue;
417           }
418           pitAttributeAssign.setOwnerAttributeAssignId(pitOwner1.getId());
419         }
420         
421         if (!GrouperUtil.isEmpty(assign.getContextId())) {
422           pitAttributeAssign.setContextId(assign.getContextId());
423         }
424                 
425         if (sendPermissionNotifications) {
426           pitAttributeAssign.setNotificationsForRolesWithPermissionChangesOnSaveOrUpdate(includeRolesWithPermissionChanges);
427         }
428 
429         pitAttributeAssign.save();
430       }
431       
432       totalProcessed++;
433     }
434     
435     if (assigns.size() > 0 && saveUpdates) {
436       showStatus("Done making " + totalProcessed + " updates");
437     }
438     
439     return totalProcessed;
440   }

1823   public long processDuplicates(GrouperDAO dao) {
1824

1826     long errorCount = 0;
1827     
1828     showStatus("\n\n" + className + ": Searching for point in time duplicates");
1829 
1830     Set<String> sourceIds = (Set<String>)GrouperUtil.callMethod(dao.getClass(), dao, "findActiveDuplicates", null, null, false, false);
1831     showStatus("Found " + sourceIds.size() + " entries that have duplicates in point in time.");
1832

1835       List<GrouperPIT> objectsList = new ArrayList<GrouperPIT>();
1836       for (GrouperPIT object : objectsSet) {
1837         if (object.isActive()) {
1838           objectsList.add(object);
1839         }
1840       }
1841       
1842       if (objectsList.size() < 2) {
1843         throw new RuntimeException("Found fewer than expected entries with sourceId=" + sourceId + ", DAO=" + className);
1844       }
1845
