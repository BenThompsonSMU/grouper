41  /**
42   * convert a messaging listener to a change log consumer
43   */
44  public class MessagingListenerToChangeLogConsumer extends MessagingListenerBase {
45  
46    /**
47     * logger 
48     */
49    private static final Log LOG = GrouperUtil.getLog(MessagingListenerToChangeLogConsumer.class);
50  
51    /**
52     * 
53     */
54    public MessagingListenerToChangeLogConsumer() {
55    }
56  
57    /**
58     * @see edu.internet2.middleware.grouper.messaging.MessagingListenerBase#processMessages(java.lang.String, java.lang.String, java.util.Collection, edu.internet2.middleware.grouper.messaging.MessagingListenerMetadata)
59     */
60    @Override
61    public void processMessages(String messageSystemName, String queue,
62        Collection<GrouperMessage> grouperMessageList,
63        MessagingListenerMetadata messagingListenerMetadata) {
64      
65      String changeLogConsumerClassName = GrouperLoaderConfig.retrieveConfig().propertyValueString("messaging.listener." 
66          + messagingListenerMetadata.getConsumerName() + ".changeLogConsumerClass");
67  
68      Class<ChangeLogConsumerBase> changeLogConsumerClass = GrouperUtil.forName(changeLogConsumerClassName);
69      ChangeLogConsumerBase changeLogConsumerBase = GrouperUtil.newInstance(changeLogConsumerClass);
70      
71      int size = GrouperUtil.length(grouperMessageList);
72      
73      List<ChangeLogEntry> changeLogEntries = new ArrayList<ChangeLogEntry>();
74      
75      Iterator<GrouperMessage> grouperMessageIterator = grouperMessageList.iterator();
76  
77      Map<Long, GrouperMessage> changeLogIdToGrouperMessage = new LinkedHashMap<Long, GrouperMessage>();
78      
79      for (int i=0;i<size;i++) {
80        
81        GrouperMessage grouperMessage = grouperMessageIterator.next();
82        
83        try {
84          Collection<ChangeLogEntry> changeLogEntriesForMessage = ChangeLogEntry.fromJsonToCollection(grouperMessage.getMessageBody());
85          
86          for (ChangeLogEntry changeLogEntry : GrouperUtil.nonNull(changeLogEntriesForMessage)) {
87            changeLogIdToGrouperMessage.put(changeLogEntry.getSequenceNumber(), grouperMessage);
88          }
89          
90          changeLogEntries.addAll(GrouperUtil.nonNull(changeLogEntriesForMessage));
91          
92        } catch (Exception e) {
93          LOG.error("Cannot convert message body to change log entry: '" 
94              + messagingListenerMetadata.getConsumerName() + "''" + grouperMessage.getMessageBody() + "'", e);
95          //dont get this message again
96          GrouperMessagingEngine.acknowledge(new GrouperMessageAcknowledgeParam().assignGrouperMessageSystemName(messageSystemName)
97              .assignQueueName(queue).addGrouperMessage(grouperMessage).assignAcknowledgeType(GrouperMessageAcknowledgeType.mark_as_processed));
98        }
99      }
100     
101     if (GrouperUtil.length(changeLogEntries) == 0) {
102       return;
103     }
104     
105     ChangeLogProcessorMetadatadata.html#ChangeLogProcessorMetadata">ChangeLogProcessorMetadata changeLogProcessorMetadata = new ChangeLogProcessorMetadata();
106     changeLogProcessorMetadata.setConsumerName(messagingListenerMetadata.getConsumerName());
107     changeLogProcessorMetadata.setHib3GrouperLoaderLog(messagingListenerMetadata.getHib3GrouperLoaderLog());
108 
109     long lastProcessed = -1;
110     
111     try {
112       lastProcessed = changeLogConsumerBase.processChangeLogEntries(changeLogEntries, changeLogProcessorMetadata);
113 
114       messagingListenerMetadata.setHadProblem(changeLogProcessorMetadata.isHadProblem());
115       messagingListenerMetadata.setRecordException(changeLogProcessorMetadata.getRecordException());
116       messagingListenerMetadata.setRecordProblemText(changeLogProcessorMetadata.getRecordProblemText());
117 
118     } catch (Exception e) {
119       messagingListenerMetadata.setHadProblem(true);
120       messagingListenerMetadata.setRecordException(e);
121       messagingListenerMetadata.setRecordProblemText(e.getLocalizedMessage());
122       lastProcessed = changeLogProcessorMetadata.getRecordExceptionSequence() -1;
123     }
124     
125     if (lastProcessed > -1) {
126       Set<String> messageIdAcked = new HashSet<String>();
127       for (Long changeLogId : changeLogIdToGrouperMessage.keySet()) {
128         GrouperMessage grouperMessage = changeLogIdToGrouperMessage.get(changeLogId);
129 
130         // multiple change log entries can be in one message
131         if (messageIdAcked.contains(grouperMessage.getId())) {
132            continue;
133         }
134         messageIdAcked.add(grouperMessage.getId());
135         
136         boolean firstError = true;
137         
138         if (changeLogId <= lastProcessed) {
139 
140           //all good, ack
141           GrouperMessagingEngine.acknowledge(new GrouperMessageAcknowledgeParam().assignGrouperMessageSystemName(messageSystemName)
142               .assignQueueName(queue).addGrouperMessage(grouperMessage).assignAcknowledgeType(GrouperMessageAcknowledgeType.mark_as_processed));
143           
144         } else {
145 
146           if (firstError) {
147             messagingListenerMetadata.setRecordExceptionId(grouperMessage.getId());
148 
149             firstError = false;
150           }
151           
152           changeLogProcessorMetadata.setHadProblem(true);
153 
154           //not processed correctly
155           GrouperMessagingEngine.acknowledge(new GrouperMessageAcknowledgeParam().assignGrouperMessageSystemName(messageSystemName)
156               .assignQueueName(queue).addGrouperMessage(grouperMessage).assignAcknowledgeType(GrouperMessageAcknowledgeType.return_to_queue));
157         }
158       }
159     }
160     
161   }
