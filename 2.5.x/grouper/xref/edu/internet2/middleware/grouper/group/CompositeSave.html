191   /**
192    * massage and validate fields
193    */
194   private void massageAndValidateFields() {
195 
196     if (!StringUtils.isBlank(this.ownerGroupId)) {
197       if (this.ownerGroup != null) {
198         if (!StringUtils.equals(this.ownerGroupId, this.ownerGroup.getId())) {
199           throw new RuntimeException("Passing in owner group id and owner group but dont match '" + this.ownerGroupId
200               + "', '" + this.ownerGroup.getId() + "', '" + this.ownerGroup.getName() + "'");
201         }
202       } else {
203         this.ownerGroup = GroupFinder.findByUuid(this.grouperSession, this.ownerGroupId, true);
204       }
205     }
206     if (!StringUtils.isBlank(this.ownerGroupName)) {
207       if (this.ownerGroup != null && !StringUtils.equals(this.ownerGroup.getName(), this.ownerGroupName)
208           && !StringUtils.equals(ownerGroup.getAlternateName(), this.ownerGroupName)) {
209         throw new RuntimeException("Passing in owner group id and owner name but dont match '" + this.ownerGroupId
210             + "', '" + this.ownerGroupName + "', '" + this.ownerGroup.getName() + "'");
211       }
212       if (this.ownerGroup == null) {
213         this.ownerGroup = GroupFinder.findByName(this.grouperSession, this.ownerGroupName, true);
214       }
215     }
216 
217     if (this.ownerGroup != null) {
218       this.ownerGroupId = this.ownerGroup.getId();
219       this.ownerGroupName = this.ownerGroup.getName();
220     } else {
221       throw new RuntimeException("ownerGroup is required");
222     }
223     
224     if (!StringUtils.isBlank(this.leftFactorGroupId)) {
225       if (this.leftFactorGroup != null) {
226         if (!StringUtils.equals(this.leftFactorGroupId, this.leftFactorGroup.getId())) {
227           throw new RuntimeException("Passing in leftFactor group id and leftFactor group but dont match '" + this.leftFactorGroupId
228               + "', '" + this.leftFactorGroup.getId() + "', '" + this.leftFactorGroup.getName() + "'");
229         }
230       } else {
231         this.leftFactorGroup = GroupFinder.findByUuid(this.grouperSession, this.leftFactorGroupId, true);
232       }
233     }
234     if (!StringUtils.isBlank(this.leftFactorGroupName)) {
235       if (this.leftFactorGroup != null && !StringUtils.equals(this.leftFactorGroup.getName(), this.leftFactorGroupName)
236           && !StringUtils.equals(leftFactorGroup.getAlternateName(), this.leftFactorGroupName)) {
237         throw new RuntimeException("Passing in leftFactor group id and leftFactor name but dont match '" + this.leftFactorGroupId
238             + "', '" + this.leftFactorGroupName + "', '" + this.leftFactorGroup.getName() + "'");
239       }
240       if (this.leftFactorGroup == null) {
241         this.leftFactorGroup = GroupFinder.findByName(this.grouperSession, this.leftFactorGroupName, true);
242       }
243     }
244 
245     if (this.leftFactorGroup != null) {
246       this.leftFactorGroupId = this.leftFactorGroup.getId();
247       this.leftFactorGroupName = this.leftFactorGroup.getName();
248     } else {
249       throw new RuntimeException("leftFactorGroup is required");
250     }
251 
252     if (!StringUtils.isBlank(this.rightFactorGroupId)) {
253       if (this.rightFactorGroup != null) {
254         if (!StringUtils.equals(this.rightFactorGroupId, this.rightFactorGroup.getId())) {
255           throw new RuntimeException("Passing in rightFactor group id and rightFactor group but dont match '" + this.rightFactorGroupId
256               + "', '" + this.rightFactorGroup.getId() + "', '" + this.rightFactorGroup.getName() + "'");
257         }
258       } else {
259         this.rightFactorGroup = GroupFinder.findByUuid(this.grouperSession, this.rightFactorGroupId, true);
260       }
261     }
262     if (!StringUtils.isBlank(this.rightFactorGroupName)) {
263       if (this.rightFactorGroup != null && !StringUtils.equals(this.rightFactorGroup.getName(), this.rightFactorGroupName)
264           && !StringUtils.equals(rightFactorGroup.getAlternateName(), this.rightFactorGroupName)) {
265         throw new RuntimeException("Passing in rightFactor group id and rightFactor name but dont match '" + this.rightFactorGroupId
266             + "', '" + this.rightFactorGroupName + "', '" + this.rightFactorGroup.getName() + "'");
267       }
268       if (this.rightFactorGroup == null) {
269         this.rightFactorGroup = GroupFinder.findByName(this.grouperSession, this.rightFactorGroupName, true);
270       }
271     }
272 
273     if (this.rightFactorGroup != null) {
274       this.rightFactorGroupId = this.rightFactorGroup.getId();
275       this.rightFactorGroupName = this.rightFactorGroup.getName();
276     } else {
277       throw new RuntimeException("rightFactorGroup is required");
278     }
279 
280     if (this.compositeType == null) {
281       throw new RuntimeException("compositeType is required");
282     }
283     
284     //default to insert or update
285     this.saveMode = (SaveMode)ObjectUtils.defaultIfNull(this.saveMode, SaveMode.INSERT_OR_UPDATE);
286 
287   }

302   /**
303    * <pre>
304    * create or update a group.  Note this will not rename a group at this time (might in future)
305    * 
306    * This is a static method since setters to Group objects persist to the DB
307    * 
308    * Steps:
309    * 
310    * 1. Find the group by groupNameToEdit
311    * 2. Internally set all the fields of the stem (no need to reset if already the same)
312    * 3. Store the group (insert or update) if needed
313    * 4. Return the group object
314    * 
315    * This runs in a tx so that if part of it fails the whole thing fails, and potentially the outer
316    * transaction too
317    * </pre>
318    * @return the group
319    * @throws StemNotFoundException
320    * @throws InsufficientPrivilegeException
321    * @throws GroupNotFoundException
322    * @throws AttributeDefNameNotFoundException
323    */
324   public Composite save() 
325         throws AttributeDefNameNotFoundException, InsufficientPrivilegeException, StemNotFoundException, 
326         GroupNotFoundException {
327 
328     // figure out fields, validate them
329     massageAndValidateFields();
330 
331     final SaveMode SAVE_MODE = CompositeSave.this.saveMode;
332 
333     final CompositeSave THIS = this;
334     
335     try {
336       GrouperTransaction.callbackGrouperTransaction(new GrouperTransactionHandler() {
337 
338         @SuppressWarnings("cast")
339         @Override
340         public Object callback(GrouperTransaction grouperTransaction)
341             throws GrouperDAOException {
342           
343           grouperTransaction.setCachingEnabled(false);
344           
345           return (Void)GrouperSession.callbackGrouperSession(THIS.grouperSession, new GrouperSessionHandler() {
346 
347               @Override
348               public Object callback(GrouperSession theGrouperSession)
349                   throws GrouperSessionException {
350                                 
351                 //see if update
352                 boolean isUpdate = THIS.saveMode == SaveMode.UPDATE;
353 
354                 THIS.composite = THIS.ownerGroup.getComposite(false);
355                 
356                 //while we are here, make sure uuid's match if passed in
357                 //not sure this can happen, if you pass in a uuid it will be used
358                 if (!StringUtils.isBlank(THIS.id) && THIS.composite != null && !StringUtils.equals(THIS.composite.getUuid(), THIS.id)) {
359                   throw new RuntimeException("UUID composite changes are not supported: new: " 
360                       + THIS.composite.getUuid() + ", old: " 
361                       + THIS.id);
362                 }
363 
364                 if (THIS.composite == null) {
365                   if (SAVE_MODE.equals(SaveMode.INSERT_OR_UPDATE) || SAVE_MODE.equals(SaveMode.INSERT)) {
366                     isUpdate = false;
367                   } else {
368                     throw new RuntimeException("There is no existing composite but the SaveMode is " + SAVE_MODE);
369                   }
370                 } else {
371                   
372                   if (SAVE_MODE.equals(SaveMode.INSERT_OR_UPDATE)) {
373                     isUpdate = true;
374                   }
375                 }
376                 
377                 //default
378                 THIS.saveResultType = SaveResultType.NO_CHANGE;
379                 //if inserting
380                 boolean needsSave = false;
381                 if (!isUpdate) {
382                   THIS.saveResultType = SaveResultType.INSERT;
383                   
384                 } else {
385 
386                   if (!StringUtils.equals(THIS.leftFactorGroup.getId(), THIS.composite.getLeftFactorUuid())) {
387                     needsSave = true;
388                   }
389                   
390                   if (!StringUtils.equals(THIS.rightFactorGroup.getId(), THIS.composite.getRightFactorUuid())) {
391                     needsSave = true;
392                   }
393 
394                   if (THIS.compositeType != THIS.composite.getType()) {
395                     needsSave = true;
396                   }
397                   
398                 }
399                 
400                 //this is an update if it is not an insert and needs an update
401                 if (needsSave && THIS.saveResultType == SaveResultType.NO_CHANGE) {
402                   THIS.saveResultType = SaveResultType.UPDATE;
403                 }
404 
405                 //only store once
406                 if (needsSave) {
407 
408                   THIS.ownerGroup.deleteCompositeMember();
409 
410                 }
411                 
412                 if (needsSave || THIS.saveResultType == SaveResultType.INSERT) {
413                   THIS.composite = THIS.ownerGroup.internal_addCompositeMember(THIS.grouperSession, THIS.compositeType, THIS.leftFactorGroup, THIS.rightFactorGroup, THIS.id);
414                 }
415                                 
416                 return null;
417               }
418 
419           });
420         }
421       });
422       return this.composite;
423     } catch (RuntimeException re) {
424 
425       GrouperUtil.injectInException(re, "Problem saving composite: " + this.ownerGroup
426           + ", left: " + this.leftFactorGroup + ", " + this.compositeType + ", right: " + this.rightFactorGroup
427           + ", thread: " + Integer.toHexString(Thread.currentThread().hashCode()));
428       
429       Throwable throwable = re.getCause();
430       if (throwable instanceof InsufficientPrivilegeException) {
431         throw (InsufficientPrivilegeException)throwable;
432       }
433       //must just be runtime
434       throw re;
435     }
436 
437   }
