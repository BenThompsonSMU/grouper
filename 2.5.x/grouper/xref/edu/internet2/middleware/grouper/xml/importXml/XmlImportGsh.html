51  /**
52   * Utility class for importing data in XML import into the Groups Registry.
53   * <p>
54   * This class reads an XML file representing all or part of a Groups Registry
55   * and updates-or-creates the equivalent {@link Stem}s, {@link Group}s and
56   * {@link Membership}s, etc (all data)
57   * <p/>
58   * @author  Chris Hyzer
59   * @version $Id: XmlImporter.java,v 1.24 2009-11-11 16:48:11 mchyzer Exp $
60   * @since   1.0
61   */
62  public class XmlImportGsh {
63  
64    /**
65     * 
66     */
67    private static final String NOPROMPT_ARG = "noprompt";
68    /**
69     * 
70     */
71    public static final String RECORD_REPORT_ARG = "recordReport";
72    
73    /** log */
74    @SuppressWarnings("unused")
75    private static final Log    LOG           = GrouperUtil.getLog(XmlImportGsh.class);
76    
77    /**
78     * Process an Xml file as the 'root' user.
79     * <p/>
80     * @param   args    args[0] = name of Xml file to process
81     * @since   1.1.0
82     */
83    public static void main(final String[] args) {
84      
85      //set this...
86      GrouperContext.createNewDefaultContext(GrouperEngineBuiltin.IMPORT, false, true);
87      if (XmlExportUtils.internal_wantsHelp(args)) {
88        System.out.println( _getUsage() );
89        //System.exit(0);
90        return;
91      }
92      final Map<String, Object> argsMap;
93      try {
94        argsMap = XmlExportUtils.internal_getXmlImportArgs(args);
95      }
96      catch (Exception e) {
97        e.printStackTrace();
98        System.err.println();
99        System.err.println( _getUsage() );
100       //System.exit(1);
101       return;
102     }
103     
104     //make sure right db
105     if(!Boolean.TRUE.equals(argsMap.get(NOPROMPT_ARG))) {
106       GrouperUtil.promptUserAboutDbChanges("import data from xml", true);
107     }
108     GrouperStartup.runFromMain = true;
109     GrouperStartup.startup();
110     
111     GrouperSession grouperSession = GrouperSession.startRootSession();
112     
113     try {
114 
115       XmlImportMain/importXml/XmlImportMain.html#XmlImportMain">XmlImportMain xmlImportMain = new XmlImportMain();
116       
117       if (Boolean.TRUE.equals(argsMap.get(RECORD_REPORT_ARG))) {
118         xmlImportMain.setRecordReport(true);
119       }
120       xmlImportMain.setRecordReport(true);
121       
122       String fileName = (String)argsMap.get(XmlExportUtils.FILE_NAME_ARG);
123       
124       if (StringUtils.isBlank(fileName) || fileName.startsWith("-")) {
125         throw new RuntimeException("Enter a file name");
126       }
127       
128       File file = new File(fileName);
129       
130       xmlImportMain.processXml(file);
131       
132       
133     } catch (Exception e) {      
134       if (GrouperUtil.getFullStackTrace(e).contains("groupTypeUuid")) {
135         System.err.println("ERROR: It appears you are trying to import an XML file that was exported from Grouper 2.1 or before. " +
136         		"This is not supported.  Please upgrade instead using the procedure documented in the release notes.");
137       } else {
138         throw new RuntimeException(e);
139       }
140     } finally {
141       GrouperSession.stopQuietly(grouperSession);
142     }
143     
144     System.exit(0);
145     return;
146   }

156             + "args:  "             + GrouperConfig.NL
157             + "      [-recordReport]"              + GrouperConfig.NL

159             + "e.g.  gsh -xmlimport f:/temp/prod.xml"                 + GrouperConfig.NL
160             +                                                                         GrouperConfig.NL
161             + "  -recordReport,     Print a file which lists each insert/update"    + GrouperConfig.NL
162             + "                     In addition to import"    + GrouperConfig.NL
163             + "  -noprompt,         Do not prompt user to confirm the database that"+ GrouperConfig.NL
164             + "                     will be updated"                                + GrouperConfig.NL
165             + "  filename,          The file to import"                             + GrouperConfig.NL
