221   /**
222    * text config for a certain locale
223    * @param locale
224    * @return the config for this user's locale
225    */
226   private static TextBundleBean retrieveTextBundleBean(Locale locale) {
227     
228     if (locale == null) {
229       //return the default
230       return GrouperConfig.retrieveConfig().textBundleDefault();
231 
232     }
233     
234     String language = StringUtils.defaultString(locale.getLanguage()).toLowerCase();
235     String country = StringUtils.defaultString(locale.getCountry()).toLowerCase();
236     
237     //see if there is a match by language and country
238     TextBundleBean textBundleBean = GrouperConfig.retrieveConfig()
239       .textBundleFromLanguageAndCountry().get(language + "_" + country);
240     
241     if (textBundleBean != null) {
242       return textBundleBean;
243     }
244     
245     //see if there is a match by language
246     textBundleBean = GrouperConfig.retrieveConfig()
247       .textBundleFromLanguage().get(language);
248   
249     if (textBundleBean != null) {
250       return textBundleBean;
251     }
252     
253     //see if there is a match by country
254     textBundleBean = GrouperConfig.retrieveConfig()
255       .textBundleFromCountry().get(country);
256   
257     if (textBundleBean != null) {
258       return textBundleBean;
259     }
260     
261     //do the default
262     return GrouperConfig.retrieveConfig().textBundleDefault();
263 
264   }

314   /**
315    * see if there is one in cache, if so, use it, if not, get from config files
316    * @return the config from file or cache
317    */
318   @Override
319   protected ConfigPropertiesCascadeBase retrieveFromConfigFileOrCache() {
320     
321     if (configFileCache == null) {
322       configFileCache = 
323         new HashMap<String, ConfigPropertiesCascadeBase>();
324     }
325     ConfigPropertiesCascadeBase configObject = configFileCache.get(this.languageCountry);
326     
327     if (configObject == null) {
328       
329       if (LOG != null && LOG.isDebugEnabled()) {
330         LOG.debug("Config file has not be created yet, will create now: " + this.getMainConfigClasspath());
331       }
332       
333       configObject = retrieveFromConfigFiles();
334       configFileCache.put(this.languageCountry, configObject);
335       
336     } else {
337       
338       //see if that much time has passed
339       if (configObject.needToCheckIfFilesNeedReloading()) {
340         
341         synchronized (configObject) {
342           
343           configObject = configFileCache.get(this.languageCountry);
344           
345           //check again in case another thread did it
346           if (configObject.needToCheckIfFilesNeedReloading()) {
347             
348             if (configObject.filesNeedReloadingBasedOnContents()) {
349               configObject = retrieveFromConfigFiles();
350               configFileCache.put(this.languageCountry, configObject);
351             }
352           }
353         }
354       }
355     }
356     
357     return configObject;
358   }

360   /**
361    * init and return the tooltip keys
362    * @return the list of keys
363    */
364   public List<String> tooltipKeys() {
365 
366     if (this.tooltipKeys == null) {
367 
368       //add properties to map
369       Properties propertiesFromConfig = properties();
370 
371       Set<Object> propertiesKeys = propertiesFromConfig.keySet();
372       Map<String, String> propertiesMap = null;
373 
374       //loop through the tooltips
375       if (propertiesKeys != null) {
376         propertiesMap = new HashMap<String, String>();
377         for (Object propertiesKeyObject : propertiesKeys) {
378           
379           String propertiesKey = (String)propertiesKeyObject;
380           
381           //see if a key
382           boolean isTerm = propertiesKey.startsWith(TERM_PREFIX);
383                     
384           //if isValue, continue on
385           if (!isTerm) {
386             continue;
387           }
388           
389           //if term then get the term
390           String term = propertiesFromConfig.getProperty(propertiesKey);
391           String termId = propertiesKey.substring(TERM_PREFIX.length());
392           
393           //strip off the .1, .2, etc if it exists
394           if (termId.matches("^.*\\.[0-9]+$")) {
395             int lastDot = termId.lastIndexOf('.');
396             termId = termId.substring(0,lastDot);
397           }
398           
399           String tooltipKey = "tooltip." + termId;
400           String tooltip = propertiesFromConfig.getProperty(tooltipKey);
401           
402           //tooltipKeys.add(term);
403           tooltip = GrouperTextContainer.convertTooltipTextToHtml(tooltip, term, false);
404 
405           //tooltipValues.add(tooltip);
406           propertiesMap.put(term, tooltip);
407         }
408         
409         propertiesMap = sortMapBySubstringFirst(propertiesMap);
410         //convert back to lists
411         this.tooltipKeys = new ArrayList<String>(propertiesMap.keySet());
412         this.tooltipValues = new ArrayList<String>(propertiesMap.values());
413         
414       }
415       
416     }
417     return this.tooltipKeys;
418   }

420   /**
421    * convert the properties configuration object to a map object (tree map)
422    * @param inputMap 
423    * @return the 
424    */
425   static Map<String, String> sortMapBySubstringFirst(Map<String, String> inputMap) {
426     if (inputMap == null) {
427       return null;
428     }
429     Map<String, String> resultMap = new TreeMap<String, String>(new Comparator<String>() {
430 
431       /**
432        * comparator that puts substrings last
433        * @param o1
434        * @param o2
435        * @return comparison
436        */
437       public int compare(String o1, String o2) {
438         if (o1 == o2) {
439           return 0;
440         }
441         if (o1 == null) {
442           return -1;
443         }
444         if (o2 == null) {
445           return 1;
446         }
447         if (o1.equals(o2)) {
448           return 0;
449         }
450         if (o1.contains(o2)) {
451           return -1;
452         }
453         if (o2.contains(o1)) {
454           return 1;
455         }
456         return o1.compareTo(o2);
457       }
458       
459     });
460     resultMap.putAll(inputMap);
461     return resultMap;
462   }
