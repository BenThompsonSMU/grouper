68    public void testFindMembership() {
69      
70      GrouperSession grouperSession = GrouperSession.startRootSession();
71      Group group1 = new GroupSave(grouperSession).assignCreateParentStemsIfNotExist(true).assignName("test:testGroup1").save();
72      Group group2 = new GroupSave(grouperSession).assignCreateParentStemsIfNotExist(true).assignName("test:testGroup2").save();
73      group1.addMember(SubjectTestHelper.SUBJ0);
74      group2.addMember(SubjectTestHelper.SUBJ1);
75      group1.addMember(group2.toSubject());
76      
77      Membership membership1 = new MembershipFinder().addGroup(group1).addSubject(SubjectTestHelper.SUBJ0).findMembership(true);
78      Membership membership2 = new MembershipFinder().addGroup(group2).addSubject(SubjectTestHelper.SUBJ1).findMembership(true);
79      
80      assertNotNull(membership1);
81      assertEquals(membership1.getGroupName(), group1.getName());
82      assertEquals(membership1.getMemberSubjectId(), SubjectTestHelper.SUBJ0_ID);
83      
84      membership1 = new MembershipFinder().addMembershipId(membership1.getUuid()).findMembership(true);
85  
86      assertNotNull(membership1);
87      assertEquals(membership1.getGroupName(), group1.getName());
88      assertEquals(membership1.getMemberSubjectId(), SubjectTestHelper.SUBJ0_ID);
89      
90      
91      assertEquals(2, new MembershipFinder().addMembershipId(membership1.getUuid()).addMembershipId(membership2.getUuid()).findMembershipsMembers().size());
92      
93    }
