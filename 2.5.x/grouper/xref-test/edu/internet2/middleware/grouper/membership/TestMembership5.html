97    public void testEffectiveMembershipsWithPrivileges() {
98      LOG.info("testEffectiveMembershipsWithPrivileges");
99      try {
100       GrouperUtil.sleep(100);
101       before  = new Date();
102       GrouperUtil.sleep(100);
103 
104       r     = R.populateRegistry(1, 11, 4);
105       gA    = r.getGroup("a", "a");
106       gB    = r.getGroup("a", "b");
107       gC    = r.getGroup("a", "c");
108       gD    = r.getGroup("a", "d");
109       gE    = r.getGroup("a", "e");
110       gF    = r.getGroup("a", "f");
111       gG    = r.getGroup("a", "g");
112       gH    = r.getGroup("a", "h");
113       gI    = r.getGroup("a", "i");
114       gJ    = r.getGroup("a", "j");
115       gK    = r.getGroup("a", "k");
116       subjA = r.getSubject("a");
117       subjB = r.getSubject("b");
118       subjC = r.getSubject("c");
119       subjD = r.getSubject("d");
120 
121       fieldMembers = Group.getDefaultList();
122       fieldUpdaters = FieldFinder.find(Field.FIELD_NAME_UPDATERS, true);
123 
124 
125       // Test 1:  Test when the last operations are adding update privileges for gA.
126       gB.addMember( subjB );
127       gB.addMember( gD.toSubject() );
128       gB.addMember( gE.toSubject() );
129       gB.grantPriv( gC.toSubject(), AccessPrivilege.UPDATE );
130       gD.addMember( subjD );
131       gD.addMember( gF.toSubject() );
132       gE.grantPriv( subjC, AccessPrivilege.UPDATE );
133       gE.grantPriv( gG.toSubject(), AccessPrivilege.UPDATE );
134       gH.addMember( gB.toSubject() );
135       gI.grantPriv( gH.toSubject(), AccessPrivilege.UPDATE );
136       gJ.addMember( gA.toSubject() );
137       gK.grantPriv( gA.toSubject(), AccessPrivilege.UPDATE );
138       gA.grantPriv( subjA, AccessPrivilege.UPDATE );
139       gA.grantPriv( gB.toSubject(), AccessPrivilege.UPDATE );
140 
141       verifyMemberships();
142 
143 
144       // clear out memberships
145       gB.deleteMember( subjB );
146       gB.deleteMember( gD.toSubject() );
147       gB.deleteMember( gE.toSubject() );
148       gB.revokePriv( gC.toSubject(), AccessPrivilege.UPDATE );
149       
150 
151       try {
152         gB.revokePriv( gC.toSubject(), AccessPrivilege.UPDATE );
153         fail("Should throw an exception about already existing");
154       } catch (RevokePrivilegeException rpe) {
155         //good
156       }
157 
158 //      gB.revokePriv( gC.toSubject(), AccessPrivilege.UPDATE );
159       //shouldnt throw an exception
160       
161       gD.deleteMember( subjD );
162       gD.deleteMember( gF.toSubject() );
163       gE.revokePriv( subjC, AccessPrivilege.UPDATE );
164       gE.revokePriv( gG.toSubject(), AccessPrivilege.UPDATE );
165       gH.deleteMember( gB.toSubject() );
166       gI.revokePriv( gH.toSubject(), AccessPrivilege.UPDATE );
167       gJ.deleteMember( gA.toSubject() );
168       gK.revokePriv( gA.toSubject(), AccessPrivilege.UPDATE );
169       gA.revokePriv( subjA, AccessPrivilege.UPDATE );
170       gA.revokePriv( gB.toSubject(), AccessPrivilege.UPDATE );
171 
172       Set<Membership> listMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldMembers);
173       T.amount("Number of list memberships", 0, listMemberships.size());
174 
175       Set<Membership> updateMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldUpdaters);
176       T.amount("Number of update privileges", 0, updateMemberships.size());
177 
178 
179 
180       // Test 2:  Test when the last operations are adding list memberships to gB.
181       gA.grantPriv( subjA, AccessPrivilege.UPDATE );
182       gA.grantPriv( gB.toSubject(), AccessPrivilege.UPDATE );
183       gB.grantPriv( gC.toSubject(), AccessPrivilege.UPDATE );
184       gD.addMember( subjD );
185       gD.addMember( gF.toSubject() );
186       gE.grantPriv( subjC, AccessPrivilege.UPDATE );
187       gE.grantPriv( gG.toSubject(), AccessPrivilege.UPDATE );
188       gH.addMember( gB.toSubject() );
189       gI.grantPriv( gH.toSubject(), AccessPrivilege.UPDATE );
190       gJ.addMember( gA.toSubject() );
191       gK.grantPriv( gA.toSubject(), AccessPrivilege.UPDATE );
192       gB.addMember( subjB );
193       gB.addMember( gD.toSubject() );
194       gB.addMember( gE.toSubject() );
195 
196       verifyMemberships();
197 
198       r.rs.stop();
199     }
200     catch (Exception e) {
201       T.e(e);
202     }
203   }
