51  /**
52   * @author shilen
53   * @version $Id: TestMemberChangeInMembership.java,v 1.2 2009-12-07 07:31:08 mchyzer Exp $
54   */
55  public class TestMemberChangeInMembership extends GrouperTest {
56  
57    R r = null;
58  
59    /**
60     * @param name
61     */
62    public TestMemberChangeInMembership(String name) {
63      super(name);
64    }
65  
66    protected void tearDown () {
67      super.tearDown();
68      if (r != null) {
69        r.rs.stop();
70      }
71    }
72    
73    /**
74     * @throws Exception
75     */
76    public void testChangingMember() throws Exception {
77      
78      r = R.populateRegistry(0, 0, 4);
79      Subject a = r.getSubject("a");
80      Subject b = r.getSubject("b");
81      Subject c = r.getSubject("c");
82      Subject d = r.getSubject("d");
83      
84      Stem root = StemFinder.findRootStem(r.rs);
85  
86      Stem stem = root.addChildStem("stem", "stem");
87      Group top1 = stem.addChildGroup("top1", "top1");
88      Group top2 = stem.addChildGroup("top2", "top2");
89  
90      Group one = stem.addChildGroup("one", "one");
91      Group two = stem.addChildGroup("two", "two");
92  
93      Group owner = stem.addChildGroup("owner", "owner");
94      Group left = stem.addChildGroup("left", "left");
95      Group right = stem.addChildGroup("right", "right");
96  
97      owner.addCompositeMember(CompositeType.COMPLEMENT, left, right);
98      left.addMember(one.toSubject());
99      left.addMember(a);
100     one.addMember(two.toSubject());
101     one.addMember(b);
102     two.addMember(c);
103     
104     top1.addMember(owner.toSubject());
105     top2.grantPriv(owner.toSubject(), AccessPrivilege.UPDATE);
106     stem.grantPriv(owner.toSubject(), NamingPrivilege.CREATE);
107     
108     int numberOfMembershipsBefore = GrouperDAOFactory.getFactory().getMembership().findAll(false).size();
109     assertTrue(two.hasMember(c));
110     assertTrue(one.hasMember(c));
111     assertTrue(left.hasMember(c));
112     assertTrue(owner.hasMember(c));
113     assertTrue(top1.hasMember(c));
114     assertTrue(top2.hasUpdate(c));
115     assertTrue(stem.hasCreate(c));
116     
117     // change member
118     Membership ms = GrouperDAOFactory.getFactory().getMembership().findByGroupOwnerAndMemberAndFieldAndType(
119         two.getUuid(), MemberFinder.findBySubject(r.rs, c, true).getUuid(), Group.getDefaultList(), MembershipType.IMMEDIATE.getTypeString(), true, true);
120     ms.setMemberUuid(MemberFinder.findBySubject(r.rs, d, true).getUuid());
121     ms.setMember(MemberFinder.findBySubject(r.rs, d, true));
122     GrouperDAOFactory.getFactory().getMembership().update(ms);
123     GrouperCacheUtils.clearAllCaches();
124     
125     int numberOfMembershipsAfter = GrouperDAOFactory.getFactory().getMembership().findAll(false).size();
126     assertEquals(numberOfMembershipsBefore, numberOfMembershipsAfter);
127     
128     assertTrue(two.hasMember(d));
129     assertTrue(one.hasMember(d));
130     assertTrue(left.hasMember(d));
131     assertTrue(owner.hasMember(d));
132     assertTrue(top1.hasMember(d));
133     assertTrue(top2.hasUpdate(d));
134     assertTrue(stem.hasCreate(d));
135     
136     assertFalse(two.hasMember(c));
137     assertFalse(one.hasMember(c));
138     assertFalse(left.hasMember(c));
139     assertFalse(owner.hasMember(c));
140     assertFalse(top1.hasMember(c));
141     assertFalse(top2.hasUpdate(c));
142     assertFalse(stem.hasCreate(c));
143   }
144 }
