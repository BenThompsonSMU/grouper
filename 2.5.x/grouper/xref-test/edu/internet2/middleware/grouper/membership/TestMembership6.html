87    public void testEffectiveMembershipsWithPrivilegesAndComposites() {
88      LOG.info("testEffectiveMembershipsWithPrivilegesAndComposites");
89      try {
90        GrouperUtil.sleep(100);
91        before  = new Date();
92        GrouperUtil.sleep(100);
93  
94        r     = R.populateRegistry(2, 10, 2);
95        gA    = r.getGroup("a", "a");
96        gB    = r.getGroup("a", "b");
97        gC    = r.getGroup("a", "c");
98        gD    = r.getGroup("a", "d");
99        gE    = r.getGroup("a", "e");
100       gF    = r.getGroup("a", "f");
101       gG    = r.getGroup("a", "g");
102       gH    = r.getGroup("a", "h");
103       gI    = r.getGroup("a", "i");
104       gJ    = r.getGroup("a", "j");
105       subjA = r.getSubject("a");
106       subjB = r.getSubject("b");
107       nsA   = r.getStem("a");
108 
109       fieldMembers = Group.getDefaultList();
110       fieldUpdaters = FieldFinder.find(Field.FIELD_NAME_UPDATERS, true);
111       fieldCreators = FieldFinder.find(Field.FIELD_NAME_CREATORS, true);
112 
113 
114       // Test 1:  Test when the last operations are adding update privileges for gH and gJ.
115       nsA.grantPriv( gE.toSubject(), NamingPrivilege.CREATE );
116       gA.addCompositeMember(CompositeType.UNION, gB, gC);
117       gA.grantPriv( subjB, AccessPrivilege.UPDATE );
118       gB.addMember( gD.toSubject() );
119       gC.addMember( subjA );
120       gE.addCompositeMember(CompositeType.UNION, gF, gA);
121       gG.grantPriv( gE.toSubject(), AccessPrivilege.UPDATE );
122       gI.addMember( gA.toSubject() );
123       gH.grantPriv( gI.toSubject(), AccessPrivilege.UPDATE );
124       gJ.grantPriv( gA.toSubject(), AccessPrivilege.UPDATE );
125 
126       verifyMemberships();
127 
128       // clear out memberships
129       nsA.revokePriv( gE.toSubject(), NamingPrivilege.CREATE );
130       gA.deleteCompositeMember();
131       gA.revokePriv( subjB, AccessPrivilege.UPDATE );
132       gB.deleteMember( gD.toSubject() );
133       gC.deleteMember( subjA );
134       gE.deleteCompositeMember();
135       gG.revokePriv( gE.toSubject(), AccessPrivilege.UPDATE );
136       gI.deleteMember( gA.toSubject() );
137       gH.revokePriv( gI.toSubject(), AccessPrivilege.UPDATE );
138       gJ.revokePriv( gA.toSubject(), AccessPrivilege.UPDATE );
139 
140       Set<Membership> listMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldMembers);
141       T.amount("Number of list memberships after test 1", 0, listMemberships.size());
142 
143       Set<Membership> updateMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldUpdaters);
144       T.amount("Number of update privileges after test 1", 0, updateMemberships.size());
145 
146       Set<Membership> createMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldCreators);
147       T.amount("Number of create privileges after test 1", 0, createMemberships.size());
148 
149 
150       // Test 2:  Test when the last operation is adding the composite for gA.
151       nsA.grantPriv( gE.toSubject(), NamingPrivilege.CREATE );
152       gA.grantPriv( subjB, AccessPrivilege.UPDATE );
153       gB.addMember( gD.toSubject() );
154       gC.addMember( subjA );
155       gE.addCompositeMember(CompositeType.UNION, gF, gA);
156       gG.grantPriv( gE.toSubject(), AccessPrivilege.UPDATE );
157       gH.grantPriv( gI.toSubject(), AccessPrivilege.UPDATE );
158       gI.addMember( gA.toSubject() );
159       gJ.grantPriv( gA.toSubject(), AccessPrivilege.UPDATE );
160       gA.addCompositeMember(CompositeType.UNION, gB, gC);
161 
162       verifyMemberships();
163 
164       // clear out memberships
165       nsA.revokePriv( gE.toSubject(), NamingPrivilege.CREATE );
166       gA.deleteCompositeMember();
167       gA.revokePriv( subjB, AccessPrivilege.UPDATE );
168       gB.deleteMember( gD.toSubject() );
169       gC.deleteMember( subjA );
170       gE.deleteCompositeMember();
171       gG.revokePriv( gE.toSubject(), AccessPrivilege.UPDATE );
172       gI.deleteMember( gA.toSubject() );
173       gH.revokePriv( gI.toSubject(), AccessPrivilege.UPDATE );
174       gJ.revokePriv( gA.toSubject(), AccessPrivilege.UPDATE );
175 
176       listMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldMembers);
177       T.amount("Number of list memberships after test 2", 0, listMemberships.size());
178 
179       updateMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldUpdaters);
180       T.amount("Number of update privileges after test 2", 0, updateMemberships.size());
181 
182       createMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldCreators);
183       T.amount("Number of create privileges after test 2", 0, createMemberships.size());
184 
185 
186       // Test 3:  Test when the last operations are adding members to gB and gC.
187       nsA.grantPriv( gE.toSubject(), NamingPrivilege.CREATE );
188       gA.addCompositeMember(CompositeType.UNION, gB, gC);
189       gA.grantPriv( subjB, AccessPrivilege.UPDATE );
190       gE.addCompositeMember(CompositeType.UNION, gF, gA);
191       gG.grantPriv( gE.toSubject(), AccessPrivilege.UPDATE );
192       gH.grantPriv( gI.toSubject(), AccessPrivilege.UPDATE );
193       gI.addMember( gA.toSubject() );
194       gJ.grantPriv( gA.toSubject(), AccessPrivilege.UPDATE );
195       gB.addMember( gD.toSubject() );
196       gC.addMember( subjA );
197 
198       verifyMemberships();
199 
200       // clear out memberships
201       nsA.revokePriv( gE.toSubject(), NamingPrivilege.CREATE );
202       gA.deleteCompositeMember();
203       gA.revokePriv( subjB, AccessPrivilege.UPDATE );
204       gB.deleteMember( gD.toSubject() );
205       gC.deleteMember( subjA );
206       gE.deleteCompositeMember();
207       gG.revokePriv( gE.toSubject(), AccessPrivilege.UPDATE );
208       gI.deleteMember( gA.toSubject() );
209       gH.revokePriv( gI.toSubject(), AccessPrivilege.UPDATE );
210       gJ.revokePriv( gA.toSubject(), AccessPrivilege.UPDATE );
211 
212       listMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldMembers);
213       T.amount("Number of list memberships after test 3", 0, listMemberships.size());
214 
215       updateMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldUpdaters);
216       T.amount("Number of update privileges after test 3", 0, updateMemberships.size());
217 
218       createMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldCreators);
219       T.amount("Number of create privileges after test 3", 0, createMemberships.size());
220 
221 
222       r.rs.stop();
223     }
224     catch (Exception e) {
225       T.e(e);
226     }
227   }

230   public  void verifyMemberships() throws Exception {
231     // SA -> gA
232     MembershipTestHelper.verifyCompositeMembership(r.rs, "SA -> gA", gA, subjA);
233 
234     // SB -> gA
235     MembershipTestHelper.verifyImmediateMembership(r.rs, "SB -> gA", gA, subjB, fieldUpdaters);
236 
237     // gD -> gB
238     MembershipTestHelper.verifyImmediateMembership(r.rs, "gD -> gB", gB, gD.toSubject(), fieldMembers);
239 
240     // SA -> gC
241     MembershipTestHelper.verifyImmediateMembership(r.rs, "SA -> gC", gC, subjA, fieldMembers);
242 
243     // SA -> gE
244     MembershipTestHelper.verifyCompositeMembership(r.rs, "SA -> gE", gE, subjA);
245 
246     // gE -> gG
247     MembershipTestHelper.verifyImmediateMembership(r.rs, "gE -> gG", gG, gE.toSubject(), fieldUpdaters);
248 
249     // SA -> gG (parent: gE -> gG) (depth: 1)
250     MembershipTestHelper.verifyEffectiveMembership(r.rs, "SA -> gG", gG, subjA, gE, 1, gG, gE.toSubject(), null, 0, fieldUpdaters);
251 
252     // gI -> gH
253     MembershipTestHelper.verifyImmediateMembership(r.rs, "gI -> gH", gH, gI.toSubject(), fieldUpdaters);
254 
255     // gA -> gH (parent: gI -> gH) (depth: 1)
256     MembershipTestHelper.verifyEffectiveMembership(r.rs, "gA -> gH", gH, gA.toSubject(), gI, 1, gH, gI.toSubject(), null, 0, fieldUpdaters);
257 
258     // SA -> gH (parent: gA -> gH) (depth: 2)
259     MembershipTestHelper.verifyEffectiveMembership(r.rs, "SA -> gH", gH, subjA, gA, 2, gH, gA.toSubject(), gI, 1, fieldUpdaters);
260 
261     // gA -> gI
262     MembershipTestHelper.verifyImmediateMembership(r.rs, "gA -> gI", gI, gA.toSubject(), fieldMembers);
263 
264     // SA -> gI (parent: gA -> gI) (depth: 1)
265     MembershipTestHelper.verifyEffectiveMembership(r.rs, "SA -> gI", gI, subjA, gA, 1, gI, gA.toSubject(), null, 0, fieldMembers);
266 
267     // gA -> gJ
268     MembershipTestHelper.verifyImmediateMembership(r.rs, "gA -> gJ", gJ, gA.toSubject(), fieldUpdaters);
269 
270     // SA -> gJ (parent: gA -> gJ) (depth: 1)
271     MembershipTestHelper.verifyEffectiveMembership(r.rs, "SA -> gJ", gJ, subjA, gA, 1, gJ, gA.toSubject(), null, 0, fieldUpdaters); 
272 
273     // gE -> nsA
274     MembershipTestHelper.verifyImmediateMembership(r.rs, "gE -> nsA", nsA, gE.toSubject(), fieldCreators);
275 
276     // SA -> nsA (parent: gE -> nsA) (depth: 1)
277     MembershipTestHelper.verifyEffectiveMembership(r.rs, "SA -> nsA", nsA, subjA, gE, 1, nsA, gE.toSubject(), null, 0, fieldCreators); 
278 
279 
280     // verify the total number of list memberships
281     Set<Membership> listMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldMembers);
282     T.amount("Number of list memberships", 6, listMemberships.size());
283 
284     // verify the total number of update privileges
285     Set<Membership> updateMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldUpdaters);
286     T.amount("Number of update privileges", 8, updateMemberships.size());
287 
288     // verify the total number of create privileges
289     Set<Membership> createMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldCreators);
290     T.amount("Number of create privileges", 2, createMemberships.size());
291   }
