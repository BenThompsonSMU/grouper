93    public void testMembershipDeletes2() {
94      LOG.info("testMembershipDeletes2");
95      try {
96        GrouperUtil.sleep(100);
97        before  = new Date();
98        GrouperUtil.sleep(100);
99  
100       r     = R.populateRegistry(2, 10, 2);
101       gA    = r.getGroup("a", "a");
102       gB    = r.getGroup("a", "b");
103       gC    = r.getGroup("a", "c");
104       gD    = r.getGroup("a", "d");
105       gE    = r.getGroup("a", "e");
106       gF    = r.getGroup("a", "f");
107       gG    = r.getGroup("a", "g");
108       gH    = r.getGroup("a", "h");
109       gI    = r.getGroup("a", "i");
110       gJ    = r.getGroup("a", "j");
111       subjA = r.getSubject("a");
112       subjB = r.getSubject("b");
113       nsA   = r.getStem("a");
114 
115       fieldMembers = Group.getDefaultList();
116       fieldUpdaters = FieldFinder.find(Field.FIELD_NAME_UPDATERS, true);
117       fieldCreators = FieldFinder.find(Field.FIELD_NAME_CREATORS, true);
118 
119 
120       // initial data
121       nsA.grantPriv( gE.toSubject(), NamingPrivilege.CREATE );
122       gA.addCompositeMember(CompositeType.UNION, gB, gC);
123       gA.grantPriv( subjB, AccessPrivilege.UPDATE );
124       gB.addMember( gD.toSubject() );
125       gC.addMember( subjA );
126       gE.addCompositeMember(CompositeType.UNION, gF, gA);
127       gG.grantPriv( gE.toSubject(), AccessPrivilege.UPDATE );
128       gI.addMember( gA.toSubject() );
129       gH.grantPriv( gI.toSubject(), AccessPrivilege.UPDATE );
130       gJ.grantPriv( gA.toSubject(), AccessPrivilege.UPDATE );
131 
132       verifyMembershipAddAndDeleteMember(gB, gG.toSubject());
133       verifyMembershipAddAndDeleteAccessPrivUpdate(gB, gI.toSubject());
134       verifyMembershipAddAndDeleteAccessPrivUpdate(gB, gD.toSubject());
135       verifyMembershipAddAndDeleteAccessPrivUpdate(gA, gA.toSubject());
136       verifyMembershipAddAndDeleteAccessPrivUpdate(gI, gA.toSubject());
137       verifyMembershipAddAndDeleteNamingPrivCreate(nsA, gI.toSubject());
138 
139 
140       r.rs.stop();
141     }
142     catch (Exception e) {
143       T.e(e);
144     }
145   }
146 
147   public void verifyMembershipAddAndDeleteMember(Group g, Subject s) throws Exception {
148     g.addMember(s);
149     g.deleteMember(s);
150     verifyMemberships();
151   }
152 
153   public void verifyMembershipAddAndDeleteAccessPrivUpdate(Group g, Subject s) throws Exception {
154     g.grantPriv(s, AccessPrivilege.UPDATE);
155     g.revokePriv(s, AccessPrivilege.UPDATE);
156     verifyMemberships();
157   }
158 
159   public void verifyMembershipAddAndDeleteNamingPrivCreate(Stem stem, Subject s) throws Exception {
160     stem.grantPriv(s, NamingPrivilege.CREATE);
161     stem.revokePriv(s, NamingPrivilege.CREATE);
162     verifyMemberships();
163   }
164 
165   public  void verifyMemberships() throws Exception {
166     // SA -> gA
167     MembershipTestHelper.verifyCompositeMembership(r.rs, "SA -> gA", gA, subjA);
168 
169     // SB -> gA
170     MembershipTestHelper.verifyImmediateMembership(r.rs, "SB -> gA", gA, subjB, fieldUpdaters);
171 
172     // gD -> gB
173     MembershipTestHelper.verifyImmediateMembership(r.rs, "gD -> gB", gB, gD.toSubject(), fieldMembers);
174 
175     // SA -> gC
176     MembershipTestHelper.verifyImmediateMembership(r.rs, "SA -> gC", gC, subjA, fieldMembers);
177 
178     // SA -> gE
179     MembershipTestHelper.verifyCompositeMembership(r.rs, "SA -> gE", gE, subjA);
180 
181     // gE -> gG
182     MembershipTestHelper.verifyImmediateMembership(r.rs, "gE -> gG", gG, gE.toSubject(), fieldUpdaters);
183 
184     // SA -> gG (parent: gE -> gG) (depth: 1)
185     MembershipTestHelper.verifyEffectiveMembership(r.rs, "SA -> gG", gG, subjA, gE, 1, gG, gE.toSubject(), null, 0, fieldUpdaters);
186 
187     // gI -> gH
188     MembershipTestHelper.verifyImmediateMembership(r.rs, "gI -> gH", gH, gI.toSubject(), fieldUpdaters);
189 
190     // gA -> gH (parent: gI -> gH) (depth: 1)
191     MembershipTestHelper.verifyEffectiveMembership(r.rs, "gA -> gH", gH, gA.toSubject(), gI, 1, gH, gI.toSubject(), null, 0, fieldUpdaters);
192 
193     // SA -> gH (parent: gA -> gH) (depth: 2)
194     MembershipTestHelper.verifyEffectiveMembership(r.rs, "SA -> gH", gH, subjA, gA, 2, gH, gA.toSubject(), gI, 1, fieldUpdaters);
195 
196     // gA -> gI
197     MembershipTestHelper.verifyImmediateMembership(r.rs, "gA -> gI", gI, gA.toSubject(), fieldMembers);
198 
199     // SA -> gI (parent: gA -> gI) (depth: 1)
200     MembershipTestHelper.verifyEffectiveMembership(r.rs, "SA -> gI", gI, subjA, gA, 1, gI, gA.toSubject(), null, 0, fieldMembers);
201 
202     // gA -> gJ
203     MembershipTestHelper.verifyImmediateMembership(r.rs, "gA -> gJ", gJ, gA.toSubject(), fieldUpdaters);
204 
205     // SA -> gJ (parent: gA -> gJ) (depth: 1)
206     MembershipTestHelper.verifyEffectiveMembership(r.rs, "SA -> gJ", gJ, subjA, gA, 1, gJ, gA.toSubject(), null, 0, fieldUpdaters); 
207 
208     // gE -> nsA
209     MembershipTestHelper.verifyImmediateMembership(r.rs, "gE -> nsA", nsA, gE.toSubject(), fieldCreators);
210 
211     // SA -> nsA (parent: gE -> nsA) (depth: 1)
212     MembershipTestHelper.verifyEffectiveMembership(r.rs, "SA -> nsA", nsA, subjA, gE, 1, nsA, gE.toSubject(), null, 0, fieldCreators); 
213 
214     // verify the total number of list memberships
215     Set<Membership> listMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldMembers);
216     T.amount("Number of list memberships", 6, listMemberships.size());
217 
218     // verify the total number of update privileges
219     Set<Membership> updateMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldUpdaters);
220     T.amount("Number of update privileges", 8, updateMemberships.size());
221 
222     // verify the total number of create privileges
223     Set<Membership> createMemberships = MembershipFinder.internal_findAllByCreatedAfter(r.rs, before, fieldCreators);
224     T.amount("Number of create privileges", 2, createMemberships.size());
225   }
