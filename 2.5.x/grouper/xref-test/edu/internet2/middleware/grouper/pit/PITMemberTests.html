94    public void testGetGroupsAtPointInTime() {
95      Member member0 = MemberFinder.findBySubject(grouperSession, SubjectTestHelper.SUBJ0, true);
96      Member member1 = MemberFinder.findBySubject(grouperSession, SubjectTestHelper.SUBJ1, true);
97      GrouperConfig.retrieveConfig().propertiesOverrideMap().put("groups.create.grant.all.read", "false");
98      GrouperConfig.retrieveConfig().propertiesOverrideMap().put("groups.create.grant.all.view", "false");
99      
100     Group group1 = edu.addChildGroup("test1", "test1");
101     Group group2 = edu.addChildGroup("test2", "test2");
102     Group group3 = edu.addChildGroup("test3", "test3");
103     Group group4 = edu.addChildGroup("test4", "test4");
104 
105     Timestamp beforeAll = getTimestampWithSleep();
106     group1.addMember(member0.getSubject());
107     group2.addMember(member0.getSubject());
108     Timestamp afterGroup2AddMember = getTimestampWithSleep();
109     group3.addMember(member0.getSubject());
110     group4.addMember(member1.getSubject());
111     ChangeLogTempToEntity.convertRecords();
112     
113     group1.deleteMember(member0.getSubject());
114     Timestamp afterGroup1DeleteMember = getTimestampWithSleep();
115     ChangeLogTempToEntity.convertRecords();
116 
117     group2.delete();
118     Timestamp afterGroup2Delete = getTimestampWithSleep();
119     ChangeLogTempToEntity.convertRecords();
120     
121     PITMember pitMember0 = GrouperDAOFactory.getFactory().getPITMember().findBySourceIdUnique(member0.getUuid(), false);
122     PITGroup pitGroup1 = GrouperDAOFactory.getFactory().getPITGroup().findBySourceIdUnique(group1.getId(), false);
123     PITGroup pitGroup2 = GrouperDAOFactory.getFactory().getPITGroup().findBySourceIdUnique(group2.getId(), false);
124     PITGroup pitGroup3 = GrouperDAOFactory.getFactory().getPITGroup().findBySourceIdUnique(group3.getId(), false);
125     
126     Set<PITGroup> results = pitMember0.getGroups(Group.getDefaultList().getUuid(), null, null, null, beforeAll, beforeAll, null);
127     assertEquals(0, results.size());
128     
129     results = pitMember0.getGroups(Group.getDefaultList().getUuid(), null, null, null, afterGroup2AddMember, afterGroup2AddMember, null);
130     assertEquals(2, results.size());
131     assertTrue(results.contains(pitGroup1));
132     assertTrue(results.contains(pitGroup2));
133     
134     results = pitMember0.getGroups(Group.getDefaultList().getUuid(), null, null, null, afterGroup1DeleteMember, afterGroup1DeleteMember, null);
135     assertEquals(2, results.size());
136     assertTrue(results.contains(pitGroup2));
137     assertTrue(results.contains(pitGroup3));
138     
139     results = pitMember0.getGroups(Group.getDefaultList().getUuid(), null, null, null, afterGroup2Delete, afterGroup2Delete, null);
140     assertEquals(1, results.size());
141     assertTrue(results.contains(pitGroup3));
142   }
