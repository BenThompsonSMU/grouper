47      TestRunner.run(new ChangeLogTypeTest("testPersistence"));

50    /**
51     * 
52     */
53    public void testPersistence() {
54      
55      //clear out
56      ChangeLogTypeDAO changeLogTypeDao = GrouperDAOFactory.getFactory().getChangeLogType();
57      changeLogTypeDao.deleteEntriesAndTypesByCategoryAndAction("a", "b");
58      
59      ChangeLogType changeLogType = new ChangeLogType("a", "b", null, "s1", "s2");
60      changeLogType.setId(GrouperUuid.getUuid());
61      changeLogTypeDao.saveOrUpdate(changeLogType);
62      
63      //update and save again
64      changeLogType.setLabelString03("s3");
65      changeLogTypeDao.saveOrUpdate(changeLogType);
66    
67      ChangeLogEntry changeLogEntry = new ChangeLogEntry();
68      changeLogEntry.setChangeLogTypeId(changeLogType.getId());
69      changeLogEntry.setContextId("whatever");
70      changeLogEntry.setString01("something");
71      changeLogEntry.setString02(GrouperUuid.getUuid());
72      
73      ChangeLogEntryDAO changeLogEntryDao = GrouperDAOFactory.getFactory().getChangeLogEntry();
74      changeLogEntryDao.save(changeLogEntry);
75      
76      //clear out
77      changeLogTypeDao.deleteEntriesAndTypesByCategoryAndAction("a", "b");
78    }
