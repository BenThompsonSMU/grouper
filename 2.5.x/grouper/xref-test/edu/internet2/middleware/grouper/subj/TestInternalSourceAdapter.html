283   public void testGrouperAllAttributes() {
284     LOG.info("testGrouperAllAttributes");
285     String id = SubjectTestHelper.SUBJ_ALL;
286     try { 
287       Subject subj  = sa.getSubject(id, true);
288       Map     attrs = subj.getAttributes();
289       Assert.assertTrue("zero attrs", attrs.size() == 1);
290       String  val   = subj.getAttributeValue("foo");
291       Assert.assertTrue(
292         "no attr (" + val + ")", val == null
293       );
294       Set     vals  = subj.getAttributeValues("foo");
295       Assert.assertTrue("zero values", GrouperUtil.length(vals) == 0);
296       Assert.assertTrue(
297         "desc == name", subj.getDescription().equals(subj.getName())
298       );
299     } 
300     catch (SubjectNotFoundException e) {
301       Assert.fail("failed to find subject: " + e.getMessage());
302     }
303     catch (SubjectNotUniqueException eSNU) {
304       T.e(eSNU);
305     }
306   } // public void testGrouperAllAttributes()
