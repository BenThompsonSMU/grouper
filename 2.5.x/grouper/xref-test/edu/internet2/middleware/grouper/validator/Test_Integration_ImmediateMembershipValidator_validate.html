86    public void testValidate_InvalidGroupInCircularCheck() {
87      try {
88        LOG.info("testValidate_InvalidGroupInCircularCheck");
89        R     r = R.getContext("grouper");
90        Group g = r.getGroup("i2mi:grouper", "grouper-dev");
91  
92        Membership _ms = new Membership();
93        _ms.setType(MembershipType.IMMEDIATE.getTypeString());
94        _ms.setDepth(0);
95        _ms.setFieldId(FieldFinder.findFieldId("members", 
96            FieldType.LIST.getType(), true));
97        _ms.setMemberUuid( g.toMember().getUuid() );
98        try {
99          ImmediateMembershipValidator.validate(_ms);
100         fail("failed to throw IllegalStateException on unknown group in circular check");
101       }
102       catch (IllegalStateException eExpected) {
103         assertTrue("threw expected IllegalStateException on unknown group in circular check", true);
104       }
105     }
106     catch (Exception e) {
107       unexpectedException(e);
108     }
109   } // public void testValidate_InvalidGroupInCircularCheck()

137   public void testValidate_Circular() {
138     try {
139       LOG.info("testValidate_Circular");
140       R     r = R.getContext("grouper");
141       Group g = r.getGroup("i2mi:grouper", "grouper-dev");
142 
143       Membership _ms = new Membership();
144       _ms.setType(MembershipType.IMMEDIATE.getTypeString());
145       _ms.setDepth(0);
146       _ms.setFieldId(FieldFinder.findFieldId("members", 
147           FieldType.LIST.getType(), true));
148       _ms.setOwnerGroupId( g.getUuid() );
149       _ms.setMemberUuid( g.toMember().getUuid() );
150       GrouperValidator v = ImmediateMembershipValidator.validate(_ms);
151       assertTrue( "v is invalid", v.isInvalid() );
152       assertEquals( "v error msg", ImmediateMembershipValidator.INVALID_CIRCULAR, v.getErrorMessage() );
153     }
154     catch (Exception e) {
155       unexpectedException(e);
156     }
157   } // public void testValidate_Circular()

159   public void testValidate_InvalidMembership() {
160     try {
161       LOG.info("testValidate_Circular");
162       R       r = R.getContext("grouper");
163       Group   g = r.getGroup("i2mi:grouper", "grouper-dev");
164       Member  m = r.getGroup("i2mi:grouper", "grouper-users").toMember();
165 
166       Membership _ms = new Membership();
167       _ms.setType(MembershipType.IMMEDIATE.getTypeString());
168       _ms.setDepth(0);
169       _ms.setFieldId(FieldFinder.findFieldId("members", 
170           FieldType.LIST.getType(), true));
171       _ms.setOwnerGroupId( g.getUuid() );
172       _ms.setMemberUuid( m.getUuid() );
173       GrouperValidator v = ImmediateMembershipValidator.validate(_ms);
174       assertTrue( "v is invalid", v.isInvalid() );
175     }
176     catch (Exception e) {
177       unexpectedException(e);
