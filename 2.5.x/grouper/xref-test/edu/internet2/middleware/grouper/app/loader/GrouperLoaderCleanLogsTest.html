50    /**
51     * 
52     */
53    public void testCleanLogs() {
54      // clear existing logs
55      HibernateSession.bySqlStatic().executeSql("delete from grouper_loader_log");
56  
57      //lets make a bunch of loader logs and change log records
58      for (int i=0;i<723;i++) {
59        Hib3GrouperLoaderLog hib3GrouploaderLog = new Hib3GrouperLoaderLog();
60        hib3GrouploaderLog.setHost("abc");
61        hib3GrouploaderLog.setStatus(GrouperLoaderStatus.CONFIG_ERROR.name());
62        hib3GrouploaderLog.store();
63  
64        ChangeLogEntry changeLogEntry = new ChangeLogEntry(true, ChangeLogTypeBuiltin.GROUP_FIELD_ADD, 
65            null, 
66            null, null, 
67            null, null, 
68            null,
69            null, 
70            null,
71            null, null
72        );
73        
74        changeLogEntry.setTempObject(false);
75        changeLogEntry.save();
76      }
77      
78      //change the dates
79      HibernateSession.bySqlStatic().executeSql("update grouper_loader_log set last_updated = ? where host = 'abc'", 
80          GrouperUtil.toListObject(new Date(0L)), HibUtils.listType(DateType.INSTANCE));
81      HibernateSession.bySqlStatic().executeSql("update grouper_change_log_entry set created_on = 1");
82  
83      int grouperLoaderLogCount = HibernateSession.bySqlStatic().select(int.class, "select count(1) from grouper_loader_log");
84      int grouperChangeLogEntryCount = HibernateSession.bySqlStatic().select(int.class, "select count(1) from grouper_change_log_entry");
85  
86      assertTrue(grouperLoaderLogCount >= 723);
87      assertTrue(grouperChangeLogEntryCount >= 723);
88  
89      GrouperConfig.retrieveConfig().propertiesOverrideMap().put("grouperBatchDeleteSelectSize", "237");
90      try {
91    
92        GrouperLoader.runOnceByJobName(GrouperSession.startRootSession(), GrouperLoaderType.MAINTENANCE_CLEAN_LOGS);
93        
94        grouperLoaderLogCount = HibernateSession.bySqlStatic().select(int.class, "select count(1) from grouper_loader_log");
95        grouperChangeLogEntryCount = HibernateSession.bySqlStatic().select(int.class, "select count(1) from grouper_change_log_entry");
96    
97        assertTrue(grouperLoaderLogCount < 100);
98        assertTrue(grouperChangeLogEntryCount < 100);
99      } finally {
100       GrouperConfig.retrieveConfig().propertiesOverrideMap().remove("grouperBatchDeleteSelectSize");
101     }
102     
103   }
