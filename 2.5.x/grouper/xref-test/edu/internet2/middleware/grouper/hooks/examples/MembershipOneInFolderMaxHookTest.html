53    /**
54     * Test method for {@link edu.internet2.middleware.grouper.hooks.examples.MembershipOneInFolderMaxHook#membershipPreAddMember(edu.internet2.middleware.grouper.hooks.beans.HooksContext, edu.internet2.middleware.grouper.hooks.beans.HooksMembershipChangeBean)}.
55     */
56    public void testGroupRemovesMembership() {
57      
58      GrouperSession grouperSession = GrouperSession.startRootSession();
59  
60      //allow all to attribute
61      MembershipOneInFolderMaxHook.membershipOneInFolderAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_READ, false);
62      MembershipOneInFolderMaxHook.membershipOneInFolderAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_UPDATE, false);
63      
64      //has the attribute
65      Stem testStem1 = new StemSave(grouperSession).assignName("test1").save();
66  
67      testStem1.grantPriv(SubjectTestHelper.SUBJ0, NamingPrivilege.CREATE);
68      testStem1.grantPriv(SubjectTestHelper.SUBJ0, NamingPrivilege.STEM);
69  
70      testStem1.grantPriv(SubjectTestHelper.SUBJ1, NamingPrivilege.CREATE);
71      testStem1.grantPriv(SubjectTestHelper.SUBJ1, NamingPrivilege.STEM);
72  
73      GrouperSession.stopQuietly(grouperSession);
74  
75      grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ0);
76  
77      Group group1A = new GroupSave(grouperSession).assignCreateParentStemsIfNotExist(true)
78          .assignName("test1:someGroupA").save();
79  
80      group1A.grantPriv(SubjectTestHelper.SUBJ1, AccessPrivilege.ADMIN, false);
81      
82      Group group1B = new GroupSave(grouperSession).assignCreateParentStemsIfNotExist(true)
83          .assignName("test1:someGroupB").save();
84  
85      group1B.grantPriv(SubjectTestHelper.SUBJ1, AccessPrivilege.ADMIN, false);
86  
87      //apply the attribute
88      MembershipOneInFolderMaxHook.assignMembershipOneInFolderAttributeDefName(testStem1);
89      
90      group1A.addMember(SubjectTestHelper.SUBJ5);
91      group1A.addMember(SubjectTestHelper.SUBJ6);
92      group1B.addMember(SubjectTestHelper.SUBJ8);
93      group1B.addMember(SubjectTestHelper.SUBJ5);
94  
95      assertFalse(group1A.hasMember(SubjectTestHelper.SUBJ5));
96      assertTrue(group1B.hasMember(SubjectTestHelper.SUBJ5));
97      assertTrue(group1A.hasMember(SubjectTestHelper.SUBJ6));
98      assertTrue(group1B.hasMember(SubjectTestHelper.SUBJ8));
99  
100     //add same again, should be fine
101     group1B.addMember(SubjectTestHelper.SUBJ5, false);
102     assertTrue(group1B.hasMember(SubjectTestHelper.SUBJ5));
103     
104     grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ1);
105 
106     //try as another user
107     group1A.addMember(SubjectTestHelper.SUBJ5);
108 
109     assertFalse(group1B.hasMember(SubjectTestHelper.SUBJ5));
110     assertTrue(group1A.hasMember(SubjectTestHelper.SUBJ5));
111     assertTrue(group1A.hasMember(SubjectTestHelper.SUBJ6));
112     assertTrue(group1B.hasMember(SubjectTestHelper.SUBJ8));
113 
114   }

119   public void testVetoIfNotAllowed() {
120     
121     GrouperSession grouperSession = GrouperSession.startRootSession();
122 
123     //allow all to attribute
124     MembershipOneInFolderMaxHook.membershipOneInFolderAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_READ, false);
125     MembershipOneInFolderMaxHook.membershipOneInFolderAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_UPDATE, false);
126     
127     //has the attribute
128     Stem testStem1 = new StemSave(grouperSession).assignName("test1").save();
129 
130     testStem1.grantPriv(SubjectTestHelper.SUBJ0, NamingPrivilege.CREATE);
131     testStem1.grantPriv(SubjectTestHelper.SUBJ0, NamingPrivilege.STEM);
132 
133     testStem1.grantPriv(SubjectTestHelper.SUBJ1, NamingPrivilege.CREATE);
134     testStem1.grantPriv(SubjectTestHelper.SUBJ1, NamingPrivilege.STEM);
135 
136     GrouperSession.stopQuietly(grouperSession);
137 
138     grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ0);
139 
140     Group group1A = new GroupSave(grouperSession).assignCreateParentStemsIfNotExist(true)
141         .assignName("test1:someGroupA").save();
142 
143     Group group1B = new GroupSave(grouperSession).assignCreateParentStemsIfNotExist(true)
144         .assignName("test1:someGroupB").save();
145 
146     //apply the attribute
147     MembershipOneInFolderMaxHook.assignMembershipOneInFolderAttributeDefName(testStem1);
148     
149     group1A.addMember(SubjectTestHelper.SUBJ5);
150     group1A.addMember(SubjectTestHelper.SUBJ6);
151     group1B.addMember(SubjectTestHelper.SUBJ8);
152     
153     grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ1);
154 
155     Group group1C = new GroupSave(grouperSession).assignCreateParentStemsIfNotExist(true)
156         .assignName("test1:someGroupC").save();
157 
158     try {
159       
160       group1C.addMember(SubjectTestHelper.SUBJ5);
161 
162     } catch (HookVeto hv) {
163       //this is a success, it is supposed to veto  
164       assertEquals(hv.getReasonKey(), MembershipOneInFolderMaxHook.HOOK_VETO_MEMBERSHIP_ONE_IN_FOLDER_CANT_DELETE_MEMBER);
165     }
166 
167     assertFalse(group1B.hasMember(SubjectTestHelper.SUBJ5));
168     assertTrue(group1A.hasMember(SubjectTestHelper.SUBJ5));
169     assertTrue(group1A.hasMember(SubjectTestHelper.SUBJ6));
170     assertTrue(group1B.hasMember(SubjectTestHelper.SUBJ8));
171 
172   }

177   public void testGroupRemovesMultipleMemberships() {
178     
179     GrouperSession grouperSession = GrouperSession.startRootSession();
180 
181     //allow all to attribute
182     MembershipOneInFolderMaxHook.membershipOneInFolderAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_READ, false);
183     MembershipOneInFolderMaxHook.membershipOneInFolderAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(SubjectFinder.findAllSubject(), AttributeDefPrivilege.ATTR_UPDATE, false);
184     
185     //has the attribute
186     Stem testStem1 = new StemSave(grouperSession).assignName("test1").save();
187 
188     testStem1.grantPriv(SubjectTestHelper.SUBJ0, NamingPrivilege.CREATE);
189     testStem1.grantPriv(SubjectTestHelper.SUBJ0, NamingPrivilege.STEM);
190 
191     testStem1.grantPriv(SubjectTestHelper.SUBJ1, NamingPrivilege.CREATE);
192     testStem1.grantPriv(SubjectTestHelper.SUBJ1, NamingPrivilege.STEM);
193 
194     GrouperSession.stopQuietly(grouperSession);
195 
196     grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ0);
197 
198     Group group1A = new GroupSave(grouperSession).assignCreateParentStemsIfNotExist(true)
199         .assignName("test1:someGroupA").save();
200 
201     group1A.grantPriv(SubjectTestHelper.SUBJ1, AccessPrivilege.ADMIN, false);
202     
203     Group group1B = new GroupSave(grouperSession).assignCreateParentStemsIfNotExist(true)
204         .assignName("test1:someGroupB").save();
205 
206     group1B.grantPriv(SubjectTestHelper.SUBJ1, AccessPrivilege.ADMIN, false);
207 
208     Group group1C = new GroupSave(grouperSession).assignCreateParentStemsIfNotExist(true)
209         .assignName("test1:someGroupC").save();
210 
211     group1C.grantPriv(SubjectTestHelper.SUBJ1, AccessPrivilege.ADMIN, false);
212 
213     group1A.addMember(SubjectTestHelper.SUBJ5);
214     group1A.addMember(SubjectTestHelper.SUBJ6);
215     group1B.addMember(SubjectTestHelper.SUBJ8);
216     group1B.addMember(SubjectTestHelper.SUBJ5);
217 
218     //apply the attribute
219     MembershipOneInFolderMaxHook.assignMembershipOneInFolderAttributeDefName(testStem1);
220 
221     group1C.addMember(SubjectTestHelper.SUBJ5);
222 
223     assertFalse(group1A.hasMember(SubjectTestHelper.SUBJ5));
224     assertFalse(group1B.hasMember(SubjectTestHelper.SUBJ5));
225     assertTrue(group1A.hasMember(SubjectTestHelper.SUBJ6));
226     assertTrue(group1B.hasMember(SubjectTestHelper.SUBJ8));
227     assertTrue(group1C.hasMember(SubjectTestHelper.SUBJ5));
228 
229     grouperSession = GrouperSession.start(SubjectTestHelper.SUBJ1);
230 
231     //try as another user
232     group1A.addMember(SubjectTestHelper.SUBJ5);
233 
234     assertFalse(group1B.hasMember(SubjectTestHelper.SUBJ5));
235     assertFalse(group1C.hasMember(SubjectTestHelper.SUBJ5));
236     assertTrue(group1A.hasMember(SubjectTestHelper.SUBJ5));
237     assertTrue(group1A.hasMember(SubjectTestHelper.SUBJ6));
238     assertTrue(group1B.hasMember(SubjectTestHelper.SUBJ8));
239 
240   }
