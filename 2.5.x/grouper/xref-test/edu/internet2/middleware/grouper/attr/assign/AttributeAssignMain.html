152   private static void attributeAssignExample() {
153     GrouperSession grouperSession = GrouperSession.startRootSession();
154     Set attributeAssignIdsAlreadyUsed = new HashSet();
155     {
156       boolean problemWithAttributeAssign = false;
157       AttributeAssignSave attributeAssignSave = new AttributeAssignSave(grouperSession);
158       attributeAssignSave.assignAttributeAssignIdsToNotUse(attributeAssignIdsAlreadyUsed);
159       attributeAssignSave.assignAttributeAssignType(AttributeAssignType.group);
160       AttributeDefName attributeDefName = AttributeDefNameFinder.findByName(
161           "test:attributeDefName0", false);
162       if (attributeDefName == null) {
163         System.out.println("Error: cant find attributeDefName: test:attributeDefName0");
164         problemWithAttributeAssign = true;
165       }
166       attributeAssignSave.assignAttributeDefName(attributeDefName);
167       Group ownerGroup = GroupFinder.findByName(grouperSession, "test:group0", false);
168       if (ownerGroup == null) {
169         System.out.println("Error: cant find group: test:group0");
170         problemWithAttributeAssign = true;
171       }
172       attributeAssignSave.assignOwnerGroup(ownerGroup);
173       attributeAssignSave.assignPutAttributeAssignIdsToNotUseSet(true);
174       if (!problemWithAttributeAssign) {
175         AttributeAssign attributeAssign = attributeAssignSave.save();
176         System.out.println("Made " + attributeAssignSave.getChangesCount() + " changes for attribute assign: " + attributeAssign.toString());
177       }
178     }
179     {
180       boolean problemWithAttributeAssign = false;
181       AttributeAssignSave attributeAssignSave = new AttributeAssignSave(grouperSession);
182       attributeAssignSave.assignAttributeAssignIdsToNotUse(attributeAssignIdsAlreadyUsed);
183       attributeAssignSave.assignAttributeAssignType(AttributeAssignType.group);
184       AttributeDefName attributeDefName = AttributeDefNameFinder.findByName(
185           "test:attributeDefName0", false);
186       if (attributeDefName == null) {
187         System.out.println("Error: cant find attributeDefName: test:attributeDefName0");
188         problemWithAttributeAssign = true;
189       }
190       attributeAssignSave.assignAttributeDefName(attributeDefName);
191       Group ownerGroup = GroupFinder.findByName(grouperSession, "test:group0", false);
192       if (ownerGroup == null) {
193         System.out.println("Error: cant find group: test:group0");
194         problemWithAttributeAssign = true;
195       }
196       attributeAssignSave.assignOwnerGroup(ownerGroup);
197       attributeAssignSave.assignPutAttributeAssignIdsToNotUseSet(true);
198       if (!problemWithAttributeAssign) {
199         AttributeAssign attributeAssign = attributeAssignSave.save();
200         System.out.println("Made " + attributeAssignSave.getChangesCount() + " changes for attribute assign: " + attributeAssign.toString());
201       }
202     }
