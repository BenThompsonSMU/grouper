506   public void testMapEquals() {
507     
508     Map<String, String> first = null;
509     Map<String, String> second = null;
510     
511     assertTrue("nulls should be equal", GrouperUtil.mapEquals(first, second));
512     
513     first = new HashMap<String, String>();
514     
515     assertTrue("null is equal to empty", GrouperUtil.mapEquals(first, second));
516     
517     first.put("key1", "value1");
518     
519     assertFalse("null is not empty to map with size", GrouperUtil.mapEquals(first, second));
520     assertTrue("map is equal to self", GrouperUtil.mapEquals(first, first));
521     
522     second = new HashMap<String, String>();
523     second.put("key1", "value2");
524     
525     assertFalse("not equal if same size, different values", GrouperUtil.mapEquals(first, second));
526     
527     second.put("key1", "value1");
528     
529     assertTrue("equal if same size, same keys/values", GrouperUtil.mapEquals(first, second));
530     
531     first.put("key2", "value2");
532     assertFalse("not equal if different size", GrouperUtil.mapEquals(first, second));
533     assertFalse("not equal if different size", GrouperUtil.mapEquals(second, first));
534     
535     second.put("key2", "value2");
536     assertTrue("equal if same size, same keys/values", GrouperUtil.mapEquals(first, second));
537     
538     second.put("key3", "value2");
539     second.remove("key2");
540     assertFalse("not equal if same size, different keys", GrouperUtil.mapEquals(first, second));
541     assertFalse("not equal if same size, different keys(reversed)", GrouperUtil.mapEquals(second, first));
542   }

544   /**
545    * see if testing for equal maps work
546    */
547   public void testMapDifferences() {
548     
549     Map<String, String> first = null;
550     Map<String, String> second = null;
551     Set<String> differences = new LinkedHashSet<String>();
552     
553     GrouperUtil.mapDifferences(first, second, differences, "attribute__");
554 
555     assertTrue("nulls should be equal", differences.size() == 0);
556     
557     first = new LinkedHashMap<String, String>();
558     GrouperUtil.mapDifferences(first, second, differences, "attribute__");
559     
560     assertEquals("null is equal to empty", 0, differences.size());
561     
562     first.put("key1", "value1");
563     GrouperUtil.mapDifferences(first, second, differences, "attribute__");
564     
565     assertEquals("First should have a size of 1", 1, first.size());
566     
567     assertEquals("null is not equal to map with size", 1, differences.size());
568     assertEquals("null is not equal to map with size", (String)differences.toArray()[0],"attribute__key1");
569 
570     differences.clear();
571     //try reversed
572     GrouperUtil.mapDifferences(second, first, differences, "attribute__");
573 
574     assertEquals("null is not equal to map with size(reversed)", 1, differences.size());
575     assertEquals("null is not equal to map with size(reversed)", (String)differences.toArray()[0],"attribute__key1");
576     
577     differences.clear();
578     GrouperUtil.mapDifferences(first, first, differences, "attribute__");
579     assertEquals("map equal to self", 0, differences.size());
580     
581     GrouperUtil.mapDifferences(first, new HashMap<String, String>(first), differences, "attribute__");
582     assertEquals("map equal to clone self", 0, differences.size());
583     
584     second = new LinkedHashMap<String, String>();
585     second.put("key1", "value2");
586
