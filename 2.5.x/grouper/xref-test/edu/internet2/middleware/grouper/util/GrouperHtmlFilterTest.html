20  package edu.internet2.middleware.grouper.util;
21  
22  import junit.textui.TestRunner;
23  import edu.internet2.middleware.grouper.helper.GrouperTest;
24  
25  
26  /**
27   *
28   */
29  public class GrouperHtmlFilterTest extends GrouperTest {
30  
31    /**
32     * 
33     * @param args
34     */
35    public static void main(String[] args) {
36      TestRunner.run(new GrouperHtmlFilterTest("testFilter"));
37    }
38    
39    /**
40     * 
41     * @param name
42     */
43    public GrouperHtmlFilterTest(String name) {
44      super(name);
45    }
46  
47    /**
48     * 
49     */
50    public void testFilter() {
51      assertEquals("hey", new GrouperHtmlFilter().filterHtml("hey"));
52      assertEquals("hey&lt;", new GrouperHtmlFilter().filterHtml("hey<"));
53      assertEquals("h&gt;ey", new GrouperHtmlFilter().filterHtml("h&gt;ey"));
54      assertEquals("hey&lt;\n&gt;", new GrouperHtmlFilter().filterHtml("hey<\n>"));
55      assertEquals("h<b>e</b>y", new GrouperHtmlFilter().filterHtml("h<b>e</b>y"));
56      assertEquals("h<BR />ey", new GrouperHtmlFilter().filterHtml("h<BR />ey"));
57      assertEquals("h<b>e</b>y", new GrouperHtmlFilter().filterHtml("h<b>e</b>y"));
58      assertEquals("h&lt;script&gt;e&lt;/script&gt;y", new GrouperHtmlFilter().filterHtml("h<script>e</script>y"));
59      assertEquals("h&lt;form&gt;e&lt;/form&gt;y", new GrouperHtmlFilter().filterHtml("h<form>e</form>y"));
60      assertEquals("h&lt;a&gt;e&lt;/a&gt;y", new GrouperHtmlFilter().filterHtml("h<a>e</a>y"));
61      assertEquals("h&lt;button&gt;e&lt;/button&gt;y", new GrouperHtmlFilter().filterHtml("h<button>e</button>y"));
62      assertEquals("h<div>e</div>y", new GrouperHtmlFilter().filterHtml("h<div>e</div>y"));
63      assertEquals("h&lt;h<div>e</div>y", new GrouperHtmlFilter().filterHtml("h<h<div>e</div>y"));
64      assertEquals("h&lt;h<div>e</div>y", new GrouperHtmlFilter().filterHtml("h&lt;h<div>e</div>y"));
65      assertEquals("h&lt;div onclick=''&gt;e</div>y", new GrouperHtmlFilter().filterHtml("h<div onclick=''>e</div>y"));
66    }
67    
68  }
