84     */
85    public void testConvertToXml() {
86      
87      XmlExportAuditType xmlExportAuditType = new XmlExportAuditType();
88      
89      xmlExportAuditType.setActionName("actionName");
90      xmlExportAuditType.setAuditCategory("auditCategory");
91      xmlExportAuditType.setContextId("contextId");
92      xmlExportAuditType.setCreatedOn("createdOn");
93      xmlExportAuditType.setHibernateVersionNumber(3L);
94      xmlExportAuditType.setId("id");
95      xmlExportAuditType.setLabelInt01("labelInt01");
96      xmlExportAuditType.setLabelInt02("labelInt02");
97      xmlExportAuditType.setLabelInt03("labelInt03");
98      xmlExportAuditType.setLabelInt04("labelInt04");
99      xmlExportAuditType.setLabelInt05("labelInt05");
100     xmlExportAuditType.setLabelString01("labelString01");
101     xmlExportAuditType.setLabelString02("labelString02");
102     xmlExportAuditType.setLabelString03("labelString03");
103     xmlExportAuditType.setLabelString04("labelString04");
104     xmlExportAuditType.setLabelString05("labelString05");
105     xmlExportAuditType.setLabelString06("labelString06");
106     xmlExportAuditType.setLabelString07("labelString07");
107     xmlExportAuditType.setLabelString08("labelString08");
108     xmlExportAuditType.setLastUpdated("lastUpdated");
109     String xml = xmlExportAuditType.toXml(new GrouperVersion(GrouperVersion.GROUPER_VERSION));
110     
111     xmlExportAuditType = XmlExportAuditType.fromXml(new GrouperVersion(GrouperVersion.GROUPER_VERSION), xml);
112     
113     assertEquals("actionName", xmlExportAuditType.getActionName());
114     assertEquals("auditCategory", xmlExportAuditType.getAuditCategory());
115     assertEquals("contextId", xmlExportAuditType.getContextId());
116     assertEquals("createdOn", xmlExportAuditType.getCreatedOn());
117     assertEquals(3L, xmlExportAuditType.getHibernateVersionNumber());
118     assertEquals("id", xmlExportAuditType.getId());
119     assertEquals("labelInt01", xmlExportAuditType.getLabelInt01());
120     assertEquals("labelInt02", xmlExportAuditType.getLabelInt02());
121     assertEquals("labelInt03", xmlExportAuditType.getLabelInt03());
122     assertEquals("labelInt04", xmlExportAuditType.getLabelInt04());
123     assertEquals("labelInt05", xmlExportAuditType.getLabelInt05());
124     assertEquals("labelString01", xmlExportAuditType.getLabelString01());
125     assertEquals("labelString02", xmlExportAuditType.getLabelString02());
126     assertEquals("labelString03", xmlExportAuditType.getLabelString03());
127     assertEquals("labelString04", xmlExportAuditType.getLabelString04());
128     assertEquals("labelString05", xmlExportAuditType.getLabelString05());
129     assertEquals("labelString06", xmlExportAuditType.getLabelString06());
130     assertEquals("labelString07", xmlExportAuditType.getLabelString07());
131     assertEquals("labelString08", xmlExportAuditType.getLabelString08());
132     assertEquals("lastUpdated", xmlExportAuditType.getLastUpdated());
133 
134   }

139   public void testConvertToAuditType() {
140     AuditType auditType = AuditTypeTest.exampleAuditType();
141     
142     XmlExportAuditType xmlExportAuditType = auditType.xmlToExportAuditType(new GrouperVersion(GrouperVersion.GROUPER_VERSION));
143 
144     //now go back
145     auditType = xmlExportAuditType.toAuditType();
146     
147     assertEquals("actionName", auditType.getActionName());
148     assertEquals("auditCategory", auditType.getAuditCategory());
149     assertEquals("contextId", auditType.getContextId());
150     assertEquals(new Long(3L), auditType.getCreatedOnDb());
151     assertEquals(new Long(4L), auditType.getHibernateVersionNumber());
152     assertEquals("id", auditType.getId());
153     assertEquals("labelInt01", auditType.getLabelInt01());
154     assertEquals("labelInt02", auditType.getLabelInt02());
155     assertEquals("labelInt03", auditType.getLabelInt03());
156     assertEquals("labelInt04", auditType.getLabelInt04());
157     assertEquals("labelInt05", auditType.getLabelInt05());
158     assertEquals("labelString01", auditType.getLabelString01());
159     assertEquals("labelString02", auditType.getLabelString02());
160     assertEquals("labelString03", auditType.getLabelString03());
