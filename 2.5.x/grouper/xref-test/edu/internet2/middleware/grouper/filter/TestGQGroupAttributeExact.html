94    public void testGroupAttributeExactFilterNothing() {
95      GrouperSession  s     = SessionHelper.getRootSession();
96      Stem            root  = StemHelper.findRootStem(s);
97      Stem            edu   = StemHelper.addChildStem(root, "edu", "education");
98      Group           i2    = StemHelper.addChildGroup(edu, "i2", "internet2");
99      Group           uofc  = StemHelper.addChildGroup(edu, "uofc", "uchicago");
100     Stem            com   = StemHelper.addChildStem(root, "com", "commercial");
101     StemHelper.addChildGroup(com, "devclue", "devclue");
102     GroupHelper.addMember(i2, uofc);
103     try {
104       GrouperQuery gq = GrouperQuery.createQuery(
105         s, new GroupAttributeExactFilter("name", "nothing", root)
106       );
107       Assert.assertTrue("groups",  gq.getGroups().size()      == 0);
108       Assert.assertTrue("members", gq.getMembers().size()     == 0);
109       Assert.assertTrue("mships",  gq.getMemberships().size() == 0);
110       Assert.assertTrue("stems",   gq.getStems().size()       == 0);
111     }
112     catch (QueryException eQ) {
113       Assert.fail("unable to query: " + eQ.getMessage());
114     }
115   }

141     GrouperSession  s     = SessionHelper.getRootSession();

143     Stem            edu   = StemHelper.addChildStem(root, "edu", "education");

162     }

181     uofc.addType(custom);
182     devclue.setAttribute("customAttribute", "String with i2 within");
183     uofc.setAttribute("customAttribute", "String with i2 within");

215   public void testFindByAttributeLegacyAttributeRoot() {
216     GrouperSession grouperSession = GrouperSession.startRootSession();
217     Stem root = StemHelper.findRootStem(grouperSession);
218     Group group1 = new GroupSave(grouperSession).assignName("test:group1").assignCreateParentStemsIfNotExist(true).save();
219     Group group2 = new GroupSave(grouperSession).assignName("test:group2").assignCreateParentStemsIfNotExist(true).save();
220     Group group3 = new GroupSave(grouperSession).assignName("test:group3").assignCreateParentStemsIfNotExist(true).save();
221 
222     GroupType type1 = GroupType.createType(grouperSession, "type1");
223     GroupType type2 = GroupType.createType(grouperSession, "type2");
224     
225     AttributeDefName type1Attr1 = type1.addAttribute(grouperSession, "type1Attr1");
226     AttributeDefName type1Attr2 = type1.addAttribute(grouperSession, "type1Attr2");
227     AttributeDefName type2Attr1 = type2.addAttribute(grouperSession, "type2Attr1");
228     AttributeDefName type2Attr2 = type2.addAttribute(grouperSession, "type2Attr2");
229     
230     group1.addType(type1);
231     group1.addType(type2);
232     group2.addType(type2);
233     
234     group1.setAttribute("type1Attr1", "test value 1");
235     group1.setAttribute("type1Attr2", "test value 2");
236     group1.setAttribute("type2Attr2", "test value 3");
237     group2.setAttribute("type2Attr2", "test value 4");
238     
239     GrouperQuery gq = GrouperQuery.createQuery(grouperSession, new GroupAttributeExactFilter("type1Attr1", "test value 1", root));
240     Assert.assertEquals(1, gq.getGroups().size());
241     Assert.assertEquals(group1.getName(), gq.getGroups().iterator().next().getName());
242     
243     gq = GrouperQuery.createQuery(grouperSession, new GroupAttributeExactFilter("type1Attr2", "test value 2", root));
244     Assert.assertEquals(1, gq.getGroups().size());
245     Assert.assertEquals(group1.getName(), gq.getGroups().iterator().next().getName());
246     
247     gq = GrouperQuery.createQuery(grouperSession, new GroupAttributeExactFilter("type2Attr2", "test value 3", root));
248     Assert.assertEquals(1, gq.getGroups().size());
249     Assert.assertEquals(group1.getName(), gq.getGroups().iterator().next().getName());
250     
251     gq = GrouperQuery.createQuery(grouperSession, new GroupAttributeExactFilter("type2Attr2", "test value 4", root));
252     Assert.assertEquals(1, gq.getGroups().size());
253     Assert.assertEquals(group2.getName(), gq.getGroups().iterator().next().getName());
254     
255     group1.setAttribute("type2Attr2", "test value 4");
256     
257     gq = GrouperQuery.createQuery(grouperSession, new GroupAttributeExactFilter("type2Attr2", "test value 4", root));
258     Assert.assertEquals(2, gq.getGroups().size());
259   }

264   public void testSecurity() {
265     GrouperSession  s     = SessionHelper.getRootSession();
266     Stem            root  = StemHelper.findRootStem(s);
267     Stem            edu   = StemHelper.addChildStem(root, "edu", "education");
268     Group           i2    = StemHelper.addChildGroup(edu, "i2", "internet2");
269     Group           uofc  = StemHelper.addChildGroup(edu, "uofc", "uchicago");
270     Group           duke  = StemHelper.addChildGroup(edu, "duke", "Duke University");
271     Stem            com   = StemHelper.addChildStem(root, "com", "commercial");
272     Group           devclue = StemHelper.addChildGroup(com, "devclue", "devclue");
273     Group           dev2 = StemHelper.addChildGroup(com, "dev2", "dev2");
274 
275     uofc.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.READ, false);
276     uofc.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.VIEW, false);
277     i2.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.READ, false);
278     i2.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.VIEW, false);
279     duke.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.READ, false);
280     duke.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.VIEW, false);
281     devclue.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.READ, false);
282     devclue.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.VIEW, false);
283     dev2.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.READ, false);
284     dev2.revokePriv(SubjectFinder.findAllSubject(), AccessPrivilege.VIEW, false);
285     
286     GroupType custom = GroupType.createType(s, "customType");
287     custom.addAttribute(s, "customAttribute", false);
288 
289     devclue.addType(custom);
290     uofc.addType(custom);
291     dev2.addType(custom);
292     devclue.setAttribute("customAttribute", "String with i2 within");
293     uofc.setAttribute("customAttribute", "String with i2 within");
294     dev2.setAttribute("customAttribute", "String with i2 within");
295 
296     devclue.grantPriv(SubjectTestHelper.SUBJ0, AccessPrivilege.GROUP_ATTR_READ);
297     custom.getAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_READ, false);
298     custom.internal_getAttributeDefForAttributes().getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_READ, false);
299     s.stop();
300     s = GrouperSession.start(SubjectTestHelper.SUBJ0);
301     GrouperQuery gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("customAttribute", "String with i2 within", root));
302     Assert.assertEquals(1, gq.getGroups().size());
303     Assert.assertEquals(devclue.getName(), gq.getGroups().iterator().next().getName());
304     
305     s.stop();
306     s = GrouperSession.startRootSession();
307     devclue.revokePriv(SubjectTestHelper.SUBJ0, AccessPrivilege.GROUP_ATTR_READ);
308     s.stop();
309     s = GrouperSession.start(SubjectTestHelper.SUBJ0);
310     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("customAttribute", "String with i2 within", root));
311     Assert.assertEquals(0, gq.getGroups().size());
312     
313     s.stop();
314     s = GrouperSession.startRootSession();
315     devclue.grantPriv(SubjectTestHelper.SUBJ0, AccessPrivilege.GROUP_ATTR_READ);
316     custom.getAttributeDefName().getAttributeDef().getPrivilegeDelegate().revokePriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_READ, true);
317     s.stop();
318     s = GrouperSession.start(SubjectTestHelper.SUBJ0);
319     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("customAttribute", "String with i2 within", root));
320     Assert.assertEquals(0, gq.getGroups().size());
321     
322     s.stop();
323     s = GrouperSession.startRootSession();
324     custom.getAttributeDefName().getAttributeDef().getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_READ, true);
325     custom.internal_getAttributeDefForAttributes().getPrivilegeDelegate().revokePriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_READ, true);
326     s.stop();
327     s = GrouperSession.start(SubjectTestHelper.SUBJ0);
328     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("customAttribute", "String with i2 within", root));
329     Assert.assertEquals(0, gq.getGroups().size());
330     
331     s.stop();
332     s = GrouperSession.startRootSession();
333     custom.internal_getAttributeDefForAttributes().getPrivilegeDelegate().grantPriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_READ, true);
334     s.stop();
335     s = GrouperSession.start(SubjectTestHelper.SUBJ0);
336     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("customAttribute", "String with i2 within", root));
337     Assert.assertEquals(1, gq.getGroups().size());
338     
339     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("etc:legacy:attribute:legacyAttribute_customAttribute", "String with i2 within", root));
340     Assert.assertEquals(1, gq.getGroups().size());
341     
342     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclue", root));
343     Assert.assertEquals(1, gq.getGroups().size());
344     
345     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclue", com));
346     Assert.assertEquals(1, gq.getGroups().size());
347     
348     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "dev", com));
349     Assert.assertEquals(0, gq.getGroups().size());
350     
351     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclux", com));
352     Assert.assertEquals(0, gq.getGroups().size());
353     
354     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclue", edu));
355     Assert.assertEquals(0, gq.getGroups().size());
356     
357     // you really only need view permissions on the group for the normal group attrs to work
358     s.stop();
359     s = GrouperSession.startRootSession();
360     devclue.revokePriv(SubjectTestHelper.SUBJ0, AccessPrivilege.GROUP_ATTR_READ);
361     custom.getAttributeDefName().getAttributeDef().getPrivilegeDelegate().revokePriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_READ, false);
362     custom.internal_getAttributeDefForAttributes().getPrivilegeDelegate().revokePriv(SubjectTestHelper.SUBJ0, AttributeDefPrivilege.ATTR_READ, false);
363 
364     s.stop();
365     s = GrouperSession.start(SubjectTestHelper.SUBJ0);
366     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclue", root));
367     Assert.assertEquals(0, gq.getGroups().size());
368     
369     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclue", com));
370     Assert.assertEquals(0, gq.getGroups().size());
371     
372     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "dev", com));
373     Assert.assertEquals(0, gq.getGroups().size());
374     
375     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclux", com));
376     Assert.assertEquals(0, gq.getGroups().size());
377     
378     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclue", edu));
379     Assert.assertEquals(0, gq.getGroups().size());
380     
381     s.stop();
382     s = GrouperSession.startRootSession();
383     devclue.grantPriv(SubjectFinder.findAllSubject(), AccessPrivilege.VIEW);
384     dev2.grantPriv(SubjectTestHelper.SUBJ0, AccessPrivilege.VIEW);
385 
386     s.stop();
387     s = GrouperSession.start(SubjectTestHelper.SUBJ0);
388     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclue", root));
389     Assert.assertEquals(1, gq.getGroups().size());
390     
391     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclue", com));
392     Assert.assertEquals(1, gq.getGroups().size());
393     
394     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "dev", root));
395     Assert.assertEquals(0, gq.getGroups().size());
396     
397     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "dev", com));
398     Assert.assertEquals(0, gq.getGroups().size());
399     
400     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclux", com));
401     Assert.assertEquals(0, gq.getGroups().size());
402     
403     gq = GrouperQuery.createQuery(s, new GroupAttributeExactFilter("extension", "devclue", edu));
404     Assert.assertEquals(0, gq.getGroups().size());
