62    public void testReplaceWithSubset() {
63      
64      UserDataObject userDataObject0 = new UserDataObject(GrouperUuid.getUuid(), System.currentTimeMillis());
65      UserDataObject userDataObject1 = new UserDataObject(GrouperUuid.getUuid(), System.currentTimeMillis()+1);
66      UserDataObject userDataObject2 = new UserDataObject(GrouperUuid.getUuid(), System.currentTimeMillis()+2);
67      UserDataObject userDataObject3 = new UserDataObject(userDataObject0.getUuid(), System.currentTimeMillis()+3);
68      
69      UserDataObject[] list = new UserDataObject[]{userDataObject0, userDataObject1, userDataObject2, userDataObject3};
70  
71      UserDataList userDataList = new UserDataList(list);
72  
73      Set<String> subset = GrouperUtil.toSet(userDataObject0.getUuid(), userDataObject1.getUuid());
74      
75      assertTrue(userDataList.replaceUserDataObjectsWithSubset(subset, 5));
76      
77      //lets check it out
78      assertEquals(2, userDataList.getList().length);
79      assertEquals(userDataObject0.getUuid(), userDataList.getList()[0].getUuid());
80      assertEquals(userDataObject0.getTimestamp(), userDataList.getList()[0].getTimestamp());
81  
82      assertEquals(userDataObject1.getUuid(), userDataList.getList()[1].getUuid());
83      assertEquals(userDataObject1.getTimestamp(), userDataList.getList()[1].getTimestamp());
84  
85      //no changes
86      assertFalse(userDataList.replaceUserDataObjectsWithSubset(subset, 5));
87  
88      
89      assertTrue(userDataList.replaceUserDataObjectsWithSubset(subset, 1));
90      assertEquals(1, userDataList.getList().length);
91      assertEquals(userDataObject0.getUuid(), userDataList.getList()[0].getUuid());
92      assertEquals(userDataObject0.getTimestamp(), userDataList.getList()[0].getTimestamp());
93      
94    }

98     */
99    public void testJsonMarshal() {
100     
101     UserDataObject userDataObject0 = new UserDataObject(GrouperUuid.getUuid(), System.currentTimeMillis());
102     UserDataObject userDataObject1 = new UserDataObject(GrouperUuid.getUuid(), System.currentTimeMillis()+1);
103     UserDataObject userDataObject2 = new UserDataObject(GrouperUuid.getUuid(), System.currentTimeMillis()+2);
104     
105     UserDataObject[] list = new UserDataObject[]{userDataObject0, userDataObject1, userDataObject2};
106 
107     UserDataList userDataList = new UserDataList(list);
108     
109     //convert to json:
110     String json = userDataList.jsonConvertTo();
111     
112     //this should be generic and not tied to the object types
113     assertFalse(json, json.toLowerCase().contains("user"));
114     
115     // this is 3 objects: {"list":[{"timestamp":1363829748082,"uuid":"f7994071e2cf4cb4b29b71016cb90a08"},{"timestamp":1363829748083,"uuid":"c92a4c8c4e7f473daf98c318e20f1355"},{"timestamp":1363829748084,"uuid":"63d66988e2a64ed38258f0a71486d21c"}]}
116     // which is size 230.  So 30 objects can comfortably fit in 4k of space
117     //System.out.println(json);
118     //System.out.println(GrouperUtil.indent(json, true));
119     
120     //lets make sure it comes back ok
121     userDataList = UserDataList.jsonMarshalFrom(json);
122     
123     assertEquals(3, GrouperUtil.length(userDataList.getList()));
124     
125     assertEquals(json, userDataObject0.getUuid(), userDataList.getList()[0].getUuid());
126     assertEquals(json, userDataObject1.getUuid(), userDataList.getList()[1].getUuid());
127     assertEquals(json, userDataObject1.getTimestamp(), userDataList.getList()[1].getTimestamp());
128     assertEquals(json, userDataObject2.getTimestamp(), userDataList.getList()[2].getTimestamp());
129     
130     //make sure no extraneous data or missing data messes it up
131     String string = "{\"list\":[{\"timestamp\":1363827648860,\"abcd\":\"qwert\"}]}";
132 
133     userDataList = UserDataList.jsonMarshalFrom(string);
134     
135     assertEquals(1, GrouperUtil.length(userDataList));
136     assertEquals(new Long(1363827648860L), userDataList.getList()[0].getTimestamp());
137     assertNull(userDataList.getList()[0].getUuid());
138 
139     list = new UserDataObject[20];
140     userDataList = new UserDataList(list);
141     
142     for (int i=0;i<20;i++) {
143       UserDataObject userDataObject = new UserDataObject(GrouperUuid.getUuid(), System.currentTimeMillis());
144       list[i] = userDataObject;
145     }
146     
147     //see how long 100 conversions takes
148     //100 conversions of array of size 3 takes 180ms
149     //100 conversions of array of size 20 takes 554ms
150     long start = System.nanoTime();
151 
152     for (int i=0;i<100;i++) {
153     
154       userDataList = new UserDataList(list);
155 
156       //convert to json:
157       json = userDataList.jsonConvertTo();
158       
159       userDataList = UserDataList.jsonMarshalFrom(json);
160       
161       assertEquals(20, GrouperUtil.length(userDataList.getList()));
162     }
163     
164     //System.out.println("100 conversions there and back took: " + ((System.nanoTime()-start) / 1000000L) + "ms");
165   }
