45    public static void dropOrgDataTable() {
46      
47      //we need to delete the test table if it is there, and create a new one
48      //drop field id col, first drop foreign keys
49      GrouperDdlUtils.changeDatabase(GrouperTestDdl.V1.getObjectName(), new DdlUtilsChangeDatabase() {
50  
51        public void changeDatabase(DdlVersionBean ddlVersionBean) {
52          
53          Database database = ddlVersionBean.getDatabase();
54  
55          {
56            Table loaderTable = database.findTable("testgrouper_loader");
57            
58            if (loaderTable != null) {
59              database.removeTable(loaderTable);
60            }
61          }
62          {
63            Table loaderGroupsTable = database.findTable("testgrouper_loader_groups");
64            
65            if (loaderGroupsTable != null) {
66              database.removeTable(loaderGroupsTable);
67            }
68            
69          }
70          
71        }
72        
73      });
74  
75    }
