112    * an AssertionFailedError is thrown with the given message.
113    */
114   static public void assertEquals(String message, Subject expected, Subject actual) {
115     if (expected == null && actual == null)
116       return;
117     if (expected != null && SubjectHelper.eq(expected,actual))
118       return;
119     failNotEquals(message, expected, actual);

284   public static String groupsString(Collection<Group> groups) {
285     
286     StringBuilder groupsString = new StringBuilder(GrouperUtil.length(groups) + " groups: " );
287 
288     int i=0;
289     
290     if (GrouperUtil.length(groups) == 0) {
291       groupsString.append(" <none>");
292       return groupsString.toString();
293     }
294     
295     for (Group current : groups) {
296 
297       if (i != 0) {
298         groupsString.append(", ");
299       }
300       
301       groupsString.append(current.getName());
302 
303       if (i>100) {
304         groupsString.append(", and " + (GrouperUtil.length(groups) - 100) + " more groups...");
305         break;
306       }
307       
308       i++;
309     }
310 
311     return groupsString.toString();
312     
313   }
